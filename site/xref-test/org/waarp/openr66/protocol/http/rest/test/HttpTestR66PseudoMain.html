41      public static RestConfiguration config;
42  
43      public static RestConfiguration getTestConfiguration() throws CryptoException, IOException {
44          RestConfiguration configuration = new RestConfiguration();
45          configuration.REST_PORT = 8088;
46          configuration.REST_SSL = false;
47          configuration.RESTHANDLERS_CRUD = new byte[RESTHANDLERS.values().length];
48          for (int i = 0; i < configuration.RESTHANDLERS_CRUD.length; i++) {
49              configuration.RESTHANDLERS_CRUD[i] = RestConfiguration.CRUD.ALL.mask;
50          }
51          configuration.REST_AUTHENTICATED = true;
52          configuration.initializeKey(new File("/opt/R66/certs/key.sha256"));
53          configuration.REST_TIME_LIMIT = 10000;
54          configuration.REST_SIGNATURE = true;
55          configuration.REST_ADDRESS = "127.0.0.1";
56          return configuration;
57      }
58  
59      public static RestConfiguration getTestConfiguration2() throws CryptoException, IOException {
60          RestConfiguration configuration = new RestConfiguration();
61          configuration.REST_PORT = 8089;
62          configuration.REST_SSL = false;
63          configuration.RESTHANDLERS_CRUD = new byte[RESTHANDLERS.values().length];
64          for (int i = 0; i < configuration.RESTHANDLERS_CRUD.length; i++) {
65              configuration.RESTHANDLERS_CRUD[i] = RestConfiguration.CRUD.READ.mask;
66          }
67          configuration.REST_AUTHENTICATED = false;
68          configuration.REST_TIME_LIMIT = 100000;
69          configuration.REST_SIGNATURE = false;
70          configuration.REST_ADDRESS = "127.0.0.1";
71          return configuration;
72      }

74      /**
75       * @param args
76       * @throws Exception
77       */
78      public static void main(String[] args) throws Exception {
79          WaarpLoggerFactory.setDefaultFactory(new WaarpSlf4JLoggerFactory(null));
80          final WaarpLogger logger = WaarpLoggerFactory
81                  .getLogger(HttpTestR66PseudoMain.class);
82          String pathTemp = "/tmp";
83          if (!R66Server.initialize(args[0])) {
84              System.err.println("Error during startup");
85              System.exit(1);
86          }
87  
88          config = getTestConfiguration();
89          HttpRestR66Handler.initialize(pathTemp);
90          HttpRestR66Handler.initializeService(config);

92          logger.warn("Server RestOpenR66 starts");
