47  /**
48   * Test Rest client response handler
49   * 
50   * Note that for testing, result is only the last "json" command, and therefore future is only
51   * validated once all items are passed in a chain.
52   * In normal condition, each step should produce: setting the RestArgument to the RestFuture and
53   * validating (error or ok) to RestFuture.
54   * 
55   * @author Frederic Bregier
56   */
57  public class HttpTestResponseHandler extends HttpRestR66ClientResponseHandler {
58      /**
59       * @param channel
60       * @throws HttpInvalidAuthenticationException
61       */
62  
63      @Override
64      protected boolean afterError(Channel channel, RestArgument ra) {
65          HttpTestRestR66Client.count.incrementAndGet();
66          WaarpSslUtility.closingSslChannel(channel);
67          return false;
68      }
69  
70      @Override
71      protected boolean afterDbGet(Channel channel, RestArgument ra) throws HttpInvalidAuthenticationException {
72          HttpTestRestR66Client.count.incrementAndGet();
73          // Update
74          HttpTestRestR66Client.updateData(channel, ra);
75          return true;
76      }
77  
78      @Override
79      protected boolean afterDbPost(Channel channel, RestArgument ra) throws HttpInvalidAuthenticationException {
80          HttpTestRestR66Client.count.incrementAndGet();
81          if (ra.getAnswer().path(DbHostAuth.Columns.ADMINROLE.name()).asBoolean()) {
82              WaarpSslUtility.closingSslChannel(channel);
83              return false;
84          }
85          // Select 1
86          HttpTestRestR66Client.readData(channel, ra);
87          return true;
88      }
89  
90      @Override
91      protected boolean afterDbPut(Channel channel, RestArgument ra) throws HttpInvalidAuthenticationException {
92          HttpTestRestR66Client.count.incrementAndGet();
93          if (ra.getAnswer().path(DbHostConfiguration.Columns.HOSTID.name()).asText().equals("hosta")) {
94              WaarpSslUtility.closingSslChannel(channel);
95              return false;
96          }
97          // Delete 1
98          HttpTestRestR66Client.deleteData(channel, ra);
99          return true;
100     }

105         WaarpSslUtility.closingSslChannel(channel);
106         return false;
107     }

113         return false;
114     }

116     @Override
117     protected boolean afterDbOptions(Channel channel, RestArgument ra) throws HttpInvalidAuthenticationException {
118         HttpTestRestR66Client.count.incrementAndGet();
119         boolean newMessage = false;
120         AtomicInteger counter = null;
121         RestFuture future = channel.attr(HttpRestClientSimpleResponseHandler.RESTARGUMENT).get();
122         if (future.getOtherObject() == null) {
123             counter = new AtomicInteger();
124             future.setOtherObject(counter);
125             JsonNode node = ra.getDetailedAllowOption();
126             if (!node.isMissingNode()) {
127                 for (JsonNode jsonNode : node) {
128                     Iterator<String> iterator = jsonNode.fieldNames();
129                     while (iterator.hasNext()) {
130                         String name = iterator.next();
131                         if (!jsonNode.path(name).path(RestArgument.REST_FIELD.JSON_PATH.field).isMissingNode()) {
132                             break;
133                         }
134                         if (name.equals(RootOptionsRestMethodHandler.ROOT)) {
135                             continue;
136                         }
137                         counter.incrementAndGet();
138                         HttpTestRestR66Client.options(channel, name);
139                         newMessage = true;
140                     }
141                 }
142             }
143         }
144         if (!newMessage) {
145             counter = (AtomicInteger) future.getOtherObject();
146             newMessage = counter.decrementAndGet() > 0;
147             if (!newMessage) {
148                 future.setOtherObject(null);
149             }
150         }
151         if (!newMessage) {
152             WaarpSslUtility.closingSslChannel(channel);
153         }
154         return newMessage;
155     }
156 
157     @Override
158     protected boolean action(Channel channel, RestArgument ra, ACTIONS_TYPE act) {
159         HttpTestRestR66Client.count.incrementAndGet();
160         boolean newMessage = false;
161         switch (act) {
162             case CreateTransfer: {
163                 // Continue with GetTransferInformation
164                 TransferRequestJsonPacket recv;
165                 try {
166                     recv = (TransferRequestJsonPacket) JsonPacket.createFromBuffer(JsonHandler.writeAsString(ra
167                             .getResults().get(0)));
168                 } catch (Exception e) {
169                     // TODO Auto-generated catch block
170                     e.printStackTrace();
171                     return newMessage;
172                 }
173                 InformationJsonPacket node = new InformationJsonPacket(recv.getSpecialId(), false, recv.getRequested());
174                 HttpTestRestR66Client.action(channel, HttpMethod.GET, RESTHANDLERS.Control.uri, node);
175                 newMessage = true;
176                 break;
177             }
178             case ExecuteBusiness:
179                 // End
180                 break;
181             case ExportConfig:
182                 // no Import in automatic test
183                 break;
184             case GetBandwidth:
185                 // End
186                 break;
187             case GetInformation:
188                 // End
189                 break;
190             case GetLog:
191                 // End
192                 break;
193             case GetTransferInformation: {
194                 // Continue with Stop in StopOrCancelTransfer
195                 ObjectNode answer = (ObjectNode) ra.getResults().get(0);
196                 StopOrCancelJsonPacket node = new StopOrCancelJsonPacket();
197                 node.setRequestUserPacket();
198                 node.setStop();
199                 node.setRequested(answer.path(DbTaskRunner.Columns.REQUESTED.name()).asText());
200                 node.setRequester(answer.path(DbTaskRunner.Columns.REQUESTER.name()).asText());
201                 node.setSpecialid(answer.path(DbTaskRunner.Columns.SPECIALID.name()).asLong());
202                 HttpTestRestR66Client.action(channel, HttpMethod.PUT, RESTHANDLERS.Control.uri, node);
203                 newMessage = true;
204                 break;
205             }
206             case ImportConfig:
207                 // End
208                 break;
209             case OPTIONS:
210                 break;
211             case RestartTransfer: {
212                 // Continue with delete transfer
213                 RestartTransferJsonPacket recv;
214                 try {
215                     recv = (RestartTransferJsonPacket) JsonPacket.createFromBuffer(JsonHandler.writeAsString(ra
216                             .getResults().get(0)));
217                 } catch (Exception e) {
218                     // TODO Auto-generated catch block
219                     e.printStackTrace();
220                     return newMessage;
221                 }
222                 try {
223                     HttpTestRestR66Client.deleteData(channel, recv.getRequested(), recv.getRequester(),
224                             recv.getSpecialid());
225                 } catch (HttpInvalidAuthenticationException e) {
226                     // TODO Auto-generated catch block
227                     e.printStackTrace();
228                 }
229                 newMessage = true;
230                 break;
231             }
232             case SetBandwidth: {
233                 // Continue with GetBandwidth
234                 BandwidthJsonPacket recv;
235                 try {
236                     recv = (BandwidthJsonPacket) JsonPacket.createFromBuffer(JsonHandler.writeAsString(ra.getResults()
237                             .get(0)));
238                 } catch (Exception e) {
239                     // TODO Auto-generated catch block
240                     e.printStackTrace();
241                     return newMessage;
242                 }
243                 recv.setSetter(false);
244                 HttpTestRestR66Client.action(channel, HttpMethod.GET, RESTHANDLERS.Bandwidth.uri, recv);
245                 newMessage = true;
246                 break;
247             }
248             case ShutdownOrBlock:
249                 // End
250                 break;
251             case StopOrCancelTransfer: {
252                 // Continue with RestartTransfer
253                 StopOrCancelJsonPacket recv;
254                 try {
255                     recv = (StopOrCancelJsonPacket) JsonPacket.createFromBuffer(JsonHandler.writeAsString(ra
256                             .getResults().get(0)));
257                 } catch (Exception e) {
258                     // TODO Auto-generated catch block
259                     e.printStackTrace();
260                     return newMessage;
261                 }
262                 RestartTransferJsonPacket node = new RestartTransferJsonPacket();
263                 node.setRequestUserPacket();
264                 node.setRequested(recv.getRequested());
265                 node.setRequester(recv.getRequester());
266                 node.setSpecialid(recv.getSpecialid());
267                 HttpTestRestR66Client.action(channel, HttpMethod.PUT, RESTHANDLERS.Control.uri, node);
268                 newMessage = true;
269                 break;
270             }
271             case GetStatus:
272                 break;
273             default:
274                 break;
275 
276         }
277         if (!newMessage) {
278             WaarpSslUtility.closingSslChannel(channel);
279         }
280         return newMessage;
281     }
