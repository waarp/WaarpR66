16  package org.waarp.openr66.protocol.http.rest.test;
17  
18  import io.netty.channel.ChannelPipeline;
19  import io.netty.channel.ChannelInitializer;
20  import io.netty.channel.socket.SocketChannel;
21  import io.netty.handler.codec.http.HttpClientCodec;
22  import io.netty.handler.codec.http.HttpContentDecompressor;
23  import io.netty.handler.ssl.SslHandler;
24  import io.netty.handler.stream.ChunkedWriteHandler;
25  
26  import org.waarp.common.crypto.ssl.WaarpSslContextFactory;
27  
28  /**
29   * Test Rest client pipeline factory
30   */
31  public class HttpTestRestClientInitializer extends ChannelInitializer<SocketChannel> {
32      private final WaarpSslContextFactory waarpSslContextFactory;
33  
34      public HttpTestRestClientInitializer(WaarpSslContextFactory waarpSslContextFactory) {
35          this.waarpSslContextFactory = waarpSslContextFactory;
36      }
37  
38      protected void initChannel(SocketChannel ch) throws Exception {
39          // Create a default pipeline implementation.
40          ChannelPipeline pipeline = ch.pipeline();
41  
42          // Enable HTTPS if necessary.
43          if (waarpSslContextFactory != null) {
44              SslHandler handler = waarpSslContextFactory.initInitializer(false, false);
45              pipeline.addLast("ssl", handler);
46          }
47  
48          pipeline.addLast("codec", new HttpClientCodec());
49          // Remove the following line if you don't want automatic content
50          // decompression.
51          pipeline.addLast("inflater", new HttpContentDecompressor());
52  
53          // to be used since huge file transfer
54          pipeline.addLast("streamer", new ChunkedWriteHandler());
55  
56          pipeline.addLast("handler", new HttpTestResponseHandler());
57      }
58  }
