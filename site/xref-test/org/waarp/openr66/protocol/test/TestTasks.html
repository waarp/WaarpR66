59      /**
60       * @param args
61       */
62      public static void main(String[] args) {
63          WaarpLoggerFactory.setDefaultFactory(new WaarpSlf4JLoggerFactory(
64                  null));
65          if (args.length < 4) {
66              System.err.println("Need config inDirectory outDirectory filename (in inDirectory)");
67              return;
68          }
69          if (!FileBasedConfiguration
70                  .setClientConfigurationFromXml(Configuration.configuration, args[0])) {
71              System.err.println("Needs a correct configuration file as first argument");
72              return;
73          }
74          String in = args[1];
75          String out = args[2];
76          String filename = args[3];
77          File file = new File(in + "/" + filename);
78          File out2 = new File(out + "/move");
79          out2.mkdirs();
80          long size = file.length();
81          String argRule = out + "/#DATE#_%s_%s_" + filename;
82          String argTransfer = "basic information";
83          R66Session session = new R66Session();
84          DbRule rule = new DbRule(null, "idRule", (String) null, TRANSFERMODE.SENDMODE.ordinal(), out, null, null, in,
85                  null, null, null, null, null, null);
86          RequestPacket requestPacket = new RequestPacket(rule.getIdRule(), rule.getMode(), filename,
87                  Configuration.BUFFERSIZEDEFAULT, (int) size / Configuration.BUFFERSIZEDEFAULT + 1,
88                  1, argTransfer, size, PartnerConfiguration.BAR_SEPARATOR_FIELD);
89          DbTaskRunner runner = null;
90          DbConstant.admin = new DbAdmin();
91          session.getAuth().specialNoSessionAuth(false, "false");
92          try {
93              runner = new DbTaskRunner(null, session, rule, false, requestPacket);
94          } catch (WaarpDatabaseException e) {
95          }
96          runner.setPostTask();
97          session.setBadRunner(runner, ErrorCode.QueryAlreadyFinished);
98  
99          // LOG
100         LogTask logTask = new LogTask(argRule, 1, argTransfer, session);
101         logTask.run();
102         try {
103             logTask.getFutureCompletion().await();
104         } catch (InterruptedException e) {
105         }
106         System.out.println("LOG: " + logTask.getFutureCompletion().isSuccess());
107         logTask = new LogTask(argRule + " " + out + "/log.txt", 3, argTransfer, session);
108         logTask.run();
109         try {
110             logTask.getFutureCompletion().await();
111         } catch (InterruptedException e) {
112         }
113         System.out.println("LOG2: " + logTask.getFutureCompletion().isSuccess());
114 
115         // COPYRENAME
116         CopyRenameTask copyRenameTask = new CopyRenameTask(argRule + "_copyrename", 0, argTransfer, session);
117         copyRenameTask.run();
118         try {
119             copyRenameTask.getFutureCompletion().await();
120         } catch (InterruptedException e) {
121         }
122         System.out.println("COPYRENAME: " + copyRenameTask.getFutureCompletion().isSuccess());
123 
124         // COPY

126         copyTask.run();
127         try {
128             copyTask.getFutureCompletion().await();
129         } catch (InterruptedException e) {
130         }
131         System.out.println("COPY: " + copyTask.getFutureCompletion().isSuccess());
132

136         try {
137             execTask.getFutureCompletion().await();
138         } catch (InterruptedException e) {
139         }

141

146             validFilePathTask.getFutureCompletion().await();
147         } catch (InterruptedException e) {
148         }
149         System.out.println("VALIDFILEPATH: " + validFilePathTask.getFutureCompletion().isSuccess());
150

152         TarTask tarTask = new TarTask(out + "/test.tar " + out, 2, argTransfer, session);
153         tarTask.run();
154         try {
155             tarTask.getFutureCompletion().await();
156         } catch (InterruptedException e) {
157         }
158         System.out.println("TAR: " + tarTask.getFutureCompletion().isSuccess());
159         tarTask = new TarTask(out + "/test.tar " + out + "/move", 1, argTransfer, session);
160         tarTask.run();
161         try {
162             tarTask.getFutureCompletion().await();
163         } catch (InterruptedException e) {
164         }
165         System.out.println("UNTAR: " + tarTask.getFutureCompletion().isSuccess());
166 
167         // ZIP
168         ZipTask zipTask = new ZipTask(out + "/test.zip " + out, 2, argTransfer, session);
169         zipTask.run();
170         try {
171             zipTask.getFutureCompletion().await();
172         } catch (InterruptedException e) {
173         }
174         System.out.println("ZIP: " + zipTask.getFutureCompletion().isSuccess());
175         zipTask = new ZipTask(out + "/test.zip " + out + "/move", 1, argTransfer, session);
176         zipTask.run();
177         try {
178             zipTask.getFutureCompletion().await();
179         } catch (InterruptedException e) {
180         }
181         System.out.println("UNZIP: " + zipTask.getFutureCompletion().isSuccess());
182

184         TranscodeTask transcodeTask = new TranscodeTask("-from UTF8 -to ISO-8859-15", 0, argTransfer, session);
185         transcodeTask.run();
186         try {
187             transcodeTask.getFutureCompletion().await();
188         } catch (InterruptedException e) {
189         }
190         System.out.println("TRANSCODE: " + transcodeTask.getFutureCompletion().isSuccess());
191

193         MoveRenameTask moveReameTask = new MoveRenameTask(argRule + "_moverename", 0, argTransfer, session);
194         moveReameTask.run();
195         try {
196             moveReameTask.getFutureCompletion().await();
197         } catch (InterruptedException e) {
198         }
199         System.out.println("MOVERENAME: " + moveReameTask.getFutureCompletion().isSuccess());
200

202         copyRenameTask.run();
203         try {
204             copyRenameTask.getFutureCompletion().await();
205         } catch (InterruptedException e) {
206         }

208

211         moveTask.run();
212         try {

214         } catch (InterruptedException e) {

226         zipTask.run();
227         try {

229         } catch (InterruptedException e) {
230         }
231         System.out.println("UNZIP: " + zipTask.getFutureCompletion().isSuccess());

235         deleteTask.run();
