35  /**
36   * Test class for JsonGenerator
37   * 
38   * @author Frederic Bregier
39   * 
40   */
41  public class TestJsonGenerator {
42      /**
43       * Internal Logger
44       */
45      static protected volatile WaarpLogger logger;
46      static int nb = 11;
47  
48      /**
49       * @param args
50       * @param rank
51       * @return True if OK
52       */
53      protected static boolean getSpecialParams(String[] args, int rank) {
54          for (int i = rank; i < args.length; i++) {
55              if (args[i].equalsIgnoreCase("-nb")) {
56                  i++;
57                  nb = Integer.parseInt(args[i]);
58              }
59          }
60          return true;
61      }
62  
63      public static void main(String[] args) {
64          WaarpLoggerFactory.setDefaultFactory(new WaarpSlf4JLoggerFactory(
65                  null));
66          if (logger == null) {
67              logger = WaarpLoggerFactory.getLogger(TestJsonGenerator.class);
68          }
69          if (args.length == 0 || !FileBasedConfiguration
70                  .setSubmitClientConfigurationFromXml(Configuration.configuration, args[0])) {
71              logger.error(Messages.getString("Configuration.NeedCorrectConfig")); //$NON-NLS-1$
72              logger.error("Wrong initialization");
73              if (DbConstant.admin != null && DbConstant.admin.isActive()) {
74                  DbConstant.admin.close();
75              }
76              System.exit(1);
77          }
78          getSpecialParams(args, 1);
79          logger.warn("Start Test Json");
80          DbPreparedStatement preparedStatement;
81          try {
82              preparedStatement = DbTaskRunner.getFilterPrepareStatement(DbConstant.admin.getSession(), nb, false,
83                      null, null, null, null, null, null,
84                      false, false, false, false, true, null);
85              preparedStatement.executeQuery();
86              String tasks = DbTaskRunner.getJson(preparedStatement, nb);
87              System.out.println(tasks);
88              preparedStatement.realClose();
89          } catch (WaarpDatabaseNoConnectionException e) {
90              e.printStackTrace();
91          } catch (WaarpDatabaseSqlException e) {
92              e.printStackTrace();
93          } catch (OpenR66ProtocolBusinessException e) {
94              // TODO Auto-generated catch block
95              e.printStackTrace();
96          }
97          System.out.println();
98          try {
99              preparedStatement = DbHostAuth.getFilterPrepareStament(DbConstant.admin.getSession(),
100                     null, null);
101             preparedStatement.executeQuery();
102             String hosts = DbHostAuth.getJson(preparedStatement, nb);
103             System.out.println(hosts);
104             preparedStatement.realClose();
105         } catch (WaarpDatabaseNoConnectionException e) {
106             e.printStackTrace();
107         } catch (WaarpDatabaseSqlException e) {
108             e.printStackTrace();
109         } catch (OpenR66ProtocolBusinessException e) {
110             // TODO Auto-generated catch block
111             e.printStackTrace();
112         }
113         System.out.println();
114         try {
115             preparedStatement = DbRule.getFilterPrepareStament(DbConstant.admin.getSession(),
116                     null, -1);
117             preparedStatement.executeQuery();
118             String rules = DbRule.getJson(preparedStatement, nb);
119             System.out.println(rules);
120             preparedStatement.realClose();
121         } catch (WaarpDatabaseNoConnectionException e) {
122             e.printStackTrace();
123         } catch (WaarpDatabaseSqlException e) {
124             e.printStackTrace();
125         } catch (OpenR66ProtocolBusinessException e) {
126             // TODO Auto-generated catch block
127             e.printStackTrace();
128         }
129     }
