18  package org.waarp.openr66.context.task;
19  
20  import java.io.File;
21  import java.io.IOException;
22  
23  import org.apache.commons.exec.CommandLine;
24  import org.apache.commons.exec.DefaultExecutor;
25  import org.apache.commons.exec.ExecuteException;
26  import org.apache.commons.exec.ExecuteWatchdog;
27  import org.apache.commons.exec.PumpStreamHandler;
28  
29  import org.waarp.common.logging.WaarpLogger;
30  import org.waarp.common.logging.WaarpLoggerFactory;
31  import org.waarp.openr66.context.ErrorCode;
32  import org.waarp.openr66.context.R66Session;
33  import org.waarp.openr66.context.task.localexec.LocalExecClient;
34  import org.waarp.openr66.protocol.configuration.Configuration;
35  
36  /**
37   * Execute an external command
38   * 
39   * 
40   * @author Frederic Bregier
41   * 
42   */
43  public class ExecTask extends AbstractExecTask {
44  
45      /**
46       * Internal Logger
47       */
48      private static final WaarpLogger logger = WaarpLoggerFactory
49              .getLogger(ExecTask.class);
50  
51      /**
52       * @param argRule
53       * @param delay
54       * @param argTransfer
55       * @param session
56       */
57      public ExecTask(String argRule, int delay, String argTransfer,
58              R66Session session) {
59          super(TaskType.EXEC, delay, argRule, argTransfer, session);
60      }
61  
62      @Override
63      public void run() {
64          /*
65           * First apply all replacements and format to argRule from context and argTransfer. Will
66           * call exec (from first element of resulting string) with arguments as the following value
67           * from the replacements. Return 0 if OK, else 1 for a warning else as an error. No change
68           * should be done in the FILENAME
69           */
70          logger.debug("Exec with " + argRule + ":" + argTransfer + " and {}",
71                  session);
72          String finalname = applyTransferSubstitutions(argRule);
73          
74          // Check if the execution will be done through LocalExec daemon
75          if (Configuration.configuration.isUseLocalExec() && useLocalExec) {
76              LocalExecClient localExecClient = new LocalExecClient();
77              if (localExecClient.connect()) {
78                  localExecClient
79                          .runOneCommand(finalname, delay, waitForValidation, futureCompletion);
80                  localExecClient.disconnect();
81                  return;
82              }// else continue
83          }
84          // Execution is done internally
85          CommandLine commandLine = buildCommandLine(finalname);
86          if (commandLine == null) {
87              return;
88          }
89  
90          DefaultExecutor defaultExecutor = new DefaultExecutor();
91          PumpStreamHandler pumpStreamHandler = new PumpStreamHandler(null, null);
92          defaultExecutor.setStreamHandler(pumpStreamHandler);
93          int[] correctValues = {
94                  0, 1 };
95          defaultExecutor.setExitValues(correctValues);
96          ExecuteWatchdog watchdog = null;
97          if (delay > 0 && waitForValidation) {
98              watchdog = new ExecuteWatchdog(delay);
99              defaultExecutor.setWatchdog(watchdog);
100         }
101         if (!waitForValidation) {
102             // Do not wait for validation
103             futureCompletion.setSuccess();
104             logger.info("Exec will start but no WAIT with {}", commandLine);
105         }
106         int status = -1;
107         try {
108             status = defaultExecutor.execute(commandLine);
109         } catch (ExecuteException e) {
110             if (e.getExitValue() == -559038737) {
111                 // Cannot run immediately so retry once
112                 try {
113                     Thread.sleep(Configuration.RETRYINMS);
114                 } catch (InterruptedException e1) {
115                 }
116                 try {
117                     status = defaultExecutor.execute(commandLine);
118                 } catch (ExecuteException e1) {
119                     try {
120                         pumpStreamHandler.stop();
121                     } catch (IOException e2) {
122                     }
123                     logger.error("Exception: " + e.getMessage() +
124                             " Exec in error with " + commandLine.toString());
125                     if (waitForValidation) {
126                         futureCompletion.setFailure(e);
127                     }
128                     return;
129                 } catch (IOException e1) {
130                     try {
131                         pumpStreamHandler.stop();
132                     } catch (IOException e2) {
133                     }
134                     logger.error("Exception: " + e.getMessage() +
135                             " Exec in error with " + commandLine.toString());
136                     if (waitForValidation) {
137                         futureCompletion.setFailure(e);
138                     }
139                     return;
140                 }
141             } else {
142                 try {
143                     pumpStreamHandler.stop();
144                 } catch (IOException e2) {
145                 }
146                 logger.error("Exception: " + e.getMessage() +
147                         " Exec in error with " + commandLine.toString());
148                 if (waitForValidation) {
149                     futureCompletion.setFailure(e);
150                 }
151                 return;
152             }
153         } catch (IOException e) {
154             try {
155                 pumpStreamHandler.stop();
156             } catch (IOException e2) {
157             }
158             logger.error("Exception: " + e.getMessage() +
159                     " Exec in error with " + commandLine.toString());
160             if (waitForValidation) {
161                 futureCompletion.setFailure(e);
162             }
163             return;
164         }
165         try {
166             pumpStreamHandler.stop();
167         } catch (IOException e2) {
168         }
169         if (defaultExecutor.isFailure(status) && watchdog != null &&
170                 watchdog.killedProcess()) {
171             // kill by the watchdoc (time out)
172             logger.error("Exec is in Time Out");
173             status = -1;
174         }
175         if (status == 0) {
176             if (waitForValidation) {
177                 futureCompletion.setSuccess();
178             }
179             logger.info("Exec OK with {}", commandLine);
180         } else if (status == 1) {
181             logger.warn("Exec in warning with " + commandLine.toString());
182             if (waitForValidation) {
183                 futureCompletion.setSuccess();
184             }
185         } else {
186             logger.error("Status: " + status + " Exec in error with " +
187                     commandLine.toString());
188             if (waitForValidation) {
189                 futureCompletion.cancel();
190             }
191         }
192     }
193 }
