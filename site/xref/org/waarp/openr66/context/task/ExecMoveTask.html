40  /**
41   * Execute an external command and Rename the file (using the new name from the result).<br>
42   * 
43   * The move of the file (if any) should be done by the external command itself.<br>
44   * <br>
45   * 
46   * waitForValidation (#NOWAIT#) must not be set since it will prevent to have the MOVE TASK to occur
47   * normally. So even if set, the #NOWAIT# will be ignored.
48   * 
49   * @author Frederic Bregier
50   * 
51   */
52  public class ExecMoveTask extends AbstractExecTask {
53      /**
54       * Internal Logger
55       */
56      private static final WaarpLogger logger = WaarpLoggerFactory
57              .getLogger(ExecMoveTask.class);
58  
59      /**
60       * @param argRule
61       * @param delay
62       * @param argTransfer
63       * @param session
64       */
65      public ExecMoveTask(String argRule, int delay, String argTransfer,
66              R66Session session) {
67          super(TaskType.EXECMOVE, delay, argRule, argTransfer, session);
68      }
69  
70      @Override
71      public void run() {
72          /*
73           * First apply all replacements and format to argRule from context and argTransfer. Will
74           * call exec (from first element of resulting string) with arguments as the following value
75           * from the replacements. Return 0 if OK, else 1 for a warning else as an error. The last
76           * line of stdout will be the new name given to the R66File in case of status 0. The
77           * previous file should be deleted by the script or will be deleted in case of status 0. If
78           * the status is 1, no change is made to the file.
79           */
80          logger.info("ExecMove with " + argRule + ":" + argTransfer + " and {}",
81                  session);
82          String finalname = applyTransferSubstitutions(argRule);
83          
84          // Force the WaitForValidation
85          waitForValidation = true;
86          if (Configuration.configuration.isUseLocalExec() && useLocalExec) {
87              LocalExecClient localExecClient = new LocalExecClient();
88              if (localExecClient.connect()) {
89                  localExecClient
90                          .runOneCommand(finalname, delay, waitForValidation, futureCompletion);
91                  LocalExecResult result = localExecClient.getLocalExecResult();
92                  move(result.getStatus(), result.getResult(), finalname);
93                  localExecClient.disconnect();
94                  return;
95              } // else continue
96          }
97  
98          CommandLine commandLine = buildCommandLine(finalname);
99          if (commandLine == null) {
100             return;
101         }
102 
103         DefaultExecutor defaultExecutor = new DefaultExecutor();
104         PipedInputStream inputStream = new PipedInputStream();
105         PipedOutputStream outputStream = null;
106         try {
107             outputStream = new PipedOutputStream(inputStream);
108         } catch (IOException e1) {
109             try {
110                 inputStream.close();
111             } catch (IOException e) {
112             }
113             logger.error("Exception: " + e1.getMessage() +
114                     " Exec in error with " + commandLine.toString(), e1);
115             futureCompletion.setFailure(e1);
116             return;
117         }
118         PumpStreamHandler pumpStreamHandler = new PumpStreamHandler(
119                 outputStream, null);
120         defaultExecutor.setStreamHandler(pumpStreamHandler);
121         int[] correctValues = {
122                 0, 1 };
123         defaultExecutor.setExitValues(correctValues);
124         ExecuteWatchdog watchdog = null;
125 
126         if (delay > 0) {
127             watchdog = new ExecuteWatchdog(delay);
128             defaultExecutor.setWatchdog(watchdog);
129         }
130         LastLineReader lastLineReader = new LastLineReader(inputStream);
131         Thread thread = new Thread(lastLineReader, "ExecRename"
132                 + session.getRunner().getSpecialId());
133         thread.setDaemon(true);
134         Configuration.configuration.getExecutorService().execute(thread);
135         int status = -1;
136         try {
137             status = defaultExecutor.execute(commandLine);
138         } catch (ExecuteException e) {
139             if (e.getExitValue() == -559038737) {
140                 // Cannot run immediately so retry once
141                 try {
142                     Thread.sleep(Configuration.RETRYINMS);
143                 } catch (InterruptedException e1) {
144                 }
145                 try {
146                     status = defaultExecutor.execute(commandLine);
147                 } catch (ExecuteException e1) {
148                     try {
149                         outputStream.close();
150                     } catch (IOException e2) {
151                     }
152                     thread.interrupt();
153                     try {
154                         inputStream.close();
155                     } catch (IOException e2) {
156                     }
157                     try {
158                         pumpStreamHandler.stop();
159                     } catch (IOException e2) {
160                     }
161                     logger.error("ExecuteException: " + e.getMessage() +
162                             " . Exec in error with " + commandLine.toString());
163                     futureCompletion.setFailure(e);
164                     return;
165                 } catch (IOException e1) {
166                     try {
167                         outputStream.close();
168                     } catch (IOException e2) {
169                     }
170                     thread.interrupt();
171                     try {
172                         inputStream.close();
173                     } catch (IOException e2) {
174                     }
175                     try {
176                         pumpStreamHandler.stop();
177                     } catch (IOException e2) {
178                     }
179                     logger.error("IOException: " + e.getMessage() +
180                             " . Exec in error with " + commandLine.toString());
181                     futureCompletion.setFailure(e);
182                     return;
183                 }
184             } else {
185                 try {
186                     outputStream.close();
187                 } catch (IOException e1) {
188                 }
189                 thread.interrupt();
190                 try {
191                     inputStream.close();
192                 } catch (IOException e1) {
193                 }
194                 try {
195                     pumpStreamHandler.stop();
196                 } catch (IOException e2) {
197                 }
198                 logger.error("ExecuteException: " + e.getMessage() +
199                         " . Exec in error with " + commandLine.toString());
200                 futureCompletion.setFailure(e);
201                 return;
202             }
203         } catch (IOException e) {
204             try {
205                 outputStream.close();
206             } catch (IOException e1) {
207             }
208             thread.interrupt();
209             try {
210                 inputStream.close();
211             } catch (IOException e1) {
212             }
213             try {
214                 pumpStreamHandler.stop();
215             } catch (IOException e2) {
216             }
217             logger.error("IOException: " + e.getMessage() +
218                     " . Exec in error with " + commandLine.toString());
219             futureCompletion.setFailure(e);
220             return;
221         }
222         try {
223             outputStream.flush();
224         } catch (IOException e) {
225         }
226         try {
227             outputStream.close();
228         } catch (IOException e) {
229         }
230         try {
231             pumpStreamHandler.stop();
232         } catch (IOException e2) {
233         }
234         try {
235             if (delay > 0) {
236                 thread.join(delay);
237             } else {
238                 thread.join();
239             }
240         } catch (InterruptedException e) {
241             Thread.currentThread().interrupt();
242         }
243         try {
244             inputStream.close();
245         } catch (IOException e1) {
246         }
247         String newname = null;
248         if (defaultExecutor.isFailure(status) && watchdog != null &&
249                 watchdog.killedProcess()) {
250             // kill by the watchdoc (time out)
251             status = -1;
252             newname = "TimeOut";
253         } else {
254             newname = lastLineReader.getLastLine();
255             if (status == 0 && (newname == null || newname.isEmpty())) {
256                 status = 1;
257             }
258         }
259         move(status, newname, commandLine.toString());
260     }
261 
262     private void move(int status, String newName, String commandLine) {
263         String newname = newName.replace('\\', '/');
264         if (status == 0) {
265             if (newname.indexOf(' ') > 0) {
266                 logger.warn("Exec returns a multiple string in final line: " +
267                         newname);
268                 // XXX FIXME: should not split String[] args = newname.split(" ");
269                 //newname = args[args.length - 1];
270             }
271             // now test if the previous file was deleted (should be)
272             File file = new File(newname);
273             if (!file.exists()) {
274                 logger.warn("New file does not exist at the end of the exec: " + newname);
275             }
276             // now replace the file with the new one
277             try {
278                 session.getFile().replaceFilename(newname, true);
279             } catch (CommandAbstractException e) {
280                 logger
281                         .warn("Exec in warning with " + commandLine,
282                                 e);
283             }
284             session.getRunner().setFileMoved(newname, true);
285             R66Result result = new R66Result(session, true, ErrorCode.CompleteOk, this.session.getRunner());
286             result.setOther(newname);
287             futureCompletion.setResult(result);
288             futureCompletion.setSuccess();
289             logger.info("Exec OK with {} returns {}", commandLine,
290                     newname);
291         } else if (status == 1) {
292             logger.warn("Exec in warning with " + commandLine +
293                     " returns " + newname);
294             session.getRunner().setErrorExecutionStatus(ErrorCode.Warning);
295             R66Result result = new R66Result(session, true, ErrorCode.Warning, this.session.getRunner());
296             result.setOther(newname);
297             futureCompletion.setResult(result);
298             futureCompletion.setSuccess();
299         } else {
300             logger.error("Status: " + status + " Exec in error with " +
301                     commandLine + " returns " + newname);
302             futureCompletion.cancel();
303         }
304     }
