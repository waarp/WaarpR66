18  package org.waarp.openr66.context.task;
19  
20  import java.io.File;
21  
22  import org.waarp.common.logging.WaarpLogger;
23  import org.waarp.common.logging.WaarpLoggerFactory;
24  import org.waarp.common.transcode.CharsetsUtil;
25  import org.waarp.common.utility.FileConvert;
26  import org.waarp.openr66.context.ErrorCode;
27  import org.waarp.openr66.context.R66Result;
28  import org.waarp.openr66.context.R66Session;
29  import org.waarp.openr66.database.data.DbTaskRunner;
30  import org.waarp.openr66.protocol.configuration.Configuration;
31  import org.waarp.openr66.protocol.exception.OpenR66ProtocolSystemException;
32  
33  /**
34   * Transcode the current file from one Charset to another Charset as specified<br>
35   * <br>
36   * Arguments are:<br>
37   * -dos2unix or -unix2dos (only one) ; optional argument, but if present -from and -to might be omitted;<br>
38   * -from charset<br>
39   * -to charset<br>
40   * -newfile newfilename ; optional argument ; if not used, will be current filename.extension ;
41   * if used, extension is ignored<br>
42   * -extension extension ; optional argument ; if not used, will be filename.transcode<br>
43   * <br>
44   * A convenient method (from Waarp Common) allows to list in html (-html), csv (-csv) or text format (-text)
45   * all the supported Charsets from your JVM. To use it, run the following command:<br>
46   * java -cp WaarpCommon-1.2.7.jar org.waarp.common.transcode.CharsetsUtil [-csv | -html | -text ]<br>
47   * <br>
48   * It could also be used as a test of transcode outside R66:<br>
49   * java -cp WaarpCommon-1.2.7.jar org.waarp.common.transcode.CharsetsUtil -from fromFilename fromCharset -to toFilename toCharset<br>
50   * 
51   * The current file is not touched and is not marked as moved.
52   * 
53   * @author Frederic Bregier
54   * 
55   */
56  public class TranscodeTask extends AbstractTask {
57      /**
58       * Internal Logger
59       */
60      private static final WaarpLogger logger = WaarpLoggerFactory
61              .getLogger(TranscodeTask.class);
62  
63      /**
64       * @param argRule
65       * @param delay
66       * @param argTransfer
67       * @param session
68       */
69      public TranscodeTask(String argRule, int delay, String argTransfer,
70              R66Session session) {
71          super(TaskType.TRANSCODE, delay, argRule, argTransfer, session);
72      }
73  
74      @Override
75      public void run() {
76          boolean success = false;
77          DbTaskRunner runner = session.getRunner();
78          String arg = argRule;
79          arg = getReplacedValue(arg, argTransfer.split(" "));
80          String[] args = arg.split(" ");
81          boolean dos2unix = false, unix2dos = false;
82          String fromCharset = null;
83          String toCharset = null;
84          String newfilename = null;
85          String extension = null;
86          for (int i = 0; i < args.length; i++) {
87              if (args[i].equalsIgnoreCase("-from")) {
88                  i++;
89                  if (i < args.length) {
90                      fromCharset = args[i];
91                  }
92              } else if (args[i].equalsIgnoreCase("-to")) {
93                  i++;
94                  if (i < args.length) {
95                      toCharset = args[i];
96                  }
97              } else if (args[i].equalsIgnoreCase("-newfile")) {
98                  i++;
99                  if (i < args.length) {
100                     newfilename = args[i];
101                 }
102             } else if (args[i].equalsIgnoreCase("-extension")) {
103                 i++;
104                 if (i < args.length) {
105                     extension = args[i];
106                 }
107             } else if (args[i].equalsIgnoreCase("-dos2unix")) {
108                 dos2unix = true;
109             } else if (args[i].equalsIgnoreCase("-unix2dos")) {
110                 unix2dos = true;
111             }
112         }
113         if (dos2unix && unix2dos) {
114             R66Result result = new R66Result(session, false, ErrorCode.Warning,
115                     runner);
116             futureCompletion.setResult(result);
117             logger.warn("Dos2Unix and Unix2Dos cannot be used simultaneously in Transcode: " + runner.toShortString());
118             futureCompletion.setFailure(new OpenR66ProtocolSystemException(
119                     "Dos2Unix and Unix2Dos cannot be used simultaneously in Transcode"));
120             return;
121         }
122         if (fromCharset == null || toCharset == null) {
123             if (!(dos2unix || unix2dos)) {
124                 R66Result result = new R66Result(session, false, ErrorCode.Warning,
125                         runner);
126                 futureCompletion.setResult(result);
127                 logger.warn("Not enough argument in Transcode: " + runner.toShortString());
128                 futureCompletion.setFailure(new OpenR66ProtocolSystemException(
129                         "Not enough argument in Transcode"));
130             } else {
131                 // only 1 of Dos2Unix/Unix2Dos
132                 FileConvert convert = new FileConvert(null, unix2dos, false, null);
133                 File from = session.getFile().getTrueFile();
134                 if (convert.convert(from, unix2dos)) {
135                     futureCompletion.setSuccess();
136                 } else {
137                     R66Result result = new R66Result(session, false, ErrorCode.Internal,
138                             runner);
139                     futureCompletion.setResult(result);
140                     logger.error("Cannot Transcode " +
141                             argRule + ":" + argTransfer + " and " + session);
142                     futureCompletion.setFailure(new OpenR66ProtocolSystemException(
143                             "Cannot Transcode file"));
144                 }
145             }
146             return;
147         }
148         File from = session.getFile().getTrueFile();
149         String finalname = newfilename;
150         if (newfilename != null) {
151             finalname = newfilename;
152         } else if (extension != null) {
153             finalname = from.getAbsolutePath() + "." + extension;
154         } else {
155             finalname = from.getAbsolutePath() + ".transcode";
156         }
157         success = CharsetsUtil.transcode(from.getAbsolutePath(), fromCharset,
158                 finalname, toCharset,
159                 Configuration.BUFFERSIZEDEFAULT);
160         if (success && (dos2unix || unix2dos)) {
161             // now convert it
162             // only 1 of Dos2Unix/Unix2Dos
163             FileConvert convert = new FileConvert(null, unix2dos, false, null);
164             File to = new File(finalname);
165             if (convert.convert(to, unix2dos)) {
166                 futureCompletion.setSuccess();
167             } else {
168                 // only warning
169                 logger.warn("Cannot Unix/Dos Transcode " + to + " : " +
170                         argRule + ":" + argTransfer + " and " + session);
171                 futureCompletion.setSuccess();
172             }
173             return;
174         }
175         if (success) {
176             futureCompletion.setSuccess();
177         } else {
178             logger.error("Cannot Transcode from " + fromCharset + " to " + toCharset + " with " +
179                     argRule + ":" + argTransfer + " and " + session);
180             futureCompletion.setFailure(new OpenR66ProtocolSystemException(
181                     "Cannot Transcode file"));
182         }
183     }
184 
185 }
