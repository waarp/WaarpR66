18  package org.waarp.openr66.context.task;
19  
20  import java.sql.Timestamp;
21  import java.text.ParseException;
22  import java.text.SimpleDateFormat;
23  import java.util.Date;
24  
25  import org.apache.commons.exec.CommandLine;
26  
27  import org.waarp.common.logging.WaarpLogger;
28  import org.waarp.common.logging.WaarpLoggerFactory;
29  import org.waarp.openr66.client.SubmitTransfer;
30  import org.waarp.openr66.context.R66Session;
31  import org.waarp.openr66.context.task.exception.OpenR66RunnerErrorException;
32  import org.waarp.openr66.database.DbConstant;
33  import org.waarp.openr66.database.data.DbTaskRunner;
34  import org.waarp.openr66.protocol.configuration.Configuration;
35  import org.waarp.openr66.protocol.utils.R66Future;
36  
37  /**
38   * Transfer task:<br>
39   * 
40   * Result of arguments will be as r66send command.<br>
41   * Format is like r66send command in any order except "-info" which should be the last item, and "-copyinfo"
42   * will copy at first place the original transfer information as the new one, while still having the possibility to add new informations through "-info":<br>
43   * "-file filepath -to requestedHost -rule rule [-md5] [-start yyyyMMddHHmmss or -delay (delay or +delay)] [-copyinfo] [-info information]" <br>
44   * <br>
45   * INFO is the only one field that can contains blank character.<br>
46   * 
47   * @author Frederic Bregier
48   * 
49   */
50  public class TransferTask extends AbstractExecTask {
51      /**
52       * Internal Logger
53       */
54      private static final WaarpLogger logger = WaarpLoggerFactory
55              .getLogger(TransferTask.class);
56  
57      /**
58       * @param argRule
59       * @param delay
60       * @param argTransfer
61       * @param session
62       */
63      public TransferTask(String argRule, int delay, String argTransfer,
64              R66Session session) {
65          super(TaskType.TRANSFER, delay, argRule, argTransfer, session);
66      }
67  
68      @Override
69      public void run() {
70          logger.info("Transfer with " + argRule + ":" + argTransfer + " and {}",
71                  session);
72          String finalname = applyTransferSubstitutions(argRule);
73  
74          finalname = finalname.replaceAll("#([A-Z]+)#", "\\${$1}");
75          CommandLine commandLine = new CommandLine("dummy");
76          commandLine.setSubstitutionMap(getSubstitutionMap());
77          commandLine.addArguments(finalname, false);
78          String[] args = commandLine.getArguments();
79  
80          if (args.length < 6) {
81              futureCompletion.setFailure(
82                      new OpenR66RunnerErrorException("Not enough argument in Transfer"));
83              return;
84          }
85          String filepath = null;
86          String requested = null;
87          String rule = null;
88          String information = null;
89          String finalInformation = null;
90          boolean isMD5 = false;
91          int blocksize = Configuration.configuration.getBLOCKSIZE();
92          Timestamp timestart = null;
93          for (int i = 0; i < args.length; i++) {
94              if (args[i].equalsIgnoreCase("-to")) {
95                  i++;
96                  requested = args[i];
97                  if (Configuration.configuration.getAliases().containsKey(requested)) {
98                      requested = Configuration.configuration.getAliases().get(requested);
99                  }
100             } else if (args[i].equalsIgnoreCase("-file")) {
101                 i++;
102                 filepath = args[i];
103             } else if (args[i].equalsIgnoreCase("-rule")) {
104                 i++;
105                 rule = args[i];
106             } else if (args[i].equalsIgnoreCase("-copyinfo")) {
107                 finalInformation = argTransfer;
108             } else if (args[i].equalsIgnoreCase("-info")) {
109                 i++;
110                 information = args[i];
111                 i++;
112                 while (i < args.length) {
113                     information += " " + args[i];
114                     i++;
115                 }
116             } else if (args[i].equalsIgnoreCase("-md5")) {
117                 isMD5 = true;
118             } else if (args[i].equalsIgnoreCase("-block")) {
119                 i++;
120                 blocksize = Integer.parseInt(args[i]);
121                 if (blocksize < 100) {
122                     logger.warn("Block size is too small: " + blocksize);
123                     blocksize = Configuration.configuration.getBLOCKSIZE();
124                 }
125             } else if (args[i].equalsIgnoreCase("-start")) {
126                 i++;
127                 SimpleDateFormat dateFormat = new SimpleDateFormat("yyyyMMddHHmmss");
128                 Date date;
129                 try {
130                     date = dateFormat.parse(args[i]);
131                     timestart = new Timestamp(date.getTime());
132                 } catch (ParseException e) {
133                 }
134             } else if (args[i].equalsIgnoreCase("-delay")) {
135                 i++;
136                 try {
137                     if (args[i].charAt(0) == '+') {
138                         timestart = new Timestamp(System.currentTimeMillis() +
139                                 Long.parseLong(args[i].substring(1)));
140                     } else {
141                         timestart = new Timestamp(Long.parseLong(args[i]));
142                     }
143                 } catch (NumberFormatException e) {
144                 }
145             }
146         }
147         if (information != null) {
148             if (finalInformation == null) {
149                 finalInformation = information;
150             } else {
151                 finalInformation += " " + information;
152             }
153         } else if (finalInformation == null) {
154             finalInformation = "noinfo";
155         }
156         R66Future future = new R66Future(true);
157         SubmitTransfer transaction = new SubmitTransfer(future,
158                 requested, filepath, rule, finalInformation, isMD5, blocksize, DbConstant.ILLEGALVALUE,
159                 timestart);
160         transaction.run();
161         future.awaitUninterruptibly();
162         futureCompletion.setResult(future.getResult());
163         DbTaskRunner runner = future.getResult().getRunner();
164         if (future.isSuccess()) {
165             logger.info("Prepare transfer in     SUCCESS     " + runner.toShortString() +
166                     "     <REMOTE>" + requested + "</REMOTE>");
167             futureCompletion.setSuccess();
168         } else {
169             if (runner != null) {
170                 if (future.getCause() == null) {
171                     futureCompletion.cancel();
172                 } else {
173                     futureCompletion.setFailure(future.getCause());
174                 }
175                 logger.error("Prepare transfer in     FAILURE      " + runner.toShortString() +
176                         "     <REMOTE>" + requested + "</REMOTE>", future.getCause());
177             } else {
178                 if (future.getCause() == null) {
179                     futureCompletion.cancel();
180                 } else {
181                     futureCompletion.setFailure(future.getCause());
182                 }
183                 logger.error("Prepare transfer in     FAILURE without any runner back",
184                         future.getCause());
185             }
186         }
187     }
