18  package org.waarp.openr66.context.task;
19  
20  import org.waarp.common.command.exception.CommandAbstractException;
21  import org.waarp.common.logging.WaarpLogger;
22  import org.waarp.common.logging.WaarpLoggerFactory;
23  import org.waarp.openr66.context.ErrorCode;
24  import org.waarp.openr66.context.R66Result;
25  import org.waarp.openr66.context.R66Session;
26  import org.waarp.openr66.protocol.exception.OpenR66ProtocolSystemException;
27  
28  /**
29   * Delete the file. The current file is no more valid.<br>
30   * No arguments are taken into account.
31   * 
32   * @author Frederic Bregier
33   * 
34   */
35  public class DeleteTask extends AbstractTask {
36      /**
37       * Internal Logger
38       */
39      private static final WaarpLogger logger = WaarpLoggerFactory
40              .getLogger(DeleteTask.class);
41  
42      /**
43       * @param argRule
44       * @param delay
45       * @param argTransfer
46       * @param session
47       */
48      public DeleteTask(String argRule, int delay, String argTransfer,
49              R66Session session) {
50          super(TaskType.DELETE, delay, argRule, argTransfer, session);
51      }
52  
53      @Override
54      public void run() {
55          logger.info("Delete file from session {}",
56                  session);
57          try {
58              if (!session.getFile().delete()) {
59                  /*
60                  logger.debug("CANNOT Delete file {} from session {}", session.getFile().getFile(),
61                  		session);
62                  R66Result result = new R66Result(session, false,
63                  		ErrorCode.FileNotFound, session.getRunner());
64                  futureCompletion.setResult(result);
65                  futureCompletion.setFailure(new OpenR66ProtocolSystemException("File not deleted: "+session.getFile().getFile()));
66                  */
67                  logger.warn("CANNOT Delete file {} from session {}", session.getFile().getFile(),
68                          session);
69                  session.getRunner().setErrorExecutionStatus(ErrorCode.Warning);
70                  futureCompletion.setSuccess();
71                  return;
72              }
73          } catch (CommandAbstractException e1) {
74              logger.debug("CANNOT Delete file from session {}",
75                      session, e1);
76              R66Result result = new R66Result(session, false,
77                      ErrorCode.FileNotFound, session.getRunner());
78              futureCompletion.setResult(result);
79              futureCompletion.setFailure(new OpenR66ProtocolSystemException(e1));
80              return;
81          }
82          futureCompletion.setSuccess();
83      }
84  
85  }
