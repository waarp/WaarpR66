33  /**
34   * Reschedule Transfer task to a time delayed by the specified number of milliseconds, if the error
35   * code is one of the specified codes and the optional intervals of date are compatible with the new
36   * time schedule<br>
37   * <br>
38   * 
39   * Result of arguments will be as following options (the two first are mandatory):<br>
40   * <br>
41   * 
42   * "-delay ms" where ms is the added number of ms on current time before retry on schedule<br>
43   * <br>
44   * "-case errorCode,errorCode,..." where errorCode is one of the following error of the current
45   * transfer (either literal or code in 1 character:<br>
46   * ConnectionImpossible(C), ServerOverloaded(l), BadAuthent(A), ExternalOp(E), TransferError(T),
47   * MD5Error(M), Disconnection(D), RemoteShutdown(r), FinalOp(F), Unimplemented(U), Shutdown(S),
48   * RemoteError(R), Internal(I), StoppedTransfer(H), CanceledTransfer(K), Warning(W), Unknown(-),
49   * QueryAlreadyFinished(Q), QueryStillRunning(s), NotKnownHost(N), QueryRemotelyUnknown(u),
50   * FileNotFound(f), CommandNotFound(c), PassThroughMode(p)<br>
51   * <br>
52   * "-between start;end" and/or "-notbetween start;end" (multiple times are allowed, start or end can
53   * be not set) and where start and stop are in the following format:<br>
54   * Yn:Mn:Dn:Hn:mn:Sn where n is a number for each time specification, each specification is
55   * optional, as Y=Year, M=Month, D=Day, H=Hour, m=minute, s=second.<br>
56   * Format can be X+n, X-n, X=n or Xn where X+-n means adding/subtracting n to current date value,
57   * while X=n or Xn means setting exact value<br>
58   * If one time specification is not set, it is based on the current date.<br>
59   * <br>
60   * "-count limit" will be the limit of retry. The current value limit is taken from the "transferInfo" internal code
61   * (not any more the "information of transfer")and not from the rule
62   * as "{"CPTLIMIT": limit}" as JSON code. Each time this function is called, the
63   * limit value will be replaced as newlimit = limit - 1 in the "transferInfo" as "{"CPTLIMIT": limit}" as JSON code.<br>
64   * To ensure correctness, the value must be in the "transferInfo" internal code since this value will be
65   * changed statically in the "transferInfo". If taken from the rule, it will be wrong since
66   * the value will never decrease. However, a value must be setup in the rule in order to reset the value
67   * when the count reach 0. <br>
68   * So in the rule, "-count resetlimit" must be present, where resetlimit will be
69   * the new value set when the limit reach 0, and in the "transferInfo" internal code,
70   * "{"CPTLIMIT": limit}" as JSON code must be present. If one is missing, the condition is not applied. <br>
71   * <br>
72   * If "-notbetween" is specified, the planned date must not be in the area.<br>
73   * If "-between" is specified, the planned date must be found in any such specified areas (could be
74   * in any of the occurrence). If not specified, it only depends on "-notbetween".<br>
75   * If none is specified, the planned date is always valid.<br>
76   * <br>
77   * 
78   * Note that if a previous called to a reschedule was done for this attempt and was successful, the
79   * following calls will be ignored.<br>
80   * <br>
81   * <B>Important note: any subsequent task will be ignored and not executed once the reschedule is accepted.</B><br>
82   * <br>
83   * In case start > end, end will be +1 day<br>
84   * In case start and end < current planned date, both will have +1 day.<br>
85   * <br>
86   * 
87   * Example: -delay 3600000 -case ConnectionImpossible,ServerOverloaded,Shutdown -notbetween
88   * H7:m0:S0;H19:m0:S0 -notbetween H1:m0:S0;H=3:m0:S0<br>
89   * means retry in case of error during initialization of connection in 1 hour if not between 7AM to
90   * 7PM and not between 1AM to 3AM.<br>
91   * 
92   * @author Frederic Bregier
93   * 
94   */
95  public class RescheduleTransferTask extends AbstractTask {
96      /**
97       * Internal Logger
98       */
99      private static final WaarpLogger logger = WaarpLoggerFactory
100             .getLogger(RescheduleTransferTask.class);
101 
102     /**
103      * Delimiter for -count option in Reschedule to be placed in the transfer info of transfer as
104      * {"CPTLIMIT": limit} where limit is an integer.
105      */
106     public static final String CPTLIMIT = "CPTLIMIT";
107     public static final String CPTTOTAL = "CPTTOTAL";
108 
109     protected long newdate = 0;
110 
111     protected Calendar newDate = null;
112 
113     protected boolean countUsed = false;
114 
115     protected int limitCount = -1;
116 
117     protected int totalCount = 0;
118 
119     protected int resetCount = -1;
120 
121     protected DbTaskRunner runner = null;

123     /**
124      * @param argRule
125      * @param delay
126      * @param argTransfer
127      * @param session
128      */
129     public RescheduleTransferTask(String argRule, int delay,
130             String argTransfer, R66Session session) {
131         super(TaskType.RESCHEDULE, delay, argRule, argTransfer, session);
132     }
133 
134     @Override
135     public void run() {
136         logger.info("Reschedule with " + argRule + ":" + argTransfer +
137                 " and {}", session);
138         runner = session.getRunner();
139         if (runner == null) {
140             futureCompletion.setFailure(new OpenR66RunnerErrorException(
141                     "No valid runner in Reschedule"));
142             return;
143         }
144         if (runner.isRescheduledTransfer()) {
145             // Already rescheduled so ignore
146             R66Result result = new R66Result(session, false, ErrorCode.Warning,
147                     runner);
148             futureCompletion.setResult(result);
149             logger.warn("Transfer already Rescheduled: " + runner.toShortString());
150             futureCompletion.setSuccess();
151             return;
152         }
153         if (runner.isSelfRequested()) {
154             // Self Requested Request so reschedule is ignored
155             R66Result result = new R66Result(session, false, ErrorCode.LoopSelfRequestedHost,
156                     runner);
157             futureCompletion.setResult(result);
158             futureCompletion.setFailure(new OpenR66RunnerErrorException(
159                     "No valid runner in Reschedule since Self Requested"));
160             return;
161         }
162         String finalname = argRule;
163         finalname = getReplacedValue(finalname, argTransfer.split(" "));
164         String[] args = finalname.split(" ");
165         if (args.length < 4) {
166             R66Result result = new R66Result(session, false, ErrorCode.Warning,
167                     runner);
168             futureCompletion.setResult(result);
169             logger.warn("Not enough argument in Reschedule: " + runner.toShortString());
170             futureCompletion.setSuccess();
171             return;
172         }
173         if (!validateArgs(args)) {
174             R66Result result = new R66Result(session, false, ErrorCode.Warning,
175                     runner);
176             futureCompletion.setResult(result);
177             logger.warn("Reschedule unallowed due to argument: " + runner.toShortString());
178             futureCompletion.setSuccess();
179             return;
180         }
181         if (countUsed) {
182             limitCount--;
183             if (limitCount >= 0) {
184                 // restart is allowed so resetting to new limitCount 
185                 resetCount = limitCount;
186             }
187             resetInformation(resetCount);
188             if (limitCount < 0) {
189                 // Must not reschedule since limit is reached
190                 try {
191                     runner.saveStatus();
192                 } catch (OpenR66RunnerErrorException e) {
193                 }
194                 R66Result result = new R66Result(session, false, ErrorCode.Warning,
195                         runner);
196                 futureCompletion.setResult(result);
197                 logger.warn("Reschedule unallowed due to limit reached: " + runner.toShortString());
198                 futureCompletion.setSuccess();
199                 return;
200             }
201         }
202         Timestamp start = new Timestamp(newdate);
203         try {
204             runner.setStart(start);
205             if (runner.restart(true)) {
206                 runner.saveStatus();
207             }
208         } catch (OpenR66RunnerErrorException e) {
209             logger.error(
210                     "Prepare transfer in     FAILURE      " +
211                             runner.toShortString() + "     <AT>" +
212                             (new Date(newdate)).toString() + "</AT>", e);
213             futureCompletion.setFailure(new OpenR66RunnerErrorException(
214                     "Reschedule failed: " + e.getMessage(), e));
215             return;
216         }
217         runner.setRescheduledTransfer();
218         R66Result result = new R66Result(session, false, ErrorCode.Warning,
219                 runner);
220         futureCompletion.setResult(result);
221         logger.warn("Reschedule transfer in     SUCCESS     " +
222                 runner.toShortString() + "     <AT>" +
223                 (new Date(newdate)).toString() + "</AT>");
224         futureCompletion.setSuccess();
225     }
226 
227     protected void resetInformation(int value) {
228         Map<String, Object> root = runner.getTransferMap();
229         root.put(CPTLIMIT, value);
230         try {
231             totalCount = (Integer) root.get(CPTTOTAL);
232             totalCount++;
233             root.put(CPTTOTAL, totalCount);
234         } catch (Exception e) {
235             totalCount = 1;
236             root.put(CPTTOTAL, totalCount);
237         }
238         runner.setTransferMap(root);
239     }

241     protected boolean validateArgs(String[] args) {
242         boolean validCode = false;
243         for (int i = 0; i < args.length; i++) {
244             if (args[i].equalsIgnoreCase("-delay")) {
245                 i++;
246                 try {
247                     newdate = Long.parseLong(args[i]);
248                 } catch (NumberFormatException e) {
249                     logger.warn("Bad Long format: args[i]");
250                     return false;
251                 }
252             } else if (args[i].equalsIgnoreCase("-case")) {
253                 i++;
254                 if (!validCode) {
255                     String[] codes = args[i].split(",");
256                     for (int j = 0; j < codes.length; j++) {
257                         ErrorCode code = ErrorCode.getFromCode(codes[j]);
258                         if (session.getLocalChannelReference().getCurrentCode() == code) {
259                             logger.debug("Code valid: " + code);
260                             validCode = true;
261                         }
262                     }
263                 }
264             } else if (args[i].equalsIgnoreCase("-count")) {
265                 i++;
266                 try {
267                     resetCount = Integer.parseInt(args[i]);
268                 } catch (NumberFormatException e) {
269                     logger.warn("ResetLimit is not an integer: " + args[i]);
270                     countUsed = false;
271                     return false;
272                 }
273                 Map<String, Object> root = runner.getTransferMap();
274                 Integer limit = null;
275                 try {
276                     limit = (Integer) root.get(CPTLIMIT);
277                 } catch (Exception e) {
278                     logger.warn("Bad Long format: CPTLIMIT", e);
279                     return false;
280                 }
281                 if (limit != null) {
282                     limitCount = limit;
283                 } else {
284                     limitCount = resetCount;
285                     root.put(CPTLIMIT, limitCount);
286                 }
287                 countUsed = true;
288             }
289         }
290         // now we have new delay plus code
291         if (!validCode) {
292             logger.warn("No valid Code found");
293             return false;
294         }
295         if (newdate <= 0) {
296             logger.warn("Delay is negative: " + newdate);
297             return false;
298         }
299         newdate += System.currentTimeMillis();
300         newDate = Calendar.getInstance();
301         newDate.setTimeInMillis(newdate);
302         boolean betweenTest = false;
303         boolean betweenResult = false;
304         for (int i = 0; i < args.length; i++) {
305             if (args[i].equalsIgnoreCase("-notbetween")) {
306                 i++;
307                 String[] elmts = args[i].split(";");
308                 boolean startModified = false;
309                 String[] values = elmts[0].split(":");
310                 Calendar start = getCalendar(values);
311                 if (start != null) {
312                     startModified = true;
313                 } else {
314                     start = Calendar.getInstance();
315                 }
316                 boolean stopModified = false;
317                 values = elmts[1].split(":");
318                 Calendar stop = getCalendar(values);
319                 if (stop != null) {
320                     stopModified = true;
321                 } else {
322                     stop = Calendar.getInstance();
323                 }
324                 logger.debug("Dates before check: Not between " + start.getTime() + " and "
325                         + stop.getTime());
326                 // Check that start < stop
327                 if (start.compareTo(stop) > 0) {
328                     // no so add 24H to stop
329                     stop.add(Calendar.DAY_OF_MONTH, 1);
330                 }
331                 // Check that start and stop > newDate (only start is checked since start <= stop)
332                 if (start.compareTo(newDate) < 0) {
333                     start.add(Calendar.DAY_OF_MONTH, 1);
334                     stop.add(Calendar.DAY_OF_MONTH, 1);
335                 }
336                 logger.debug("Dates after check: Not between " + start.getTime() + " and "
337                         + stop.getTime());
338                 if (!startModified) {
339                     if (newDate.compareTo(stop) < 0) {
340                         logger.debug("newDate: " + newDate.getTime() + " Should not be between "
341                                 + start.getTime() + " and " + stop.getTime());
342                         return false;
343                     }
344                 } else if (start.compareTo(newDate) < 0) {
345                     if ((!stopModified) || (newDate.compareTo(stop) < 0)) {
346                         logger.debug("newDate: " + newDate.getTime() + " Should not be between "
347                                 + start.getTime() + " and " + stop.getTime());
348                         return false;
349                     }
350                 }
351             } else if (args[i].equalsIgnoreCase("-between")) {
352                 i++;
353                 betweenTest = true;
354                 String[] elmts = args[i].split(";");
355                 boolean startModified = false;
356                 String[] values = elmts[0].split(":");
357                 Calendar start = getCalendar(values);
358                 if (start != null) {
359                     startModified = true;
360                 } else {
361                     start = Calendar.getInstance();
362                 }
363                 boolean stopModified = false;
364                 values = elmts[1].split(":");
365                 Calendar stop = getCalendar(values);
366                 if (stop != null) {
367                     stopModified = true;
368                 } else {
369                     stop = Calendar.getInstance();
370                 }
371                 logger.debug("Dates before check: Between " + start.getTime() + " and "
372                         + stop.getTime());
373                 // Check that start < stop
374                 if (start.compareTo(stop) > 0) {
375                     // no so add 24H to stop
376                     stop.add(Calendar.DAY_OF_MONTH, 1);
377                 }
378                 // Check that start and stop > newDate (only start is checked since start <= stop)
379                 if (start.compareTo(newDate) < 0) {
380                     start.add(Calendar.DAY_OF_MONTH, 1);
381                     stop.add(Calendar.DAY_OF_MONTH, 1);
382                 }
383                 logger.debug("Dates before check: Between " + start.getTime() + " and "
384                         + stop.getTime());
385                 if (!startModified) {
386                     if (newDate.compareTo(stop) < 0) {
387                         logger.debug("newDate: " + newDate.getTime() + " is between "
388                                 + start.getTime() + " and " + stop.getTime());
389                         betweenResult = true;
390                     }
391                 } else if (start.compareTo(newDate) < 0) {
392                     if ((!stopModified) || (newDate.compareTo(stop) < 0)) {
393                         logger.debug("newDate: " + newDate.getTime() + " is between "
394                                 + start.getTime() + " and " + stop.getTime());
395                         betweenResult = true;
396                     }
397                 }
398             }
399         }
400         if (betweenTest) {
401             logger.debug("Since between is specified, do we found newDate: " + newDate.getTime()
402                     + " Result: " + betweenResult);
403             return betweenResult;
404         }
405         logger.debug("newDate: " + newDate.getTime() + " rescheduled");
406         return true;
407     }

409     /**
410      * 
411      * @param values
412      *            as X+n or X-n or X=n or Xn where X=Y/M/D/H/m/s, n=number and +/- meaning
413      *            adding/subtracting from current date and = meaning specific set value
414      * @return the Calendar if any specification, or null if no calendar specified
415      */
416     private Calendar getCalendar(String[] values) {
417         Calendar newcal = Calendar.getInstance();
418         boolean isModified = false;
419         for (int j = 0; j < values.length; j++) {
420             if (values[j].length() > 1) {
421                 int addvalue = 0; // will be different of 0
422                 int value = -1; // will be >= 0
423                 switch (values[j].charAt(0)) {
424                     case '+':
425                         try {
426                             addvalue = Integer.parseInt(values[j].substring(2));
427                         } catch (NumberFormatException e) {
428                             continue;
429                         }
430                         break;
431                     case '-':
432                         try {
433                             addvalue = Integer.parseInt(values[j].substring(1));
434                         } catch (NumberFormatException e) {
435                             continue;
436                         }
437                         break;
438                     case '=':
439                         try {
440                             value = Integer.parseInt(values[j].substring(2));
441                         } catch (NumberFormatException e) {
442                             continue;
443                         }
444                         break;
445                     default: // no sign
446                         try {
447                             value = Integer.parseInt(values[j].substring(1));
448                         } catch (NumberFormatException e) {
449                             continue;
450                         }
451                 }
452                 switch (values[j].charAt(0)) {
453                     case 'Y':
454                         if (value >= 0) {
455                             newcal.set(Calendar.YEAR, value);
456                         } else {
457                             newcal.add(Calendar.YEAR, addvalue);
458                         }
459                         isModified = true;
460                         break;
461                     case 'M':
462                         if (value >= 0) {
463                             newcal.set(Calendar.MONTH, value);
464                         } else {
465                             newcal.add(Calendar.MONTH, addvalue);
466                         }
467                         isModified = true;
468                         break;
469                     case 'D':
470                         if (value >= 0) {
471                             newcal.set(Calendar.DAY_OF_MONTH, value);
472                         } else {
473                             newcal.add(Calendar.DAY_OF_MONTH, addvalue);
474                         }
475                         isModified = true;
476                         break;
477                     case 'H':
478                         if (value >= 0) {
479                             newcal.set(Calendar.HOUR_OF_DAY, value);
480                         } else {
481                             newcal.add(Calendar.HOUR_OF_DAY, addvalue);
482                         }
483                         isModified = true;
484                         break;
485                     case 'm':
486                         if (value >= 0) {
487                             newcal.set(Calendar.MINUTE, value);
488                         } else {
489                             newcal.add(Calendar.MINUTE, addvalue);
490                         }
491                         isModified = true;
492                         break;
493                     case 'S':
494                         if (value >= 0) {
495                             newcal.set(Calendar.SECOND, value);
496                         } else {
497                             newcal.add(Calendar.SECOND, addvalue);
498                         }
499                         isModified = true;
500                         break;
501                 }
502             }
503         }
504         if (isModified)
505             return newcal;
506         return null;
507     }
