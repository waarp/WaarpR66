18  package org.waarp.openr66.context.task;
19  
20  import org.waarp.common.logging.WaarpLogger;
21  import org.waarp.common.logging.WaarpLoggerFactory;
22  import org.waarp.openr66.context.ErrorCode;
23  import org.waarp.openr66.context.R66Result;
24  import org.waarp.openr66.context.R66Session;
25  import org.waarp.openr66.context.task.exception.OpenR66RunnerException;
26  import org.waarp.openr66.database.data.DbTaskRunner;
27  
28  /**
29   * This task checks some properties relative to the File according to the following argument:<br>
30   * - SIZE LT/GT/LTE/GTE/EQ number : to check the file size according to a limit (less than, greater than, less than or equal,
31   * greater than or equal, equal)<br>
32   * - DFCHECK : to check if the future received file size is compatible with the space left on both working and received
33   * directories (from general configuration or rule configuration)<br>
34   * <br>
35   * For instance "SIZE LT 10000000 SIZE GT 1000 DFCHECK" will test that the current file size is less than 10 MB (base 10),
36   * and greater than 1000 bytes, and that the working and received directories have enough space to receive the file.
37   * 
38   * @author Frederic Bregier
39   * 
40   */
41  public class FileCheckTask extends AbstractTask {
42      /**
43       * Internal Logger
44       */
45      private static final WaarpLogger logger = WaarpLoggerFactory
46              .getLogger(FileCheckTask.class);
47  
48      private static enum FC_COMMAND {
49          SIZE, DFCHECK
50      };
51  
52      private static enum FC_ARGSSIZE {
53          LT, GT, LTE, GTE, EQ
54      };
55  
56      /**
57       * @param argRule
58       * @param delay
59       * @param argTransfer
60       * @param session
61       */
62      public FileCheckTask(String argRule, int delay, String argTransfer,
63              R66Session session) {
64          super(TaskType.CHKFILE, delay, argRule, argTransfer, session);
65      }
66  
67      @Override
68      public void run() {
69          DbTaskRunner runner = session.getRunner();
70          if (runner == null) {
71              // no information so in error
72              logger.error("No Runner task so cannot check file: " +
73                      session.toString());
74              futureCompletion.setFailure(new OpenR66RunnerException("No Runner task so cannot check File"));
75              return;
76          }
77          long supposelength = runner.getOriginalSize();
78          if (supposelength <= 0) {
79              // no information but could be not an error so ignore
80              logger.warn("No file size known: " +
81                      session.toString());
82              futureCompletion.setSuccess();
83              return;
84          }
85          String[] commands = argRule.split(" ");
86          int current = 0;
87          while (current < commands.length) {
88              if (commands[current].equalsIgnoreCase(FC_COMMAND.SIZE.name())) {
89                  current++;
90                  FC_ARGSSIZE arg = null;
91                  try {
92                      arg = FC_ARGSSIZE.valueOf(commands[current]);
93                  } catch (IllegalArgumentException e) {
94                      arg = null;
95                  }
96                  if (arg != null) {
97                      current++;
98                      try {
99                          long tocompare = Long.parseLong(commands[current]);
100                         current++;
101                         boolean result = true;
102                         switch (arg) {
103                             case EQ:
104                                 result = supposelength == tocompare;
105                                 break;
106                             case GT:
107                                 result = supposelength > tocompare;
108                                 break;
109                             case GTE:
110                                 result = supposelength >= tocompare;
111                                 break;
112                             case LT:
113                                 result = supposelength < tocompare;
114                                 break;
115                             case LTE:
116                                 result = supposelength <= tocompare;
117                                 break;
118                             default:
119                                 result = true; // ??
120                                 break;
121                         }
122                         logger.debug("DEBUG: " + supposelength + " " + arg.name() + " " + tocompare);
123                         if (result == false) {
124                             // error so stop
125                             logger.error("File length is incompatible with specified SIZE comparizon: " +
126                                     supposelength + " NOT " + arg.name() + " " + tocompare);
127                             futureCompletion.setResult(new R66Result(session, false, ErrorCode.SizeNotAllowed, runner));
128                             futureCompletion.setFailure(new OpenR66RunnerException("File size incompatible"));
129                             return;
130                         }
131                     } catch (NumberFormatException e) {
132                         // ignore and continue
133                     }
134                 }
135             } else if (commands[current].equalsIgnoreCase(FC_COMMAND.DFCHECK.name())) {
136                 current++;
137                 long freesize = runner.freespace(session, true);
138                 if (freesize > 0 && supposelength > freesize) {
139                     // error so stop
140                     logger.error("File length is incompatible with available space in Working directory: " +
141                             supposelength + " > " + freesize);
142                     futureCompletion.setResult(new R66Result(session, false, ErrorCode.SizeNotAllowed, runner));
143                     futureCompletion.setFailure(new OpenR66RunnerException(
144                             "File size incompatible with Working directory"));
145                     return;
146                 }
147                 logger.debug("DEBUG: " + supposelength + " < " + freesize);
148                 freesize = runner.freespace(session, false);
149                 if (freesize > 0 && supposelength > freesize) {
150                     // error so stop
151                     logger.error("File length is incompatible with available space in Recv directory: " +
152                             supposelength + " > " + freesize);
153                     futureCompletion.setResult(new R66Result(session, false, ErrorCode.SizeNotAllowed, runner));
154                     futureCompletion
155                             .setFailure(new OpenR66RunnerException("File size incompatible with Recv directory"));
156                     return;
157                 }
158                 logger.debug("DEBUG: " + supposelength + " < " + freesize);
159             } else {
160                 // ignore and continue
161                 current++;
162             }
163         }
164         logger.debug("DEBUG: End of check " + supposelength);
165         futureCompletion.setSuccess();
166         return;
167     }
