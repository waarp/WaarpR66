18  package org.waarp.openr66.context.task;
19  
20  import java.io.File;
21  
22  import org.waarp.common.logging.WaarpLogger;
23  import org.waarp.common.logging.WaarpLoggerFactory;
24  import org.waarp.openr66.context.R66Session;
25  import org.waarp.openr66.protocol.exception.OpenR66ProtocolSystemException;
26  import org.waarp.openr66.protocol.utils.FileUtils;
27  
28  /**
29   * Copy and Rename task
30   * 
31   * @author Frederic Bregier
32   * 
33   */
34  public class CopyRenameTask extends AbstractTask {
35      /**
36       * Internal Logger
37       */
38      private static final WaarpLogger logger = WaarpLoggerFactory
39              .getLogger(CopyRenameTask.class);
40  
41      /**
42       * @param argRule
43       * @param delay
44       * @param argTransfer
45       * @param session
46       */
47      public CopyRenameTask(String argRule, int delay, String argTransfer,
48              R66Session session) {
49          super(TaskType.COPYRENAME, delay, argRule, argTransfer, session);
50      }
51  
52      @Override
53      public void run() {
54          String finalname = argRule;
55          finalname = getReplacedValue(finalname, argTransfer.split(" ")).trim().replace('\\', '/');
56          logger.info("Copy and Rename to " + finalname + " with " + argRule +
57                  ":" + argTransfer + " and {}", session);
58          File from = session.getFile().getTrueFile();
59          File to = new File(finalname);
60          try {
61              FileUtils.copy(from, to, false, false);
62          } catch (OpenR66ProtocolSystemException e1) {
63              logger.error("Copy and Rename to " + finalname + " with " +
64                      argRule + ":" + argTransfer + " and " + session, e1);
65              futureCompletion.setFailure(new OpenR66ProtocolSystemException(e1));
66              return;
67          }
68          futureCompletion.setSuccess();
69      }
70  
71  }
