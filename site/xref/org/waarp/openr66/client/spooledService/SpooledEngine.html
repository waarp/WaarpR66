40  /**
41   * Engine used to start and stop the SpooledDirectory service
42   * 
43   * @author Frederic Bregier
44   *
45   */
46  public class SpooledEngine extends EngineAbstract {
47      /**
48       * Internal Logger
49       */
50      private static final WaarpLogger logger = WaarpLoggerFactory
51              .getLogger(SpooledEngine.class);
52  
53      static final WaarpFuture closeFuture = new WaarpFuture(true);
54  
55      @Override
56      public void run() {
57          String config = SystemPropertyUtil.get(R66SystemProperties.OPENR66_CONFIGFILE);
58          if (config == null) {
59              logger.error("Cannot find " + R66SystemProperties.OPENR66_CONFIGFILE + " parameter for SpooledEngine");
60              closeFuture.cancel();
61              shutdown();
62              return;
63          }
64          Configuration.configuration.getShutdownConfiguration().serviceFuture = closeFuture;
65          try {
66              Properties prop = new Properties();
67              FileInputStream in = new FileInputStream(config);
68              try {
69                  prop.load(in);
70                  ArrayList<String> array = new ArrayList<String>();
71                  for (Object okey : prop.keySet()) {
72                      String key = (String) okey;
73                      String val = prop.getProperty(key);
74                      if (key.equals("xmlfile")) {
75                          if (val != null && !val.trim().isEmpty()) {
76                              array.add(0, val);
77                          } else {
78                              throw new Exception("Initialization in error: missing xmlfile");
79                          }
80                      } else {
81                          array.add("-" + key);
82                          if (val != null && !val.trim().isEmpty()) {
83                              array.add(val);
84                          }
85                      }
86                  }
87                  if (!SpooledDirectoryTransfer.initialize((String[]) array.toArray(new String[] {}), false)) {
88                      throw new Exception("Initialization in error");
89                  }
90              } finally {
91                  in.close();
92              }
93          } catch (Throwable e) {
94              logger.error("Cannot start SpooledDirectory", e);
95              closeFuture.cancel();
96              shutdown();
97              return;
98          }
99          logger.warn("SpooledDirectory Service started with " + config);
100     }
101 
102     @Override
103     public void shutdown() {
104         R66ShutdownHook.shutdownWillStart();
105         logger.info("Shutdown");
106         for (SpooledDirectoryTransfer spooled : SpooledDirectoryTransfer.list) {
107             spooled.stop();
108         }
109         Configuration.configuration.setTIMEOUTCON(Configuration.configuration.getTIMEOUTCON() / 10);
110         try {
111             while (!SpooledDirectoryTransfer.executorService.awaitTermination(Configuration.configuration.getTIMEOUTCON(),
112                     TimeUnit.MILLISECONDS)) {
113                 Thread.sleep(Configuration.configuration.getTIMEOUTCON());
114             }
115         } catch (InterruptedException e) {
116         }
117         for (SpooledDirectoryTransfer spooledDirectoryTransfer : SpooledDirectoryTransfer.list) {
118             logger.warn(Messages.getString("SpooledDirectoryTransfer.58") + spooledDirectoryTransfer.name + ": "
119                     + spooledDirectoryTransfer.getSent()
120                     + " success, " + spooledDirectoryTransfer.getError()
121                     + Messages.getString("SpooledDirectoryTransfer.60")); //$NON-NLS-1$
122         }
123         SpooledDirectoryTransfer.list.clear();
124         logger.info("Shutdown network");
125         SpooledDirectoryTransfer.networkTransactionStatic.closeAll(false);
126         logger.info("All");
127         ChannelUtils.startShutdown();
128         closeFuture.setSuccess();
129         logger.info("SpooledDirectory Service stopped");
130     }
131 
132     @Override
133     public boolean isShutdown() {
134         return closeFuture.isDone();
135     }
136 
137     @Override
138     public boolean waitShutdown() throws InterruptedException {
139         closeFuture.await();
140         logger.info("Shutdown on going: " + closeFuture.isSuccess());
141         return closeFuture.isSuccess();
142     }
143 }
