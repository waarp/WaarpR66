37  /**
38   * Client to submit a transfer
39   * 
40   * @author Frederic Bregier
41   * 
42   */
43  public class SubmitTransfer extends AbstractTransfer {
44  
45      public SubmitTransfer(R66Future future, String remoteHost,
46              String filename, String rulename, String fileinfo, boolean isMD5, int blocksize,
47              long id,
48              Timestamp starttime) {
49          super(SubmitTransfer.class,
50                  future, filename, rulename, fileinfo, isMD5, remoteHost, blocksize, id, starttime);
51      }
52  
53      public void run() {
54          if (logger == null) {
55              logger = WaarpLoggerFactory.getLogger(SubmitTransfer.class);
56          }
57          if (!DbConstant.admin.isActive()) {
58              logger.debug("Client not connected");
59              R66Result result = new R66Result(new OpenR66DatabaseGlobalException("No database connexion"), null, true,
60                      ErrorCode.Internal, null);
61              future.setResult(result);
62              future.setFailure(result.getException());
63              return;
64          }
65          long srcId = id;
66          DbTaskRunner taskRunner = this.initRequest();
67          if (taskRunner == null) {
68              logger.debug("Cannot prepare task");
69              if (future.isFailed() && future.getResult() != null) {
70                  return;
71              }
72              R66Result result = new R66Result(new OpenR66DatabaseGlobalException(), null, true,
73                      ErrorCode.Internal, taskRunner);
74              future.setResult(result);
75              future.setFailure(result.getException());
76              return;
77          }
78          if (srcId != DbConstant.ILLEGALVALUE) {
79              // Resubmit call, some checks are needed
80              if (!taskRunner.restart(true)) {
81                  // cannot be done from there => must be done through IHM
82                  logger.debug("Cannot prepare task from there. IHM must be used");
83                  R66Result result = new R66Result(
84                          new OpenR66DatabaseGlobalException("Cannot prepare task from there. IHM must be used"),
85                          null, true,
86                          ErrorCode.Internal, taskRunner);
87                  future.setResult(result);
88                  future.setFailure(result.getException());
89                  return;
90              }
91          } else {
92              taskRunner.changeUpdatedInfo(AbstractDbData.UpdatedInfo.TOSUBMIT);
93          }
94          if (!taskRunner.forceSaveStatus()) {
95              try {
96                  if (!taskRunner.specialSubmit()) {
97                      logger.debug("Cannot prepare task");
98                      R66Result result = new R66Result(new OpenR66DatabaseGlobalException("Cannot prepare Task"), null,
99                              true,
100                             ErrorCode.Internal, taskRunner);
101                     future.setResult(result);
102                     future.setFailure(result.getException());
103                     return;
104                 }
105             } catch (WaarpDatabaseException e) {
106                 logger.debug("Cannot prepare task");
107                 R66Result result = new R66Result(new OpenR66DatabaseGlobalException("Cannot prepare Task"), null, true,
108                         ErrorCode.Internal, taskRunner);
109                 future.setResult(result);
110                 future.setFailure(result.getException());
111                 return;
112             }
113         }
114         R66Result result = new R66Result(null, false, ErrorCode.InitOk, taskRunner);
115         future.setResult(result);
116         future.setSuccess();
117     }
118 
119     /**
120      * 
121      * @param args
122      *            configuration file, the remoteHost Id, the file to transfer, the rule, file
123      *            transfer information as arguments and optionally isMD5=1 for true or 0 for
124      *            false(default) and the blocksize if different than default
125      */
126     public static void main(String[] args) {
127         WaarpLoggerFactory.setDefaultFactory(new WaarpSlf4JLoggerFactory(null));
128         if (logger == null) {
129             logger = WaarpLoggerFactory.getLogger(SubmitTransfer.class);
130         }
131         if (!getParams(args, true)) {
132             logger.error(Messages.getString("Configuration.WrongInit")); //$NON-NLS-1$
133             if (!OutputFormat.isQuiet()) {
134                 System.out.println(Messages.getString("Configuration.WrongInit")); //$NON-NLS-1$
135             }
136             if (DbConstant.admin != null && DbConstant.admin.isActive()) {
137                 DbConstant.admin.close();
138             }
139             ChannelUtils.stopLogger();
140             System.exit(1);
141         }
142         R66Future future = new R66Future(true);
143         SubmitTransfer transaction = new SubmitTransfer(future,
144                 rhost, localFilename, rule, fileInfo, ismd5, block, idt,
145                 ttimestart);
146         transaction.normalInfoAsWarn = snormalInfoAsWarn;
147         transaction.run();
148         future.awaitUninterruptibly();
149         DbTaskRunner runner = future.getResult().getRunner();
150         OutputFormat outputFormat = new OutputFormat(SubmitTransfer.class.getSimpleName(), args);
151         if (future.isSuccess()) {
152             outputFormat.setValue(FIELDS.status.name(), 0);
153             outputFormat.setValue(FIELDS.statusTxt.name(),
154                     Messages.getString("SubmitTransfer.3") + Messages.getString("RequestInformation.Success")); //$NON-NLS-1$
155             outputFormat.setValue(FIELDS.remote.name(), rhost);
156             outputFormat.setValueString(runner.getJson());
157             if (transaction.normalInfoAsWarn) {
158                 logger.warn(outputFormat.loggerOut());
159             } else {
160                 logger.info(outputFormat.loggerOut());
161             }
162             if (!OutputFormat.isQuiet()) {
163                 outputFormat.sysout();
164             }
165         } else {
166             outputFormat.setValue(FIELDS.status.name(), 2);
167             if (runner == null) {
168                 outputFormat.setValue(FIELDS.statusTxt.name(),
169                         Messages.getString("SubmitTransfer.3") + Messages.getString("Transfer.FailedNoId")); //$NON-NLS-1$
170                 outputFormat.setValue(FIELDS.remote.name(), rhost);
171             } else {
172                 outputFormat.setValue(FIELDS.statusTxt.name(),
173                         Messages.getString("SubmitTransfer.3") + Messages.getString("RequestInformation.Failure")); //$NON-NLS-1$
174                 outputFormat.setValue(FIELDS.remote.name(), rhost);
175                 outputFormat.setValueString(runner.getJson());
176             }
177             logger.error(outputFormat.loggerOut(), future.getCause());
178             if (future.getCause() != null) {
179                 outputFormat.setValue(FIELDS.error.name(), future.getCause().getMessage());
180             }
181             if (!OutputFormat.isQuiet()) {
182                 outputFormat.sysout();
183             }
184             DbConstant.admin.close();
185             ChannelUtils.stopLogger();
186             System.exit(future.getResult().getCode().ordinal());
187         }
188         DbConstant.admin.close();
189         System.exit(0);
190     }
