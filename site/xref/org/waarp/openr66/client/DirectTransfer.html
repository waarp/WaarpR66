41  /**
42   * Direct Transfer from a client with or without database connection
43   * 
44   * @author Frederic Bregier
45   * 
46   */
47  public class DirectTransfer extends AbstractTransfer {
48      protected final NetworkTransaction networkTransaction;
49  
50      public DirectTransfer(R66Future future, String remoteHost,
51              String filename, String rulename, String fileinfo, boolean isMD5, int blocksize,
52              long id,
53              NetworkTransaction networkTransaction) {
54          // no starttime since it is direct (blocking request, no delay)
55          super(DirectTransfer.class,
56                  future, filename, rulename, fileinfo, isMD5, remoteHost, blocksize, id, null);
57          this.networkTransaction = networkTransaction;
58      }
59  
60      /**
61       * Prior to call this method, the pipeline and NetworkTransaction must have been initialized. It
62       * is the responsibility of the caller to finish all network resources.
63       */
64      public void run() {
65          if (logger == null) {
66              logger = WaarpLoggerFactory.getLogger(DirectTransfer.class);
67          }
68          DbTaskRunner taskRunner = this.initRequest();
69          if (taskRunner == null) {
70              // already an error from there
71              return;
72          }
73          ClientRunner runner = new ClientRunner(networkTransaction, taskRunner, future);
74          OpenR66ProtocolNotYetConnectionException exc = null;
75          for (int i = 0; i < Configuration.RETRYNB; i++) {
76              try {
77                  runner.runTransfer();
78                  exc = null;
79                  break;
80              } catch (OpenR66RunnerErrorException e) {
81                  logger.debug("Cannot Transfer", e);
82                  future.setResult(new R66Result(e, null, true,
83                          ErrorCode.Internal, taskRunner));
84                  future.setFailure(e);
85                  return;
86              } catch (OpenR66ProtocolNoConnectionException e) {
87                  logger.debug("Cannot Connect", e);
88                  future.setResult(new R66Result(e, null, true,
89                          ErrorCode.ConnectionImpossible, taskRunner));
90                  finalizeInErrorTransferRequest(runner, taskRunner, ErrorCode.ConnectionImpossible);
91                  // since no connection : just forget it
92                  if (nolog || taskRunner.shallIgnoreSave()) {
93                      try {
94                          taskRunner.delete();
95                      } catch (WaarpDatabaseException e1) {
96                      }
97                  }
98                  future.setFailure(e);
99                  return;
100             } catch (OpenR66ProtocolPacketException e) {
101                 logger.debug("Bad Protocol", e);
102                 future.setResult(new R66Result(e, null, true,
103                         ErrorCode.TransferError, taskRunner));
104                 future.setFailure(e);
105                 return;
106             } catch (OpenR66ProtocolNotYetConnectionException e) {
107                 logger.debug("Not Yet Connected", e);
108                 exc = e;
109                 continue;
110             }
111         }
112         if (exc != null) {
113             taskRunner.setLocalChannelReference(new LocalChannelReference());
114             logger.debug("Cannot Connect", exc);
115             future.setResult(new R66Result(exc, null, true,
116                     ErrorCode.ConnectionImpossible, taskRunner));
117             // since no connection : just forget it
118             if (nolog || taskRunner.shallIgnoreSave()) {
119                 try {
120                     taskRunner.delete();
121                 } catch (WaarpDatabaseException e1) {
122                 }
123             }
124             future.setFailure(exc);
125             return;
126         }
127     }

129     public static void main(String[] args) {
130         WaarpLoggerFactory.setDefaultFactory(new WaarpSlf4JLoggerFactory(null));
131         if (logger == null) {
132             logger = WaarpLoggerFactory.getLogger(DirectTransfer.class);
133         }
134         if (!getParams(args, false)) {
135             logger.error(Messages.getString("Configuration.WrongInit")); //$NON-NLS-1$
136             if (!OutputFormat.isQuiet()) {
137                 System.out.println(Messages.getString("Configuration.WrongInit")); //$NON-NLS-1$
138             }
139             if (DbConstant.admin != null && DbConstant.admin.isActive()) {
140                 DbConstant.admin.close();
141             }
142             ChannelUtils.stopLogger();
143             System.exit(2);
144         }
145         long time1 = System.currentTimeMillis();
146         R66Future future = new R66Future(true);
147 
148         Configuration.configuration.pipelineInit();
149         NetworkTransaction networkTransaction = new NetworkTransaction();
150         try {
151             DirectTransfer transaction = new DirectTransfer(future,
152                     rhost, localFilename, rule, fileInfo, ismd5, block, idt,
153                     networkTransaction);
154             transaction.normalInfoAsWarn = snormalInfoAsWarn;
155             logger.debug("rhost: " + rhost + ":" + transaction.remoteHost);
156             transaction.run();
157             future.awaitUninterruptibly();
158             long time2 = System.currentTimeMillis();
159             logger.debug("finish transfer: " + future.isSuccess());
160             long delay = time2 - time1;
161             R66Result result = future.getResult();
162             OutputFormat outputFormat = new OutputFormat(DirectTransfer.class.getSimpleName(), args);
163             if (future.isSuccess()) {
164                 if (result.getRunner().getErrorInfo() == ErrorCode.Warning) {
165                     outputFormat.setValue(FIELDS.status.name(), 1);
166                     outputFormat.setValue(FIELDS.statusTxt.name(),
167                             Messages.getString("Transfer.Status") + Messages.getString("RequestInformation.Warned")); //$NON-NLS-1$
168                 } else {
169                     outputFormat.setValue(FIELDS.status.name(), 0);
170                     outputFormat.setValue(FIELDS.statusTxt.name(),
171                             Messages.getString("Transfer.Status") + Messages.getString("RequestInformation.Success")); //$NON-NLS-1$
172                 }
173                 outputFormat.setValue(FIELDS.remote.name(), rhost);
174                 outputFormat.setValueString(result.getRunner().getJson());
175                 outputFormat.setValue("filefinal", (result.getFile() != null ? result.getFile().toString() : "no file"));
176                 outputFormat.setValue("delay", delay);
177                 if (transaction.normalInfoAsWarn) {
178                     logger.warn(outputFormat.loggerOut());
179                 } else {
180                     logger.info(outputFormat.loggerOut());
181                 }
182                 if (!OutputFormat.isQuiet()) {
183                     outputFormat.sysout();
184                 }
185                 if (nolog || result.getRunner().shallIgnoreSave()) {
186                     // In case of success, delete the runner
187                     try {
188                         result.getRunner().delete();
189                     } catch (WaarpDatabaseException e) {
190                         logger.warn("Cannot apply nolog to     " + result.getRunner().toShortString(),
191                                 e);
192                     }
193                 }
194                 networkTransaction.closeAll();
195                 System.exit(0);
196             } else {
197                 if (result == null || result.getRunner() == null) {
198                     outputFormat.setValue(FIELDS.status.name(), 2);
199                     outputFormat.setValue(FIELDS.statusTxt.name(), Messages.getString("Transfer.FailedNoId")); //$NON-NLS-1$
200                     outputFormat.setValue(FIELDS.remote.name(), rhost);
201                     logger.error(outputFormat.loggerOut(), future.getCause());
202                     outputFormat.setValue(FIELDS.error.name(), future.getCause().getMessage());
203                     if (!OutputFormat.isQuiet()) {
204                         outputFormat.sysout();
205                     }
206                     networkTransaction.closeAll();
207                     System.exit(ErrorCode.Unknown.ordinal());
208                 }
209                 if (result.getRunner().getErrorInfo() == ErrorCode.Warning) {
210                     outputFormat.setValue(FIELDS.status.name(), 1);
211                     outputFormat.setValue(FIELDS.statusTxt.name(),
212                             Messages.getString("Transfer.Status") + Messages.getString("RequestInformation.Warned")); //$NON-NLS-1$
213                 } else {
214                     outputFormat.setValue(FIELDS.status.name(), 2);
215                     outputFormat.setValue(FIELDS.statusTxt.name(),
216                             Messages.getString("Transfer.Status") + Messages.getString("RequestInformation.Failure")); //$NON-NLS-1$
217                 }
218                 outputFormat.setValue(FIELDS.remote.name(), rhost);
219                 outputFormat.setValueString(result.getRunner().getJson());
220                 if (result.getRunner().getErrorInfo() == ErrorCode.Warning) {
221                     logger.warn(outputFormat.loggerOut(), future.getCause());
222                 } else {
223                     logger.error(outputFormat.loggerOut(), future.getCause());
224                 }
225                 outputFormat.setValue(FIELDS.error.name(), future.getCause().getMessage());
226                 if (!OutputFormat.isQuiet()) {
227                     outputFormat.sysout();
228                 }
229                 networkTransaction.closeAll();
230                 System.exit(result.getCode().ordinal());
231             }
232         } catch (Throwable e) {
233             logger.error("Exception", e);
234         } finally {
235             logger.debug("finish transfer: " + future.isDone() + ":" + future.isSuccess());
236             networkTransaction.closeAll();
237             // In case something wrong append
238             if (future.isDone() && future.isSuccess()) {
239                 System.exit(0);
240             } else {
241                 System.exit(66);
242             }
243         }
244     }
