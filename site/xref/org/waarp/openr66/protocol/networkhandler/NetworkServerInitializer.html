18  package org.waarp.openr66.protocol.networkhandler;
19  
20  import java.util.concurrent.TimeUnit;
21  
22  import io.netty.channel.ChannelHandlerContext;
23  import io.netty.channel.ChannelInitializer;
24  import io.netty.channel.ChannelPipeline;
25  import io.netty.channel.socket.SocketChannel;
26  import io.netty.handler.timeout.IdleStateHandler;
27  import io.netty.handler.traffic.ChannelTrafficShapingHandler;
28  import io.netty.handler.traffic.GlobalTrafficShapingHandler;
29  
30  import org.waarp.openr66.protocol.configuration.Configuration;
31  import org.waarp.openr66.protocol.exception.OpenR66ProtocolNetworkException;
32  import org.waarp.openr66.protocol.networkhandler.packet.NetworkPacketCodec;
33  
34  /**
35   * NetworkServer pipeline (Requester side)
36   * 
37   * @author Frederic Bregier
38   */
39  public class NetworkServerInitializer extends ChannelInitializer<SocketChannel> {
40  
41      public static final String TIMEOUT = "timeout";
42      public static final String READTIMEOUT = "readTimeout";
43      public static final String LIMITGLOBAL = "GLOBALLIMIT";
44      public static final String LIMITCHANNEL = "CHANNELLIMIT";
45  
46      protected boolean server = false;
47  
48      public NetworkServerInitializer(boolean server) {
49          this.server = server;
50      }
51  
52      @Override
53      protected void initChannel(SocketChannel ch) throws Exception {
54          final ChannelPipeline pipeline = ch.pipeline();
55          pipeline.addLast(TIMEOUT, new IdleStateHandler(0, 0,
56                      Configuration.configuration.getTIMEOUTCON(),
57                      TimeUnit.MILLISECONDS));
58          // Global limitation
59  	GlobalTrafficShapingHandler handler =
60                  Configuration.configuration.getGlobalTrafficShapingHandler();
61  	if (handler == null) {
62  		throw new OpenR66ProtocolNetworkException(
63  			"Error at pipeline initialization,"
64  			+ " GlobalTrafficShapingHandler configured.");
65  	}
66          pipeline.addLast(LIMITGLOBAL, handler);
67          // Per channel limitation
68          pipeline.addLast(LIMITCHANNEL,
69                  new ChannelTrafficShapingHandler(
70                      Configuration.configuration.getServerChannelWriteLimit(),
71                      Configuration.configuration.getServerChannelReadLimit(),
72                      Configuration.configuration.getDelayLimit()));
73          pipeline.addLast("codec", new NetworkPacketCodec());
74          pipeline.addLast(Configuration.configuration.getHandlerGroup(),
75                  "handler", new NetworkServerHandler(this.server));
76      }
77  }
