18  package org.waarp.openr66.protocol.networkhandler;
19  
20  import io.netty.util.internal.ConcurrentSet;
21  
22  import java.util.Set;
23  
24  /**
25   * Client NetworkChannelReference attached to one HostId.
26   * 
27   * 
28   * 
29   * This class is used to keep information for one HostID when it connects as client to the current
30   * Host. As one Id can be shared or one can use direct send, so having a connection by request, this
31   * class is useful when one wants to know who is connected and how many times.
32   * 
33   * 
34   * @author Frederic Bregier
35   * 
36   */
37  public class ClientNetworkChannels {
38  
39      private final String hostId;
40      private final Set<NetworkChannelReference> networkChannelReferences = new ConcurrentSet<NetworkChannelReference>();
41  
42      public ClientNetworkChannels(String hostId) {
43          this.hostId = hostId;
44      }
45  
46      public void add(NetworkChannelReference networkChannelReference) {
47          networkChannelReferences.add(networkChannelReference);
48          networkChannelReference.clientNetworkChannels = this;
49          networkChannelReference.setHostId(this.hostId);
50      }
51  
52      public void remove(NetworkChannelReference networkChannelReference) {
53          networkChannelReferences.remove(networkChannelReference);
54      }
55  
56      public boolean isEmpty() {
57          return networkChannelReferences.isEmpty();
58      }
59  
60      public int size() {
61          return networkChannelReferences.size();
62      }
63  
64      public boolean shutdownAllNetworkChannels() {
65          boolean status = false;
66          for (NetworkChannelReference networkChannelReference : networkChannelReferences) {
67              NetworkTransaction.shuttingDownNetworkChannel(networkChannelReference);
68              status = true;
69          }
70          networkChannelReferences.clear();
71          return status;
72      }
73  
74      public String getHostId() {
75          return hostId;
76      }
77  }
