38  /**
39   * Packet Decoder
40   * 
41   * @author Frederic Bregier
42   */
43  public class NetworkPacketCodec extends ByteToMessageCodec<NetworkPacket> {
44      /**
45       * Internal Logger
46       */
47      private static final WaarpLogger logger = WaarpLoggerFactory.getLogger(NetworkPacketCodec.class);
48  
49  
50      @Override
51      protected void decode(ChannelHandlerContext ctx, ByteBuf buf, List<Object> out) throws Exception {
52          // Make sure if the length field was received.
53          if (buf.readableBytes() < 4) {
54              // The length field was not received yet - return null.
55              // This method will be invoked again when more packets are
56              // received and appended to the buffer.
57              return;
58          }
59          // Mark the current buffer position
60          buf.markReaderIndex();
61          // Read the length field
62          final int length = buf.readInt();
63          if (length < 9) {
64              throw new OpenR66ProtocolPacketException("Incorrect decode first field in Network Packet: " + length
65                      + " < 9");
66          }
67          if (buf.readableBytes() < length) {
68              buf.resetReaderIndex();
69              return;
70          }
71          // Now we can read the two Ids
72          final int localId = buf.readInt();
73          final int remoteId = buf.readInt();
74          final byte code = buf.readByte();
75          int readerInder = buf.readerIndex();
76          ByteBuf buffer = buf.slice(readerInder, length - 9);
77          buffer.retain();
78          buf.skipBytes(length - 9);
79          NetworkPacket networkPacket = new NetworkPacket(localId, remoteId, code, buffer);
80          logger.trace("received network packet {}", networkPacket);
81          if (code == LocalPacketFactory.KEEPALIVEPACKET) {
82              KeepAlivePacket keepAlivePacket = (KeepAlivePacket)
83                      LocalPacketCodec.decodeNetworkPacket(networkPacket.getBuffer());
84              if (keepAlivePacket.isToValidate()) {
85                  keepAlivePacket.validate();
86                  NetworkPacket response =
87                          new NetworkPacket(ChannelUtils.NOCHANNEL,
88                                  ChannelUtils.NOCHANNEL, keepAlivePacket, null);
89                  NetworkChannelReference nc = NetworkTransaction.getImmediateNetworkChannel(ctx.channel());
90                  if (nc != null) {
91                      nc.useIfUsed();
92                  }
93                  ctx.writeAndFlush(response.getNetworkPacket());
94                  buffer.release();
95              }
96              // Replaced by a NoOp packet
97              networkPacket = new NetworkPacket(localId, remoteId, new NoOpPacket(), null);
98              NetworkServerHandler nsh = (NetworkServerHandler) ctx.pipeline().last();
99              nsh.setKeepAlivedSent();
100         }
101         out.add(networkPacket);
102     }
103 
104     @Override
105     protected void encode(ChannelHandlerContext ctx, NetworkPacket msg, ByteBuf out) throws Exception {
106         final NetworkPacket packet = msg;
107         logger.trace("sending network packet {}", packet);
108         final ByteBuf finalBuf = packet.getNetworkPacket();
109         out.writeBytes(finalBuf);
110         finalBuf.release();
111         //msg.getBuffer().release();
112     }
113 
114 }
