18  package org.waarp.openr66.protocol.http.adminssl;
19  
20  import io.netty.channel.ChannelPipeline;
21  import io.netty.channel.ChannelInitializer;
22  import io.netty.channel.socket.SocketChannel;
23  import io.netty.handler.codec.http.HttpContentCompressor;
24  import io.netty.handler.codec.http.HttpObjectAggregator;
25  import io.netty.handler.codec.http.HttpServerCodec;
26  import io.netty.handler.ssl.SslHandler;
27  import io.netty.handler.stream.ChunkedWriteHandler;
28  
29  import org.waarp.openr66.protocol.configuration.Configuration;
30  
31  /**
32   * @author Frederic Bregier
33   * 
34   */
35  public class HttpSslInitializer extends ChannelInitializer<SocketChannel> {
36      private boolean useHttpCompression = false;
37  
38      public HttpSslInitializer(boolean useHttpCompression) {
39          this.useHttpCompression = useHttpCompression;
40      }
41  
42      @Override
43      protected void initChannel(SocketChannel ch) throws Exception {
44          final ChannelPipeline pipeline = ch.pipeline();
45          // Add SSL handler first to encrypt and decrypt everything.
46          SslHandler sslhandler = Configuration.getWaarpSslContextFactory().initInitializer(true, false);
47          pipeline.addLast("ssl", sslhandler);
48  
49          pipeline.addLast("decoder", new HttpServerCodec());
50          pipeline.addLast("aggregator", new HttpObjectAggregator(1048576));
51          pipeline.addLast("streamer", new ChunkedWriteHandler());
52          if (useHttpCompression) {
53              pipeline.addLast("deflater", new HttpContentCompressor());
54          }
55          pipeline.addLast("handler", new HttpSslHandler());
56      }
57  
58  }
