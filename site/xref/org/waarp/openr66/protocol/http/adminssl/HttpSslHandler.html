251     private String readFileHeader(String filename) {
252         String value;
253         try {
254             value = WaarpStringUtils.readFileException(filename);
255         } catch (InvalidArgumentException e) {
256             logger.error("Error while trying to open: " + filename, e);
257             return "";
258         } catch (FileTransferException e) {
259             logger.error("Error while trying to read: " + filename, e);
260             return "";
261         }
262         StringBuilder builder = new StringBuilder(value);
263         WaarpStringUtils.replace(builder, REPLACEMENT.XXXLOCALXXX.toString(),
264                 Integer.toString(
265                         Configuration.configuration.getLocalTransaction().
266                                 getNumberLocalChannel()) + " " + Thread.activeCount());
267         WaarpStringUtils.replace(builder, REPLACEMENT.XXXNETWORKXXX.toString(),
268                 Integer.toString(
269                         DbAdmin.getNbConnection() - Configuration.getNBDBSESSION()));
270         WaarpStringUtils.replace(builder, REPLACEMENT.XXXHOSTIDXXX.toString(),
271                 Configuration.configuration.getHOST_ID());
272         if (authentHttp.isAuthenticated()) {
273             WaarpStringUtils.replace(builder, REPLACEMENT.XXXADMINXXX.toString(),
274                     Messages.getString("HttpSslHandler.1")); //$NON-NLS-1$
275         } else {
276             WaarpStringUtils.replace(builder, REPLACEMENT.XXXADMINXXX.toString(),
277                     Messages.getString("HttpSslHandler.0")); //$NON-NLS-1$
278         }
279         TrafficCounter trafficCounter =
280                 Configuration.configuration.getGlobalTrafficShapingHandler().trafficCounter();
281         WaarpStringUtils.replace(builder, REPLACEMENT.XXXBANDWIDTHXXX.toString(),
282                 Messages.getString("HttpSslHandler.IN") + (trafficCounter.lastReadThroughput() >> 20) + //$NON-NLS-1$
283                 Messages.getString("HttpSslHandler.MOPS") + //$NON-NLS-1$
284                 Messages.getString("HttpSslHandler.OUT") + //$NON-NLS-1$
285                 (trafficCounter.lastWriteThroughput() >> 20) +
286                 Messages.getString("HttpSslHandler.MOPS")); //$NON-NLS-1$
287         WaarpStringUtils.replace(builder, REPLACEMENT.XXXLIMITROWXXX.toString(),
288                 "" + getLIMITROW());
289         WaarpStringUtils.replace(builder, REPLACEMENT.XXXLANGXXX.toString(), lang);
290         return builder.toString();
291     }

379     private String Listing() {
380         getParams();
381         if (params == null) {
382             String head = REQUEST.Listing.readHeader(this);
383             head = resetOptionTransfer(head, "", "", "", "", "", "",
384                     false, false, false, false, true);
385             String end = REQUEST.Listing.readEnd().replace(XXXRESULTXXX, "");
386             return head + end;
387         }
388         String head = REQUEST.Listing.readHeader(this);
389         String body0, body, body1;
390         body0 = body1 = body = "";
391         String endText = "";
392         List<String> parms = params.get("ACTION");
393         if (parms != null) {
394             body0 = REQUEST.Listing.readBodyHeader();
395             String parm = parms.get(0);
396             boolean isNotReload = ! "Reload".equalsIgnoreCase(parm);
397             if ("Filter".equalsIgnoreCase(parm) || ! isNotReload) {
398                 String startid = getTrimValue("startid");
399                 String stopid = getTrimValue("stopid");
400                 if (isNotReload && startid != null && stopid == null) {
401                     stopid = Long.MAX_VALUE + "";
402                 }
403                 if (isNotReload && stopid != null && startid == null) {
404                     startid = (DbConstant.ILLEGALVALUE + 1) + "";
405                 }
406                 String start = getValue("start");
407                 String stop = getValue("stop");
408                 String rule = getTrimValue("rule");
409                 String req = getTrimValue("req");
410                 boolean pending, transfer, error, done, all;
411                 pending = params.containsKey("pending");
412                 transfer = params.containsKey("transfer");
413                 error = params.containsKey("error");
414                 done = params.containsKey("done");
415                 all = params.containsKey("all");
416                 if (pending && transfer && error && done) {
417                     all = true;
418                 } else if (!(pending || transfer || error || done)) {
419                     all = true;
420                 }
421                 Timestamp tstart = WaarpStringUtils.fixDate(start);
422                 if (tstart != null) {
423                     start = tstart.toString();
424                 }
425                 Timestamp tstop = WaarpStringUtils.fixDate(stop, tstart);
426                 if (tstop != null) {
427                     stop = tstop.toString();
428                 }
429                 Long idstart = null;
430                 body = REQUEST.Listing.readBody();
431                 String seeAll = checkAuthorizedToSeeAll();
432                 DbPreparedStatement preparedStatement = null;
433                 try {
434                     preparedStatement =
435                             DbTaskRunner.getFilterPrepareStatement(dbSession, getLIMITROW(), false,
436                                     startid, stopid, tstart, tstop, rule, req,
437                                     pending, transfer, error, done, all, seeAll);
438                     preparedStatement.executeQuery();
439                     StringBuilder builder = new StringBuilder();
440                     int i = 0;
441                     while (preparedStatement.getNext()) {
442                         try {
443                             i++;
444                             DbTaskRunner taskRunner = DbTaskRunner
445                                     .getFromStatement(preparedStatement);
446                             if (isNotReload) {
447                                 long specid = taskRunner.getSpecialId();
448                                 if (idstart == null || idstart > specid) {
449                                     idstart = specid;
450                                 }
451                             }
452                             LocalChannelReference lcr =
453                                     Configuration.configuration.getLocalTransaction().
454                                             getFromRequest(taskRunner.getKey());
455                             builder.append(taskRunner.toSpecializedHtml(
456                                     authentHttp,
457                                     body,
458                                     lcr != null ? Messages.getString("HttpSslHandler.Active") : Messages
459                                             .getString("HttpSslHandler.NotActive")));
460                             if (i > getLIMITROW()) {
461                                 break;
462                             }
463                         } catch (WaarpDatabaseException e) {
464                             // try to continue if possible
465                             logger.warn("An error occurs while accessing a Runner: {}",
466                                     e.getMessage());
467                             endText += e.getMessage() + "</BR>";
468                             continue;
469                         }
470                     }
471                     preparedStatement.realClose();
472                     body = builder.toString();
473                 } catch (WaarpDatabaseException e) {
474                     if (preparedStatement != null) {
475                         preparedStatement.realClose();
476                     }
477                     logger.warn("OpenR66 Web Error {}", e.getMessage());
478                 }
479                 head = resetOptionTransfer(head, startid == null ? (idstart != null ? idstart.toString() : "")
480                         : startid,
481                         stopid == null ? "" : stopid, start, stop,
482                         rule == null ? "" : rule, req == null ? "" : req,
483                         pending, transfer, error, done, all);
484             } else {
485                 head = resetOptionTransfer(head, "", "", "", "", "", "",
486                         false, false, false, false, true);
487             }
488             body1 = REQUEST.Listing.readBodyEnd();
489         } else {
490             head = resetOptionTransfer(head, "", "", "", "", "", "",
491                     false, false, false, false, true);
492         }
493         String end = REQUEST.Listing.readEnd().replace(XXXRESULTXXX, endText);
494         return head + body0 + body + body1 + end;
495     }
