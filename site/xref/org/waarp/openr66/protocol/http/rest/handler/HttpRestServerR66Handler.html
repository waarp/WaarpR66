52  /**
53   * Server Http REST interface: http://host/server?... + ShutdownOrBlockJsonPacket as PUT or no Json body but GET to get Current
54   * Status in Json
55   * 
56   * @author "Frederic Bregier"
57   *
58   */
59  public class HttpRestServerR66Handler extends HttpRestAbstractR66Handler {
60  
61      public static final String BASEURI = "server";
62      /**
63       * Internal Logger
64       */
65      private static final WaarpLogger logger = WaarpLoggerFactory
66              .getLogger(HttpRestServerR66Handler.class);
67  
68      public HttpRestServerR66Handler(RestConfiguration config, METHOD... methods) {
69          super(BASEURI, config, METHOD.OPTIONS);
70          setIntersectionMethods(methods, METHOD.GET, METHOD.PUT);
71      }
72  
73      @Override
74      public void endParsingRequest(HttpRestHandler handler, RestArgument arguments, RestArgument result, Object body)
75              throws HttpIncorrectRequestException, HttpInvalidAuthenticationException {
76          logger.debug("debug: {} ### {}", arguments, result);
77          if (body != null) {
78              logger.debug("Obj: {}", body);
79          }
80          handler.setWillClose(false);
81          ServerActions serverHandler = ((HttpRestR66Handler) handler).getServerHandler();
82          if (arguments.getMethod() == METHOD.GET) {
83              // status Json
84              result.setCommand(ACTIONS_TYPE.GetStatus.name());
85              Configuration.configuration.getMonitoring().run(0, true);
86              ObjectNode node = Configuration.configuration.getMonitoring().exportAsJson(true);
87              result.setDetail("Current Waarp R66 status");
88              result.addResult(node);
89              setOk(handler, result, null, HttpResponseStatus.OK);
90              return;
91          }
92          // now action according to body
93          JsonPacket json = (JsonPacket) body;
94          if (json == null) {
95              result.setDetail("not enough information");
96              setError(handler, result, HttpResponseStatus.BAD_REQUEST);
97              return;
98          }
99          result.getAnswer().put(AbstractDbData.JSON_MODEL, RESTHANDLERS.Server.name());
100         try {
101             if (json instanceof ShutdownOrBlockJsonPacket) {//
102                 ShutdownOrBlockJsonPacket node = (ShutdownOrBlockJsonPacket) json;
103                 result.setCommand(ACTIONS_TYPE.ShutdownOrBlock.name());
104                 byte[] key = node.getKey();
105                 if (node.isShutdownOrBlock()) {
106                     // Shutdown
107                     serverHandler.shutdown(key, node.isRestartOrBlock());
108                     result.setDetail("Shutdown on going");
109                     setOk(handler, result, json, HttpResponseStatus.OK);
110                 } else {
111                     // Block
112                     serverHandler.blockRequest(key, node.isRestartOrBlock());
113                     node.setComment((node.isRestartOrBlock() ? "Block" : "Unblock") + " new request");
114                     result.setDetail((node.isRestartOrBlock() ? "Block" : "Unblock") + " new request");
115                     setOk(handler, result, json, HttpResponseStatus.OK);
116                 }
117             } else {
118                 logger.info("Validation is ignored: " + json);
119                 result.setDetail("Unknown command");
120                 setError(handler, result, json, HttpResponseStatus.PRECONDITION_FAILED);
121             }
122         } catch (OpenR66ProtocolNotAuthenticatedException e) {
123             throw new HttpInvalidAuthenticationException(e);
124         } catch (OpenR66ProtocolBusinessException e) {
125             throw new HttpIncorrectRequestException(e);
126         } catch (OpenR66ProtocolShutdownException e) {
127             R66ShutdownHook.shutdownWillStart();
128             logger.warn(Messages.getString("LocalServerHandler.0") + //$NON-NLS-1$
129                     serverHandler.getSession().getAuth().getUser());
130             // dont'close, thread will do
131             ChannelUtils.startShutdown();
132         }
133     }
134 
135     protected ArrayNode getDetailedAllow() {
136         ArrayNode node = JsonHandler.createArrayNode();
137 
138         if (this.methods.contains(METHOD.PUT)) {
139             ShutdownOrBlockJsonPacket node3 = new ShutdownOrBlockJsonPacket();
140             node3.setComment("Shutdown Or Block request (PUT)");
141             node3.setKey("Key".getBytes());
142             ObjectNode node2;
143             ArrayNode node1 = JsonHandler.createArrayNode();
144             try {
145                 node1.add(node3.createObjectNode());
146                 node2 = RestArgument.fillDetailedAllow(METHOD.PUT, this.path, ACTIONS_TYPE.ShutdownOrBlock.name(),
147                         node3.createObjectNode(), node1);
148                 node.add(node2);
149             } catch (OpenR66ProtocolPacketException e1) {
150             }
151         }
152         if (this.methods.contains(METHOD.GET)) {
153             ArrayNode node1 = JsonHandler.createArrayNode();
154             node1.add(Configuration.configuration.getMonitoring().exportAsJson(true));
155             ObjectNode node2 = RestArgument.fillDetailedAllow(METHOD.GET, this.path, ACTIONS_TYPE.GetStatus.name(),
156                     null, node1);
157             node.add(node2);
158         }
159 
160         ObjectNode node2 = RestArgument.fillDetailedAllow(METHOD.OPTIONS, this.path, COMMAND_TYPE.OPTIONS.name(), null,
161                 null);
162         node.add(node2);
163 
164         return node;
165     }
