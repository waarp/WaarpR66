18  package org.waarp.openr66.protocol.localhandler;
19  
20  import io.netty.channel.ChannelPipeline;
21  import io.netty.channel.ChannelInitializer;
22  import io.netty.channel.local.LocalChannel;
23  
24  import org.waarp.openr66.protocol.localhandler.packet.LocalPacketCodec;
25  
26  /**
27   * Pipeline Factory for Local Server
28   * 
29   * @author Frederic Bregier
30   */
31  class LocalServerInitializer extends ChannelInitializer<LocalChannel> {
32  
33      @Override
34      protected void initChannel(LocalChannel ch) throws Exception {
35          final ChannelPipeline pipeline = ch.pipeline();
36          pipeline.addLast("codec", new LocalPacketCodec());
37          pipeline.addLast("handler", new LocalServerHandler());
38      }
39  
40  }
