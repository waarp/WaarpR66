18  package org.waarp.openr66.protocol.localhandler.packet;
19  
20  import io.netty.buffer.ByteBuf;
21  import io.netty.buffer.Unpooled;
22  import org.waarp.common.digest.FilesystemBasedDigest;
23  import org.waarp.openr66.database.data.DbHostAuth;
24  import org.waarp.openr66.protocol.configuration.Configuration;
25  import org.waarp.openr66.protocol.configuration.PartnerConfiguration;
26  import org.waarp.openr66.protocol.exception.OpenR66ProtocolNoSslException;
27  import org.waarp.openr66.protocol.exception.OpenR66ProtocolPacketException;
28  import org.waarp.openr66.protocol.localhandler.LocalChannelReference;
29  import org.waarp.openr66.protocol.utils.R66Versions;
30  
31  /**
32   * Request Authentication class
33   * 
34   * header = "hostId" middle = "key bytes" end = localId + way + (optional version: could be a JSON on the form version.{})
35   * 
36   * @author frederic bregier
37   */
38  public class AuthentPacket extends AbstractLocalPacket {
39      private static final byte ASKVALIDATE = 0;
40  
41      private static final byte ANSWERVALIDATE = 1;
42  
43      private final Integer localId;
44  
45      private byte way;
46  
47      private String version;
48  
49      private String hostId;
50  
51      private byte[] key;
52  
53      /**
54       * @param headerLength
55       * @param middleLength
56       * @param endLength
57       * @param buf
58       * @return the new AuthentPacket from buffer
59       * @throws OpenR66ProtocolPacketException
60       */
61      public static AuthentPacket createFromBuffer(int headerLength,
62              int middleLength, int endLength, ByteBuf buf)
63              throws OpenR66ProtocolPacketException {
64          if (headerLength - 1 <= 0) {
65              throw new OpenR66ProtocolPacketException("Not enough data");
66          }
67          if (middleLength <= 0) {
68              throw new OpenR66ProtocolPacketException("Not enough data");
69          }
70          if (endLength < 5) {
71              throw new OpenR66ProtocolPacketException("Not enough data");
72          }
73          final byte[] bheader = new byte[headerLength - 1];
74          final byte[] bmiddle = new byte[middleLength];
75          if (headerLength - 1 > 0) {
76              buf.readBytes(bheader);
77          }
78          if (middleLength > 0) {
79              buf.readBytes(bmiddle);
80          }
81          // end part
82          Integer newId = buf.readInt();
83          byte valid = buf.readByte();
84          String version = R66Versions.V2_4_12.getVersion(); // first base reference where it is unacceptable
85          if (endLength > 5) {
86              // version
87              byte[] bversion = new byte[endLength - 5];
88              buf.readBytes(bversion);
89              version = new String(bversion);
90          }
91          final String sheader = new String(bheader);
92          return new AuthentPacket(sheader, bmiddle, newId, valid, version);
93      }
94  
95      /**
96       * @param hostId
97       * @param key
98       * @param newId
99       * @param valid
100      * @param version
101      */
102     private AuthentPacket(String hostId, byte[] key, Integer newId, byte valid, String version) {
103         this.hostId = hostId;
104         this.key = key;
105         localId = newId;
106         way = valid;
107         Configuration.configuration.getVersions().put(hostId, new PartnerConfiguration(hostId, version));
108         this.version = version;
109     }
110 
111     /**
112      * @param hostId
113      * @param key
114      * @param newId
115      */
116     public AuthentPacket(String hostId, byte[] key, Integer newId) {
117         this.hostId = hostId;
118         this.key = key;
119         localId = newId;
120         way = ASKVALIDATE;
121         Configuration.configuration.getVersions().putIfAbsent(hostId, new PartnerConfiguration(hostId));
122         version = Configuration.configuration.getVersions().get(hostId).toString();
123     }
124 
125     @Override
126     public void createEnd(LocalChannelReference lcr) throws OpenR66ProtocolPacketException {
127         byte[] bversion = version != null ? version.getBytes() : null;
128         end = Unpooled.buffer(5 + (version != null ? bversion.length : 0));
129         end.writeInt(localId);
130         end.writeByte(way);
131         if (version != null) {
132             end.writeBytes(bversion);
133         }
134     }
135 
136     @Override
137     public void createHeader(LocalChannelReference lcr) throws OpenR66ProtocolPacketException {
138         if (hostId == null) {
139             throw new OpenR66ProtocolPacketException("Not enough data");
140         }
141         header = Unpooled.wrappedBuffer(hostId.getBytes());
142     }
143 
144     @Override
145     public void createMiddle(LocalChannelReference lcr) throws OpenR66ProtocolPacketException {
146         if (key == null) {
147             throw new OpenR66ProtocolPacketException("Not enough data");
148         }
149         middle = Unpooled.wrappedBuffer(key);
150     }
151 
152     @Override
153     public byte getType() {
154         return LocalPacketFactory.AUTHENTPACKET;
155     }
156 
157     @Override
158     public String toString() {
159         return "AuthentPacket: " + hostId + " " + localId + " " + way + " " + version;
160     }
161 
162     /**
163      * @return the hostId
164      */
165     public String getHostId() {
166         return hostId;
167     }
168 
169     /**
170      * @return the key
171      */
172     public byte[] getKey() {
173         return key;
174     }
175 
176     /**
177      * @return the localId
178      */
179     public Integer getLocalId() {
180         return localId;
181     }
182 
183     /**
184      * @return True if this packet is to be validated
185      */
186     public boolean isToValidate() {
187         return way == ASKVALIDATE;
188     }
189 
190     /**
191      * Validate the connection
192      */
193     public void validate(boolean isSSL) {
194         way = ANSWERVALIDATE;
195         DbHostAuth auth = isSSL ? Configuration.configuration.getHOST_SSLAUTH() : Configuration.configuration.getHOST_AUTH();
196         try {
197             hostId = Configuration.configuration.getHostId(isSSL);
198         } catch (OpenR66ProtocolNoSslException e) {
199             hostId = Configuration.configuration.getHOST_ID();
200             auth = Configuration.configuration.getHOST_AUTH();
201         }
202         key = FilesystemBasedDigest.passwdCrypt(auth.getHostkey());
203         Configuration.configuration.getVersions().putIfAbsent(hostId, new PartnerConfiguration(hostId));
204         version = Configuration.configuration.getVersions().get(hostId).toString();
205         header = null;
206         middle = null;
207         end = null;
208     }
209 }
