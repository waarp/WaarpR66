18  package org.waarp.openr66.protocol.localhandler.packet;
19  
20  import io.netty.buffer.ByteBuf;
21  import io.netty.buffer.Unpooled;
22  import org.waarp.openr66.protocol.exception.OpenR66ProtocolPacketException;
23  import org.waarp.openr66.protocol.localhandler.LocalChannelReference;
24  
25  /**
26   * No Op class
27   * 
28   * header = empty middle = empty end = empty
29   * 
30   * @author frederic bregier
31   */
32  public class NoOpPacket extends AbstractLocalPacket {
33  
34      /**
35       * @param headerLength
36       * @param middleLength
37       * @param endLength
38       * @param buf
39       * @return the new EndTransferPacket from buffer
40       * @throws OpenR66ProtocolPacketException
41       */
42      public static NoOpPacket createFromBuffer(int headerLength,
43              int middleLength, int endLength, ByteBuf buf)
44              throws OpenR66ProtocolPacketException {
45          return new NoOpPacket();
46      }
47  
48      @Override
49      public void createEnd(LocalChannelReference lcr) {
50          end = Unpooled.EMPTY_BUFFER;
51      }
52  
53      @Override
54      public void createHeader(LocalChannelReference lcr) {
55          header = Unpooled.EMPTY_BUFFER;
56      }
57  
58      @Override
59      public void createMiddle(LocalChannelReference lcr) {
60          middle = Unpooled.EMPTY_BUFFER;
61      }
62  
63      @Override
64      public byte getType() {
65          return LocalPacketFactory.NOOPPACKET;
66      }
67  
68      @Override
69      public String toString() {
70          return "NoOpPacket";
71      }
72  
73  }
