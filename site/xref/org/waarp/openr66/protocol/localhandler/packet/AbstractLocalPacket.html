18  package org.waarp.openr66.protocol.localhandler.packet;
19  
20  import io.netty.buffer.ByteBuf;
21  import io.netty.buffer.Unpooled;
22  
23  import org.waarp.openr66.protocol.exception.OpenR66ProtocolPacketException;
24  import org.waarp.openr66.protocol.localhandler.LocalChannelReference;
25  
26  /**
27   * This class represents Abstract Packet with its header, middle and end parts. A Packet is composed
28   * of one Header part, one Middle part (data), and one End part. Header: length field (4 bytes) =
29   * Middle length field (4 bytes), End length field (4 bytes), type field (1 byte), ...<br>
30   * Middle: (Middle length field bytes)<br>
31   * End: (End length field bytes) = code status field (4 bytes), ...<br>
32   * 
33   * @author frederic bregier
34   */
35  public abstract class AbstractLocalPacket {
36      protected ByteBuf header;
37  
38      protected ByteBuf middle;
39  
40      protected ByteBuf end;
41  
42      public AbstractLocalPacket(ByteBuf header, ByteBuf middle,
43              ByteBuf end) {
44          this.header = header;
45          this.middle = middle;
46          this.end = end;
47      }
48  
49      public AbstractLocalPacket() {
50          header = null;
51          middle = null;
52          end = null;
53      }
54  
55      /**
56       * Prepare the Header buffer
57       * 
58       * @throws OpenR66ProtocolPacketException
59       */
60      public abstract void createHeader(LocalChannelReference lcr) throws OpenR66ProtocolPacketException;
61  
62      /**
63       * Prepare the Middle buffer
64       * 
65       * @throws OpenR66ProtocolPacketException
66       */
67      public abstract void createMiddle(LocalChannelReference lcr) throws OpenR66ProtocolPacketException;
68  
69      /**
70       * Prepare the End buffer
71       * 
72       * @throws OpenR66ProtocolPacketException
73       */
74      public abstract void createEnd(LocalChannelReference lcr) throws OpenR66ProtocolPacketException;
75  
76      /**
77       * 
78       * @return the type of Packet
79       */
80      public abstract byte getType();
81  
82      @Override
83      public abstract String toString();
84  
85      /**
86       * @param lcr
87       *            the LocalChannelReference in use
88       * @return the ByteBuf as LocalPacket
89       * @throws OpenR66ProtocolPacketException
90       */
91      public ByteBuf getLocalPacket(LocalChannelReference lcr) throws OpenR66ProtocolPacketException {
92          final ByteBuf buf = Unpooled.buffer(4 * 3 + 1);// 3 header
93          // lengths+type
94          if (header == null) {
95              createHeader(lcr);
96          }
97          final ByteBuf newHeader = header != null ? header
98                  : Unpooled.EMPTY_BUFFER;
99          final int headerLength = 4 * 2 + 1 + newHeader.readableBytes();
100         if (middle == null) {
101             createMiddle(lcr);
102         }
103         final ByteBuf newMiddle = middle != null ? middle
104                 : Unpooled.EMPTY_BUFFER;
105         final int middleLength = newMiddle.readableBytes();
106         if (end == null) {
107             createEnd(lcr);
108         }
109         final ByteBuf newEnd = end != null ? end
110                 : Unpooled.EMPTY_BUFFER;
111         final int endLength = newEnd.readableBytes();
112         buf.writeInt(headerLength);
113         buf.writeInt(middleLength);
114         buf.writeInt(endLength);
115         buf.writeByte(getType());
116         final ByteBuf ByteBuf = Unpooled.wrappedBuffer(
117                 buf, newHeader, newMiddle, newEnd);
118         return ByteBuf;
119     }
120 
121     public void clear() {
122         if (header != null) {
123             if (header.release()) {
124                 header = null;
125             }
126         }
127         if (middle != null) {
128             if (middle.release()) {
129                 middle = null;
130             }
131         }
132         if (end != null) {
133             if (end.release()) {
134                 end = null;
135             }
136         }
137     }
138 
139     public void retain() {
140         if (header != null) {
141             header.retain();
142         }
143         if (middle != null) {
144             middle.retain();
145         }
146         if (end != null) {
147             end.retain();
148         }
149     }
150 }
