179         this.pastLimit = pastLimit;

183         } else {
184             if (DbConstant.admin.isActive()) {

189                 }
190             } else {
191                 dbSession = DbConstant.admin.getSession();
192             }
193         }
194         this.initialize();
195     }

269             return;
270         }

273             // Overall status including past, future and current transfers

284             countStepAllTransfer.realClose();

287             countStepTransfer.realClose();
288             countStepPosttask.realClose();

345     /**
346      * 
347      * @return False if too early, else return True
348      */
349     private boolean reCompute() {
350         long limitDate = System.currentTimeMillis();
351         if (limitDate < lastTry + minimalDelay) {
352             // too early
353             return false;
354         }
355         lastTry = limitDate;
356         return true;
357     }

366     public void run(long nbSecond, boolean detail) {
367         synchronized (trafficCounter) {
368             long limitDate = System.currentTimeMillis();
369             long nbMs = pastLimit;
370             if (nbSecond <= 0) {
371                 nbMs = pastLimit;
372             } else {
373                 nbMs = nbSecond * 1000;
374             }
375             if (dbSession != null && dbSession.isDisActive()) {
376                 dbSession.checkConnectionNoException();
377             }
378             if (dbSession == null || dbSession.isDisActive()) {
379                 nbNetworkConnection =
380                         Configuration.configuration.getHttpChannelGroup().size() +
381                                 Configuration.configuration.getServerChannelGroup().size();
382                 bandwidthIn = trafficCounter.lastReadThroughput() >> 7;// B/s -> Kb/s
383                 bandwidthOut = trafficCounter.lastWriteThroughput() >> 7;
384                 nbThread = Thread.activeCount();
385                 secondsRunning = (limitDate - startMonitor) / 1000;
386 
387                 if (!reCompute()) {
388                     // too early
389                     return;
390                 }
391                 limitDate -= nbMs;
392                 currentLimit = limitDate;
393                 // Update value
394                 // Overall status including past, future and current transfers
395                 nbCountInfoToSubmit = CommanderNoDb.todoList.size();
396                 if (Configuration.configuration.getInternalRunner() != null) {
397                     nbCountInfoRunning = Configuration.configuration.getInternalRunner()
398                             .nbInternalRunner();
399                 } else {
400                     nbCountInfoRunning = Configuration.configuration.getServerChannelGroup().size();
401                 }
402                 // Current situation of all transfers, running or not
403                 nbCountAllRunningStep = nbCountInfoRunning;
404             } else {
405                 nbNetworkConnection = DbAdmin.getNbConnection();
406                 bandwidthIn = trafficCounter.lastReadThroughput() >> 7;// B/s -> Kb/s

408                 nbThread = Thread.activeCount();
