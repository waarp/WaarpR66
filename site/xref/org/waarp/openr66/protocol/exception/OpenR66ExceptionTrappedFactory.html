18  package org.waarp.openr66.protocol.exception;
19  
20  import java.io.IOException;
21  import java.net.BindException;
22  import java.net.ConnectException;
23  import java.nio.channels.CancelledKeyException;
24  import java.nio.channels.ClosedChannelException;
25  import java.nio.channels.NotYetConnectedException;
26  import java.util.concurrent.RejectedExecutionException;
27  
28  import javax.net.ssl.SSLException;
29  
30  import io.netty.channel.Channel;
31  import io.netty.channel.ChannelException;
32  import org.waarp.common.logging.WaarpLogger;
33  import org.waarp.common.logging.WaarpLoggerFactory;
34  import org.waarp.openr66.protocol.configuration.Configuration;
35  import org.waarp.openr66.protocol.utils.R66ShutdownHook;
36  
37  /**
38   * Class that filter exceptions
39   * 
40   * @author frederic bregier
41   */
42  public class OpenR66ExceptionTrappedFactory {
43      /**
44       * Internal Logger
45       */
46      private static final WaarpLogger logger = WaarpLoggerFactory
47              .getLogger(OpenR66ExceptionTrappedFactory.class);
48  
49      /**
50       * @param channel
51       * @param throwable
52       * @return the OpenR66Exception corresponding to the ExceptionEvent, or null if the exception
53       *         should be ignored
54       */
55      public static OpenR66Exception getExceptionFromTrappedException(
56              Channel channel, Throwable throwable) {
57          final Throwable e1 = throwable;
58          if (e1 instanceof ConnectException) {
59              final ConnectException e2 = (ConnectException) e1;
60              logger.debug("Connection impossible since {} with Channel {}", e2
61                      .getMessage(), channel);
62              return new OpenR66ProtocolNoConnectionException(
63                      "Connection impossible", e2);
64          } else if (e1 instanceof ChannelException) {
65              final ChannelException e2 = (ChannelException) e1;
66              logger.info("Connection (example: timeout) impossible since {} with Channel {}",
67                      e2.getMessage(), channel);
68              return new OpenR66ProtocolNetworkException(
69                      "Connection (example: timeout) impossible", e2);
70          } else if (e1 instanceof CancelledKeyException) {
71              final CancelledKeyException e2 = (CancelledKeyException) e1;
72              logger.error("Connection aborted since {}", e2.getMessage());
73              // Is it really what we should do ?
74              // Yes, No action
75              return null;
76          } else if (e1 instanceof ClosedChannelException) {
77              logger.debug("Connection closed before end");
78              return new OpenR66ProtocolBusinessNoWriteBackException(
79                      "Connection closed before end", e1);
80          } else if (e1 instanceof IllegalMonitorStateException) {
81              logger.debug("Try to release a lock incorrectly", e1);
82              return new OpenR66ProtocolBusinessNoWriteBackException(
83                      "Ignored exception", e1);
84          } else if (e1 instanceof OpenR66ProtocolBusinessCancelException) {
85              final OpenR66ProtocolBusinessCancelException e2 = (OpenR66ProtocolBusinessCancelException) e1;
86              logger.debug("Request is canceled: {}", e2.getMessage());
87              return e2;
88          } else if (e1 instanceof OpenR66ProtocolBusinessStopException) {
89              final OpenR66ProtocolBusinessStopException e2 = (OpenR66ProtocolBusinessStopException) e1;
90              logger.debug("Request is stopped: {}", e2.getMessage());
91              return e2;
92          } else if (e1 instanceof OpenR66ProtocolBusinessQueryAlreadyFinishedException) {
93              final OpenR66ProtocolBusinessQueryAlreadyFinishedException e2 =
94                      (OpenR66ProtocolBusinessQueryAlreadyFinishedException) e1;
95              logger.debug("Request is already finished: {}", e2.getMessage());
96              return e2;
97          } else if (e1 instanceof OpenR66ProtocolBusinessQueryStillRunningException) {
98              final OpenR66ProtocolBusinessQueryStillRunningException e2 =
99                      (OpenR66ProtocolBusinessQueryStillRunningException) e1;
100             logger.debug("Request is still running: {}", e2.getMessage());
101             return e2;
102         } else if (e1 instanceof OpenR66ProtocolBusinessRemoteFileNotFoundException) {
103             final OpenR66ProtocolBusinessRemoteFileNotFoundException e2 =
104                     (OpenR66ProtocolBusinessRemoteFileNotFoundException) e1;
105             logger.debug("Remote server did not find file: {}", e2.getMessage());
106             return e2;
107         } else if (e1 instanceof OpenR66ProtocolBusinessNoWriteBackException) {
108             final OpenR66ProtocolBusinessNoWriteBackException e2 = (OpenR66ProtocolBusinessNoWriteBackException) e1;
109             logger.error("Command Error Reply: {}", e2.getMessage());
110             return e2;
111         } else if (e1 instanceof OpenR66ProtocolShutdownException) {
112             final OpenR66ProtocolShutdownException e2 = (OpenR66ProtocolShutdownException) e1;
113             logger.debug("Command Shutdown {}", e2.getMessage());
114             return e2;
115         } else if (e1 instanceof OpenR66Exception) {
116             final OpenR66Exception e2 = (OpenR66Exception) e1;
117             logger.debug("Command Error Reply: {}", e2.getMessage());
118             return e2;
119         } else if (e1 instanceof BindException) {
120             final BindException e2 = (BindException) e1;
121             logger.debug("Address already in use {}", e2.getMessage());
122             return new OpenR66ProtocolNetworkException(
123                     "Address already in use", e2);
124         } else if (e1 instanceof NotYetConnectedException) {
125             final NotYetConnectedException e2 = (NotYetConnectedException) e1;
126             logger.debug("Timeout occurs {}", e2.getMessage());
127             return new OpenR66ProtocolNetworkException("Timeout occurs", e2);
128         } else if (e1 instanceof ConnectException) {
129             final ConnectException e2 = (ConnectException) e1;
130             logger.debug("Timeout occurs {}", e2.getMessage());
131             return new OpenR66ProtocolNetworkException("Timeout occurs", e2);
132         } else if (e1 instanceof NullPointerException) {
133             final NullPointerException e2 = (NullPointerException) e1;
134             logger.error("Null pointer Exception", e2);
135             return new OpenR66ProtocolSystemException("Null Pointer Exception",
136                     e2);
137         } else if (e1 instanceof SSLException) {
138             final SSLException e2 = (SSLException) e1;
139             logger.debug("Connection aborted since SSL Error {} with Channel {}", e2
140                     .getMessage(), channel);
141             return new OpenR66ProtocolBusinessNoWriteBackException("SSL Connection aborted", e2);
142         } else if (e1 instanceof IOException) {
143             final IOException e2 = (IOException) e1;
144             logger.debug("Connection aborted since {} with Channel {}", e2
145                     .getMessage(), channel);
146             if (channel.isActive()) {
147                 return new OpenR66ProtocolSystemException("Connection aborted due to "
148                         + e2.getMessage(), e2);
149             } else {
150                 return new OpenR66ProtocolBusinessNoWriteBackException("Connection aborted due to "
151                         + e2.getMessage(), e2);
152             }
153         } else if (e1 instanceof RejectedExecutionException) {
154             final RejectedExecutionException e2 = (RejectedExecutionException) e1;
155             logger.debug("Connection aborted since {} with Channel {}", e2
156                     .getMessage(), channel);
157             if (channel.isActive()) {
158                 return new OpenR66ProtocolSystemException("Execution aborted", e2);
159             } else {
160                 return new OpenR66ProtocolBusinessNoWriteBackException("Execution aborted", e2);
161             }
162         } else if (e1 instanceof OutOfMemoryError) {
163             final OpenR66ProtocolShutdownException e2 = new OpenR66ProtocolShutdownException(
164                     "Restart since OOME raized", e1);
165             logger.debug("Force Shutdown and Restart : {}", e2.getMessage());
166             if (Configuration.configuration.getR66Mib() != null) {
167                 Configuration.configuration.getR66Mib().notifyWarning(
168                         "OOME so shutdown and restart", e1.getMessage());
169             }
170             R66ShutdownHook.setRestart(true);
171             return e2;
172         } else {
173             logger.error("Unexpected exception from Outband" +
174                     " Ref Channel: " + channel.toString(), e1);
175         }
176         if (Configuration.configuration.getR66Mib() != null) {
177             Configuration.configuration.getR66Mib().notifyWarning(
178                     "Unexpected exception", e1.getMessage());
179         }
180         return new OpenR66ProtocolSystemException("Unexpected exception: " + e1.getMessage(), e1);
181     }
182 }
