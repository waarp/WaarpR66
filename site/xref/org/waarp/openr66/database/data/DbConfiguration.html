221     /**
222      * @param dbSession
223      * @param hostid
224      * @param rg

227      *            Write Global Limit
228      * @param rs
229      *            Read Session Limit

231      *            Write Session Limit
232      * @param del

235     public DbConfiguration(DbSession dbSession, String hostid, long rg, long wg, long rs,
236             long ws, long del) {
237         super(dbSession);
238         this.hostid = hostid;
239         readgloballimit = (rg / 10) * 10;
240         writegloballimit = (wg / 10) * 10;
241         readsessionlimit = (rs / 10) * 10;
242         writesessionlimit = (ws / 10) * 10;
243         delayllimit = del;
244         setToArray();
245         isSaved = false;
246     }

289     @Override
290     public void delete() throws WaarpDatabaseException {
291         if (dbSession == null) {
292             dbR66ConfigurationHashMap.remove(this.hostid);
293             isSaved = false;
294             return;
295         }
296         super.delete();
297     }
298 
299     @Override
300     public void insert() throws WaarpDatabaseException {
301         if (isSaved) {
302             return;
303         }
304         if (dbSession == null) {
305             dbR66ConfigurationHashMap.put(this.hostid, this);
306             isSaved = true;
307             if (this.updatedInfo == UpdatedInfo.TOSUBMIT.ordinal()) {
308                 CommanderNoDb.todoList.add(this);
309             }
310             return;
311         }
312         super.insert();
313     }
314 
315     @Override
316     public boolean exist() throws WaarpDatabaseException {
317         if (dbSession == null) {
318             return dbR66ConfigurationHashMap.containsKey(hostid);
319         }
320         return super.exist();
321     }
322 
323     @Override
324     public void select() throws WaarpDatabaseException {
325         if (dbSession == null) {
326             DbConfiguration conf = dbR66ConfigurationHashMap.get(this.hostid);
327             if (conf == null) {
328                 throw new WaarpDatabaseNoDataException("No row found");
329             } else {
330                 // copy info
331                 for (int i = 0; i < allFields.length; i++) {
332                     allFields[i].setValue(conf.allFields[i].getValue());
333                 }
334                 setFromArray();
335                 isSaved = true;
336                 return;
337             }
338         }
339         super.select();
340     }

365     /**
366      * For instance from Commander when getting updated information
367      * 
368      * @param preparedStatement
369      * @return the next updated Configuration
370      * @throws WaarpDatabaseNoConnectionException
371      * @throws WaarpDatabaseSqlException
372      */
373     public static DbConfiguration getFromStatement(DbPreparedStatement preparedStatement)
374             throws WaarpDatabaseNoConnectionException, WaarpDatabaseSqlException {
375         DbConfiguration dbConfiguration = new DbConfiguration(preparedStatement.getDbSession());
376         dbConfiguration.getValues(preparedStatement, dbConfiguration.allFields);
377         dbConfiguration.setFromArray();
378         dbConfiguration.isSaved = true;
379         return dbConfiguration;
380     }

382     /**
383      * 
384      * @return the DbPreparedStatement for getting Updated Object
385      * @throws WaarpDatabaseNoConnectionException
386      * @throws WaarpDatabaseSqlException
387      */
388     public static DbPreparedStatement getUpdatedPrepareStament(DbSession session)
389             throws WaarpDatabaseNoConnectionException, WaarpDatabaseSqlException {
390         String request = "SELECT " + selectAllFields;
391         request += " FROM " + table +
392                 " WHERE " + Columns.UPDATEDINFO.name() + " = " +
393                 AbstractDbData.UpdatedInfo.TOSUBMIT.ordinal() +
394                 " AND " + Columns.HOSTID.name() + " = '" + Configuration.configuration.getHOST_ID()
395                 + "'";
396         DbPreparedStatement prep = new DbPreparedStatement(session, request);
397         return prep;
398     }

400     /**
401      * 
402      * @param session
403      * @param hostid
404      * @param limitBandwith
405      *            0 for no limit, > 0 for one limit, < 0 for no filter
406      * @return the preparedStatement with the filter
407      * @throws WaarpDatabaseNoConnectionException
408      * @throws WaarpDatabaseSqlException
409      */
410     public static DbPreparedStatement getFilterPrepareStament(DbSession session,
411             String hostid, long limitBandwith)
412             throws WaarpDatabaseNoConnectionException, WaarpDatabaseSqlException {
413         DbPreparedStatement preparedStatement = new DbPreparedStatement(session);
414         String request = "SELECT " + selectAllFields + " FROM " + table;
415         String condition = null;
416         if (hostid != null) {
417             condition = " WHERE " + Columns.HOSTID.name() + " LIKE '%" + hostid + "%' ";
418         }
419         if (limitBandwith >= 0) {
420             if (condition == null) {
421                 condition = " WHERE ";
422             } else {
423                 condition += " AND ";
424             }
425             if (limitBandwith == 0) {
426                 condition += "(" + Columns.READGLOBALLIMIT + " == 0 AND " + Columns.READSESSIONLIMIT + " == 0 AND "
427                         + Columns.WRITEGLOBALLIMIT + " == 0 AND " + Columns.WRITESESSIONLIMIT + " == 0)";
428             } else {
429                 condition += "(" + Columns.READGLOBALLIMIT + " > " + limitBandwith + " OR " + Columns.READSESSIONLIMIT
430                         + " > " + limitBandwith + " OR "
431                         + Columns.WRITEGLOBALLIMIT + " > " + limitBandwith + " OR " + Columns.WRITESESSIONLIMIT + " > "
432                         + limitBandwith + ")";
433             }
434         }
435         if (condition != null) {
436             preparedStatement.createPrepareStatement(request + condition +
437                     " ORDER BY " + Columns.HOSTID.name());
438         } else {
439             preparedStatement.createPrepareStatement(request +
440                     " ORDER BY " + Columns.HOSTID.name());
441         }
442         return preparedStatement;
443     }

445     @Override
446     public void changeUpdatedInfo(UpdatedInfo info) {
447         if (updatedInfo != info.ordinal()) {
448             updatedInfo = info.ordinal();
449             allFields[Columns.UPDATEDINFO.ordinal()].setValue(updatedInfo);
450             isSaved = false;
451         }
452     }
