255     /**
256      * @param dbSession
257      * @param hostid
258      * @param address
259      * @param port
260      * @param isSSL
261      * @param hostkey
262      * @param adminrole
263      * @param isClient
264      */
265     public DbHostAuth(DbSession dbSession, String hostid, String address, int port,
266             boolean isSSL, byte[] hostkey, boolean adminrole, boolean isClient) {
267         super(dbSession);
268         this.hostid = hostid;
269         this.address = address;
270         this.port = port;
271         this.isSsl = isSSL;
272         if (hostkey == null) {
273             this.hostkey = null;
274         } else {
275             try {
276                 // Save as crypted with the local Key and HEX
277                 this.hostkey = Configuration.configuration.getCryptoKey().cryptToHex(hostkey)
278                         .getBytes(WaarpStringUtils.UTF8);
279             } catch (Exception e) {
280                 this.hostkey = new byte[0];
281             }
282         }
283         this.adminrole = adminrole;
284         this.isClient = isClient;
285         if (this.port < 0) {
286             this.isClient = true;
287             this.address = DEFAULT_CLIENT_ADDRESS;
288         }
289         setToArray();
290         isSaved = false;
291     }

434     @Override
435     public void select() throws WaarpDatabaseException {
436         if (dbSession == null) {
437             DbHostAuth host = dbR66HostAuthHashMap.get(this.hostid);
438             if (host == null) {
439                 throw new WaarpDatabaseNoDataException("No row found");
440             } else {
441                 // copy info
442                 for (int i = 0; i < allFields.length; i++) {
443                     allFields[i].setValue(host.allFields[i].getValue());
444                 }
445                 setFromArray();
446                 isSaved = true;
447                 return;
448             }
449         }
450         DbPreparedStatement preparedStatement = new DbPreparedStatement(
451                 dbSession);
452         try {
453             preparedStatement.createPrepareStatement("SELECT " +
454                     selectAllFields + " FROM " + table + " WHERE " +
455                     getWherePrimaryKey());
456             setPrimaryKey();
457             setValues(preparedStatement, primaryKey);
458             preparedStatement.executeQuery();
459             if (preparedStatement.getNext()) {
460                 getValues(preparedStatement, allFields);
461                 setFromArray();
462                 isSaved = true;
463             } else {
464                 throw new WaarpDatabaseNoDataException("No row found");
465             }
466         } finally {
467             preparedStatement.realClose();
468         }
469     }

505     /**
506      * Get All DbHostAuth from database or from internal hashMap in case of no database support
507      * 
508      * @param dbSession
509      *            may be null
510      * @return the array of DbHostAuth
511      * @throws WaarpDatabaseNoConnectionException
512      * @throws WaarpDatabaseSqlException
513      */
514     public static DbHostAuth[] getAllHosts(DbSession dbSession)
515             throws WaarpDatabaseNoConnectionException, WaarpDatabaseSqlException {
516         if (dbSession == null) {
517             DbHostAuth[] result = new DbHostAuth[0];
518             return dbR66HostAuthHashMap.values().toArray(result);
519         }
520         String request = "SELECT " + selectAllFields;
521         request += " FROM " + table;
522         ArrayList<DbHostAuth> dbArrayList = new ArrayList<DbHostAuth>();
523         DbPreparedStatement preparedStatement = new DbPreparedStatement(dbSession, request);
524         try {
525             preparedStatement.executeQuery();
526             while (preparedStatement.getNext()) {
527                 DbHostAuth hostAuth = getFromStatement(preparedStatement);
528                 dbArrayList.add(hostAuth);
529             }
530         } finally {
531             preparedStatement.realClose();
532         }
533         DbHostAuth[] result = new DbHostAuth[0];
534         return dbArrayList.toArray(result);
535     }

570     /**
571      * 
572      * @param session
573      * @param host
574      * @param addr
575      * @param ssl
576      * @param active
577      * @return the DbPreparedStatement according to the filter
578      * @throws WaarpDatabaseNoConnectionException
579      * @throws WaarpDatabaseSqlException
580      */
581     public static DbPreparedStatement getFilterPrepareStament(DbSession session,
582             String host, String addr, boolean ssl, boolean active)
583             throws WaarpDatabaseNoConnectionException, WaarpDatabaseSqlException {
584         DbPreparedStatement preparedStatement = new DbPreparedStatement(session);
585         String request = "SELECT " + selectAllFields + " FROM " + table + " WHERE ";
586         String condition = null;
587         if (host != null) {
588             condition = Columns.HOSTID.name() + " LIKE '%" + host + "%' ";
589         }
590         if (addr != null) {
591             if (condition != null) {
592                 condition += " AND ";
593             } else {
594                 condition = "";
595             }
596             condition += Columns.ADDRESS.name() + " LIKE '%" + addr + "%' ";
597         }
598         if (condition != null) {
599             condition += " AND ";
600         } else {
601             condition = "";
602         }
603         condition += Columns.ISSSL.name() + " = ? AND ";
604         condition += Columns.ISACTIVE.name() + " = ? ";
605         preparedStatement.createPrepareStatement(request + condition +
606                 " ORDER BY " + Columns.HOSTID.name());
607         try {
608             preparedStatement.getPreparedStatement().setBoolean(1, ssl);
609             preparedStatement.getPreparedStatement().setBoolean(2, active);
610         } catch (SQLException e) {
611             preparedStatement.realClose();
612             throw new WaarpDatabaseSqlException(e);
613         }
614         return preparedStatement;
615     }

699     /**
700      * Is the given key a valid one
701      * 
702      * @param newkey
703      * @return True if the key is valid (or any key is valid)
704      */
705     public boolean isKeyValid(byte[] newkey) {
706         // It is valid to not have a key
707         if (this.hostkey == null) {
708             return true;
709         }
710         // Check before if any key is passed or if account is active
711         if (newkey == null || !isActive) {
712             return false;
713         }
714         try {
715             return FilesystemBasedDigest.equalPasswd(
716                     Configuration.configuration.getCryptoKey().decryptHexInBytes(this.hostkey),
717                     newkey);
718         } catch (Exception e) {
719             logger.debug("Error while checking key", e);
720             return false;
721         }
722     }

805     private static String getVersion(String host) {
806         String remoteHost = host;
807         String alias = "";
808         if (Configuration.configuration.getAliases().containsKey(remoteHost)) {
809             remoteHost = Configuration.configuration.getAliases().get(remoteHost);
810             alias += "(Alias: " + remoteHost + ") ";
811         }
812         if (Configuration.configuration.getReverseAliases().containsKey(remoteHost)) {
813             String alias2 = "(ReverseAlias: ";
814             String[] list = Configuration.configuration.getReverseAliases().get(remoteHost);
815             boolean found = false;
816             for (String string : list) {
817                 if (string.equals(host)) {
818                     continue;
819                 }
820                 found = true;
821                 alias2 += string + " ";
822             }
823             if (found) {
824                 alias += alias2 + ") ";
825             }
826         }
827         if (Configuration.configuration.getBusinessWhiteSet().contains(remoteHost)) {
828             alias += "(Business: Allowed) ";
829         }
830         if (Configuration.configuration.getRoles().containsKey(remoteHost)) {
831             RoleDefault item = Configuration.configuration.getRoles().get(remoteHost);
832             alias += "(Role: " + item.toString() + ") ";
833         }
834         return alias + (Configuration.configuration.getVersions().containsKey(remoteHost) ?
835                 Configuration.configuration.getVersions().get(remoteHost).toString() :
836                 "Version Unknown");
837     }

844                 + " isProxified: " + isProxified + " ("

849     /**
850      * Write selected DbHostAuth to a Json String
851      * 
852      * @param preparedStatement
853      * @return the associated Json String
854      * @throws WaarpDatabaseNoConnectionException
855      * @throws WaarpDatabaseSqlException
856      * @throws OpenR66ProtocolBusinessException
857      */
858     public static String getJson(DbPreparedStatement preparedStatement, int limit)
859             throws WaarpDatabaseNoConnectionException, WaarpDatabaseSqlException,
860             OpenR66ProtocolBusinessException {
861         ArrayNode arrayNode = JsonHandler.createArrayNode();
862         try {
863             preparedStatement.executeQuery();
864             int nb = 0;
865             while (preparedStatement.getNext()) {
866                 DbHostAuth host = DbHostAuth
867                         .getFromStatement(preparedStatement);
868                 ObjectNode node = host.getInternalJson();
869                 arrayNode.add(node);
870                 nb++;
871                 if (nb >= limit) {
872                     break;
873                 }
874             }
875         } finally {
876             preparedStatement.realClose();
877         }
878         return JsonHandler.writeAsString(arrayNode);
879     }
880     private ObjectNode getInternalJson() {
881         ObjectNode node = getJson();
882         try {
883             node.put(Columns.HOSTKEY.name(), Configuration.configuration.getCryptoKey().decryptHexInString(
884                     new String(hostkey, WaarpStringUtils.UTF8)));
885         } catch (Exception e1) {
886             node.put(Columns.HOSTKEY.name(), "");
887         }
888         int nb = 0;
889         try {
890             nb = NetworkTransaction.nbAttachedConnection(getSocketAddress(), getHostid());
891         } catch (Exception e) {
892             nb = -1;
893         }
894         node.put("Connection", nb); 
895         node.put("Version", getVersion(hostid).replace("\"", "").replace(",", ", "));
896         return node;
897     }

911      * @return the runner in Html format specified by body by replacing all instance of fields

913     public String toSpecializedHtml(R66Session session, String body, boolean crypted) {
914         StringBuilder builder = new StringBuilder(body);
915         WaarpStringUtils.replace(builder, "XXXHOSTXXX", hostid);
916         WaarpStringUtils.replace(builder, "XXXADDRXXX", address);
917         WaarpStringUtils.replace(builder, "XXXPORTXXX", Integer.toString(port));
918         if (crypted) {
919             WaarpStringUtils.replace(builder, "XXXKEYXXX", new String(hostkey, WaarpStringUtils.UTF8));
920         } else {
921             try {
922                 WaarpStringUtils.replace(builder, "XXXKEYXXX",
923                         Configuration.configuration.getCryptoKey().decryptHexInString(new String(
924                                 this.hostkey, WaarpStringUtils.UTF8)));
925             } catch (Exception e) {
926                 WaarpStringUtils.replace(builder, "XXXKEYXXX", "BAD DECRYPT");
927             }
928         }
929         WaarpStringUtils.replace(builder, "XXXSSLXXX", isSsl ? "checked" : "");
930         WaarpStringUtils.replace(builder, "XXXADMXXX", adminrole ? "checked" : "");
931         WaarpStringUtils.replace(builder, "XXXISCXXX", isClient ? "checked" : "");
932         WaarpStringUtils.replace(builder, "XXXISAXXX", isActive ? "checked" : "");
933         WaarpStringUtils.replace(builder, "XXXISPXXX", isProxified ? "checked" : "");
934         WaarpStringUtils.replace(builder, "XXXVERSIONXXX", getVersion(hostid).replace(",", ", "));
935         int nb = 0;
936         try {
937             nb = NetworkTransaction.nbAttachedConnection(getSocketAddress(), getHostid());
938         } catch (Exception e) {
939             nb = -1;
940         }
941         WaarpStringUtils.replace(builder, "XXXCONNXXX", (nb > 0)
942                 ? "(" + nb + " Connected) " : "");
943         return builder.toString();

947      * 
948      * @param session
949      * @return True if any of the server has the isProxified property
950      */
951     public static boolean hasProxifiedHosts(DbSession session) {
952         if (session == null) {
953             for (DbHostAuth host : dbR66HostAuthHashMap.values()) {
954                 if (host.isProxified) {
955                     return true;
956                 }
957             }
958             return false;
959         }
960         DbPreparedStatement preparedStatement = null;
961         int val = 0;
962         try {
963             preparedStatement = new DbPreparedStatement(session,
964                     "SELECT count(*) FROM " + table + " WHERE " + Columns.ISPROXIFIED + " = " + true);
965             preparedStatement.executeQuery();
966             preparedStatement.getNext();
967             val = preparedStatement.getResultSet().getInt(1);
968         } catch (WaarpDatabaseNoConnectionException e) {
969             return false;
970         } catch (WaarpDatabaseSqlException e) {
971             return false;
972         } catch (SQLException e) {
973             return false;
974         } finally {
975             if (preparedStatement != null) {
976                 preparedStatement.realClose();
977             }
978         }
979         return val > 0;
980     }
