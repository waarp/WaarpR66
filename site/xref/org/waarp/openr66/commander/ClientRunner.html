113     @Override
114     public void run() {
115         if (Configuration.configuration.isShutdown()) {
116             taskRunner.changeUpdatedInfo(UpdatedInfo.TOSUBMIT);
117             taskRunner.forceSaveStatus();
118             return;
119         }
120         try {
121             if (activeRunners != null) {
122                 activeRunners.add(this);
123             }
124             // fix for SelfRequest
125             if (taskRunner.isSelfRequest()) {
126                 taskRunner.setSenderByRequestToValidate(false);
127             }
128             R66Future transfer;
129             try {
130                 transfer = this.runTransfer();
131             } catch (OpenR66RunnerErrorException e) {
132                 logger.error("Runner Error: {} {}", e.getMessage(),
133                         taskRunner.toShortString());
134                 return;
135             } catch (OpenR66ProtocolNoConnectionException e) {
136                 logger.error("No connection Error {}", e.getMessage());
137                 if (localChannelReference != null) {
138                     localChannelReference.setErrorMessage(
139                             ErrorCode.ConnectionImpossible.mesg,
140                             ErrorCode.ConnectionImpossible);
141                 }
142                 taskRunner.setErrorTask(localChannelReference);
143                 try {
144                     taskRunner.forceSaveStatus();
145                     taskRunner.run();
146                 } catch (OpenR66RunnerErrorException e1) {
147                     this.changeUpdatedInfo(UpdatedInfo.INERROR,
148                             ErrorCode.ConnectionImpossible, true);
149                 }
150                 return;
151             } catch (OpenR66ProtocolPacketException e) {
152                 logger.error("Protocol Error", e);
153                 return;
154             } catch (OpenR66ProtocolNotYetConnectionException e) {
155                 logger.warn("No connection warning {}", e.getMessage());
156                 return;
157             }
158             R66Result result = transfer.getResult();
159             if (result != null) {
160                 if (result.getCode() == ErrorCode.QueryAlreadyFinished) {
161                     logger.warn(Messages.getString("Transfer.Status")
162                             +
163                             (transfer.isSuccess() ? Messages.getString("RequestInformation.Success") : Messages
164                                     .getString("RequestInformation.Failure")) +
165                             "     " + ErrorCode.QueryAlreadyFinished.mesg +
166                             ":" +
167                             (result != null ? result.toString() : "no result"));
168                 } else {
169                     if (transfer.isSuccess()) {
170                         logger.info(Messages.getString("Transfer.Status")
171                                 + Messages.getString("RequestInformation.Success") + "     " +
172                                 (result != null ? result.toString()
173                                         : "no result"));
174                     } else {
175                         logger.error(Messages.getString("Transfer.Status")
176                                 + Messages.getString("RequestInformation.Failure") + "     " +
177                                 (result != null ? result.toString()
178                                         : "no result"));
179                     }
180                 }
181             } else {
182                 if (transfer.isSuccess()) {
183                     logger.warn(Messages.getString("Transfer.Status")
184                             + Messages.getString("RequestInformation.Success") + "     no result");
185                 } else {
186                     logger.error(Messages.getString("Transfer.Status")
187                             + Messages.getString("RequestInformation.Failure") + "     no result");
188                 }
189             }
190             transfer = null;
191             Thread.currentThread().setName(
192                     "Finished_" + Thread.currentThread().getName());
193         } finally {
194             if (activeRunners != null) {
195                 activeRunners.remove(this);
196             }
197         }
198     }

200     /**
201      * 
202      * @param runner
203      * @param limit
204      * @return True if the task was run less than limit, else False
205      */
206     public boolean incrementTaskRunnerTry(DbTaskRunner runner, int limit) {
207         String key = runner.getKey();
208         Integer tries = taskRunnerRetryHashMap.get(key);
209         logger.debug("try to find integer: " + tries);
210         if (tries == null) {
211             tries = Integer.valueOf(1);
212         } else {
213             tries = tries + 1;
214         }
215         if (limit <= tries) {
216             taskRunnerRetryHashMap.remove(key);
217             return false;
218         } else {
219             taskRunnerRetryHashMap.put(key, tries);
220             return true;
221         }
222     }

224     /**
225      * True transfer run (can be called directly to enable exception outside any executors)
226      * 
227      * @return The R66Future of the transfer operation
228      * @throws OpenR66RunnerErrorException
229      * @throws OpenR66ProtocolNoConnectionException
230      * @throws OpenR66ProtocolPacketException
231      * @throws OpenR66ProtocolNotYetConnectionException
232      */
233     public R66Future runTransfer() throws OpenR66RunnerErrorException,
234             OpenR66ProtocolNoConnectionException,
235             OpenR66ProtocolPacketException,
236             OpenR66ProtocolNotYetConnectionException {
237         logger.debug("Start attempt Transfer");
238         localChannelReference = initRequest();
239         try {
240             localChannelReference.getFutureValidRequest().await();
241         } catch (InterruptedException e) {
242         }
243         if (localChannelReference.getFutureValidRequest().isSuccess()) {
244             return finishTransfer(localChannelReference);
245         } else if (localChannelReference.getFutureValidRequest().getResult() != null &&
246                 localChannelReference.getFutureValidRequest().getResult().getCode() == ErrorCode.ServerOverloaded) {
247             return tryAgainTransferOnOverloaded(true, localChannelReference);
248         } else
249             return finishTransfer(localChannelReference);
250     }

252     /**
253      * In case an overloaded signal is returned by the requested
254      * 
255      * @param retry
256      *            if True, it will retry in case of overloaded remote server, else it just stops
257      * @param localChannelReference
258      * @return The R66Future of the transfer operation
259      * @throws OpenR66RunnerErrorException
260      * @throws OpenR66ProtocolNoConnectionException
261      * @throws OpenR66ProtocolPacketException
262      * @throws OpenR66ProtocolNotYetConnectionException
263      */
264     public R66Future tryAgainTransferOnOverloaded(boolean retry,
265             LocalChannelReference localChannelReference)
266             throws OpenR66RunnerErrorException,
267             OpenR66ProtocolNoConnectionException,
268             OpenR66ProtocolPacketException,
269             OpenR66ProtocolNotYetConnectionException {
270         if (this.localChannelReference == null) {
271             this.localChannelReference = localChannelReference;
272         }
273         boolean incRetry = incrementTaskRunnerTry(taskRunner,
274                 Configuration.RETRYNB);
275         logger.debug("tryAgainTransferOnOverloaded: " + retry + ":" + incRetry);
276         switch (taskRunner.getUpdatedInfo()) {
277             case DONE:
278             case INERROR:
279             case INTERRUPTED:
280                 break;
281             default:
282                 this.changeUpdatedInfo(UpdatedInfo.INERROR,
283                         ErrorCode.ServerOverloaded, true);
284         }
285         // redo if possible
286         if (retry && incRetry) {
287             try {
288                 Thread.sleep(Configuration.configuration.getConstraintLimitHandler()
289                         .getSleepTime());
290             } catch (InterruptedException e) {
291             }
292             return runTransfer();
293         } else {
294             if (localChannelReference == null) {
295                 taskRunner
296                         .setLocalChannelReference(new LocalChannelReference());
297             }
298             taskRunner.getLocalChannelReference().setErrorMessage(
299                     ErrorCode.ConnectionImpossible.mesg,
300                     ErrorCode.ConnectionImpossible);
301             this.taskRunner.setErrorTask(localChannelReference);
302             this.taskRunner.run();
303             throw new OpenR66ProtocolNoConnectionException(
304                     "End of retry on ServerOverloaded");
305         }
306     }

308     /**
309      * Finish the transfer (called at the end of runTransfer)
310      * 
311      * @param localChannelReference
312      * @return The R66Future of the transfer operation
313      * @throws OpenR66ProtocolNotYetConnectionException
314      * @throws OpenR66ProtocolPacketException
315      * @throws OpenR66ProtocolNoConnectionException
316      * @throws OpenR66RunnerErrorException
317      */
318     public R66Future finishTransfer(LocalChannelReference localChannelReference)
319             throws OpenR66RunnerErrorException {
320         if (this.localChannelReference == null) {
321             this.localChannelReference = localChannelReference;
322         }
323         R66Future transfer = localChannelReference.getFutureRequest();
324         try {
325             transfer.await();
326         } catch (InterruptedException e1) {
327         }
328         taskRunnerRetryHashMap.remove(taskRunner.getKey());
329         logger.info("Request done with {}", (transfer.isSuccess() ? "success"
330                 : "error"));
331         localChannelReference.getLocalChannel().close();
332         // now reload TaskRunner if it still exists (light client can forget it)
333         boolean isSender = taskRunner.isSender();
334         if (transfer.isSuccess()) {
335             try {
336                 taskRunner.select();
337             } catch (WaarpDatabaseException e) {
338                 logger.debug("Not a problem but cannot find at the end the task");
339                 taskRunner.setFrom(transfer.getRunner());
340             }
341             taskRunner.setSender(isSender);
342             this.changeUpdatedInfo(UpdatedInfo.DONE, ErrorCode.CompleteOk, false);
343         } else {
344             try {
345                 taskRunner.select();
346             } catch (WaarpDatabaseException e) {
347                 logger.debug("Not a problem but cannot find at the end the task");
348                 taskRunner.setFrom(transfer.getRunner());
349             }
350             taskRunner.setSender(isSender);
351             // Case when we were interrupted
352             if (transfer.getResult() == null) {
353                 switch (taskRunner.getUpdatedInfo()) {
354                     case DONE:
355                         R66Result ok = new R66Result(null, true,
356                                 ErrorCode.CompleteOk, taskRunner);
357                         transfer.setResult(ok);
358                         transfer.setSuccess();
359                         this.changeUpdatedInfo(UpdatedInfo.DONE,
360                                 ErrorCode.CompleteOk, false);
361                         break;
362                     case INERROR:
363                     case INTERRUPTED:
364                     default:
365                         R66Result error = new R66Result(null, true,
366                                 ErrorCode.Internal, taskRunner);
367                         transfer.setResult(error);
368                         transfer.cancel();
369                         this.changeUpdatedInfo(UpdatedInfo.INERROR,
370                                 ErrorCode.Internal, false);
371                 }
372                 return transfer;
373             }
374             if (transfer.getResult().getCode() == ErrorCode.QueryAlreadyFinished) {
375                 // check if post task to execute
376                 logger.warn("WARN QueryAlreadyFinished:     " +
377                         transfer.toString() + "     " +
378                         taskRunner.toShortString());
379                 try {
380                     TransferUtils.finalizeTaskWithNoSession(taskRunner,
381                             localChannelReference);
382                 } catch (OpenR66RunnerErrorException e) {
383                     this.taskRunner.changeUpdatedInfo(UpdatedInfo.INERROR);
384                     this.taskRunner.forceSaveStatus();
385                 }
386             } else {
387                 switch (taskRunner.getUpdatedInfo()) {
388                     case DONE:
389                     case INERROR:
390                     case INTERRUPTED:
391                     case TOSUBMIT:
392                         break;
393                     default:
394                         this.changeUpdatedInfo(UpdatedInfo.INERROR,
395                                 transfer.getResult().getCode(), false);
396                 }
397             }
398         }
399         return transfer;
400     }

402     /**
403      * Initialize the request
404      * 
405      * @return the localChannelReference holding the transfer request
406      * @throws OpenR66ProtocolNoConnectionException
407      * @throws OpenR66RunnerErrorException
408      * @throws OpenR66ProtocolPacketException
409      * @throws OpenR66ProtocolNotYetConnectionException
410      */
411     public LocalChannelReference initRequest()
412             throws OpenR66ProtocolNoConnectionException,
413             OpenR66RunnerErrorException, OpenR66ProtocolPacketException,
414             OpenR66ProtocolNotYetConnectionException {
415         this.changeUpdatedInfo(UpdatedInfo.RUNNING, ErrorCode.Running, true);
416         long id = taskRunner.getSpecialId();
417         String tid;
418         if (id == DbConstant.ILLEGALVALUE) {
419             tid = taskRunner.getRuleId() + "_" + taskRunner.getMode() +
420                     "_NEWTRANSFER";
421         } else {
422             tid = taskRunner.getRuleId() + "_" + taskRunner.getMode() + "_" +
423                     id;
424         }
425         Thread.currentThread().setName(tid);
426         logger.debug("Will run {}", this.taskRunner);
427         boolean restartPost = false;
428         if (taskRunner.getGloballaststep() == TASKSTEP.POSTTASK.ordinal()) {
429             // Send a validation to requested
430             if (!taskRunner.isSelfRequested()) {
431                 // restart
432                 restartPost = true;
433             }
434         }
435         if (taskRunner.isSelfRequested()) {
436             // Don't have to restart a task for itself (or should use requester)
437             logger.warn("Requested host cannot initiate itself the request");
438             this.changeUpdatedInfo(UpdatedInfo.INERROR,
439                     ErrorCode.LoopSelfRequestedHost, true);
440             throw new OpenR66ProtocolNoConnectionException(
441                     "Requested host cannot initiate itself the request");
442         }
443         DbHostAuth host = R66Auth.getServerAuth(DbConstant.admin.getSession(),
444                 taskRunner.getRequested());
445         if (host == null) {
446             logger.error("Requested host cannot be found: " +
447                     taskRunner.getRequested());
448             this.changeUpdatedInfo(UpdatedInfo.INERROR, ErrorCode.NotKnownHost, true);
449             throw new OpenR66ProtocolNoConnectionException(
450                     "Requested host cannot be found " +
451                             taskRunner.getRequested());
452         }
453         if (host.isClient()) {
454             logger.debug("Cannot initiate a connection with a client: {}", host);
455             this.changeUpdatedInfo(UpdatedInfo.INERROR,
456                     ErrorCode.ConnectionImpossible, true);
457             throw new OpenR66ProtocolNoConnectionException(
458                     "Cannot connect to client " + host.toString());
459         }
460         SocketAddress socketAddress = host.getSocketAddress();
461         boolean isSSL = host.isSsl();
462 
463         LocalChannelReference localChannelReference = networkTransaction
464                 .createConnectionWithRetry(socketAddress, isSSL, futureRequest);
465         taskRunner.setLocalChannelReference(localChannelReference);
466         if (localChannelReference == null) {
467             // propose to redo
468             // See if reprogramming is ok (not too many tries)
469             String retry;
470             if (incrementTaskRunnerTry(taskRunner, Configuration.RETRYNB)) {
471                 logger.debug("Will retry since Cannot connect to {}", host);
472                 retry = " but will retry";
473                 // now wait
474                 try {
475                     Thread.sleep(Configuration.configuration.getDelayRetry());
476                 } catch (InterruptedException e) {
477                     logger.debug(
478                             "Will not retry since limit of connection attemtps is reached for {}",
479                             host);
480                     retry = " and retries limit is reached so stop here";
481                     this.changeUpdatedInfo(UpdatedInfo.INERROR,
482                             ErrorCode.ConnectionImpossible, true);
483                     taskRunner
484                             .setLocalChannelReference(new LocalChannelReference());
485                     throw new OpenR66ProtocolNoConnectionException(
486                             "Cannot connect to server " + host.toString() + retry);
487                 }
488                 this.changeUpdatedInfo(UpdatedInfo.TOSUBMIT,
489                         ErrorCode.ConnectionImpossible, true);
490                 throw new OpenR66ProtocolNotYetConnectionException(
491                         "Cannot connect to server " + host.toString() + retry);
492             } else {
493                 logger.debug(
494                         "Will not retry since limit of connection attemtps is reached for {}",
495                         host);
496                 retry = " and retries limit is reached so stop here";
497                 this.changeUpdatedInfo(UpdatedInfo.INERROR,
498                         ErrorCode.ConnectionImpossible, true);
499                 taskRunner
500                         .setLocalChannelReference(new LocalChannelReference());
501                 // set this server as being in shutdown status
502                 NetworkTransaction.proposeShutdownNetworkChannel(socketAddress);
503                 throw new OpenR66ProtocolNoConnectionException(
504                         "Cannot connect to server " + host.toString() + retry);
505             }
506         }
507         socketAddress = null;
508         if (handler != null) {
509             localChannelReference.setRecvThroughHandler(handler);
510         }
511         localChannelReference.setSendThroughMode(isSendThroughMode);
512         if (restartPost) {
513             RequestPacket request = taskRunner.getRequest();
514             logger.debug("Will send request {} ", request);
515             localChannelReference.setClientRunner(this);
516             localChannelReference.sessionNewState(R66FiniteDualStates.REQUESTR);
517             try {
518                 ChannelUtils.writeAbstractLocalPacket(localChannelReference,
519                         request, true);
520             } catch (OpenR66ProtocolPacketException e) {
521                 // propose to redo
522                 logger.warn("Cannot transfer request to " + host.toString());
523                 this.changeUpdatedInfo(UpdatedInfo.INTERRUPTED,
524                         ErrorCode.Internal, true);
525                 localChannelReference.getLocalChannel().close();
526                 localChannelReference = null;
527                 host = null;
528                 request = null;
529                 throw e;
530             }
531             logger.debug("Wait for request to {}", host);
532             request = null;
533             host = null;
534             return localChannelReference;
535         }
536         // If Requester is NOT Sender, and if TransferTask then decrease now if
537         // possible the rank
538         if (!taskRunner.isSender() &&
539                 (taskRunner.getGloballaststep() == TASKSTEP.TRANSFERTASK
540                         .ordinal())) {
541             logger.debug(
542                     "Requester is not Sender so decrease if possible the rank {}",
543                     taskRunner);
544             taskRunner.restartRank();
545             taskRunner.forceSaveStatus();
546             logger.debug(
547                     "Requester is not Sender so new rank is " +
548                             taskRunner.getRank() + " {}", taskRunner);
549         }
550         RequestPacket request = taskRunner.getRequest();
551         request.setLimit(localChannelReference.getChannelLimit(
552                     taskRunner.isSender()));
553         logger.debug("Will send request {} {}", request, localChannelReference);
554         localChannelReference.setClientRunner(this);
555         localChannelReference.sessionNewState(R66FiniteDualStates.REQUESTR);
556         try {
557             ChannelUtils.writeAbstractLocalPacket(localChannelReference,
558                     request, true);
559         } catch (OpenR66ProtocolPacketException e) {
560             // propose to redo
561             logger.warn("Cannot transfer request to " + host.toString());
562             this.changeUpdatedInfo(UpdatedInfo.INTERRUPTED, ErrorCode.Internal, true);
563             localChannelReference.getLocalChannel().close();
564             localChannelReference = null;
565             host = null;
566             request = null;
567             throw e;
568         }
569         logger.debug("Wait for request to {}", host);
570         request = null;
571         host = null;
572         return localChannelReference;
573     }

575     /**
576      * Change the UpdatedInfo of the current runner
577      * 
578      * @param info
579      */
580     public void changeUpdatedInfo(AbstractDbData.UpdatedInfo info,
581             ErrorCode code, boolean force) {
582         this.taskRunner.changeUpdatedInfo(info);
583         this.taskRunner.setErrorExecutionStatus(code);
584         if (force) {
585             this.taskRunner.forceSaveStatus();
586         } else {
587             try {
588                 this.taskRunner.saveStatus();
589             } catch (OpenR66RunnerErrorException e) {
590             }
591         }
592     }
