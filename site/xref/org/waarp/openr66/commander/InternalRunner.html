39  /**
40   * This class launch and control the Commander and enable TaskRunner job submissions
41   * 
42   * @author Frederic Bregier
43   * 
44   */
45  public class InternalRunner {
46      /**
47       * Internal Logger
48       */
49      private static final WaarpLogger logger = WaarpLoggerFactory
50              .getLogger(InternalRunner.class);
51  
52      private final ScheduledExecutorService scheduledExecutorService;
53      private ScheduledFuture<?> scheduledFuture;
54      private CommanderInterface commander = null;
55      private volatile boolean isRunning = true;
56      private final ThreadPoolExecutor threadPoolExecutor;
57      private final NetworkTransaction networkTransaction;
58  
59      /**
60       * Create the structure to enable submission by database
61       * 
62       * @throws WaarpDatabaseNoConnectionException
63       * @throws WaarpDatabaseSqlException
64       */
65      public InternalRunner() throws WaarpDatabaseNoConnectionException, WaarpDatabaseSqlException {
66          if (DbConstant.admin.isActive()) {
67              commander = new Commander(this, true);
68          } else {
69              commander = new CommanderNoDb(this, true);
70          }
71          scheduledExecutorService = Executors.newSingleThreadScheduledExecutor(new WaarpThreadFactory("InternalRunner"));
72          isRunning = true;
73          BlockingQueue<Runnable> workQueue = new LinkedBlockingQueue<Runnable>();
74          threadPoolExecutor = new ThreadPoolExecutor(Configuration.configuration.getRUNNER_THREAD(), Configuration.configuration.getRUNNER_THREAD(),
75                  1000, TimeUnit.MILLISECONDS, workQueue);
76          scheduledFuture = scheduledExecutorService.scheduleWithFixedDelay(commander,
77                  Configuration.configuration.getDelayCommander(),
78                  Configuration.configuration.getDelayCommander(), TimeUnit.MILLISECONDS);
79          networkTransaction = new NetworkTransaction();
80      }
81  
82      public NetworkTransaction getNetworkTransaction() {
83          return networkTransaction;
84      }
85  
86      /**
87       * Submit a task
88       * 
89       * @param taskRunner
90       */
91      public void submitTaskRunner(DbTaskRunner taskRunner) {
92          if (isRunning || !Configuration.configuration.isShutdown()) {
93              if (threadPoolExecutor.getActiveCount() < 
94                      Configuration.configuration.getRUNNER_THREAD()) {
95                  logger.debug("Will run {}", taskRunner);
96                  ClientRunner runner = new ClientRunner(networkTransaction, taskRunner, null);
97                  if (taskRunner.isSendThrough() && (taskRunner.isRescheduledTransfer()
98                          || taskRunner.isPreTaskStarting())) {
99                      runner.setSendThroughMode();
100                     taskRunner.checkThroughMode();
101                 }
102                 runner.setDaemon(true);
103                 // create the client, connect and run
104                 threadPoolExecutor.execute(runner);
105             } else {
106                 // too many current active threads
107                 logger.debug("Task rescheduled {}", taskRunner);
108             }
109         }
110     }
111 
112     /**
113      * First step while shutting down the service
114      */
115     public void prepareStopInternalRunner() {
116         isRunning = false;
117         scheduledFuture.cancel(false);
118         scheduledExecutorService.shutdown();
119         threadPoolExecutor.shutdown();
120     }
121 
122     /**
123      * This should be called when the server is shutting down, after stopping active requests if
124      * possible.
125      */
126     public void stopInternalRunner() {
127         isRunning = false;
128         logger.info("Stopping Commander and Runner Tasks");
129         scheduledFuture.cancel(false);
130         scheduledExecutorService.shutdownNow();
131         threadPoolExecutor.shutdownNow();
132         networkTransaction.closeAll(false);
133     }
134 
135     public int nbInternalRunner() {
136         return threadPoolExecutor.getActiveCount();
137     }
138 
139     public void reloadInternalRunner()
140             throws WaarpDatabaseNoConnectionException, WaarpDatabaseSqlException {
141         scheduledFuture.cancel(false);
142         if (commander != null) {
143             commander.finalize();
144         }
145         if (DbConstant.admin.isActive()) {
146             commander = new Commander(this);
147         } else {
148             commander = new CommanderNoDb(this);
149         }
150         scheduledFuture = scheduledExecutorService.scheduleWithFixedDelay(commander,
151                 Configuration.configuration.getDelayCommander(),
152                 Configuration.configuration.getDelayCommander(), TimeUnit.MILLISECONDS);
153     }
154 }
