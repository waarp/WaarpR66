17  package org.waarp.gateway.kernel.exec;
18  
19  import org.waarp.common.command.exception.Reply421Exception;
20  import org.waarp.common.future.WaarpFuture;
21  
22  /**
23   * NoTaskExecutor class. No action will be taken.
24   * 
25   * 
26   * 
27   * @author Frederic Bregier
28   * 
29   */
30  public class NoTaskExecutor extends AbstractExecutor {
31      private final WaarpFuture futureCompletion;
32  
33      /**
34       * 
35       * @param command
36       * @param delay
37       * @param futureCompletion
38       */
39      public NoTaskExecutor(String command, long delay, WaarpFuture futureCompletion) {
40          this.futureCompletion = futureCompletion;
41      }
42  
43      public void run() throws Reply421Exception {
44          futureCompletion.setSuccess();
45      }
46  }
