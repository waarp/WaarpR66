40  /**
41   * R66PreparedTransferExecutor class. If the command starts with "REFUSED", the command will be
42   * refused for execution. If "REFUSED" is set, the command "RETR" or "STOR" like operations will be
43   * stopped at starting of command.
44   * 
45   * 
46   * 
47   * Format is like r66send command in any order except "-info" which should be the last item:<br>
48   * "-to Host -file FILE -rule RULE [-md5] [-nolog] [-start yyyyMMddHHmmss or -delay (delay or +delay)] [-info INFO]" <br>
49   * <br>
50   * INFO is the only one field that can contains blank character.<br>
51   * <br>
52   * The following replacement are done dynamically before the command is executed:<br>
53   * - #BASEPATH# is replaced by the full path for the root of FTP Directory<br>
54   * - #FILE# is replaced by the current file path relative to FTP Directory (so #BASEPATH##FILE# is
55   * the full path of the file)<br>
56   * - #USER# is replaced by the username<br>
57   * - #ACCOUNT# is replaced by the account<br>
58   * - #COMMAND# is replaced by the command issued for the file<br>
59   * - #SPECIALID# is replaced by the FTP id of the transfer (whatever in or out)<br>
60   * - #UUID# is replaced by a special UUID globally unique for the transfer, in general to be placed in -info part (for instance ##UUID## giving #uuid#)<br>
61   * <br>
62   * So for instance
63   * "-to Host -file #BASEPATH##FILE# -rule RULE [-md5] [-nolog] [-delay +delay]  [-info ##UUID## #USER# #ACCOUNT# #COMMAND# INFO]" <br>
64   * will be a standard use of this function.
65   * 
66   * @author Frederic Bregier
67   * 
68   */
69  public class R66PreparedTransferExecutor extends AbstractExecutor {
70      /**
71       * Internal Logger
72       */
73      private static final WaarpLogger logger = WaarpLoggerFactory
74              .getLogger(R66PreparedTransferExecutor.class);
75  
76      protected final WaarpFuture future;
77  
78      protected String filename = null;
79  
80      protected String rulename = null;
81  
82      protected String fileinfo = null;
83  
84      protected boolean isMD5 = false;
85  
86      protected boolean nolog = false;
87  
88      protected Timestamp timestart = null;
89  
90      protected String remoteHost = null;
91  
92      protected int blocksize = Configuration.configuration.getBLOCKSIZE();;
93  
94      protected DbSession dbsession;
95  
96      /**
97       * 
98       * @param command
99       * @param delay
100      * @param futureCompletion
101      */
102     public R66PreparedTransferExecutor(String command, long delay,
103             WaarpFuture futureCompletion) {
104         String args[] = command.split(" ");
105         for (int i = 0; i < args.length; i++) {
106             if (args[i].equalsIgnoreCase("-to")) {
107                 i++;
108                 remoteHost = args[i];
109                 if (Configuration.configuration.getAliases().containsKey(remoteHost)) {
110                     remoteHost = Configuration.configuration.getAliases().get(remoteHost);
111                 }
112             } else if (args[i].equalsIgnoreCase("-file")) {
113                 i++;
114                 filename = args[i];
115             } else if (args[i].equalsIgnoreCase("-rule")) {
116                 i++;
117                 rulename = args[i];
118             } else if (args[i].equalsIgnoreCase("-info")) {
119                 i++;
120                 fileinfo = args[i];
121                 i++;
122                 while (i < args.length) {
123                     fileinfo += " " + args[i];
124                     i++;
125                 }
126             } else if (args[i].equalsIgnoreCase("-md5")) {
127                 isMD5 = true;
128             } else if (args[i].equalsIgnoreCase("-block")) {
129                 i++;
130                 blocksize = Integer.parseInt(args[i]);
131                 if (blocksize < 100) {
132                     logger.warn("Block size is too small: " + blocksize);
133                     blocksize = Configuration.configuration.getBLOCKSIZE();
134                 }
135             } else if (args[i].equalsIgnoreCase("-nolog")) {
136                 nolog = true;
137                 i++;
138             } else if (args[i].equalsIgnoreCase("-start")) {
139                 i++;
140                 SimpleDateFormat dateFormat = new SimpleDateFormat("yyyyMMddHHmmss");
141                 Date date;
142                 try {
143                     date = dateFormat.parse(args[i]);
144                     timestart = new Timestamp(date.getTime());
145                 } catch (ParseException e) {
146                 }
147             } else if (args[i].equalsIgnoreCase("-delay")) {
148                 i++;
149                 if (args[i].charAt(0) == '+') {
150                     timestart = new Timestamp(System.currentTimeMillis() +
151                             Long.parseLong(args[i].substring(1)));
152                 } else {
153                     timestart = new Timestamp(Long.parseLong(args[i]));
154                 }
155             }
156         }
157         if (fileinfo == null) {
158             fileinfo = "noinfo";
159         }
160         this.future = futureCompletion;
161     }
162 
163     /**
164      * @param dbsession
165      *            the dbsession to set
166      */
167     public void setDbsession(DbSession dbsession) {
168         this.dbsession = dbsession;
169     }
170 
171     public void run() throws CommandAbstractException {
172         String message = "R66Prepared with -to " + remoteHost + " -rule " +
173                 rulename + " -file " + filename + " -nolog: " + nolog +
174                 " -isMD5: " + isMD5 + " -info " + fileinfo;
175         if (remoteHost == null || rulename == null || filename == null) {
176             logger.error("Mandatory argument is missing: -to " + remoteHost +
177                     " -rule " + rulename + " -file " + filename);
178             throw new Reply421Exception("Mandatory argument is missing\n    " + message);
179         }
180         logger.debug(message);
181         DbRule rule;
182         try {
183             rule = new DbRule(rulename);
184         } catch (WaarpDatabaseException e) {
185             logger.error("Cannot get Rule: " + rulename + " since {}\n    " +
186                     message, e.getMessage());
187             throw new Reply421Exception("Cannot get Rule: " +
188                     rulename + "\n    " + message);
189         }
190         int mode = rule.getMode();
191         if (isMD5) {
192             mode = RequestPacket.getModeMD5(mode);
193         }
194         String sep = PartnerConfiguration.getSeparator(remoteHost);
195         long originalSize = -1;
196         if (RequestPacket.isSendMode(mode) && !RequestPacket.isThroughMode(mode)) {
197             File file = new File(filename);
198             if (file.canRead()) {
199                 originalSize = file.length();
200             }
201         }
202         RequestPacket request = new RequestPacket(rulename, mode, filename,
203                 blocksize, 0, DbConstant.ILLEGALVALUE, fileinfo, originalSize, sep);
204         // Not isRecv since it is the requester, so send => isRetrieve is true
205         boolean isRetrieve = !RequestPacket.isRecvMode(request.getMode());
206         logger.debug("Will prepare: {}", request);
207         DbTaskRunner taskRunner;
208         try {
209             taskRunner = new DbTaskRunner(rule, isRetrieve, request,
210                     remoteHost, timestart);
211         } catch (WaarpDatabaseException e) {
212             logger.error("Cannot get new task since {}\n    " + message, e
213                     .getMessage());
214             throw new Reply421Exception("Cannot get new task\n    " + message);
215         }
216         taskRunner.changeUpdatedInfo(AbstractDbData.UpdatedInfo.TOSUBMIT);
217         if (!taskRunner.forceSaveStatus()) {
218             try {
219                 if (!taskRunner.specialSubmit()) {
220                     logger.error("Cannot prepare task: " + message);
221                     throw new Reply421Exception("Cannot get new task\n    " + message);
222                 }
223             } catch (WaarpDatabaseException e) {
224                 logger.error("Cannot prepare task since {}\n    " + message, e
225                         .getMessage());
226                 throw new Reply421Exception("Cannot get new task\n    " + message);
227             }
228         }
229         logger.debug("R66PreparedTransfer prepared: {}", request);
230         future.setSuccess();
231     }
