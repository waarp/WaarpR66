40  /**
41   * R66PreparedTransferExecutor class. If the command starts with "REFUSED", the command will be
42   * refused for execution. If "REFUSED" is set, the command "RETR" or "STOR" like operations will be
43   * stopped at starting of command.
44   * 
45   * 
46   * 
47   * Format is like r66send command in any order except "-info" which should be the last item:<br>
48   * "-to Host -file FILE -rule RULE [-md5] [-nolog] [-start yyyyMMddHHmmss or -delay (delay or +delay)] [-info INFO]"
49   * <br>
50   * <br>
51   * INFO is the only one field that can contains blank character.<br>
52   * <br>
53   * The following replacement are done dynamically before the command is executed:<br>
54   * - #BASEPATH# is replaced by the full path for the root of FTP Directory<br>
55   * - #FILE# is replaced by the current file path relative to FTP Directory (so #BASEPATH##FILE# is
56   * the full path of the file)<br>
57   * - #USER# is replaced by the username<br>
58   * - #ACCOUNT# is replaced by the account<br>
59   * - #COMMAND# is replaced by the command issued for the file<br>
60   * <br>
61   * So for instance
62   * "-to Host -file #BASEPATH##FILE# -rule RULE [-md5] [-nolog] [-delay +delay]  [-info #USER# #ACCOUNT# #COMMAND# INFO]"
63   * <br>
64   * will be a standard use of this function.
65   * 
66   * @author Frederic Bregier
67   * 
68   */
69  public class R66PreparedTransferExecutor extends AbstractExecutor {

96  	/**
97  	 * 
98  	 * @param command
99  	 * @param delay
100 	 * @param futureCompletion
101 	 */
102 	public R66PreparedTransferExecutor(String command, long delay,
103 			WaarpFuture futureCompletion) {
104 		String args[] = command.split(" ");
105 		for (int i = 0; i < args.length; i++) {
106 			if (args[i].equalsIgnoreCase("-to")) {
107 				i++;
108 				remoteHost = args[i];
109 			} else if (args[i].equalsIgnoreCase("-file")) {
110 				i++;
111 				filename = args[i];
112 			} else if (args[i].equalsIgnoreCase("-rule")) {
113 				i++;
114 				rulename = args[i];
115 			} else if (args[i].equalsIgnoreCase("-info")) {
116 				i++;
117 				fileinfo = args[i];
118 				i++;
119 				while (i < args.length) {
120 					fileinfo += " " + args[i];
121 					i++;
122 				}
123 			} else if (args[i].equalsIgnoreCase("-md5")) {
124 				isMD5 = true;
125 			} else if (args[i].equalsIgnoreCase("-block")) {
126 				i++;
127 				blocksize = Integer.parseInt(args[i]);
128 				if (blocksize < 100) {
129 					logger.warn("Block size is too small: " + blocksize);
130 					blocksize = Configuration.configuration.BLOCKSIZE;
131 				}
132 			} else if (args[i].equalsIgnoreCase("-nolog")) {
133 				nolog = true;
134 				i++;
135 			} else if (args[i].equalsIgnoreCase("-start")) {
136 				i++;
137 				SimpleDateFormat dateFormat = new SimpleDateFormat("yyyyMMddHHmmss");
138 				Date date;
139 				try {
140 					date = dateFormat.parse(args[i]);
141 					timestart = new Timestamp(date.getTime());
142 				} catch (ParseException e) {
143 				}
144 			} else if (args[i].equalsIgnoreCase("-delay")) {
145 				i++;
146 				if (args[i].charAt(0) == '+') {
147 					timestart = new Timestamp(System.currentTimeMillis() +
148 							Long.parseLong(args[i].substring(1)));
149 				} else {
150 					timestart = new Timestamp(Long.parseLong(args[i]));
151 				}
152 			}
153 		}
154 		if (fileinfo == null) {
155 			fileinfo = "noinfo";
156 		}
157 		this.future = futureCompletion;
158 	}

168 	public void run() throws CommandAbstractException {
169 		String message = "R66Prepared with -to " + remoteHost + " -rule " +
170 				rulename + " -file " + filename + " -nolog: " + nolog +
171 				" -isMD5: " + isMD5 + " -info " + fileinfo;
172 		if (remoteHost == null || rulename == null || filename == null) {
173 			logger.error("Mandatory argument is missing: -to " + remoteHost +
174 					" -rule " + rulename + " -file " + filename);
175 			throw new Reply421Exception("Mandatory argument is missing\n    " + message);
176 		}
177 		logger.debug(message);
178 		DbRule rule;
179 		try {
180 			rule = new DbRule(dbsession, rulename);
181 		} catch (WaarpDatabaseException e) {
182 			logger.error("Cannot get Rule: " + rulename + " since {}\n    " +
183 					message, e.getMessage());
184 			throw new Reply421Exception("Cannot get Rule: " +
185 					rulename + "\n    " + message);
186 		}
187 		int mode = rule.mode;
188 		if (isMD5) {
189 			mode = RequestPacket.getModeMD5(mode);
190 		}
191 		String sep = PartnerConfiguration.getSeparator(remoteHost);
192 		long originalSize = -1;
193 		if (RequestPacket.isSendMode(mode) && ! RequestPacket.isThroughMode(mode)) {
194 			File file = new File(filename);
195 			if (file.canRead()) {
196 				originalSize = file.length();
197 			}
198 		}
199 		RequestPacket request = new RequestPacket(rulename, mode, filename,
200 				blocksize, 0, DbConstant.ILLEGALVALUE, fileinfo, originalSize, sep);
201 		// Not isRecv since it is the requester, so send => isRetrieve is true
202 		boolean isRetrieve = !RequestPacket.isRecvMode(request.getMode());
203 		logger.debug("Will prepared: " + request.toString());
204 		DbTaskRunner taskRunner;
205 		try {
206 			taskRunner = new DbTaskRunner(dbsession, rule, isRetrieve, request,
207 					remoteHost, timestart);
208 		} catch (WaarpDatabaseException e) {
209 			logger.error("Cannot get new task since {}\n    " + message, e
210 					.getMessage());
211 			throw new Reply421Exception("Cannot get new task\n    " + message);
212 		}
213 		taskRunner.changeUpdatedInfo(AbstractDbData.UpdatedInfo.TOSUBMIT);
214 		try {
215 			taskRunner.update();
216 		} catch (WaarpDatabaseException e) {
217 			logger.error("Cannot prepare task since {}\n    " + message, e
218 					.getMessage());
219 			throw new Reply421Exception("Cannot prepare task\n    " + message);
220 		}
221 		future.setSuccess();
222 	}
