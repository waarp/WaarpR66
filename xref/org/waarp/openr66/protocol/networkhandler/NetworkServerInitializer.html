18  package org.waarp.openr66.protocol.networkhandler;
19  
20  import java.util.concurrent.TimeUnit;
21  
22  import io.netty.channel.ChannelInitializer;
23  import io.netty.channel.ChannelPipeline;
24  import io.netty.channel.socket.SocketChannel;
25  import io.netty.handler.timeout.IdleStateHandler;
26  import io.netty.handler.traffic.ChannelTrafficShapingHandler;
27  
28  import org.waarp.openr66.protocol.configuration.Configuration;
29  import org.waarp.openr66.protocol.exception.OpenR66ProtocolNoDataException;
30  import org.waarp.openr66.protocol.networkhandler.packet.NetworkPacketCodec;
31  
32  /**
33   * NetworkServer pipeline (Requester side)
34   * 
35   * @author Frederic Bregier
36   */
37  public class NetworkServerInitializer extends ChannelInitializer<SocketChannel> {
38      public static final String TIMEOUT = "timeout";
39      public static final String READTIMEOUT = "readTimeout";
40      public static final String LIMIT = "LIMIT";
41      public static final String LIMITCHANNEL = "LIMITCHANNEL";
42  
43      protected boolean server = false;
44  
45      public NetworkServerInitializer(boolean server) {
46          this.server = server;
47      }
48  
49      @Override
50      protected void initChannel(SocketChannel ch) throws Exception {
51          final ChannelPipeline pipeline = ch.pipeline();
52          pipeline.addLast("codec", new NetworkPacketCodec());
53          pipeline.addLast(TIMEOUT, new IdleStateHandler(0, 0, Configuration.configuration.getTIMEOUTCON(),
54                  TimeUnit.MILLISECONDS));
55          GlobalTrafficHandler handler = Configuration.configuration.getGlobalTrafficShapingHandler();
56          if (handler != null) {
57              pipeline.addLast(LIMIT, handler);
58          }
59          ChannelTrafficShapingHandler trafficChannel = null;
60          try {
61              trafficChannel = Configuration.configuration.newChannelTrafficShapingHandler();
62              if (trafficChannel != null) {
63                  pipeline.addLast(LIMITCHANNEL, trafficChannel);
64              }
65          } catch (OpenR66ProtocolNoDataException e) {
66          }
67          pipeline.addLast(Configuration.configuration.getHandlerGroup(), "handler",
68                  new NetworkServerHandler(this.server));
69      }
70  
71  }
