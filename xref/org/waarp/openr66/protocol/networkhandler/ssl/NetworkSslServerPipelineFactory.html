43  public class NetworkSslServerPipelineFactory implements ChannelPipelineFactory {
44  	private final boolean isClient;
45  	public static WaarpSslContextFactory waarpSslContextFactory;
46  	public static WaarpSecureKeyStore WaarpSecureKeyStore;
47  	private final ExecutorService executorService;
48  	/**
49  	 * Global HashedWheelTimer
50  	 */
51  	public HashedWheelTimer timer = (HashedWheelTimer) Configuration.configuration
52  			.getTimerClose();
53  
54  	/**
55  	 * 
56  	 * @param isClient
57  	 *            True if this Factory is to be used in Client mode
58  	 */
59  	public NetworkSslServerPipelineFactory(boolean isClient, ExecutorService executor) {
60  		super();
61  		this.isClient = isClient;
62  		this.executorService = executor;
63  	}
64  
65  	@Override
66  	public ChannelPipeline getPipeline() {
67  		final ChannelPipeline pipeline = Channels.pipeline();
68  		// Add SSL handler first to encrypt and decrypt everything.
69  		SslHandler sslHandler = null;
70  		if (isClient) {
71  			// Not server: no clientAuthent, no renegotiation
72  			sslHandler = 
73  					waarpSslContextFactory.initPipelineFactory(false,
74  							false, false, executorService);
75  			sslHandler.setIssueHandshake(true);
76  		} else {
77  			// Server: no renegotiation still, but possible clientAuthent
78  			sslHandler = 
79  					waarpSslContextFactory.initPipelineFactory(true,
80  							waarpSslContextFactory.needClientAuthentication(),
81  							true, executorService);
82  		}
83  		pipeline.addLast("ssl", sslHandler);
84  
85  		pipeline.addLast("codec", new NetworkPacketCodec());
86  		GlobalTrafficShapingHandler handler = Configuration.configuration
87  				.getGlobalTrafficShapingHandler();
88  		if (handler != null) {
89  			pipeline.addLast(NetworkServerPipelineFactory.LIMIT, handler);
90  		}
91  		ChannelTrafficShapingHandler trafficChannel = null;
92  		try {
93  			trafficChannel =
94  					Configuration.configuration
95  							.newChannelTrafficShapingHandler();
96  			pipeline.addLast(NetworkServerPipelineFactory.LIMITCHANNEL, trafficChannel);
97  		} catch (OpenR66ProtocolNoDataException e) {
98  		}
99  		pipeline.addLast("pipelineExecutor", new ExecutionHandler(
100 				Configuration.configuration.getServerPipelineExecutor()));
101 
102 		pipeline.addLast(NetworkServerPipelineFactory.TIMEOUT,
103 				new IdleStateHandler(timer,
104 						0, 0,
105 						Configuration.configuration.TIMEOUTCON,
106 						TimeUnit.MILLISECONDS));
107 		pipeline.addLast("handler", new NetworkSslServerHandler(!this.isClient));
108 		return pipeline;
109 	}
