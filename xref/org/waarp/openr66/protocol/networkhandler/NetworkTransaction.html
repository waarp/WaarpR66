170 	private static ReentrantLock getChannelLock(SocketAddress socketAddress) {
171 		lock.lock();
172 		try {
173 			if (socketAddress == null) {
174 				// should not
175 				logger.info("SocketAddress empty here !");
176 				return lock;
177 			}
178 			Integer hash = socketAddress.hashCode();
179 			ReentrantLock socketLock = reentrantLockOnSocketAddressConcurrentHashMap.get(hash);
180 			if (socketLock == null) {
181 				socketLock = new ReentrantLock(true);
182 				reentrantLockOnSocketAddressConcurrentHashMap.put(hash, socketLock);
183 			}
184 			return socketLock;
185 		} finally {
186 			lock.unlock();
187 		}
188 	}
189 
190 	private static void removeChannelLock(SocketAddress socketAddress) {
191 		lock.lock();
192 		try {
193 			if (socketAddress == null) {
194 				// should not
195 				logger.info("SocketAddress empty here !");
196 				return;
197 			}
198 			Integer hash = socketAddress.hashCode();
199 			reentrantLockOnSocketAddressConcurrentHashMap.remove(hash);
200 		} finally {
201 			lock.unlock();
202 		}
203 	}

205 	/**
206 	 * Create a connection to the specified socketAddress with multiple retries
207 	 * 
208 	 * @param socketAddress
209 	 * @param isSSL
210 	 * @param futureRequest
211 	 * @return the LocalChannelReference
212 	 */
213 	public LocalChannelReference createConnectionWithRetry(SocketAddress socketAddress,
214 			boolean isSSL, R66Future futureRequest) {
215 		LocalChannelReference localChannelReference = null;
216 		OpenR66Exception lastException = null;
217 		for (int i = 0; i < Configuration.RETRYNB; i++) {
218 			try {
219 				localChannelReference =
220 						createConnection(socketAddress, isSSL, futureRequest);
221 				break;
222 			} catch (OpenR66ProtocolRemoteShutdownException e1) {
223 				lastException = e1;
224 				localChannelReference = null;
225 				break;
226 			} catch (OpenR66ProtocolNoConnectionException e1) {
227 				lastException = e1;
228 				localChannelReference = null;
229 				break;
230 			} catch (OpenR66ProtocolNetworkException e1) {
231 				// Can retry
232 				lastException = e1;
233 				localChannelReference = null;
234 				try {
235 					Thread.sleep(Configuration.WAITFORNETOP);
236 				} catch (InterruptedException e) {
237 					break;
238 				}
239 			}
240 		}
241 		if (localChannelReference == null) {
242 			logger.debug("Cannot connect : {}", lastException.getMessage());
243 		} else if (lastException != null) {
244 			logger.debug("Connection retried since {}", lastException.getMessage());
245 		}
246 		return localChannelReference;
247 	}

249 	/**
250 	 * Create a connection to the specified socketAddress
251 	 * 
252 	 * @param socketAddress
253 	 * @param isSSL
254 	 * @param futureRequest
255 	 * @return the LocalChannelReference
256 	 * @throws OpenR66ProtocolNetworkException
257 	 * @throws OpenR66ProtocolRemoteShutdownException
258 	 * @throws OpenR66ProtocolNoConnectionException
259 	 */
260 	public LocalChannelReference createConnection(SocketAddress socketAddress, boolean isSSL,
261 			R66Future futureRequest)
262 			throws OpenR66ProtocolNetworkException,
263 			OpenR66ProtocolRemoteShutdownException,
264 			OpenR66ProtocolNoConnectionException {
265 		NetworkChannel networkChannel = null;
266 		LocalChannelReference localChannelReference = null;
267 		boolean ok = false;
268 		// check valid limit on server side only (could be the initiator but not a client)
269 		if (!Configuration.configuration.HOST_AUTH.isClient()) {
270 			boolean valid = false;
271 			for (int i = 0; i < Configuration.RETRYNB * 2; i++) {
272 				if (Configuration.configuration.constraintLimitHandler.checkConstraintsSleep(i)) {
273 					logger.debug("Constraints exceeded: " + i);
274 				} else {
275 					logger.debug("Constraints NOT exceeded");
276 					valid = true;
277 					break;
278 				}
279 			}
280 			if (!valid) {
281 				// Limit is locally exceeded
282 				logger.debug("Overloaded local system");
283 				throw new OpenR66ProtocolNetworkException(
284 						"Cannot connect to remote server due to local overload");
285 			}
286 		}
287 		try {
288 			networkChannel = createNewConnection(socketAddress, isSSL);
289 			localChannelReference = createNewClient(networkChannel, futureRequest);
290 			ok = true;
291 		} finally {
292 			if (!ok) {
293 				if (networkChannel != null) {
294 					removeNetworkChannel(networkChannel.channel, null, null);
295 				}
296 			}
297 		}
298 		if (localChannelReference.getFutureValidateStartup().isDone() &&
299 				localChannelReference.getFutureValidateStartup().isSuccess()) {
300 			sendValidationConnection(localChannelReference);
301 		} else {
302 			OpenR66ProtocolNetworkException exc =
303 					new OpenR66ProtocolNetworkException("Startup is invalid");
304 			logger.debug("Startup is Invalid", exc);
305 			R66Result finalValue = new R66Result(
306 					exc, null, true, ErrorCode.ConnectionImpossible, null);
307 			localChannelReference.invalidateRequest(finalValue);
308 			Channels.close(localChannelReference.getLocalChannel());
309 			throw exc;
310 		}
311 		return localChannelReference;
312 	}

314 	/**
315 	 * 
316 	 * @param socketServerAddress
317 	 * @param isSSL
318 	 * @return the NetworkChannel
319 	 * @throws OpenR66ProtocolNetworkException
320 	 * @throws OpenR66ProtocolRemoteShutdownException
321 	 * @throws OpenR66ProtocolNoConnectionException
322 	 */
323 	private NetworkChannel createNewConnection(SocketAddress socketServerAddress, boolean isSSL)
324 			throws OpenR66ProtocolNetworkException,
325 			OpenR66ProtocolRemoteShutdownException,
326 			OpenR66ProtocolNoConnectionException {
327 		ReentrantLock socketLock = getChannelLock(socketServerAddress);
328 		socketLock.lock();
329 		try {
330 			if (!isAddressValid(socketServerAddress)) {
331 				throw new OpenR66ProtocolRemoteShutdownException(
332 						"Cannot connect to remote server since it is shutting down");
333 			}
334 			NetworkChannel networkChannel;
335 			try {
336 				networkChannel = getRemoteChannel(socketServerAddress);
337 			} catch (OpenR66ProtocolNoDataException e1) {
338 				networkChannel = null;
339 			}
340 			if (networkChannel != null) {
341 				networkChannel.count.incrementAndGet();
342 				logger.info("Already Connected: {}", networkChannel);
343 				return networkChannel;
344 			}
345 			ChannelFuture channelFuture = null;
346 			for (int i = 0; i < Configuration.RETRYNB; i++) {
347 				try {
348 					if (isSSL) {
349 						if (Configuration.configuration.HOST_SSLID != null) {
350 							channelFuture = clientSslBootstrap.connect(socketServerAddress);
351 						} else {
352 							throw new OpenR66ProtocolNoConnectionException("No SSL support");
353 						}
354 					} else {
355 						channelFuture = clientBootstrap.connect(socketServerAddress);
356 					}
357 				} catch (ChannelPipelineException e) {
358 					throw new OpenR66ProtocolNoConnectionException(
359 							"Cannot connect to remote server due to a channel exception");
360 				}
361 				try {
362 					channelFuture.await();
363 				} catch (InterruptedException e1) {
364 				}
365 				if (channelFuture.isSuccess()) {
366 					final Channel channel = channelFuture.getChannel();
367 					if (isSSL) {
368 						if (!NetworkSslServerHandler.isSslConnectedChannel(channel)) {
369 							logger.debug("KO CONNECT since SSL handshake is over");
370 							Channels.close(channel);
371 							throw new OpenR66ProtocolNoConnectionException(
372 									"Cannot finish connect to remote server");
373 						}
374 					}
375 					networkChannelGroup.add(channel);
376 					return putRemoteChannel(channel);
377 				} else {
378 					try {
379 						Thread.sleep(Configuration.WAITFORNETOP);
380 					} catch (InterruptedException e) {
381 					}
382 					if (!channelFuture.isDone()) {
383 						throw new OpenR66ProtocolNoConnectionException(
384 								"Cannot connect to remote server due to interruption");
385 					}
386 					if (channelFuture.getCause() instanceof ConnectException) {
387 						logger.debug("KO CONNECT:" +
388 								channelFuture.getCause().getMessage());
389 						throw new OpenR66ProtocolNoConnectionException(
390 								"Cannot connect to remote server", channelFuture
391 										.getCause());
392 					} else {
393 						logger.debug("KO CONNECT but retry", channelFuture
394 								.getCause());
395 					}
396 				}
397 			}
398 			throw new OpenR66ProtocolNetworkException(
399 					"Cannot connect to remote server", channelFuture.getCause());
400 		} finally {
401 			socketLock.unlock();
402 		}
403 	}

406 	 * 
407 	 * @param channel
408 	 * @param futureRequest
409 	 * @return the LocalChannelReference
410 	 * @throws OpenR66ProtocolNetworkException
411 	 * @throws OpenR66ProtocolRemoteShutdownException
412 	 */
413 	private LocalChannelReference createNewClient(NetworkChannel networkChannel,
414 			R66Future futureRequest)
415 			throws OpenR66ProtocolNetworkException,
416 			OpenR66ProtocolRemoteShutdownException {
417 		if (!networkChannel.channel.isConnected()) {
418 			throw new OpenR66ProtocolNetworkException(
419 					"Network channel no more connected");
420 		}
421 		LocalChannelReference localChannelReference = null;
422 		try {
423 			localChannelReference = Configuration.configuration
424 					.getLocalTransaction().createNewClient(networkChannel.channel,
425 							ChannelUtils.NOCHANNEL, futureRequest);
426 		} catch (OpenR66ProtocolSystemException e) {
427 			// check if the channel has other attached local channels
428 			// NO since done in caller (createConnection)
429 			/*
430 			 * logger.info("Try to Close Network"); removeNetworkChannel(networkChannel.channel,
431 			 * null);
432 			 */
433 			throw new OpenR66ProtocolNetworkException(
434 					"Cannot connect to local channel", e);
435 		}
436 		return localChannelReference;
437 	}

458 	/**
459 	 * Send a validation of connection with Authentication
460 	 * 
461 	 * @param localChannelReference
462 	 * @throws OpenR66ProtocolNetworkException
463 	 * @throws OpenR66ProtocolRemoteShutdownException
464 	 */
465 	private void sendValidationConnection(
466 			LocalChannelReference localChannelReference)
467 			throws OpenR66ProtocolNetworkException,
468 			OpenR66ProtocolRemoteShutdownException {
469 		AuthentPacket authent;
470 		try {
471 			authent = new AuthentPacket(
472 					Configuration.configuration.getHostId(
473 							localChannelReference.getNetworkServerHandler().isSsl()),
474 					FilesystemBasedDigest.passwdCrypt(
475 							Configuration.configuration.HOST_AUTH.getHostkey()),
476 					localChannelReference.getLocalId());
477 		} catch (OpenR66ProtocolNoSslException e1) {
478 			R66Result finalValue = new R66Result(
479 					new OpenR66ProtocolSystemException("No SSL support", e1),
480 					localChannelReference.getSession(), true, ErrorCode.ConnectionImpossible, null);
481 			logger.error("Authent is Invalid due to no SSL: {}", e1.getMessage());
482 			if (localChannelReference.getRemoteId() != ChannelUtils.NOCHANNEL) {
483 				ConnectionErrorPacket error = new ConnectionErrorPacket(
484 						"Cannot connect to localChannel since no SSL is supported", null);
485 				try {
486 					ChannelUtils.writeAbstractLocalPacket(localChannelReference, error, true);
487 				} catch (OpenR66ProtocolPacketException e) {
488 				}
489 			}
490 			localChannelReference.invalidateRequest(finalValue);
491 			Channels.close(localChannelReference.getLocalChannel());
492 			throw new OpenR66ProtocolNetworkException(e1);
493 		}
494 		logger.debug("Will send request of connection validation");
495 		localChannelReference.sessionNewState(AUTHENTR);
496 		try {
497 			ChannelUtils.writeAbstractLocalPacket(localChannelReference, authent, true);
498 		} catch (OpenR66ProtocolPacketException e) {
499 			R66Result finalValue = new R66Result(
500 					new OpenR66ProtocolSystemException("Wrong Authent Protocol", e),
501 					localChannelReference.getSession(), true, ErrorCode.ConnectionImpossible, null);
502 			logger.error("Authent is Invalid due to protocol: {}", e.getMessage());
503 			localChannelReference.invalidateRequest(finalValue);
504 			if (localChannelReference.getRemoteId() != ChannelUtils.NOCHANNEL) {
505 				ConnectionErrorPacket error = new ConnectionErrorPacket(
506 						"Cannot connect to localChannel since Authent Protocol is invalid", null);
507 				try {
508 					ChannelUtils.writeAbstractLocalPacket(localChannelReference, error, true);
509 				} catch (OpenR66ProtocolPacketException e1) {
510 				}
511 			}
512 			Channels.close(localChannelReference.getLocalChannel());
513 			throw new OpenR66ProtocolNetworkException("Bad packet", e);
514 		}
515 		R66Future future = localChannelReference.getFutureValidateConnection();
516 		if (future.isFailed()) {
517 			logger.debug("Will close NETWORK channel since Future cancelled: {}",
518 					future);
519 			R66Result finalValue = new R66Result(
520 					new OpenR66ProtocolSystemException(
521 							"Out of time or Connection invalid during Authentication"),
522 					localChannelReference.getSession(), true, ErrorCode.ConnectionImpossible, null);
523 			logger.warn("Authent is Invalid due to: {} {}", finalValue.exception.getMessage(),
524 					future.toString());
525 			localChannelReference.invalidateRequest(finalValue);
526 			if (localChannelReference.getRemoteId() != ChannelUtils.NOCHANNEL) {
527 				ConnectionErrorPacket error = new ConnectionErrorPacket(
528 						"Cannot connect to localChannel with Out of Time", null);
529 				try {
530 					ChannelUtils.writeAbstractLocalPacket(localChannelReference, error, true);
531 				} catch (OpenR66ProtocolPacketException e) {
532 				}
533 			}
534 			Channels.close(localChannelReference.getLocalChannel());
535 			throw new OpenR66ProtocolNetworkException(
536 					"Cannot validate connection: " + future.getResult(), future
537 							.getCause());
538 		}
539 	}

592 	/**
593 	 * Close all Network Ttransaction
594 	 */
595 	public void closeAll() {
596 		logger.debug("close All Network Channels");
597 		try {
598 			Thread.sleep(Configuration.RETRYINMS * 2);
599 		} catch (InterruptedException e) {
600 		}
601 		if (!Configuration.configuration.isServer) {
602 			R66ShutdownHook.shutdownHook.launchFinalExit();
603 		}
604 		closeRetrieveExecutors();
605 		networkChannelGroup.close().awaitUninterruptibly();
606 		clientBootstrap.releaseExternalResources();
607 		clientSslBootstrap.releaseExternalResources();
608 		channelClientFactory.releaseExternalResources();
609 		try {
610 			Thread.sleep(Configuration.WAITFORNETOP);
611 		} catch (InterruptedException e) {
612 		}
613 		DbAdmin.closeAllConnection();
614 		Configuration.configuration.clientStop();
615 		logger.debug("Last action before exit");
616 		ChannelUtils.stopLogger();
617 	}
