18  package org.waarp.openr66.protocol.networkhandler;
19  
20  import java.util.concurrent.ConcurrentLinkedQueue;
21  import java.util.concurrent.atomic.AtomicInteger;
22  
23  import org.jboss.netty.channel.Channel;
24  import org.jboss.netty.channel.Channels;
25  import org.waarp.openr66.protocol.exception.OpenR66ProtocolRemoteShutdownException;
26  
27  /**
28   * NetworkChannel object to keep Network channel open while some local channels are attached to it.
29   * 
30   * @author Frederic Bregier
31   * 
32   */
33  public class NetworkChannel {
34  	/**
35  	 * Number of active Local Channel referencing this Network Channel
36  	 */
37  	public volatile AtomicInteger count = new AtomicInteger(1);
38  	/**
39  	 * Does this Network Channel is in shutdown
40  	 */
41  	public volatile boolean isShuttingDown = false;
42  	/**
43  	 * Associated LocalChannel
44  	 */
45  	public ConcurrentLinkedQueue<Channel> localChannels =
46  			new ConcurrentLinkedQueue<Channel>();
47  	/**
48  	 * Network Channel
49  	 */
50  	public final Channel channel;
51  	/**
52  	 * Remote Host Id
53  	 */
54  	public String hostId;
55  	/**
56  	 * Last Time in ms this channel was used by a LocalChannel
57  	 */
58  	public long lastTimeUsed = System.currentTimeMillis();
59  
60  	public NetworkChannel(Channel networkChannel) {
61  		this.channel = networkChannel;
62  	}
63  
64  	synchronized public void add(Channel localChannel)
65  			throws OpenR66ProtocolRemoteShutdownException {
66  		if (isShuttingDown) {
67  			throw new OpenR66ProtocolRemoteShutdownException("Current NetworkChannel is closed");
68  		}
69  		lastTimeUsed = System.currentTimeMillis();
70  		localChannels.add(localChannel);
71  	}
72  	
73  	/**
74  	 * To set the last time used
75  	 */
76  	public void use() {
77  		lastTimeUsed = System.currentTimeMillis();
78  	}
79  
80  	synchronized public void remove(Channel localChannel) {
81  		if (localChannel.isConnected()) {
82  			Channels.close(localChannel);
83  		}
84  		localChannels.remove(localChannel);
85  	}
86  
87  	synchronized public void shutdownAllLocalChannels() {
88  		isShuttingDown = true;
89  		count.set(0);
90  		Channel localChannel = localChannels.poll();
91  		while (localChannel != null) {
92  			Channels.close(localChannel);
93  			localChannel = localChannels.poll();
94  		}
95  	}
96  	
97  	@Override
98  	public String toString() {
99  		return "NC: " + hostId+":"+ channel.isConnected() + " " +
100 				channel.getRemoteAddress() + " Count: " + count;
101 	}
102 
103 	/*
104 	 * (non-Javadoc)
105 	 * @see java.lang.Object#equals(java.lang.Object)
106 	 */
107 	@Override
108 	public boolean equals(Object obj) {
109 		if (obj instanceof NetworkChannel) {
110 			NetworkChannel obj2 = (NetworkChannel) obj;
111 			return (obj2.channel.getId() == this.channel.getId());
112 		}
113 		return false;
114 	}
115 
116 	/*
117 	 * (non-Javadoc)
118 	 * @see java.lang.Object#hashCode()
119 	 */
120 	@Override
121 	public int hashCode() {
122 		return this.channel.getId();
123 	}

125 }
