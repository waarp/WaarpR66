600 	public Configuration() {
601 		// Init signal handler
602 		shutdownConfiguration.timeout = TIMEOUTCON;
603 		new R66ShutdownHook(shutdownConfiguration);
604 		computeNbThreads();
605 		// Init FiniteStates
606 		R66FiniteDualStates.initR66FiniteStates();
607 		isExecuteErrorBeforeTransferAllowed = SystemPropertyUtil.getBoolean(R66SystemProperties.OPENR66_EXECUTEBEFORETRANSFERRED, true);
608 		boolean useSpaceSeparator = SystemPropertyUtil.getBoolean(R66SystemProperties.OPENR66_USESPACESEPARATOR, false);
609 		if (useSpaceSeparator) {
610 			PartnerConfiguration.SEPARATOR_FIELD = PartnerConfiguration.BLANK_SEPARATOR_FIELD;
611 		}
612 		isHostProxyfied = SystemPropertyUtil.getBoolean(R66SystemProperties.OPENR66_ISHOSTPROXYFIED, false);
613 		warnOnStartup = SystemPropertyUtil.getBoolean(R66SystemProperties.OPENR66_STARTUP_WARNING, true);
614 		chrootChecked = SystemPropertyUtil.getBoolean(R66SystemProperties.OPENR66_CHROOT_CHECKED, true);
615 		blacklistBadAuthent = SystemPropertyUtil.getBoolean(R66SystemProperties.OPENR66_BLACKLIST_BADAUTHENT, true);
616 		maxfilenamelength = SystemPropertyUtil.getInt(R66SystemProperties.OPENR66_FILENAME_MAXLENGTH, 255);
617 		if (isHostProxyfied) {
618 			blacklistBadAuthent = false;
619 		}
620 	}

629 	/**
630 	 * Configure the pipeline for client (to be called ony once)
631 	 */
632 	public void pipelineInit() {
633 		if (configured) {
634 			return;
635 		}
636 		localTransaction = new LocalTransaction();
637 		InternalLoggerFactory.setDefaultFactory(InternalLoggerFactory
638 				.getDefaultFactory());
639 		objectSizeEstimator = new NetworkPacketSizeEstimator();
640 		httpPipelineInit();
641 		if (warnOnStartup) {
642 			logger.warn("Server Thread: " + SERVER_THREAD + " Client Thread: " + CLIENT_THREAD
643 				+ " Runner Thread: " + RUNNER_THREAD);
644 		} else {
645 			logger.info("Server Thread: " + SERVER_THREAD + " Client Thread: " + CLIENT_THREAD
646 					+ " Runner Thread: " + RUNNER_THREAD);
647 		}
648 		serverPipelineExecutor = new OrderedMemoryAwareThreadPoolExecutor(
649 				CLIENT_THREAD, maxGlobalMemory / 10, maxGlobalMemory, 1000,
650 				TimeUnit.MILLISECONDS, objectSizeEstimator,
651 				new WaarpThreadFactory("ServerExecutor"));
652 		localPipelineExecutor = new OrderedMemoryAwareThreadPoolExecutor(
653 				CLIENT_THREAD * 100, maxGlobalMemory / 10, maxGlobalMemory,
654 				1000, TimeUnit.MILLISECONDS, new LocalPacketSizeEstimator(),
655 				new WaarpThreadFactory("LocalExecutor"));
656 		localClientPipelineExecutor = new OrderedMemoryAwareThreadPoolExecutor(
657 				CLIENT_THREAD * 100, maxGlobalMemory / 10, maxGlobalMemory,
658 				1000, TimeUnit.MILLISECONDS, new LocalPacketSizeEstimator(),
659 				new WaarpThreadFactory("LocalClientExecutor"));
660 		if (useLocalExec) {
661 			LocalExecClient.initialize();
662 		}
663 		configured = true;
664 	}

672 	/**
673 	 * Startup the server
674 	 * 
675 	 * @throws WaarpDatabaseSqlException
676 	 * @throws WaarpDatabaseNoConnectionException
677 	 */
678 	public void serverStartup() throws WaarpDatabaseNoConnectionException,
679 			WaarpDatabaseSqlException {
680 		isServer = true;
681 		if (blacklistBadAuthent) {
682 			blacklistBadAuthent = ! DbHostAuth.hasProxifiedHosts(DbConstant.admin.session);
683 		}
684 		shutdownConfiguration.timeout = TIMEOUTCON;
685 		R66ShutdownHook.addShutdownHook();
686 		logger.debug("Use NoSSL: "+useNOSSL+" Use SSL: "+useSSL);
687 		if ((!useNOSSL) && (!useSSL)) {
688 			logger.error(Messages.getString("Configuration.NoSSL")); //$NON-NLS-1$
689 			System.exit(-1);
690 		}
691 		pipelineInit();
692 		r66Startup();
693 		startHttpSupport();
694 		startMonitoring();
695 		if (logger.isDebugEnabled()) {
696 			// XXX FIXME for debug
697 			launchInFixedDelay(new UsageStatistic(), 10, TimeUnit.SECONDS);
698 		}

1008 		newReadLimit = readSessionLimit > 1024 ? readSessionLimit
1009 				: serverChannelReadLimit;
1010 		if (readSessionLimit <= 0) {
1011 			newReadLimit = 0;
1012 		}
1013 		serverChannelReadLimit = newReadLimit;
