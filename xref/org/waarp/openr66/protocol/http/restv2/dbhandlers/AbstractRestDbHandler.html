22  package org.waarp.openr66.protocol.http.restv2.dbhandlers;
23  
24  import io.cdap.http.AbstractHttpHandler;
25  import io.netty.handler.codec.http.HttpMethod;
26  import io.netty.handler.codec.http.HttpRequest;
27  
28  import static io.netty.handler.codec.http.HttpMethod.*;
29  import static org.waarp.gateway.kernel.rest.RestConfiguration.CRUD;
30  
31  /**
32   * This abstract class represents a handler for the WaarpR66 REST API. A handler
33   * can handle one or multiple entry points with multiple methods each.
34   *
35   * To assign an entry point to a handler, use the {@link javax.ws.rs.Path}
36   * annotation with the URI as value. This annotation can also be used on methods
37   * of the handler to create entry points below the handler's one.
38   *
39   * For each entry point, all the desired HTTP methods must be assigned a
40   * corresponding Java method using the appropriate {@link javax.ws.rs.HttpMethod}.
41   * These methods must all have an {@link HttpRequest} and  a
42   * {@link io.cdap.http.HttpResponder} as their first 2 parameters, which
43   * correspond respectively to the request and its reply. The methods may also
44   * have other parameters marked with the {@link javax.ws.rs.PathParam} annotation,
45   * meaning that these parameters correspond to the {id} given in the URI of
46   * the request. Some parameters may also be marked with the
47   * {@link javax.ws.rs.QueryParam} an annotation, in which case the parameter
48   * marked is referring to the query parameter whose name is given in the
49   * annotation's parameter. A query parameter can be given a default value
50   * with the {@link javax.ws.rs.DefaultValue} which will be assigned to the
51   * parameter when the request does not contain a value for it.
52   *
53   * These methods should also be annotated with the {@link javax.ws.rs.Consumes}
54   * annotation to indicate the expected content type of the request.
55   * Finally, each method should also be annotated with the
56   * {@link RequiredRole} annotation
57   * to make the method only accessible to users with the specified
58   * {@link org.waarp.common.role.RoleDefault.ROLE} or higher.
59   */
60  public abstract class AbstractRestDbHandler extends AbstractHttpHandler {
61  
62      /** A byte mask defining which HTTP methods are allowed on the handler. */
63      protected final byte crud;
64  
65      /**
66       * Initializes the handler with the given CRUD mask.
67       *
68       * @param crud the CRUD mask for this handler
69       */
70      protected AbstractRestDbHandler(byte crud) {
71          this.crud = crud;
72      }
73  
74      /**
75       * Checks whether the {@link HttpRequest} given can be made on the handler in
76       * accordance with the handler's CRUD configuration.
77       *
78       * @param request the HTTP request to check
79       * @return        {@code true} if the request is active, {@code false} otherwise.
80       */
81      public boolean checkCRUD(HttpRequest request) {
82          HttpMethod method = request.method();
83          if (method.equals(GET)) {
84              return CRUD.READ.isValid(crud);
85          }
86          else if (method.equals(POST)) {
87              return CRUD.CREATE.isValid(crud);
88          }
89          else if (method.equals(DELETE)) {
90              return CRUD.DELETE.isValid(crud);
91          }
92          else if (method.equals(PUT)) {
93              return CRUD.UPDATE.isValid(crud);
94          }
95          else return method.equals(OPTIONS);
96      }
97  }
