46  import static com.sun.org.apache.xalan.internal.xsltc.trax.TransformerFactoryImpl.INDENT_NUMBER;

48  import static javax.xml.transform.OutputKeys.OMIT_XML_DECLARATION;

51  /** A series of utility methods for serializing and deserializing XML. */
52  public final class XmlUtils {
53  
54      /** Prevents the default constructor from being called. */
55      private XmlUtils() throws InstantiationException {
56          throw new InstantiationException(this.getClass().getName() +
57                  " cannot be instantiated.");
58      }
59  
60  
61      //######################### PUBLIC METHODS #################################
62  
63      /**
64       * Converts a serializable Java object into XML format as a String.
65       *
66       * @param object the object to convert to XML
67       * @return       the object's representation in XML
68       * @throws InternalServerErrorException if an unexpected error occurred
69       */
70      public static String objectToXml(XmlSerializable object) {
71          try {
72              StringWriter writer = new StringWriter();
73              JAXBContext context = JAXBContext.newInstance(object.getClass());
74              Marshaller marshaller = context.createMarshaller();
75              marshaller.setProperty(Marshaller.JAXB_FRAGMENT, true);
76              marshaller.marshal(object, writer);
77  
78              return writer.toString();
79          } catch (JAXBException e) {
80              throw new InternalServerErrorException(e);
81          }
82      }
83  
84      /**
85       * Converts an XML String into a serializable Java object.
86       *
87       * @param xml  the string to convert into an object
88       * @param clazz the class of the serializable object
89       * @return     the deserialized Java object
90       * @throws InternalServerErrorException if an unexpected error occurred
91       */
92      public static <T extends XmlSerializable> T xmlToObject(String xml,
93                                                              Class<T> clazz) {
94          try {
95              StringReader reader = new StringReader(xml);
96              StreamSource source = new StreamSource(reader);
97              JAXBContext context = JAXBContext.newInstance(clazz);
98              Unmarshaller unmarshaller = context.createUnmarshaller();
99  
100             return unmarshaller.unmarshal(source, clazz).getValue();
101         } catch (JAXBException e) {
102             throw new InternalServerErrorException(e);
103         }
104     }
105 
106     /**
107      * Saves an XML String to a file at the given location.
108      *
109      * @param xml      the XML String
110      * @param filePath the path where to save the XML file
111      * @throws InternalServerErrorException if an unexpected error occurred
112      */
113     public static void saveXML(String xml, String filePath) {
114         try {
115             FileWriter fileWriter = new FileWriter(filePath, false);
116             String formattedXML = pretty(xml);
117             fileWriter.write(formattedXML);
118             fileWriter.flush();
119             fileWriter.close();
120         } catch (IOException e) {
121             throw new InternalServerErrorException(e);
122         }
123     }
124 
125     /**
126      * Loads an XML file into a String.
127      *
128      * @param filePath the path of the XML file to load
129      * @return         the content of the XML file
130      * @throws InternalServerErrorException if an unexpected error occurred
131      */
132     public static String loadXML(String filePath) {
133         try {
134             BufferedReader buff = new BufferedReader(new FileReader(filePath));
135             StringBuilder stringBuilder = new StringBuilder();
136             String line;
137             while ((line = buff.readLine()) != null) {
138                 stringBuilder.append(line.trim());
139             }
140             return stringBuilder.toString();
141         } catch (FileNotFoundException e) {
142             throw new RestErrorException(RestErrors.FILE_NOT_FOUND(filePath));
143         } catch (IOException e) {
144             throw new InternalServerErrorException(e);
145         }
146     }
147 
148     /**
149      * Saves a serializable Java object to an XML file at the given location.
150      *
151      * @param object    the object to save as XML
152      * @param filePath  the path where to save the XML file
153      * @throws InternalServerErrorException if an unexpected error occurred
154      */
155     public static void saveObject(XmlSerializable object, String filePath) {
156 
157         String xml = objectToXml(object);
158         saveXML(xml, filePath);
159     }
160 
161     /**
162      * Loads the given XML file into a corresponding serializable Java object.
163      *
164      * @param filePath path of the file to load
165      * @param clazz    class of the target Java object
166      * @return         the deserialized XML object
167      * @throws InternalServerErrorException if an unexpected error occurred
168      */
169     public static <T extends XmlSerializable> T loadObject(String filePath,
170                                                            Class<T> clazz) {
171 
172         String xml = loadXML(filePath);
173         return xmlToObject(xml, clazz);
174     }
175 
176     //######################### PRIVATE METHODS #################################
177 
178     /**
179      * Formats an unformatted XML String into a human readable one.
180      *
181      * @param input   The unformatted XML String.
182      * @return      The XML String in human readable format.
183      * @throws InternalServerErrorException if an unexpected error occurred
184      */
185     private static String pretty(String input) {
186         try {
187             Source xmlInput = new StreamSource(new StringReader(input));
188             StringWriter stringWriter = new StringWriter();
189             StreamResult xmlOutput = new StreamResult(stringWriter);
190             TransformerFactory transformerFactory = TransformerFactory.newInstance();
191             transformerFactory.setAttribute(INDENT_NUMBER, 2);
192             Transformer transformer = transformerFactory.newTransformer();
193             transformer.setOutputProperty(INDENT, "yes");
194             transformer.setOutputProperty(OMIT_XML_DECLARATION, "yes");
195 
196             transformer.transform(xmlInput, xmlOutput);
197             return xmlOutput.getWriter().toString();
198         } catch (TransformerConfigurationException e) {
199             throw new InternalServerErrorException(e);
200         } catch (TransformerException e) {
201             throw new InternalServerErrorException(e);
202         }
203     }
204 }
