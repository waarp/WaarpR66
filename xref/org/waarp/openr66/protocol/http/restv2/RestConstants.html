32  /**
33   * A list of all constants of the RESTv2 API.
34   * <p>
35   * This includes the URI of all the entry points, the name of HTTP headers
36   * specific to the API, the name of the JSON objects' fields, and
37   * a {@link DAOFactory} to create DAOs for access to the database.
38   */
39  public final class RestConstants {
40  
41      static {
42          DAOFactory.initialize(ConnectionFactory.getInstance());
43          DAO_FACTORY = DAOFactory.getInstance();
44      }
45  
46      /** Makes the default constructor of this utility class inaccessible. */
47      private RestConstants() throws InstantiationException {
48          throw new InstantiationException(this.getClass().getName() +
49                  " cannot be instantiated.");
50      }

78      /** Root directory of the API. */

81      /** Name of the URI parameter containing the id of an entry in a collection. */
82      public static final String URI_ID = "id";

85      private static final String ID_PARAMETER = "/{"+ URI_ID + "}";
