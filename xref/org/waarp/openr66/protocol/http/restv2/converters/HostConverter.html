43  /**
44   * A collection of utility methods to convert {@link Host} objects
45   * to their corresponding {@link ObjectNode} and vice-versa.
46   */
47  public final class HostConverter {
48  
49      /** Makes the default constructor of this utility class inaccessible. */
50      private HostConverter() throws InstantiationException {
51          throw new InstantiationException(this.getClass().getName() +
52                  " cannot be instantiated.");
53      }

55      //########################### INNER CLASSES ################################
56  
57      /** Represents all the possible ways to sort a list of host objects. */
58      public enum Order {
59          /** By hostID, in ascending order. */
60          ascId(new Comparator<Host>() {
61              @Override
62              public int compare(Host t1, Host t2) {
63                  return t1.getHostid().compareTo(t2.getHostid());
64              }
65          }),
66          /** By hostID, in descending order. */
67          descId(new Comparator<Host>() {
68              @Override
69              public int compare(Host t1, Host t2) {
70                  return -t1.getHostid().compareTo(t2.getHostid());
71              }
72          }),
73          /** By address, in ascending order. */
74          ascAddress(new Comparator<Host>() {
75              @Override
76              public int compare(Host t1, Host t2) {
77                  return t1.getAddress().compareTo(t2.getAddress());
78              }
79          }),
80          /** By address, in descending order. */
81          descAddress(new Comparator<Host>() {
82              @Override
83              public int compare(Host t1, Host t2) {
84                  return -t1.getAddress().compareTo(t2.getAddress());
85              }
86          });
87  
88          /** The comparator used to sort the list of RestHost objects. */
89          public final Comparator<Host> comparator;
90  
91          Order(Comparator<Host> comparator) {
92              this.comparator = comparator;
93          }
94      }

98      /**
99       * Converts the given {@link Host} object into an {@link ObjectNode}.
100      *
101      * @param host the host to convert
102      * @return     the converted ObjectNode
103      */
104     public static ObjectNode hostToNode(Host host) {
105         ObjectNode node = new ObjectNode(JsonNodeFactory.instance);
106         node.put(HOST_NAME, host.getHostid());
107         node.put(ADDRESS, host.getAddress());
108         node.put(PORT, host.getPort());
109         node.put(PASSWORD, host.getHostkey());
110         node.put(IS_SSL, host.isSSL());
111         node.put(IS_CLIENT, host.isClient());
112         node.put(IS_ADMIN, host.isAdmin());
113         node.put(IS_ACTIVE, host.isActive());
114         node.put(IS_PROXY, host.isProxified());
115 
116         return node;
117     }
118 
119     /**
120      * Converts the given {@link ObjectNode} into a {@link Host} object.
121      *
122      * @param object the ObjectNode to convert
123      * @return       the corresponding Host object
124      * @throws RestErrorException if the given ObjectNode does not represent
125      *                            a Host object
126      * @throws InternalServerErrorException if an unexpected error occurred
127      */
128     public static Host nodeToNewHost(ObjectNode object) {
129         Host emptyHost = new Host(null, null, -1, null, false, false, false,
130                 false, true);
131 
132         return nodeToUpdatedHost(object, emptyHost);
133     }
134 
135     /**
136      * Returns the given {@link Host} object updated with the values defined
137      * in the {@link ObjectNode} parameter. All fields missing in the JSON
138      * object will stay unchanged in the updated host entry.
139      *
140      * @param object  the ObjectNode to convert.
141      * @param oldHost the host entry to update.
142      * @return        the updated host entry
143      * @throws RestErrorException if the given ObjectNode does not represent
144      *                            a Host object
145      * @throws InternalServerErrorException if an unexpected error occurred
146      */
147     public static Host nodeToUpdatedHost(ObjectNode object, Host oldHost) {
148 
149         List<RestError> errors = new ArrayList<RestError>();
150 
151         Iterator<Map.Entry<String, JsonNode>> fields = object.fields();
152         while (fields.hasNext()) {
153             Map.Entry<String, JsonNode> field = fields.next();
154             String name = field.getKey();
155             JsonNode value = field.getValue();
156 
157             if (name.equalsIgnoreCase(HOST_NAME)) {
158                 if (value.isTextual()) {
159                     if (oldHost.getHostid() == null) {
160                         oldHost.setHostid(value.asText());
161                     } else if (!oldHost.getHostid().equals(value.asText())) {
162                         errors.add(FIELD_NOT_ALLOWED(HOST_NAME));
163                     }
164                 } else {
165                     errors.add(ILLEGAL_FIELD_VALUE(name, value.toString()));
166                 }
167             }
168             else if (name.equalsIgnoreCase(ADDRESS)) {
169                 if (value.isTextual()) {
170                     oldHost.setAddress(value.asText());
171                 } else {
172                     errors.add(ILLEGAL_FIELD_VALUE(name, value.toString()));
173                 }
174             }
175             else if (name.equalsIgnoreCase(PORT)) {
176                 if (value.canConvertToInt() && value.asInt() >= 0 && value.asInt() < 65536) {
177                     oldHost.setPort(value.asInt());
178                 } else {
179                     errors.add(ILLEGAL_FIELD_VALUE(name, value.toString()));
180                 }
181             }
182             else if (name.equalsIgnoreCase(PASSWORD)) {
183                 if (value.isTextual()) {
184                     oldHost.setHostkey(encryptPassword(value.asText()));
185                 } else {
186                     errors.add(ILLEGAL_FIELD_VALUE(name, value.toString()));
187                 }
188             }
189             else if (name.equalsIgnoreCase(IS_SSL)) {
190                 if (value.isBoolean()) {
191                     oldHost.setSSL(value.asBoolean());
192                 } else {
193                     errors.add(ILLEGAL_FIELD_VALUE(name, value.toString()));
194                 }
195             }
196             else if (name.equalsIgnoreCase(IS_CLIENT)) {
197                 if (value.isBoolean()) {
198                     oldHost.setClient(value.asBoolean());
199                 } else {
200                     errors.add(ILLEGAL_FIELD_VALUE(name, value.toString()));
201                 }
202             }
203             else if (name.equalsIgnoreCase(IS_ADMIN)) {
204                 if (value.isBoolean()) {
205                     oldHost.setAdmin(value.asBoolean());
206                 } else {
207                     errors.add(ILLEGAL_FIELD_VALUE(name, value.toString()));
208                 }
209             }
210             else if (name.equalsIgnoreCase(IS_ACTIVE)) {
211                 if (value.isBoolean()) {
212                     oldHost.setActive(value.asBoolean());
213                 } else {
214                     errors.add(ILLEGAL_FIELD_VALUE(name, value.toString()));
215                 }
216             }
217             else if (name.equalsIgnoreCase(IS_PROXY)) {
218                 if (value.isBoolean()) {
219                     oldHost.setProxified(value.asBoolean());
220                 } else {
221                     errors.add(ILLEGAL_FIELD_VALUE(name, value.toString()));
222                 }
223             }
224             else {
225                 errors.add(UNKNOWN_FIELD(name));
226             }
227         }
228 
229         errors.addAll(checkRequiredFields(oldHost));
230 
231         if (errors.isEmpty()) {
232             return oldHost;
233         } else {
234             throw new RestErrorException(errors);
235         }
236     }

240     /**
241      * Encrypts the given password String using the server's cryptographic key.
242      *
243      * @param password the password to encrypt
244      * @return         the encrypted password
245      * @throws InternalServerErrorException If an error occurred when encrypting
246      *                                      the password.
247      */
248     private static byte[] encryptPassword(String password) {
249         try {
250             return configuration.getCryptoKey().cryptToHex(password).getBytes();
251         } catch (Exception e) {
252             throw new InternalServerErrorException("Failed to encrypt the host password", e);
253         }
254     }
255 
256     /**
257      * List all missing required fields.
258      * This method returns a list of {@link RestError} representing all
259      * the errors encountered when checking the given host's required  fields.
260      * If all required fields have indeed been initialized, an empty list
261      * is returned.
262      *
263      * @param host the host entry to check
264      * @return     the list of encountered errors
265      */
266     private static List<RestError> checkRequiredFields(Host host) {
267         List<RestError> errors = new ArrayList<RestError>();
268         if (host.getHostid() == null || host.getHostid().isEmpty()) {
269             errors.add(MISSING_FIELD(HOST_NAME));
270         }
271         if (host.getAddress() == null || host.getAddress().isEmpty()) {
272             errors.add(MISSING_FIELD(ADDRESS));
273         }
274         if (host.getPort() == -1) {
275             errors.add(MISSING_FIELD(PORT));
276         }
277         if (host.getHostkey() == null || host.getHostkey().length == 0) {
278             errors.add(MISSING_FIELD(PASSWORD));
279         }
280 
281         return errors;
282     }
