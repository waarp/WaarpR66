68      /** All the possible ways to order a list of transfer objects. */
69      public enum Order {
70          /** By transferId, in ascending order. */
71          ascId(ID_FIELD, true),
72          /** By transferId, in descending order. */
73          descId(ID_FIELD, false),
74          /** By fileName, in ascending order. */
75          ascFile(ORIGINAL_NAME_FIELD, true),
76          /** By fileName, in descending order. */
77          descFile(ORIGINAL_NAME_FIELD, false),
78          /** By starting date, in ascending order. */
79          ascStart(TRANSFER_START_FIELD, true),
80          /** By starting date, in descending order. */
81          descStart(TRANSFER_START_FIELD, false),
82          /** By end date, in ascending order. */
83          ascStop(TRANSFER_STOP_FIELD, true),
84          /** By end date, in descending order. */
85          descStop(TRANSFER_STOP_FIELD, false);
86  
87          /** The name of the database column used for sorting. */
88          public final String column;
89          /** If the order is ascending or descending. */
90          public final boolean ascend;
91  
92          Order(String column, boolean ascend) {
93              this.column = column;
94              this.ascend = ascend;
95          }
96      }

101     /**
102      * Returns an {@link ObjectNode} representing the {@link Transfer} object
103      * given as parameter.
104      *
105      * @param transfer the Transfer object to serialize
106      * @return         the corresponding ObjectNode
107      */
108     public static ObjectNode transferToNode(Transfer transfer) {
109         ObjectNode node = new ObjectNode(JsonNodeFactory.instance);
110         node.put(TRANSFER_ID, transfer.getId());
111         node.put(GLOBAL_STEP, transfer.getGlobalStep().toString());
112         node.put(GLOBAL_LAST_STEP, transfer.getLastGlobalStep().toString());
113         node.put(STEP, transfer.getStep());
114         node.put(RANK, transfer.getRank());
115         node.put(UPDATED_INFO, transfer.getUpdatedInfo().toString());
116         node.put(STEP_STATUS, transfer.getStepStatus().toString());
117         node.put(ERROR_CODE, transfer.getInfoStatus().code);
118         node.put(ERROR_MESSAGE, transfer.getInfoStatus().getMesg());
119         node.put(ORIGINAL_FILENAME, transfer.getOriginalName());
120         node.put(FILENAME, transfer.getFilename());
121         node.put(RULE, transfer.getRule());
122         node.put(BLOCK_SIZE, transfer.getBlockSize());
123         node.put(FILE_INFO, transfer.getFileInfo());
124         node.put(TRANSFER_INFO, transfer.getTransferInfo());
125         node.put(START, new DateTime(transfer.getStart()).toString());
126         node.put(STOP, new DateTime(transfer.getStop()).toString());
127         node.put(REQUESTED, transfer.getRequested());
128         node.put(REQUESTER, transfer.getRequester());
129         node.put(RETRIEVE, transfer.getRetrieveMode());
130 
131         return node;
132     }

134     /**
135      * Initialize a {@link Transfer} object using the values of the given
136      * {@link ObjectNode}.
137      *
138      * @param object the ObjectNode to convert
139      * @return       the new Transfer object
140      * @throws RestErrorException if the given ObjectNode does not represent
141      *                            a Transfer object
142      * @throws InternalServerErrorException if an unexpected error occurred
143      */
144     public static Transfer nodeToNewTransfer(ObjectNode object) {
145         Transfer defaultTransfer = new Transfer(null, null, -1, false, null, null, 65536);
146         defaultTransfer.setRequester(SERVER_NAME);
147         defaultTransfer.setOwnerRequest(SERVER_NAME);
148         defaultTransfer.setBlockSize(65536);
149         defaultTransfer.setTransferInfo("");
150         defaultTransfer.setStart(new Timestamp(DateTime.now().getMillis()));
151         Transfer transfer = parseNode(object, defaultTransfer);
152 
153         ModeTrans mode;
154         RuleDAO ruleDAO = null;
155         try {
156             ruleDAO = DAO_FACTORY.getRuleDAO();
157             Rule rule = ruleDAO.select(transfer.getRule());
158             mode = ModeTrans.fromCode(rule.getMode());
159         } catch (DAOException e) {
160             throw new InternalServerErrorException(e);
161         } finally {
162             if (ruleDAO != null) {
163                 ruleDAO.close();
164             }
165         }
166 
167         transfer.setRetrieveMode(mode == ModeTrans.receive || mode == ModeTrans.receiveMD5);
168         transfer.setTransferMode(mode.code);
169         transfer.setFileInfo("");
170         transfer.setStop(transfer.getStart());
171         transfer.setUpdatedInfo(UpdatedInfo.TOSUBMIT);
172 
173         return transfer;
174     }

179     /**
180      * Tells if the given rule exists in the database.
181      *
182      * @param rule the name of the rule
183      * @return     {@code true} if the rule exists, {@code false} otherwise.
184      */
185     private static boolean ruleExists(String rule) {
186         RuleDAO ruleDAO = null;
187         try {
188             ruleDAO = DAO_FACTORY.getRuleDAO();
189             return ruleDAO.exist(rule);
190         } catch (DAOException e) {
191             throw new InternalServerErrorException(e);
192         } finally {
193             if (ruleDAO != null) {
194                 ruleDAO.close();
195             }
196         }
197     }

219     /**
220      * Tells if the given host is allowed to use given rule.
221      *
222      * @param host the name of the host
223      * @param rule the name of the rule
224      * @return {@code true} if the host is allowed to use the rule, {@code false}
225      *         otherwise
226      */
227     private static boolean canUseRule(String host, String rule) {
228         RuleDAO ruleDAO = null;
229         try {
230             ruleDAO = DAO_FACTORY.getRuleDAO();
231             List<String> hostIds = ruleDAO.select(rule).getHostids();
232             return !hostIds.isEmpty() && !hostIds.contains(host);
233         } catch (DAOException e) {
234             throw new InternalServerErrorException(e);
235         } finally {
236             if (ruleDAO != null) {
237                 ruleDAO.close();
238             }
239         }
240     }

242     /**
243      * Returns a list of {@link RestError} corresponding to all the fields required
244      * to initialize a transfer that are missing from the given {@link Transfer}
245      * object. If no fields are missing, an empty list is returned.
246      *
247      * @param transfer the Transfer object to check.
248      * @return         the list of all missing fields
249      */
250     private static List<RestError> checkRequiredFields(Transfer transfer) {
251         List<RestError> errors = new ArrayList<RestError>();
252         if (transfer.getRule() == null || transfer.getRule().isEmpty()) {
253             errors.add(MISSING_FIELD(RULE));
254         }
255         if (transfer.getOriginalName() == null || transfer.getOriginalName().isEmpty()) {
256             errors.add(MISSING_FIELD(FILENAME));
257         }
258         if (transfer.getRequested() == null || transfer.getRequested().isEmpty()) {
259             errors.add(MISSING_FIELD(REQUESTED));
260         }
261 
262         return errors;
263     }

265     /**
266      * Fills the fields of the given {@link Transfer} object with the values
267      * extracted from the {@link ObjectNode} parameter, and returns the result.
268      *
269      * @param object   the ObjectNode from which the values should be extracted
270      * @param transfer the Transfer object whose fields will be filled
271      * @return         the filled Transfer object
272      * @throws RestErrorException if the given ObjectNode does not represent
273      *                            a Transfer object.
274      */
275     private static Transfer parseNode(ObjectNode object, Transfer transfer) {
276         List<RestError> errors = new ArrayList<RestError>();
277 
278         Iterator<Map.Entry<String, JsonNode>> fields = object.fields();
279         while (fields.hasNext()) {
280             Map.Entry<String, JsonNode> field = fields.next();
281             String name = field.getKey();
282             JsonNode value = field.getValue();
283 
284             if (name.equalsIgnoreCase(RULE)) {
285                 if (value.isTextual()) {
286                     if (ruleExists(value.asText())) {
287                         transfer.setRule(value.asText());
288                     } else {
289                         errors.add(RestErrors.UNKNOWN_RULE(value.asText()));
290                     }
291                 } else {
292                     errors.add(ILLEGAL_FIELD_VALUE(name, value.toString()));
293                 }
294             }
295             else if (name.equalsIgnoreCase(FILENAME)) {
296                 if (value.isTextual()) {
297                     transfer.setOriginalName(value.asText());
298                     transfer.setFilename(value.asText());
299                 } else {
300                     errors.add(ILLEGAL_FIELD_VALUE(name, value.toString()));
301                 }
302             }
303             else if (name.equalsIgnoreCase(REQUESTED)) {
304                 if (value.isTextual()) {
305                     if (hostExists(value.asText())) {
306                         transfer.setRequested(value.asText());
307                     } else {
308                         errors.add(RestErrors.UNKNOWN_HOST(value.asText()));
309                     }
310                 } else {
311                     errors.add(ILLEGAL_FIELD_VALUE(name, value.toString()));
312                 }
313             }
314             else if (name.equalsIgnoreCase(BLOCK_SIZE)) {
315                 if (value.canConvertToInt() && value.asInt() > 0) {
316                     transfer.setBlockSize(value.asInt());
317                 } else {
318                     errors.add(ILLEGAL_FIELD_VALUE(name, value.toString()));
319                 }
320             }
321             else if (name.equalsIgnoreCase(FILE_INFO)) {
322                 if (value.isTextual()) {
323                     transfer.setFileInfo(value.asText());
324                 } else {
325                     errors.add(ILLEGAL_FIELD_VALUE(name, value.toString()));
326                 }
327             }
328             else if (name.equalsIgnoreCase(START)) {
329                 if (value.isTextual()) {
330                     try {
331                         DateTime start = DateTime.parse(value.asText());
332                         if (start.isBeforeNow()) {
333                             errors.add(ILLEGAL_FIELD_VALUE(name, value.asText()));
334                         } else {
335                             transfer.setStart(new Timestamp(start.getMillis()));
336                         }
337                     } catch (IllegalArgumentException e) {
338                         errors.add(ILLEGAL_FIELD_VALUE(name, value.asText()));
339                     }
340                 } else {
341                     errors.add(ILLEGAL_FIELD_VALUE(name, value.toString()));
342                 }
343             }
344         }
345 
346         //check that both hosts are allowed to use the transfer rule
347         String rule = transfer.getRule();
348         String requested = transfer.getRequested();
349         String requester = transfer.getRequester();
350         if (rule != null && !requested.isEmpty() && canUseRule(requested, rule)) {
351             errors.add(RULE_NOT_ALLOWED(requested, rule));
352         }
353         if (rule != null && !requester.isEmpty() && canUseRule(requester, rule)) {
354             errors.add(RULE_NOT_ALLOWED(requester, rule));
355         }
356 
357         errors.addAll(checkRequiredFields(transfer));
358 
359         if (errors.isEmpty()) {
360             return transfer;
361         } else {
362             throw new RestErrorException(errors);
363         }
364     }
