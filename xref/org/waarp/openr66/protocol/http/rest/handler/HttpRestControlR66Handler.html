59  /**
60   * Transfer Http REST interface: http://host/control?... +
61   * InformationJsonPacket (should be on Transfer only) RestartTransferJsonPacket StopOrCancelJsonPacket TransferRequestJsonPacket
62   * as GET PUT PUT POST
63   * 
64   * @author "Frederic Bregier"
65   *
66   */
67  public class HttpRestControlR66Handler extends HttpRestAbstractR66Handler {
68  
69      public static final String BASEURI = "control";
70      /**
71       * Internal Logger
72       */
73      private static final WaarpLogger logger = WaarpLoggerFactory
74              .getLogger(HttpRestControlR66Handler.class);
75  
76      public HttpRestControlR66Handler(RestConfiguration config, METHOD... methods) {
77          super(BASEURI, config, METHOD.OPTIONS);
78          setIntersectionMethods(methods, METHOD.GET, METHOD.PUT, METHOD.POST);
79      }
80  
81      @Override
82      public void endParsingRequest(HttpRestHandler handler, RestArgument arguments, RestArgument result, Object body)
83              throws HttpIncorrectRequestException, HttpInvalidAuthenticationException {
84          logger.debug("debug: {} ### {}", arguments, result);
85          if (body != null) {
86              logger.debug("Obj: {}", body);
87          }
88          handler.setWillClose(false);
89          ServerActions serverHandler = ((HttpRestR66Handler) handler).getServerHandler();
90          R66Session session = serverHandler.getSession();
91          // now action according to body
92          JsonPacket json = (JsonPacket) body;
93          if (json == null) {
94              result.setDetail("not enough information");
95              setError(handler, result, HttpResponseStatus.BAD_REQUEST);
96              return;
97          }
98          result.getAnswer().put(AbstractDbData.JSON_MODEL, RESTHANDLERS.Control.name());
99          METHOD method = arguments.getMethod();
100         try {
101             if (json instanceof InformationJsonPacket && method == METHOD.GET) {//
102                 InformationJsonPacket node = (InformationJsonPacket) json;
103                 if (node.isIdRequest()) {
104                     result.setCommand(ACTIONS_TYPE.GetTransferInformation.name());
105                     ValidPacket validPacket = null;
106                     if (node.isIdRequest()) {
107                         validPacket = serverHandler.informationRequest(node.getId(), node.isTo(), node.getRulename(),
108                                 true);
109                     } else {
110                         validPacket = serverHandler.informationFile(node.getRequest(), node.getRulename(),
111                                 node.getFilename(), true);
112                     }
113                     if (validPacket != null) {
114                         ObjectNode resp = JsonHandler.getFromString(validPacket.getSheader());
115                         handler.setStatus(HttpResponseStatus.OK);
116                         result.setResult(HttpResponseStatus.OK);
117                         result.getResults().add(resp);
118                     } else {
119                         result.setDetail("Error during information request");
120                         setError(handler, result, HttpResponseStatus.NOT_ACCEPTABLE);
121                     }
122                 } else {
123                     result.setCommand(ACTIONS_TYPE.GetInformation.name());
124                     result.setDetail("Error: FileInformation is not applicable with URI " + BASEURI);
125                     setError(handler, result, HttpResponseStatus.NOT_ACCEPTABLE);
126                 }
127             } else if (json instanceof RestartTransferJsonPacket && method == METHOD.PUT) {//
128                 result.setCommand(ACTIONS_TYPE.RestartTransfer.name());
129                 RestartTransferJsonPacket node = (RestartTransferJsonPacket) json;
130                 R66Result r66result = serverHandler.requestRestart(node.getRequested(), node.getRequester(),
131                         node.getSpecialid(), node.getRestarttime());
132                 if (serverHandler.isCodeValid(r66result.getCode())) {
133                     result.setDetail("Restart Transfer done");
134                     setOk(handler, result, node, HttpResponseStatus.OK);
135                 } else {
136                     result.setDetail("Restart Transfer in error");
137                     setError(handler, result, node, HttpResponseStatus.NOT_ACCEPTABLE);
138                 }
139             } else if (json instanceof StopOrCancelJsonPacket && method == METHOD.PUT) {//
140                 result.setCommand(ACTIONS_TYPE.StopOrCancelTransfer.name());
141                 StopOrCancelJsonPacket node = (StopOrCancelJsonPacket) json;
142                 R66Result resulttest;
143                 if (node.getRequested() == null || node.getRequester() == null
144                         || node.getSpecialid() == DbConstant.ILLEGALVALUE) {
145                     ErrorCode code = ErrorCode.CommandNotFound;
146                     resulttest = new R66Result(session, true,
147                             code, session.getRunner());
148                     result.setDetail("Not enough argument passed to identify a transfer");
149                     setError(handler, result, node, HttpResponseStatus.NOT_FOUND);
150                 } else {
151                     String reqd = node.getRequested();
152                     String reqr = node.getRequester();
153                     long id = node.getSpecialid();
154                     resulttest = serverHandler.stopOrCancel(node.getRequestUserPacket(), reqd, reqr, id);
155                     result.setDetail(resulttest.getCode().mesg);
156                     setOk(handler, result, node, HttpResponseStatus.OK);
157                 }
158             } else if (json instanceof TransferRequestJsonPacket && method == METHOD.POST) {
159                 result.setCommand(ACTIONS_TYPE.CreateTransfer.name());
160                 TransferRequestJsonPacket node = (TransferRequestJsonPacket) json;
161                 R66Result r66result = serverHandler.transferRequest(node);
162                 if (serverHandler.isCodeValid(r66result.getCode())) {
163                     result.setDetail("New Transfer registered");
164                     setOk(handler, result, node, HttpResponseStatus.OK);
165                 } else {
166                     result.setDetail("New Transfer cannot be registered");
167                     setError(handler, result, HttpResponseStatus.NOT_ACCEPTABLE);
168                 }
169             } else {
170                 logger.info("Validation is ignored: " + json);
171                 result.setDetail("Unknown command");
172                 setError(handler, result, json, HttpResponseStatus.PRECONDITION_FAILED);
173             }
174         } catch (OpenR66ProtocolNotAuthenticatedException e) {
175             throw new HttpInvalidAuthenticationException(e);
176         } catch (OpenR66ProtocolPacketException e) {
177             throw new HttpIncorrectRequestException(e);
178         } catch (OpenR66ProtocolNoDataException e) {
179             throw new HttpIncorrectRequestException(e);
180         }
181     }

183     protected ArrayNode getDetailedAllow() {
184         ArrayNode node = JsonHandler.createArrayNode();
185 
186         if (this.methods.contains(METHOD.GET)) {
187             InformationJsonPacket node3 = new InformationJsonPacket(Long.MIN_VALUE, false, "remoteHost");
188             node3.setComment("Information on Transfer request (GET)");
189             ArrayNode node1 = JsonHandler.createArrayNode();
190             ObjectNode node1b = JsonHandler.createObjectNode();
191             node1b.put(DbTaskRunner.JSON_MODEL, DbTaskRunner.class.getSimpleName());
192             DbValue[] values = DbTaskRunner.getAllType();
193             for (DbValue dbValue : values) {
194                 node1b.put(dbValue.getColumn(), dbValue.getType());
195             }
196             node1.add(node1b);
197             ObjectNode node2;
198             try {
199                 node2 = RestArgument.fillDetailedAllow(METHOD.GET, this.path,
200                         ACTIONS_TYPE.GetTransferInformation.name(), node3.createObjectNode(), node1);
201                 node.add(node2);
202             } catch (OpenR66ProtocolPacketException e1) {
203             }
204         }
205         if (this.methods.contains(METHOD.PUT)) {
206             RestartTransferJsonPacket node4 = new RestartTransferJsonPacket();
207             node4.setRequestUserPacket();
208             node4.setComment("Restart Transfer request (PUT)");
209             node4.setRequested("Requested host");
210             node4.setRequester("Requester host");
211             node4.setRestarttime(new Date());
212             ArrayNode node1 = JsonHandler.createArrayNode();
213             try {
214                 node1.add(node4.createObjectNode());
215                 ObjectNode node2 = RestArgument.fillDetailedAllow(METHOD.PUT, this.path,
216                         ACTIONS_TYPE.RestartTransfer.name(), node4.createObjectNode(), node1);
217                 node.add(node2);
218             } catch (OpenR66ProtocolPacketException e1) {
219             }
220             StopOrCancelJsonPacket node5 = new StopOrCancelJsonPacket();
221             node5.setRequestUserPacket();
222             node5.setComment("Stop Or Cancel request (PUT)");
223             node5.setRequested("Requested host");
224             node5.setRequester("Requester host");
225             node1 = JsonHandler.createArrayNode();
226             try {
227                 node1.add(node5.createObjectNode());
228                 ObjectNode node2 = RestArgument.fillDetailedAllow(METHOD.PUT, this.path,
229                         ACTIONS_TYPE.StopOrCancelTransfer.name(), node5.createObjectNode(), node1);
230                 node.add(node2);
231             } catch (OpenR66ProtocolPacketException e1) {
232             }
233         }
234         if (this.methods.contains(METHOD.POST)) {
235             TransferRequestJsonPacket node6 = new TransferRequestJsonPacket();
236             node6.setRequestUserPacket();
237             node6.setComment("Transfer Request (POST)");
238             node6.setFilename("Filename");
239             node6.setFileInformation("File information");
240             node6.setRequested("Requested host");
241             node6.setRulename("Rulename");
242             node6.setStart(new Date());
243             ArrayNode node1 = JsonHandler.createArrayNode();
244             try {
245                 node1.add(node6.createObjectNode());
246                 ObjectNode node2 = RestArgument.fillDetailedAllow(METHOD.POST, this.path,
247                         ACTIONS_TYPE.CreateTransfer.name(), node6.createObjectNode(), node1);
248                 node.add(node2);
249             } catch (OpenR66ProtocolPacketException e1) {
250             }
251         }
252 
253         ObjectNode node2 = RestArgument.fillDetailedAllow(METHOD.OPTIONS, this.path, COMMAND_TYPE.OPTIONS.name(), null,
254                 null);
255         node.add(node2);
256 
257         return node;
258     }
