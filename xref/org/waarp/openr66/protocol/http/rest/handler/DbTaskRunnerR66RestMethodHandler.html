60      public static enum FILTER_ARGS {
61          LIMIT("number"),
62          ORDERBYID("boolean"),
63          STARTID("transfer id"),
64          STOPID("transfer id"),
65          IDRULE("rule name"),
66          PARTNER("partner (requester or requested) name"),
67          PENDING("boolean"),
68          INTRANSFER("boolean"),
69          INERROR("boolean"),
70          DONE("boolean"),
71          ALLSTATUS("boolean"),
72          STARTTRANS("Date in ISO 8601 format or ms"),
73          STOPTRANS("Date in ISO 8601 format or ms");
74  
75          public String type;
76  
77          FILTER_ARGS(String type) {
78              this.type = type;
79          }
80      }

90      protected DbTaskRunner getItem(HttpRestHandler handler, RestArgument arguments,
91              RestArgument result, Object body) throws HttpIncorrectRequestException,
92              HttpInvalidAuthenticationException, HttpNotFoundRequestException {
93          ObjectNode arg = arguments.getUriArgs().deepCopy();
94          arg.setAll(arguments.getBody());
95          try {
96              JsonNode node = RestArgument.getId(arg);
97              long id;
98              if (node.isMissingNode()) {
99                  // shall not be but continue however
100                 id = arg.path(DbTaskRunner.Columns.SPECIALID.name()).asLong();
101             } else {
102                 id = node.asLong();
103             }
104             return new DbTaskRunner(id,
105                     arg.path(DbTaskRunner.Columns.REQUESTER.name()).asText(),
106                     arg.path(DbTaskRunner.Columns.REQUESTED.name()).asText(),
107                     arg.path(DbTaskRunner.Columns.OWNERREQ.name()).asText());
108         } catch (WaarpDatabaseException e) {
109             throw new HttpNotFoundRequestException("Issue while reading from database " + arg, e);
110         }
111     }

126     @Override
127     protected DbPreparedStatement getPreparedStatement(HttpRestHandler handler,
128             RestArgument arguments, RestArgument result, Object body)
129             throws HttpIncorrectRequestException, HttpInvalidAuthenticationException {
130         ObjectNode arg = arguments.getUriArgs().deepCopy();
131         arg.setAll(arguments.getBody());
132         int limit = arg.path(FILTER_ARGS.LIMIT.name()).asInt(0);
133         boolean orderBySpecialId = arg.path(FILTER_ARGS.ORDERBYID.name()).asBoolean(false);
134         JsonNode node = arg.path(FILTER_ARGS.STARTID.name());
135         String startid = null;
136         if (!node.isMissingNode()) {
137             startid = node.asText();
138         }
139         if (startid == null || startid.isEmpty()) {
140             startid = null;
141         }
142         node = arg.path(FILTER_ARGS.STOPID.name());
143         String stopid = null;
144         if (!node.isMissingNode()) {
145             stopid = node.asText();
146         }
147         if (stopid == null || stopid.isEmpty()) {
148             stopid = null;
149         }
150         String rule = arg.path(FILTER_ARGS.IDRULE.name()).asText();
151         if (rule == null || rule.isEmpty()) {
152             rule = null;
153         }
154         String req = arg.path(FILTER_ARGS.PARTNER.name()).asText();
155         if (req == null || req.isEmpty()) {
156             req = null;
157         }
158         String owner = arg.path(DbTaskRunner.Columns.OWNERREQ.name()).asText();
159         if (owner == null || owner.isEmpty()) {
160             owner = null;
161         }
162         boolean pending = arg.path(FILTER_ARGS.PENDING.name()).asBoolean(false);
163         boolean transfer = arg.path(FILTER_ARGS.INTRANSFER.name()).asBoolean(false);
164         boolean error = arg.path(FILTER_ARGS.INERROR.name()).asBoolean(false);
165         boolean done = arg.path(FILTER_ARGS.DONE.name()).asBoolean(false);
166         boolean all = arg.path(FILTER_ARGS.ALLSTATUS.name()).asBoolean(false);
167         Timestamp start = null;
168         node = arg.path(FILTER_ARGS.STARTTRANS.name());
169         if (!node.isMissingNode()) {
170             long val = node.asLong();
171             if (val == 0) {
172                 DateTime received = DateTime.parse(node.asText());
173                 val = received.getMillis();
174             }
175             start = new Timestamp(val);
176         }
177         Timestamp stop = null;
178         node = arg.path(FILTER_ARGS.STOPTRANS.name());
179         if (!node.isMissingNode()) {
180             long val = node.asLong();
181             if (val == 0) {
182                 DateTime received = DateTime.parse(node.asText());
183                 val = received.getMillis();
184             }
185             stop = new Timestamp(val);
186         }
187         try {
188             return DbTaskRunner.getFilterPrepareStatement(handler.getDbSession(),
189                     limit, orderBySpecialId, startid, stopid, start, stop, rule, req,
190                     pending, transfer, error, done, all, owner);
191         } catch (WaarpDatabaseNoConnectionException e) {
192             throw new HttpIncorrectRequestException("Issue while reading from database", e);
193         } catch (WaarpDatabaseSqlException e) {
194             throw new HttpIncorrectRequestException("Issue while reading from database", e);
195         }
196     }

198     @Override
199     protected DbTaskRunner getItemPreparedStatement(DbPreparedStatement statement)
200             throws HttpIncorrectRequestException, HttpNotFoundRequestException {
201         try {
202             return DbTaskRunner.getFromStatementNoDbRule(statement);
203         } catch (WaarpDatabaseNoConnectionException e) {
204             throw new HttpIncorrectRequestException("Issue while selecting from database", e);
205         } catch (WaarpDatabaseSqlException e) {
206             throw new HttpNotFoundRequestException("Issue while selecting from database", e);
207         }
208     }

210     @Override
211     protected ArrayNode getDetailedAllow() {
212         ArrayNode node = JsonHandler.createArrayNode();
213 
214         ObjectNode node1 = JsonHandler.createObjectNode();
215         node1.put(DbTaskRunner.JSON_MODEL, DbTaskRunner.class.getSimpleName());
216         DbValue[] values = DbTaskRunner.getAllType();
217         for (DbValue dbValue : values) {
218             node1.put(dbValue.getColumn(), dbValue.getType());
219         }
220 
221         ObjectNode node2;
222         ObjectNode node3 = JsonHandler.createObjectNode();
223         if (this.methods.contains(METHOD.GET)) {
224             node3.put(DbTaskRunner.Columns.SPECIALID.name(), "Special Id as LONG in URI as " + this.path + "/id");
225             node3.put(DbTaskRunner.Columns.REQUESTER.name(), "Partner as requester as VARCHAR");
226             node3.put(DbTaskRunner.Columns.REQUESTED.name(), "Partner as requested as VARCHAR");
227             node3.put(DbTaskRunner.Columns.OWNERREQ.name(), "Owner of this request (optional) as VARCHAR");
228             node2 = RestArgument.fillDetailedAllow(METHOD.GET, this.path + "/id", COMMAND_TYPE.GET.name(),
229                     node3, node1);
230             node.add(node2);
231 
232             node3 = JsonHandler.createObjectNode();
233             for (FILTER_ARGS arg : FILTER_ARGS.values()) {
234                 node3.put(arg.name(), arg.type);
235             }
236             node3.put(DbTaskRunner.Columns.OWNERREQ.name(), "Owner of this request (optional) as VARCHAR");
237             node2 = RestArgument.fillDetailedAllow(METHOD.GET, this.path, COMMAND_TYPE.MULTIGET.name(),
238                     node3, JsonHandler.createArrayNode().add(node1));
239             node.add(node2);
240         }
241         if (this.methods.contains(METHOD.PUT)) {
242             node3 = JsonHandler.createObjectNode();
243             node3.put(DbTaskRunner.Columns.SPECIALID.name(), "Special Id as LONG in URI as " + this.path + "/id");
244             node3.put(DbTaskRunner.Columns.REQUESTER.name(), "Partner as requester as VARCHAR");
245             node3.put(DbTaskRunner.Columns.REQUESTED.name(), "Partner as requested as VARCHAR");
246             node3.put(DbTaskRunner.Columns.OWNERREQ.name(), "Owner of this request (optional) as VARCHAR");
247             for (DbValue dbValue : values) {
248                 if (dbValue.getColumn().equalsIgnoreCase(DbTaskRunner.Columns.IDRULE.name())) {
249                     continue;
250                 }
251                 node3.put(dbValue.getColumn(), dbValue.getType());
252             }
253             node2 = RestArgument.fillDetailedAllow(METHOD.PUT, this.path + "/id", COMMAND_TYPE.UPDATE.name(),
254                     node3, node1);
255             node.add(node2);
256         }
257         if (this.methods.contains(METHOD.DELETE)) {
258             node3 = JsonHandler.createObjectNode();
259             node3.put(DbTaskRunner.Columns.SPECIALID.name(), "Special Id as LONG in URI as " + this.path + "/id");
260             node3.put(DbTaskRunner.Columns.REQUESTER.name(), "Partner as requester as VARCHAR");
261             node3.put(DbTaskRunner.Columns.REQUESTED.name(), "Partner as requested as VARCHAR");
262             node3.put(DbTaskRunner.Columns.OWNERREQ.name(), "Owner of this request (optional) as VARCHAR");
263             node2 = RestArgument.fillDetailedAllow(METHOD.DELETE, this.path + "/id", COMMAND_TYPE.DELETE.name(),
264                     node3, node1);
265             node.add(node2);
266         }
267         if (this.methods.contains(METHOD.POST)) {
268             node3 = JsonHandler.createObjectNode();
269             for (DbValue dbValue : values) {
270                 node3.put(dbValue.getColumn(), dbValue.getType());
271             }
272             node2 = RestArgument.fillDetailedAllow(METHOD.POST, this.path, COMMAND_TYPE.CREATE.name(),
273                     node3, node1);
274             node.add(node2);
275         }
276         node2 = RestArgument.fillDetailedAllow(METHOD.OPTIONS, this.path, COMMAND_TYPE.OPTIONS.name(), null, null);
277         node.add(node2);
278 
279         return node;
280     }

287     @Override
288     protected void checkAuthorization(HttpRestHandler handler, RestArgument arguments,
289             RestArgument result, METHOD method) throws HttpForbiddenRequestException {
290         HttpRestR66Handler r66handler = (HttpRestR66Handler) handler;
291         R66Session session = r66handler.getServerHandler().getSession();
292         if (!session.getAuth().isValidRole(ROLE.SYSTEM)) {
293             throw new HttpForbiddenRequestException("Partner must have System role");
294         }
295     }
