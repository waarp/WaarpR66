78      public static enum RESTHANDLERS {
79          DbHostAuth(DbHostAuthR66RestMethodHandler.BASEURI, org.waarp.openr66.database.data.DbHostAuth.class),
80          DbRule(DbRuleR66RestMethodHandler.BASEURI, org.waarp.openr66.database.data.DbRule.class),
81          DbTaskRunner(DbTaskRunnerR66RestMethodHandler.BASEURI, org.waarp.openr66.database.data.DbTaskRunner.class),
82          DbHostConfiguration(DbHostConfigurationR66RestMethodHandler.BASEURI,
83                  org.waarp.openr66.database.data.DbHostConfiguration.class),
84          DbConfiguration(DbConfigurationR66RestMethodHandler.BASEURI,
85                  org.waarp.openr66.database.data.DbConfiguration.class),
86          Bandwidth(HttpRestBandwidthR66Handler.BASEURI, null),
87          Business(HttpRestBusinessR66Handler.BASEURI, null),
88          Config(HttpRestConfigR66Handler.BASEURI, null),
89          Information(HttpRestInformationR66Handler.BASEURI, null),
90          Log(HttpRestLogR66Handler.BASEURI, null),
91          Server(HttpRestServerR66Handler.BASEURI, null),
92          Control(HttpRestControlR66Handler.BASEURI, null);
93  
94          public String uri;
95          @SuppressWarnings("rawtypes")
96          public Class clasz;
97  
98          @SuppressWarnings("rawtypes")
99          RESTHANDLERS(String uri, Class clasz) {
100             this.uri = uri;
101             this.clasz = clasz;
102         }

104         public static RESTHANDLERS getRESTHANDLER(String baseUri) {
105             for (RESTHANDLERS resthandler : RESTHANDLERS.values()) {
106                 if (resthandler.uri.equals(baseUri)) {
107                     return resthandler;
108                 }
109             }
110             return null;
111         }

117     public static void defaultHandlers() {
118         synchronized (defaultConfiguration) {
119             if (defaultConfiguration.restHashMap.isEmpty()) {
120                 defaultConfiguration.REST_AUTHENTICATED = true;
121                 defaultConfiguration.RESTHANDLERS_CRUD = new byte[RESTHANDLERS.values().length];
122                 for (int i = 0; i < defaultConfiguration.RESTHANDLERS_CRUD.length; i++) {
123                     defaultConfiguration.RESTHANDLERS_CRUD[i] = 0x0F;

159     protected static METHOD[] getMethods(byte check) {
160         List<METHOD> methods = new ArrayList<METHOD>();
161         if (RestConfiguration.CRUD.CREATE.isValid(check)) {
162             methods.add(METHOD.POST);
163         }
164         if (RestConfiguration.CRUD.READ.isValid(check)) {
165             methods.add(METHOD.GET);
166         }
167         if (RestConfiguration.CRUD.UPDATE.isValid(check)) {
168             methods.add(METHOD.PUT);
169         }
170         if (RestConfiguration.CRUD.DELETE.isValid(check)) {
171             methods.add(METHOD.DELETE);
172         }
173         return methods.toArray(new METHOD[0]);
174     }

258     @Override
259     protected void checkConnection(ChannelHandlerContext ctx) throws HttpInvalidAuthenticationException {
260         logger.debug("Request: {} ### {}", arguments, response);
261         String user = null;
262         String key = null;
263         if (restConfiguration.REST_AUTHENTICATED) {
264             user = arguments.getXAuthUser();
265             if (user == null || user.isEmpty()) {
266                 status = HttpResponseStatus.UNAUTHORIZED;
267                 throw new HttpInvalidAuthenticationException("Empty Authentication");
268             }
269             DbHostAuth host;
270             try {
271                 host = new DbHostAuth(user);
272                 key = new String(host.getHostkey(), WaarpStringUtils.UTF8);
273             } catch (WaarpDatabaseException e) {
274                 // might be global Admin
275                 if (user.equals(Configuration.configuration.getADMINNAME())) {
276                     key = new String(Configuration.configuration.getSERVERADMINKEY(), WaarpStringUtils.UTF8);
277                 }
278             }
279             if (key == null || key.isEmpty()) {
280                 status = HttpResponseStatus.UNAUTHORIZED;
281                 throw new HttpInvalidAuthenticationException("Wrong Authentication");
282             }
283             if (restConfiguration.REST_SIGNATURE) {
284                 arguments.checkBaseAuthent(restConfiguration.hmacSha256, key, restConfiguration.REST_TIME_LIMIT);
285             } else {
286                 arguments.checkTime(restConfiguration.REST_TIME_LIMIT);
287             }
288         } else {
289             // User set only for right access, not for signature check
290             user = Configuration.configuration.getADMINNAME();
291             if (restConfiguration.REST_SIGNATURE) {
292                 arguments.checkBaseAuthent(restConfiguration.hmacSha256, null, restConfiguration.REST_TIME_LIMIT);
293             } else {
294                 arguments.checkTime(restConfiguration.REST_TIME_LIMIT);
295             }
296         }
297         getServerHandler().newSession();
298         R66Session session = getServerHandler().getSession();
299         if (!restConfiguration.REST_AUTHENTICATED) {
300             // Default is Admin
301             session.getAuth().specialNoSessionAuth(true, Configuration.configuration.getHOST_SSLID());
302         } else {
303             // we have one DbSession per connection, only after authentication
304             DbSession temp = getDbSessionFromUser().get(user);
305             if (temp == null) {
306                 try {
307                     temp = new DbSession(DbConstant.admin, false);
308                     getDbSessionFromUser().put(user, temp);
309                 } catch (WaarpDatabaseNoConnectionException e) {
310                 }
311             }
312             if (temp != null) {
313                 temp.useConnection();
314                 this.dbSession = temp;
315             }
316             try {
317                 session.getAuth().connectionHttps(getDbSession(), user,
318                         FilesystemBasedDigest.passwdCrypt(key.getBytes(WaarpStringUtils.UTF8)));
319             } catch (Reply530Exception e) {
320                 status = HttpResponseStatus.UNAUTHORIZED;
321                 throw new HttpInvalidAuthenticationException("Wrong Authentication", e);
322             } catch (Reply421Exception e) {
323                 status = HttpResponseStatus.SERVICE_UNAVAILABLE;
324                 throw new HttpInvalidAuthenticationException("Service unavailable", e);
325             }
326         }
327         arguments.setXAuthRole(session.getAuth().getRole());
328         arguments.methodFromUri();
329         arguments.methodFromHeader();
330     }

347     /**
348      * Initialize the REST service (server side) for one restConfiguration
349      * 
350      * @param restConfiguration
351      */
352     public static void initializeService(RestConfiguration restConfiguration) {
353         instantiateHandlers(restConfiguration);
354         if (group == null) {
355             group = Configuration.configuration.getHttpChannelGroup();
356         }
357         // Configure the server.
358         ServerBootstrap httpBootstrap = new ServerBootstrap();
359         WaarpNettyUtil.setServerBootstrap(httpBootstrap, Configuration.configuration.getHttpBossGroup(),
360                 Configuration.configuration.getHttpWorkerGroup(), (int) Configuration.configuration.getTIMEOUTCON());
361         // Set up the event pipeline factory.
362         if (restConfiguration.REST_SSL) {
363             httpBootstrap.childHandler(new HttpRestR66Initializer(false, Configuration.getWaarpSslContextFactory(),
364                     restConfiguration));
365         } else {
366             httpBootstrap.childHandler(new HttpRestR66Initializer(false, null, restConfiguration));
367         }
368         // Bind and start to accept incoming connections.
369         ChannelFuture future;
370         if (restConfiguration != null && restConfiguration.REST_ADDRESS != null
371                 && !restConfiguration.REST_ADDRESS.isEmpty()) {
372             future = httpBootstrap.bind(new InetSocketAddress(restConfiguration.REST_ADDRESS,
373                     restConfiguration.REST_PORT));
374         } else {
375             future = httpBootstrap.bind(new InetSocketAddress(restConfiguration.REST_PORT));
376         }
377         try {
378             future.await();
379         } catch (InterruptedException e) {
380         }
381         if (future.isSuccess()) {
382             group.add(future.channel());
383         }
384     }
