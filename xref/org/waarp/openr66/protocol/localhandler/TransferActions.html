93      /**
94       * Finalize a request initialization in error
95       *
96       * @param channel
97       * @param code
98       * @param runner
99       * @param e1
100      * @param packet
101      * @throws OpenR66ProtocolPacketException
102      */
103     private final void endInitRequestInError(Channel channel, ErrorCode code, DbTaskRunner runner,
104             OpenR66Exception e1, RequestPacket packet) throws OpenR66ProtocolPacketException {
105         logger.error("TaskRunner initialisation in error: " + code.getMesg() + " " + session
106                 + " {} runner {}",
107                 e1 != null ? e1.getMessage() : "no exception",
108                 (runner != null ? runner.toShortString() : "no runner"));
109         logger.debug("DEBUG Full stack", e1);
110         localChannelReference.invalidateRequest(new R66Result(
111                 e1, session, true, code, null));
112 
113         if (packet.isToValidate()) {
114             // / answer with a wrong request since runner is not set on remote host
115             if (runner != null) {
116                 if (runner.isSender()) {
117                     // In case Wildcard was used
118                     logger.debug("New FILENAME: {}", runner.getOriginalFilename());
119                     packet.setFilename(runner.getOriginalFilename());
120                     logger.debug("Rank set: " + runner.getRank());
121                     packet.setRank(runner.getRank());
122                 } else {
123                     logger.debug("Rank set: " + runner.getRank());
124                     packet.setRank(runner.getRank());
125                 }
126             }
127             packet.validate();
128             packet.setCode(code.code);
129             session.newState(ERROR);
130             ChannelUtils.writeAbstractLocalPacket(localChannelReference, packet, true);
131         } else {
132             session.newState(ERROR);
133             ErrorPacket error = new ErrorPacket(
134                     "TaskRunner initialisation in error: " + e1
135                             .getMessage() + " for " + packet.toString() + " since " + code.getMesg(),
136                     code.getCode(), ErrorPacket.FORWARDCLOSECODE);
137             ChannelUtils.writeAbstractLocalPacket(localChannelReference, error, true);
138         }
139         session.setStatus(47);
140         ChannelCloseTimer.closeFutureChannel(channel);
141     }

143     /**
144      * Receive a request of Transfer
145      *
146      * @param channel
147      * @param packet
148      * @throws OpenR66ProtocolNoDataException
149      * @throws OpenR66ProtocolPacketException
150      * @throws OpenR66ProtocolBusinessException
151      * @throws OpenR66ProtocolSystemException
152      * @throws OpenR66RunnerErrorException
153      */
154     public void request(LocalChannel channel, RequestPacket packet)
155             throws OpenR66ProtocolNoDataException, OpenR66ProtocolPacketException,
156             OpenR66RunnerErrorException, OpenR66ProtocolSystemException,
157             OpenR66ProtocolBusinessException {
158         session.setStatus(99);
159         if (!session.isAuthenticated()) {
160             session.setStatus(48);
161             throw new OpenR66ProtocolNotAuthenticatedException(
162                     Messages.getString("LocalServerHandler.3")); //$NON-NLS-1$
163         }
164         if (packet.isToValidate()) {
165             session.newState(REQUESTR);
166         }
167         // XXX validLimit only on requested side
168         if (packet.isToValidate()) {
169             if (Configuration.configuration.isShutdown()) {
170                 logger.warn(Messages.getString("LocalServerHandler.7") //$NON-NLS-1$
171                         + packet.getRulename() + " from " + session.getAuth().toString());
172                 session.setStatus(100);
173                 endInitRequestInError(channel,
174                         ErrorCode.ServerOverloaded, null,
175                         new OpenR66ProtocolNotYetConnectionException(
176                                 "All new Request blocked"), packet);
177                 session.setStatus(100);
178                 return;
179             }
180             if (Configuration.configuration.getConstraintLimitHandler().checkConstraints()) {
181                 if (Configuration.configuration.getR66Mib() != null) {
182                     Configuration.configuration.getR66Mib().
183                             notifyOverloaded("Rule: " + packet.getRulename() + " from "
184                                     + session.getAuth().toString(),
185                                     Configuration.configuration.getConstraintLimitHandler().lastAlert);
186                 }
187                 logger.warn(Messages.getString("LocalServerHandler.8") //$NON-NLS-1$
188                         + packet.getRulename()
189                         + " while "
190                         + Configuration.configuration.getConstraintLimitHandler().lastAlert +
191                         " from " + session.getAuth().toString());
192                 session.setStatus(100);
193                 endInitRequestInError(channel,
194                         ErrorCode.ServerOverloaded, null,
195                         new OpenR66ProtocolNotYetConnectionException(
196                                 "Limit exceeded " + Configuration.configuration.getConstraintLimitHandler().lastAlert),
197                         packet);
198                 session.setStatus(100);
199                 return;
200             }
201         } else if (packet.getCode() == ErrorCode.ServerOverloaded.code) {
202             // XXX unvalid limit on requested host received
203             logger.info("TaskRunner initialisation in error: " + ErrorCode.ServerOverloaded.getMesg());
204             localChannelReference.invalidateRequest(new R66Result(
205                     null, session, true, ErrorCode.ServerOverloaded, null));
206             session.setStatus(101);
207             ChannelCloseTimer.closeFutureChannel(channel);
208             return;
209         }
210         DbRule rule;
211         try {
212             rule = new DbRule(packet.getRulename());
213         } catch (WaarpDatabaseException e) {
214             logger.info("Rule is unknown: " + packet.getRulename() + " {}", e.getMessage());
215             session.setStatus(49);
216             endInitRequestInError(channel,
217                     ErrorCode.QueryRemotelyUnknown, null,
218                     new OpenR66ProtocolBusinessException(
219                             Messages.getString("LocalServerHandler.9") + //$NON-NLS-1$
220                                     packet.getRulename()), packet);
221             return;
222         }
223         int blocksize = packet.getBlocksize();
224         if (packet.isToValidate()) {
225             if (!rule.checkHostAllow(session.getAuth().getUser())) {
226                 session.setStatus(30);
227                 throw new OpenR66ProtocolNotAuthenticatedException(
228                         Messages.getString("LocalServerHandler.10")); //$NON-NLS-1$
229             }
230             // Check if the blocksize is greater than local value
231             if (Configuration.configuration.getBLOCKSIZE() < blocksize) {
232                 blocksize = Configuration.configuration.getBLOCKSIZE();
233                 String sep = localChannelReference.getPartner().getSeperator();
234                 packet = new RequestPacket(packet.getRulename(), packet.getMode(),
235                         packet.getFilename(), blocksize, packet.getRank(),
236                         packet.getSpecialId(), packet.getFileInformation(), packet.getOriginalSize(), sep);
237             }
238         }
239         if (!RequestPacket.isCompatibleMode(rule.getMode(), packet.getMode())) {
240             // not compatible Rule and mode in request
241             throw new OpenR66ProtocolNotAuthenticatedException(
242                     Messages.getString("LocalServerHandler.12") + rule.getMode() + " vs " //$NON-NLS-1$
243                             + packet.getMode());
244         }
245         session.setBlockSize(blocksize);
246         DbTaskRunner runner;
247         // requested
248         boolean isRetrieve = DbTaskRunner.getSenderByRequestPacket(packet);
249         if (packet.getSpecialId() != DbConstant.ILLEGALVALUE) {
250             // Reload or create
251             String requested = DbTaskRunner.getRequested(session, packet);
252             String requester = DbTaskRunner.getRequester(session, packet);
253             logger.debug("DEBUG: " + packet.getSpecialId() + ":" + isRetrieve);
254             if (packet.isToValidate()) {
255                 // Id could be a creation or a reload
256                 // Try reload
257                 try {
258                     runner = new DbTaskRunner(session, rule, packet.getSpecialId(),
259                             requester, requested);
260                     // Patch to prevent self request to be stored by sender
261                     boolean ignoreSave = runner.shallIgnoreSave();
262                     runner.setSender(isRetrieve);
263                     logger.debug("DEBUG: " + runner.getSpecialId() + ":" + ignoreSave + ":" + runner.shallIgnoreSave()
264                             + ":" + isRetrieve);
265                     if (ignoreSave && !runner.shallIgnoreSave() && !runner.checkFromDbForSubmit()) {
266                         // Since status changed, it means that object should be created and not reloaded
267                         // But in case of submit, item already exist so shall be loaded from database
268                         throw new WaarpDatabaseNoDataException("False load, must reopen and create DbTaskRunner");
269                     }
270                 } catch (WaarpDatabaseNoDataException e) {
271                     // Reception of request from requester host
272                     try {
273                         runner = new DbTaskRunner(session, rule, isRetrieve, packet);
274                         logger.debug("Runner before any action: {} {}", runner.shallIgnoreSave(), runner);
275                     } catch (WaarpDatabaseException e1) {
276                         session.setStatus(33);
277                         endInitRequestInError(channel, ErrorCode.QueryRemotelyUnknown,
278                                 null, new OpenR66DatabaseGlobalException(e), packet);
279                         return;
280                     }
281                 } catch (WaarpDatabaseException e) {
282                     session.setStatus(34);
283                     endInitRequestInError(channel, ErrorCode.QueryRemotelyUnknown, null,
284                             new OpenR66DatabaseGlobalException(e), packet);
285                     return;
286                 }
287                 if (runner.isAllDone()) {
288                     // truly an error since done
289                     session.setStatus(31);
290                     endInitRequestInError(channel,
291                             ErrorCode.QueryAlreadyFinished, runner,
292                             new OpenR66ProtocolBusinessQueryAlreadyFinishedException(
293                                     Messages.getString("LocalServerHandler.13") //$NON-NLS-1$
294                                             +
295                                             packet.getSpecialId()), packet);
296                     return;
297                 }
298                 LocalChannelReference lcr =
299                         Configuration.configuration.getLocalTransaction().
300                                 getFromRequest(
301                                         requested + " " + requester + " "
302                                                 + packet.getSpecialId());
303                 if (lcr != null) {
304                     // truly an error since still running
305                     session.setStatus(32);
306                     endInitRequestInError(channel,
307                             ErrorCode.QueryStillRunning, runner,
308                             new OpenR66ProtocolBusinessQueryStillRunningException(
309                                     Messages.getString("LocalServerHandler.14") //$NON-NLS-1$
310                                             +
311                                             packet.getSpecialId()), packet);
312                     return;
313                 }
314                 logger.debug("Runner before any action: {} {}", runner.shallIgnoreSave(), runner);
315                 // ok to restart
316                 try {
317                     if (runner.restart(false)) {
318                         runner.saveStatus();
319                     }
320                 } catch (OpenR66RunnerErrorException e) {
321                 }
322                 // Change the SpecialID! => could generate an error ?
323                 packet.setSpecialId(runner.getSpecialId());
324             } else {
325                 // Id should be a reload
326                 try {
327                     runner = new DbTaskRunner(
328                             session, rule, packet.getSpecialId(),
329                             requester, requested);
330                 } catch (WaarpDatabaseException e) {
331                     if (localChannelReference.getDbSession() == null) {
332                         // Special case of no database client
333                         try {
334                             runner = new DbTaskRunner(
335                                     session, rule, isRetrieve, packet);
336                             logger.debug("Runner before any action: {} {}", runner.shallIgnoreSave(), runner);
337                         } catch (WaarpDatabaseException e1) {
338                             session.setStatus(35);
339                             endInitRequestInError(channel, ErrorCode.QueryRemotelyUnknown, null,
340                                     new OpenR66DatabaseGlobalException(e1), packet);
341                             return;
342                         }
343                     } else {
344                         endInitRequestInError(channel, ErrorCode.QueryRemotelyUnknown, null,
345                                 new OpenR66DatabaseGlobalException(e), packet);
346                         session.setStatus(36);
347                         return;
348                     }
349                 }
350                 runner.setSender(isRetrieve);
351                 // FIX check for SelfRequest
352                 if (runner.isSelfRequest()) {
353                     runner.setFilename(runner.getOriginalFilename());
354                 }
355                 if (!runner.isSender()) {
356                     logger.debug("New filename ? :" + packet.getFilename());
357                     runner.setOriginalFilename(packet.getFilename());
358                     if (runner.getRank() == 0) {
359                         runner.setFilename(packet.getFilename());
360                     }
361                 }
362                 logger.debug("Runner before any action: {} {}", runner.shallIgnoreSave(), runner);
363                 try {
364                     if (runner.restart(false)) {
365                         if (!runner.isSelfRequest()) {
366                             runner.saveStatus();
367                         }
368                     }
369                 } catch (OpenR66RunnerErrorException e) {
370                 }
371             }
372         } else {
373             // Very new request
374             // should not be the case (the requester should always set the id)
375             logger.error("NO TransferID specified: SHOULD NOT BE THE CASE");
376             try {
377                 runner = new DbTaskRunner(session, rule, isRetrieve, packet);
378             } catch (WaarpDatabaseException e) {
379                 session.setStatus(37);
380                 endInitRequestInError(channel, ErrorCode.QueryRemotelyUnknown, null,
381                         new OpenR66DatabaseGlobalException(e), packet);
382                 return;
383             }
384             packet.setSpecialId(runner.getSpecialId());
385         }
386         logger.debug("Runner before any action: {} {}", runner.shallIgnoreSave(), runner);
387         // Check now if request is a valid one
388         if (packet.getCode() != ErrorCode.InitOk.code) {
389             // not valid so create an error from there
390             ErrorCode code = ErrorCode.getFromCode("" + packet.getCode());
391             session.setBadRunner(runner, code);
392             if (!runner.shallIgnoreSave()) {
393                 runner.saveStatus();
394             }
395             session.newState(ERROR);
396             logger.error("Bad runner at startup {} {}", packet, session);
397             ErrorPacket errorPacket = new ErrorPacket(code.getMesg(),
398                     code.getCode(), ErrorPacket.FORWARDCLOSECODE);
399             errorMesg(channel, errorPacket);
400             return;
401         }
402         // Receiver can specify a rank different from database
403         if (runner.isSender()) {
404             logger.debug("Rank was: " + runner.getRank() + " -> " + packet.getRank());
405             runner.setRankAtStartup(packet.getRank());
406         } else {
407             if (runner.getRank() > packet.getRank()) {
408                 logger.debug("Recv Rank was: " + runner.getRank() + " -> " + packet.getRank());
409                 // if receiver, change only if current rank is upper proposed rank
410                 runner.setRankAtStartup(packet.getRank());
411             }
412             if (packet.getOriginalSize() > 0) {
413                 runner.setOriginalSize(packet.getOriginalSize());
414             }
415         }
416         logger.debug("Filesize: " + packet.getOriginalSize() + ":" + runner.isSender());
417         boolean shouldInformBack = false;
418         try {
419             session.setRunner(runner);
420             // Fix to ensure that recv request are not trying to access to not chroot files
421             if (Configuration.configuration.isChrootChecked() && packet.isToValidate() && runner.isSender()) {
422                 session.startup(true);
423             } else {
424                 session.startup(false);
425             }
426             if (runner.isSender() && !runner.isSendThrough()) {
427                 if (packet.getOriginalSize() != runner.getOriginalSize()) {
428                     packet.setOriginalSize(runner.getOriginalSize());
429                     shouldInformBack = true;
430                     logger.debug("Filesize2: " + packet.getOriginalSize() + ":" + runner.isSender());
431                 }
432             }
433         } catch (OpenR66RunnerErrorException e) {
434             try {
435                 runner.saveStatus();
436             } catch (OpenR66RunnerErrorException e1) {
437                 logger.error("Cannot save Status: " + runner, e1);
438             }
439             if (runner.getErrorInfo() == ErrorCode.InitOk ||
440                     runner.getErrorInfo() == ErrorCode.PreProcessingOk ||
441                     runner.getErrorInfo() == ErrorCode.TransferOk) {
442                 runner.setErrorExecutionStatus(ErrorCode.ExternalOp);
443             }
444             logger.error("PreTask in error {}", e.getMessage(), e);
445             errorToSend("PreTask in error: " + e.getMessage(), runner.getErrorInfo(), channel, 38);
446             return;
447         }
448         logger.debug("Filesize: " + packet.getOriginalSize() + ":" + runner.isSender());
449         if (! shouldInformBack) {
450             shouldInformBack = ! packet.getFileInformation().equals(runner.getFileInformation());
451         }
452         if (runner.isFileMoved() && runner.isSender() && runner.isInTransfer()
453                 && runner.getRank() == 0 && (!packet.isToValidate())) {
454             // File was moved during PreTask and very beginning of the transfer
455             // and the remote host has already received the request packet
456             // => Informs the receiver of the new name
457             sendFilenameFilesizeChanging(packet, runner, "Will send a modification of filename due to pretask: ",
458                     "Change Filename by Pre action on sender");
459         } else if ((!packet.getFilename().equals(runner.getOriginalFilename()))
460                 && runner.isSender() && runner.isInTransfer()
461                 && runner.getRank() == 0 && (!packet.isToValidate())) {
462             // File was modify at the very beginning (using wildcards)
463             // and the remote host has already received the request packet
464             // => Informs the receiver of the new name
465             sendFilenameFilesizeChanging(packet, runner, "Will send a modification of filename due to wildcard: ",
466                     "Change Filename by Wildcard on sender");
467         } else if (runner.isSelfRequest() && runner.isSender() && runner.isInTransfer()
468                 && runner.getRank() == 0 && (!packet.isToValidate())) {
469             // FIX SelfRequest
470             // File could be modified at the very beginning (using wildcards)
471             // and the remote host has already received the request packet
472             // => Informs the receiver of the new name
473             sendFilenameFilesizeChanging(packet, runner,
474                     "Will send a modification of filename due to wildcard in SelfMode: ",
475                     "Change Filename by Wildcard on sender in SelfMode");
476         } else if (shouldInformBack && (!packet.isToValidate())) {
477             // Was only for (shouldInformBack)
478             // File length is now known, so inform back
479             sendFilenameFilesizeChanging(packet, runner, "Will send a modification of filesize or fileInfo: ",
480                     "Change Filesize / FileInfo on sender");
481         }
482         session.setReady(true);
483         Configuration.configuration.getLocalTransaction().setFromId(runner, localChannelReference);
484 
485         // Set read/write limit
486         long remoteLimit = packet.getLimit();
487         long localLimit = localChannelReference.getChannelLimit(
488                 runner.isSender());
489         // Take the minimum speed
490         logger.trace("Received limit {}", packet.getLimit());
491         logger.trace("Local limit {}", localChannelReference.getChannelLimit(runner.isSender()));
492         if (localLimit <= 0) {
493             localLimit = remoteLimit;
494         } else if (remoteLimit > 0 && remoteLimit < localLimit) {
495             localLimit = remoteLimit;
496         }
497         localChannelReference.setChannelLimit(runner.isSender(), localLimit);
498         packet.setLimit(localLimit);
499 
500         // inform back
501         if (packet.isToValidate()) {
502             if (Configuration.configuration.getMonitoring() != null) {
503                 Configuration.configuration.getMonitoring().lastInActiveTransfer =
504                         System.currentTimeMillis();
505             }
506             if (runner.isSender()) {
507                 // In case Wildcard was used
508                 logger.debug("New FILENAME: {}", runner.getOriginalFilename());
509                 packet.setFilename(runner.getOriginalFilename());
510                 logger.debug("Rank set: " + runner.getRank());
511                 packet.setRank(runner.getRank());
512             } else {
513                 logger.debug("Rank set: " + runner.getRank());
514                 packet.setRank(runner.getRank());
515             }
516             packet.validate();
517             session.newState(REQUESTD);
518             ChannelUtils.writeAbstractLocalPacket(localChannelReference, packet, true);
519         } else {
520             session.newState(REQUESTD);
521             // requester => might be a client
522             // Save the runner into the session and validate the request so begin transfer
523             session.getLocalChannelReference().getFutureRequest().setRunner(runner);
524             localChannelReference.getFutureValidRequest().setSuccess();
525             if (Configuration.configuration.getMonitoring() != null) {
526                 Configuration.configuration.getMonitoring().lastOutActiveTransfer =
527                         System.currentTimeMillis();
528             }
529         }
530         // if retrieve => START the retrieve operation except if in Send Through mode
531         if (runner.isSender()) {
532             if (runner.isSendThrough()) {
533                 // it is legal to send data from now
534                 logger.debug("Now ready to continue with send through");
535                 localChannelReference.validateEndTransfer(
536                         new R66Result(session, false, ErrorCode.PreProcessingOk, runner));
537             } else {
538                 // Automatically send data now
539                 logger.debug("Now ready to continue with runRetrieve");
540                 NetworkTransaction.runRetrieve(session, channel);
541             }
542         }
543         session.setStatus(39);
544     }

546     /**
547      * Send a Filename/Filesize change to the partner
548      *
549      * @param packet
550      * @param runner
551      * @throws OpenR66ProtocolPacketException
552      */
553     private final void sendFilenameFilesizeChanging(RequestPacket packet, DbTaskRunner runner, String debug, String info)
554             throws OpenR66ProtocolPacketException {
555         logger.debug(debug + runner.getFilename());
556         session.newState(VALID);
557         if (localChannelReference.getPartner().useJson()) {
558             RequestJsonPacket request = new RequestJsonPacket();
559             request.setComment(info);
560             request.setFilename(runner.getFilename());
561             request.setFilesize(packet.getOriginalSize());
562             String infoTransfer = runner.getFileInformation();
563             if (infoTransfer != null && ! infoTransfer.equals(packet.getFileInformation())) {
564                 request.setFileInfo(runner.getFileInformation());
565             }
566             JsonCommandPacket validPacket = new JsonCommandPacket(request,
567                     LocalPacketFactory.REQUESTPACKET);
568             ChannelUtils.writeAbstractLocalPacket(localChannelReference,
569                     validPacket, true);
570         } else {
571             String infoTransfer = runner.getFileInformation();
572             ValidPacket validPacket;
573             if (infoTransfer != null && ! infoTransfer.equals(packet.getFileInformation())
574                     && localChannelReference.getPartner().changeFileInfoEnabled()) {
575                 validPacket = new ValidPacket(info,
576                     runner.getFilename() + PartnerConfiguration.BAR_SEPARATOR_FIELD + packet.getOriginalSize()
577                     + PartnerConfiguration.BAR_SEPARATOR_FIELD + packet.getFileInformation(),
578                     LocalPacketFactory.REQUESTPACKET);

619     /**
620      * Receive a data block
621      *
622      * @param channel
623      * @param packet
624      * @throws OpenR66ProtocolNotAuthenticatedException
625      * @throws OpenR66ProtocolBusinessException
626      * @throws OpenR66ProtocolPacketException
627      */
628     public void data(Channel channel, DataPacket packet)
629             throws OpenR66ProtocolNotAuthenticatedException,
630             OpenR66ProtocolBusinessException, OpenR66ProtocolPacketException {
631         logger.trace("receiving data block {}", packet.getPacketRank());
632         if (!session.isAuthenticated()) {
633             logger.debug("Not authenticated while Data received");
634             packet.clear();
635             throw new OpenR66ProtocolNotAuthenticatedException(
636                     "Not authenticated while Data received");
637         }
638         if (!session.isReady()) {
639             logger.debug("No request prepared");
640             packet.clear();
641             throw new OpenR66ProtocolBusinessException("No request prepared");
642         }
643         if (session.getRunner().isSender()) {
644             logger.debug("Not in receive MODE but receive a packet");
645             packet.clear();
646             throw new OpenR66ProtocolBusinessException(
647                     "Not in receive MODE but receive a packet");
648         }
649         if (!session.getRunner().continueTransfer()) {
650             logger.debug("EndTransfer failed ? " + localChannelReference.getFutureEndTransfer().isFailed());
651             if (localChannelReference.getFutureEndTransfer().isFailed()) {
652                 // nothing to do since already done
653                 session.setStatus(94);
654                 packet.clear();
655                 return;
656             }
657             errorToSend("Transfer in error due previously aborted transmission", ErrorCode.TransferError, channel, 95);
658             packet.clear();
659             return;
660         }
661         if (packet.getPacketRank() != session.getRunner().getRank()) {
662             logger.debug("Issue on rank: " + packet.getPacketRank() + ":" + session.getRunner().getRank());
663             if (!session.addError()) {
664                 // cannot continue
665                 logger.error(Messages.getString("LocalServerHandler.15") + packet.getPacketRank() + " : " + //$NON-NLS-1$
666                         session.getRunner().getRank() + " from {}", session.getRunner());
667                 errorToSend("Too much Bad Rank in transmission: " +
668                         packet.getPacketRank(), ErrorCode.TransferError, channel, 96);
669                 packet.clear();
670                 return;
671             }
672             // Fix the rank if possible
673             if (packet.getPacketRank() < session.getRunner().getRank()) {
674                 logger.debug("Bad RANK: " + packet.getPacketRank() + " : " +
675                         session.getRunner().getRank());
676                 session.getRunner().setRankAtStartup(packet.getPacketRank());
677                 session.getRestart().restartMarker(
678                         session.getRunner().getBlocksize() *
679                                 session.getRunner().getRank());
680                 try {
681                     session.getFile().restartMarker(session.getRestart());
682                 } catch (CommandAbstractException e) {
683                     logger.error("Bad RANK: " + packet.getPacketRank() + " : " +
684                             session.getRunner().getRank());
685                     errorToSend("Bad Rank in transmission even after retry: " +
686                             packet.getPacketRank(), ErrorCode.TransferError, channel, 96);
687                     packet.clear();
688                     return;
689                 }
690             } else {
691                 // really bad
692                 logger.error("Bad RANK: " + packet.getPacketRank() + " : " +
693                         session.getRunner().getRank());
694                 errorToSend("Bad Rank in transmission: " +
695                         packet.getPacketRank() + " > " +
696                         session.getRunner().getRank(), ErrorCode.TransferError, channel, 20);
697                 packet.clear();
698                 return;
699             }
700         }
701         // Check global size
702         long originalSize = session.getRunner().getOriginalSize();
703         if (originalSize >= 0) {
704             if (session.getRunner().getBlocksize() * (session.getRunner().getRank() - 1) > originalSize) {
705                 // cannot continue
706                 logger.error(Messages.getString("LocalServerHandler.16") + packet.getPacketRank() + " : " + //$NON-NLS-1$
707                         (originalSize / session.getRunner().getBlocksize() + 1) + " from {}", session.getRunner());
708                 errorToSend("Too much data transferred: " +
709                         packet.getPacketRank(), ErrorCode.TransferError, channel, 96);
710                 packet.clear();
711                 return;
712             }
713         }
714         // if MD5 check MD5
715         if (RequestPacket.isMD5Mode(session.getRunner().getMode())) {
716             logger.debug("AlgoDigest: "
717                     + (localChannelReference.getPartner() != null ? localChannelReference.getPartner().getDigestAlgo()
718                             : "usual algo"));
719             if (!packet.isKeyValid(localChannelReference.getPartner().getDigestAlgo())) {
720                 // Wrong packet
721                 logger.error(
722                         Messages.getString("LocalServerHandler.17"), packet, localChannelReference.getPartner().getDigestAlgo().name); //$NON-NLS-1$
723                 errorToSend("Transfer in error due to bad Hash on data packet ("
724                         + localChannelReference.getPartner().getDigestAlgo().name + ")",
725                         ErrorCode.MD5Error, channel, 21);
726                 packet.clear();
727                 return;
728             }
729         }
730         if (Configuration.configuration.isGlobalDigest()) {
731             if (globalDigest == null) {
732                 try {
733                     // check if first block, since if not, digest will be only partial
734                     if (session.getRunner().getRank() > 0) {
735                         localChannelReference.setPartialHash();
736                     }
737                     if (localChannelReference.getPartner() != null) {
738                         if (localChannelReference.getPartner().useFinalHash()) {
739                             DigestAlgo algo = localChannelReference.getPartner().getDigestAlgo();
740                             if (algo != Configuration.configuration.getDigest()) {
741                                 globalDigest = new FilesystemBasedDigest(algo);
742                                 localDigest = new FilesystemBasedDigest(Configuration.configuration.getDigest());
743                             }
744                         }
745                     }
746                     if (globalDigest == null) {
747                         globalDigest = new FilesystemBasedDigest(Configuration.configuration.getDigest());
748                         localDigest = null;
749                     }
750                 } catch (NoSuchAlgorithmException e) {
751                 }
752                 logger.debug("GlobalDigest: " + localChannelReference.getPartner().getDigestAlgo() + " different? "
753                         + (localDigest != null));
754             }
755             FileUtils.computeGlobalHash(globalDigest, packet.getData());
756             if (localDigest != null) {
757                 FileUtils.computeGlobalHash(localDigest, packet.getData());
758             }
759         }
760         DataBlock dataBlock = new DataBlock();
761         if (session.getRunner().isRecvThrough() && localChannelReference.isRecvThroughMode()) {
762             try {
763                 localChannelReference.getRecvThroughHandler().writeByteBuf(packet.getData());
764                 session.getRunner().incrementRank();
765                 if (packet.getPacketRank() % 100 == 1) {
766                     logger.debug("Good RANK: " + packet.getPacketRank() + " : " +
767                             session.getRunner().getRank());
768                 }
769             } finally {
770                 packet.clear();
771             }
772         } else {
773             dataBlock.setBlock(packet.getData());
774             try {
775                 session.getFile().writeDataBlock(dataBlock);
776                 session.getRunner().incrementRank();
777                 if (packet.getPacketRank() % 100 == 1) {
778                     logger.debug("Good RANK: " + packet.getPacketRank() + " : " +
779                             session.getRunner().getRank());
780                 }
781             } catch (FileTransferException e) {
782                 errorToSend("Transfer in error",
783                         ErrorCode.TransferError, channel, 22);
784                 return;
785             } finally {
786                 dataBlock.clear();
787                 packet.clear();
788             }
789         }
790     }

792     /**
793      * Receive an End of Transfer
794      *
795      * @param channel
796      * @param packet
797      * @throws OpenR66RunnerErrorException
798      * @throws OpenR66ProtocolSystemException
799      * @throws OpenR66ProtocolNotAuthenticatedException
800      */
801     public void endTransfer(Channel channel, EndTransferPacket packet)
802             throws OpenR66RunnerErrorException, OpenR66ProtocolSystemException,
803             OpenR66ProtocolNotAuthenticatedException {
804         if (!session.isAuthenticated()) {
805             throw new OpenR66ProtocolNotAuthenticatedException(
806                     "Not authenticated while EndTransfer received");
807         }
808         // Check end of transfer
809         long originalSize = session.getRunner().getOriginalSize();
810         logger.debug("OSize: " + originalSize + " isSender: " + session.getRunner().isSender());
811         if (packet.isToValidate()) {
812             // check if possible originalSize
813             if (originalSize > 0) {
814                 try {
815                     if (!session.getRunner().isRecvThrough() && session.getFile().length() != originalSize ||
816                             session.getFile().length() == 0) {
817                         R66Result result = new R66Result(new OpenR66RunnerErrorException(
818                                 Messages.getString("LocalServerHandler.18")), //$NON-NLS-1$
819                                 session, true, ErrorCode.TransferError, session.getRunner());
820                         try {
821                             session.setFinalizeTransfer(false, result);
822                         } catch (OpenR66RunnerErrorException e) {
823                         } catch (OpenR66ProtocolSystemException e) {
824                         }
825                         ErrorPacket error = new ErrorPacket(
826                                 "Final size in error, transfer in error and rank should be reset to 0",
827                                 ErrorCode.TransferError.getCode(), ErrorPacket.FORWARDCLOSECODE);
828                         try {
829                             ChannelUtils.writeAbstractLocalPacket(localChannelReference, error, true);
830                         } catch (OpenR66ProtocolPacketException e) {
831                         }
832                         session.setStatus(23);
833                         ChannelCloseTimer.closeFutureChannel(channel);
834                         return;
835                     }
836                 } catch (CommandAbstractException e) {
837                     // ignore
838                 }
839             }
840             // check if possible Global Digest
841             String hash = packet.getOptional();
842             logger.debug("GlobalDigest: " + localChannelReference.getPartner().getDigestAlgo() + " different? "
843                     + (localDigest != null) + " remoteHash? " + (hash != null));
844             if (hash != null && globalDigest != null) {
845                 String localhash = FilesystemBasedDigest.getHex(globalDigest.Final());
846                 globalDigest = null;
847                 if (!localhash.equalsIgnoreCase(hash)) {
848                     // bad global Hash
849                     //session.getRunner().setRankAtStartup(0);
850                     R66Result result = new R66Result(new OpenR66RunnerErrorException(
851                             Messages.getString("LocalServerHandler.19") + //$NON-NLS-1$
852                                     localChannelReference.getPartner().getDigestAlgo().name + ")"),
853                             session, true, ErrorCode.MD5Error, session.getRunner());
854                     try {
855                         session.setFinalizeTransfer(false, result);
856                     } catch (OpenR66RunnerErrorException e) {
857                     } catch (OpenR66ProtocolSystemException e) {
858                     }
859                     ErrorPacket error = new ErrorPacket(
860                             "Global Hash in error, transfer in error and rank should be reset to 0 (using " +
861                                     localChannelReference.getPartner().getDigestAlgo().name + ")",
862                             ErrorCode.MD5Error.getCode(), ErrorPacket.FORWARDCLOSECODE);
863                     try {
864                         ChannelUtils.writeAbstractLocalPacket(localChannelReference, error, true);
865                     } catch (OpenR66ProtocolPacketException e) {
866                     }
867                     session.setStatus(23);
868                     ChannelCloseTimer.closeFutureChannel(channel);
869                     return;
870                 } else {
871                     if (localDigest != null) {
872                         localhash = FilesystemBasedDigest.getHex(localDigest.Final());
873                     }
874                     localChannelReference.setHashComputeDuringTransfer(localhash);
875                     logger.debug("Global digest ok");
876                 }
877             } else if (globalDigest != null) {
878                 String localhash = null;
879                 if (localDigest != null) {
880                     localhash = FilesystemBasedDigest.getHex(localDigest.Final());
881                 } else {
882                     localhash = FilesystemBasedDigest.getHex(globalDigest.Final());
883                 }
884                 globalDigest = null;
885                 localChannelReference.setHashComputeDuringTransfer(localhash);
886             }
887             localDigest = null;
888             globalDigest = null;
889             session.newState(ENDTRANSFERS);
890             if (!localChannelReference.getFutureRequest().isDone()) {
891                 // Finish with post Operation
892                 R66Result result = new R66Result(session, false,
893                         ErrorCode.TransferOk, session.getRunner());
894                 session.newState(ENDTRANSFERR);
895                 try {
896                     session.setFinalizeTransfer(true, result);
897                 } catch (OpenR66RunnerErrorException e) {
898                     session.newState(ERROR);
899                     ErrorPacket error = null;
900                     if (localChannelReference.getFutureRequest().getResult() != null) {
901                         result = localChannelReference.getFutureRequest().getResult();
902                         error = new ErrorPacket(
903                                 "Error while finalizing transfer: " + result.getMessage(),
904                                 result.getCode().getCode(), ErrorPacket.FORWARDCLOSECODE);
905                     } else {
906                         error = new ErrorPacket(
907                                 "Error while finalizing transfer",
908                                 ErrorCode.FinalOp.getCode(), ErrorPacket.FORWARDCLOSECODE);
909                     }
910                     try {
911                         ChannelUtils.writeAbstractLocalPacket(localChannelReference, error, true);
912                     } catch (OpenR66ProtocolPacketException e1) {
913                     }
914                     session.setStatus(23);
915                     ChannelCloseTimer.closeFutureChannel(channel);
916                     return;
917                 } catch (OpenR66ProtocolSystemException e) {
918                     session.newState(ERROR);
919                     ErrorPacket error = null;
920                     if (localChannelReference.getFutureRequest().getResult() != null) {
921                         result = localChannelReference.getFutureRequest().getResult();
922                         error = new ErrorPacket(
923                                 "Error while finalizing transfer: " + result.getMessage(),
924                                 result.getCode().getCode(), ErrorPacket.FORWARDCLOSECODE);
925                     } else {
926                         error = new ErrorPacket(
927                                 "Error while finalizing transfer",
928                                 ErrorCode.FinalOp.getCode(), ErrorPacket.FORWARDCLOSECODE);
929                     }
930                     try {
931                         ChannelUtils.writeAbstractLocalPacket(localChannelReference, error, true);
932                     } catch (OpenR66ProtocolPacketException e1) {
933                     }
934                     session.setStatus(23);
935                     ChannelCloseTimer.closeFutureChannel(channel);
936                     return;
937                 }
938                 // Now can send validation
939                 packet.validate();
940                 try {
941                     ChannelUtils.writeAbstractLocalPacket(localChannelReference,
942                             packet, false);
943                 } catch (OpenR66ProtocolPacketException e) {
944                     // ignore
945                 }
946             } else {
947                 // in error due to a previous status (like bad MD5)
948                 logger
949                         .error(Messages.getString("LocalServerHandler.20")); //$NON-NLS-1$
950                 session.setStatus(23);
951                 channel.close();
952                 return;
953             }
954         } else {
955             session.newState(ENDTRANSFERR);
956             if (!localChannelReference.getFutureRequest().isDone()) {
957                 // Validation of end of transfer
958                 R66Result result = new R66Result(session, false,
959                         ErrorCode.TransferOk, session.getRunner());
960                 try {
961                     session.setFinalizeTransfer(true, result);
962                 } catch (OpenR66RunnerErrorException e) {
963                     session.newState(ERROR);
964                     ErrorPacket error = null;
965                     if (localChannelReference.getFutureRequest().getResult() != null) {
966                         result = localChannelReference.getFutureRequest().getResult();
967                         error = new ErrorPacket(
968                                 "Error while finalizing transfer: " + result.getMessage(),
969                                 result.getCode().getCode(), ErrorPacket.FORWARDCLOSECODE);
970                     } else {
971                         error = new ErrorPacket(
972                                 "Error while finalizing transfer",
973                                 ErrorCode.FinalOp.getCode(), ErrorPacket.FORWARDCLOSECODE);
974                     }
975                     try {
976                         ChannelUtils.writeAbstractLocalPacket(localChannelReference, error, true);
977                     } catch (OpenR66ProtocolPacketException e1) {
978                     }
979                     session.setStatus(23);
980                     ChannelCloseTimer.closeFutureChannel(channel);
981                     return;
982                 } catch (OpenR66ProtocolSystemException e) {
983                     session.newState(ERROR);
984                     ErrorPacket error = null;
985                     if (localChannelReference.getFutureRequest().getResult() != null) {
986                         result = localChannelReference.getFutureRequest().getResult();
987                         error = new ErrorPacket(
988                                 "Error while finalizing transfer: " + result.getMessage(),
989                                 result.getCode().getCode(), ErrorPacket.FORWARDCLOSECODE);
990                     } else {
991                         error = new ErrorPacket(
992                                 "Error while finalizing transfer",
993                                 ErrorCode.FinalOp.getCode(), ErrorPacket.FORWARDCLOSECODE);
994                     }
995                     try {
996                         ChannelUtils.writeAbstractLocalPacket(localChannelReference, error, true);
997                     } catch (OpenR66ProtocolPacketException e1) {
998                     }
999                     session.setStatus(23);
1000                     ChannelCloseTimer.closeFutureChannel(channel);
1001                     return;
1002                 }
1003             }
1004         }
1005     }

1008      * Receive an End of Request
1009      *
1010      * @param channel
1011      * @param packet
1012      * @throws OpenR66RunnerErrorException
1013      * @throws OpenR66ProtocolSystemException
1014      * @throws OpenR66ProtocolNotAuthenticatedException
1015      */
1016     public void endRequest(Channel channel, EndRequestPacket packet) {
1017         // Validate the last post action on a transfer from receiver remote host
1018         logger.info("Valid Request {} Packet {}",
1019                 localChannelReference,
1020                 packet);
1021         DbTaskRunner runner = session.getRunner();
1022         logger.debug("Runner endRequest: " + (session.getRunner() != null));
1023         if (runner != null) {
1024             runner.setAllDone();
1025             try {
1026                 runner.saveStatus();
1027             } catch (OpenR66RunnerErrorException e) {
1028                 // ignore
1029             }
1030             runner.clean();
1031         }
1032         String optional = null;
1033         if (session.getExtendedProtocol()) {
1034             optional = packet.getOptional();
1035         }
1036         if (!localChannelReference.getFutureRequest().isDone()) {
1037             // end of request
1038             R66Future transfer = localChannelReference.getFutureEndTransfer();
1039             try {
1040                 transfer.await();
1041             } catch (InterruptedException e) {
1042             }
1043             if (transfer.isSuccess()) {
1044                 if (session.getExtendedProtocol() && session.getBusinessObject() != null) {
1045                     if (session.getBusinessObject().getInfo(session) == null) {
1046                         session.getBusinessObject().setInfo(session, optional);
1047                     } else {
1048                         String temp = session.getBusinessObject().getInfo(session);
1049                         session.getBusinessObject().setInfo(session, optional);
1050                         optional = temp;
1051                     }
1052                 } else if (session.getExtendedProtocol() &&
1053                         transfer.getResult().getOther() == null && optional != null) {
1054                     transfer.getResult().setOther(optional);
1055                 }
1056                 localChannelReference.validateRequest(transfer.getResult());
1057             }
1058         }
1059         session.setStatus(1);
1060         if (packet.isToValidate()) {
1061             session.newState(ENDREQUESTS);
1062             packet.validate();
1063             if (session.getExtendedProtocol()) {
1064                 packet.setOptional(optional);
1065             }
1066             session.newState(ENDREQUESTR);
1067             try {
1068                 ChannelUtils.writeAbstractLocalPacket(localChannelReference,
1069                         packet, true);
1070             } catch (OpenR66ProtocolPacketException e) {
1071             }
1072         } else {
1073             session.newState(ENDREQUESTR);
1074         }
1075         if (runner != null && (runner.isSelfRequested() || runner.isSelfRequest())) {
1076             ChannelCloseTimer.closeFutureChannel(channel);
1077         }
1078     }
