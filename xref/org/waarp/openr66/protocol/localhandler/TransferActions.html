90  	/**
91  	 * Finalize a request initialization in error
92  	 * 
93  	 * @param channel
94  	 * @param code
95  	 * @param runner
96  	 * @param e1
97  	 * @param packet
98  	 * @throws OpenR66ProtocolPacketException
99  	 */
100 	private final void endInitRequestInError(Channel channel, ErrorCode code, DbTaskRunner runner,
101 			OpenR66Exception e1, RequestPacket packet) throws OpenR66ProtocolPacketException {
102 		logger.error("TaskRunner initialisation in error: " + code.mesg + " " + session
103 				+ " {} runner {}",
104 				e1 != null ? e1.getMessage() : "no exception",
105 				(runner != null ? runner.toShortString() : "no runner"));
106 		logger.debug("DEBUG Full stack", e1);
107 		localChannelReference.invalidateRequest(new R66Result(
108 				e1, session, true, code, null));
109 
110 		if (packet.isToValidate()) {
111 			// / answer with a wrong request since runner is not set on remote host
112 			if (runner != null) {
113 				if (runner.isSender()) {
114 					// In case Wildcard was used
115 					logger.debug("New FILENAME: {}", runner.getOriginalFilename());
116 					packet.setFilename(runner.getOriginalFilename());
117 					logger.debug("Rank set: " + runner.getRank());
118 					packet.setRank(runner.getRank());
119 				} else {
120 					logger.debug("Rank set: " + runner.getRank());
121 					packet.setRank(runner.getRank());
122 				}
123 			}
124 			packet.validate();
125 			packet.setCode(code.code);
126 			session.newState(ERROR);
127 			ChannelUtils.writeAbstractLocalPacket(localChannelReference, packet, true);
128 		} else {
129 			session.newState(ERROR);
130 			ErrorPacket error = new ErrorPacket(
131 					"TaskRunner initialisation in error: " + e1
132 							.getMessage() + " for " + packet.toString() + " since " + code.mesg,
133 					code.getCode(), ErrorPacket.FORWARDCLOSECODE);
134 			ChannelUtils.writeAbstractLocalPacket(localChannelReference, error, true);
135 		}
136 		session.setStatus(47);
137 		ChannelCloseTimer.closeFutureChannel(channel);
138 	}

140 	/**
141 	 * Receive a request of Transfer
142 	 * 
143 	 * @param channel
144 	 * @param packet
145 	 * @throws OpenR66ProtocolNoDataException
146 	 * @throws OpenR66ProtocolPacketException
147 	 * @throws OpenR66ProtocolBusinessException
148 	 * @throws OpenR66ProtocolSystemException
149 	 * @throws OpenR66RunnerErrorException
150 	 */
151 	public void request(Channel channel, RequestPacket packet)
152 			throws OpenR66ProtocolNoDataException, OpenR66ProtocolPacketException,
153 			OpenR66RunnerErrorException, OpenR66ProtocolSystemException,
154 			OpenR66ProtocolBusinessException {
155 		session.setStatus(99);
156 		if (!session.isAuthenticated()) {
157 			session.setStatus(48);
158 			throw new OpenR66ProtocolNotAuthenticatedException(
159 					Messages.getString("LocalServerHandler.3")); //$NON-NLS-1$
160 		}
161 		if (packet.isToValidate()) {
162 			session.newState(REQUESTR);
163 		}
164 		// XXX validLimit only on requested side
165 		if (packet.isToValidate()) {
166 			if (Configuration.configuration.isShutdown) {
167 				logger.warn(Messages.getString("LocalServerHandler.7") //$NON-NLS-1$
168 						+ packet.getRulename() + " from " + session.getAuth().toString());
169 				session.setStatus(100);
170 				endInitRequestInError(channel,
171 						ErrorCode.ServerOverloaded, null,
172 						new OpenR66ProtocolNotYetConnectionException(
173 								"All new Request blocked"), packet);
174 				session.setStatus(100);
175 				return;
176 			}
177 			if (Configuration.configuration.constraintLimitHandler.checkConstraints()) {
178 				if (Configuration.configuration.r66Mib != null) {
179 					Configuration.configuration.r66Mib.
180 							notifyOverloaded("Rule: " + packet.getRulename() + " from "
181 									+ session.getAuth().toString(),
182 									Configuration.configuration.constraintLimitHandler.lastAlert);
183 				}
184 				logger.warn(Messages.getString("LocalServerHandler.8") //$NON-NLS-1$
185 						+ packet.getRulename() + " while " + Configuration.configuration.constraintLimitHandler.lastAlert +
186 						" from " + session.getAuth().toString());
187 				session.setStatus(100);
188 				endInitRequestInError(channel,
189 						ErrorCode.ServerOverloaded, null,
190 						new OpenR66ProtocolNotYetConnectionException(
191 								"Limit exceeded "+Configuration.configuration.constraintLimitHandler.lastAlert), packet);
192 				session.setStatus(100);
193 				return;
194 			}
195 		} else if (packet.getCode() == ErrorCode.ServerOverloaded.code) {
196 			// XXX unvalid limit on requested host received
197 			logger.info("TaskRunner initialisation in error: " + ErrorCode.ServerOverloaded.mesg);
198 			localChannelReference.invalidateRequest(new R66Result(
199 					null, session, true, ErrorCode.ServerOverloaded, null));
200 			session.setStatus(101);
201 			ChannelCloseTimer.closeFutureChannel(channel);
202 			return;
203 		}
204 		DbRule rule;
205 		try {
206 			rule = new DbRule(localChannelReference.getDbSession(), packet.getRulename());
207 		} catch (WaarpDatabaseException e) {
208 			logger.info("Rule is unknown: " + packet.getRulename() + " {}", e.getMessage());
209 			session.setStatus(49);
210 			endInitRequestInError(channel,
211 					ErrorCode.QueryRemotelyUnknown, null,
212 					new OpenR66ProtocolBusinessException(
213 							Messages.getString("LocalServerHandler.9") + //$NON-NLS-1$
214 									packet.getRulename()), packet);
215 			return;
216 		}
217 		int blocksize = packet.getBlocksize();
218 		if (packet.isToValidate()) {
219 			if (!rule.checkHostAllow(session.getAuth().getUser())) {
220 				session.setStatus(30);
221 				throw new OpenR66ProtocolNotAuthenticatedException(
222 						Messages.getString("LocalServerHandler.10")); //$NON-NLS-1$
223 			}
224 			// Check if the blocksize is greater than local value
225 			if (Configuration.configuration.BLOCKSIZE < blocksize) {
226 				blocksize = Configuration.configuration.BLOCKSIZE;
227 				String sep = localChannelReference.getPartner().getSeperator();
228 				packet = new RequestPacket(packet.getRulename(), packet.getMode(),
229 						packet.getFilename(), blocksize, packet.getRank(),
230 						packet.getSpecialId(), packet.getFileInformation(), packet.getOriginalSize(), sep);
231 			}
232 		}
233 		if (!RequestPacket.isCompatibleMode(rule.mode, packet.getMode())) {
234 			// not compatible Rule and mode in request
235 			throw new OpenR66ProtocolNotAuthenticatedException(
236 					Messages.getString("LocalServerHandler.12") + rule.mode + " vs " //$NON-NLS-1$
237 							+ packet.getMode());
238 		}
239 		session.setBlockSize(blocksize);
240 		DbTaskRunner runner;
241 		// requested
242 		boolean isRetrieve = DbTaskRunner.getSenderByRequestPacket(packet);
243 		if (packet.getSpecialId() != DbConstant.ILLEGALVALUE) {
244 			// Reload or create
245 			String requested = DbTaskRunner.getRequested(session, packet);
246 			String requester = DbTaskRunner.getRequester(session, packet);
247 			logger.debug("DEBUG: "+packet.getSpecialId()+":"+isRetrieve);
248 			if (packet.isToValidate()) {
249 				// Id could be a creation or a reload
250 				// Try reload
251 				try {
252 					runner = new DbTaskRunner(localChannelReference.getDbSession(),
253 							session, rule, packet.getSpecialId(),
254 							requester, requested);
255 					// Patch to prevent self request to be stored by sender
256 					boolean ignoreSave = runner.shallIgnoreSave();
257 					runner.setSender(isRetrieve);
258 					logger.debug("DEBUG: "+runner.getSpecialId()+":"+ignoreSave+":"+runner.shallIgnoreSave()+":"+isRetrieve);
259 					if (ignoreSave && ! runner.shallIgnoreSave() && ! runner.checkFromDbForSubmit()) {
260 						// Since status changed, it means that object should be created and not reloaded
261 						// But in case of submit, item already exist so shall be loaded from database
262 						throw new WaarpDatabaseNoDataException("False load, must reopen and create DbTaskRunner");
263 					}
264 				} catch (WaarpDatabaseNoDataException e) {
265 					// Reception of request from requester host
266 					try {
267 						runner = new DbTaskRunner(localChannelReference.getDbSession(),
268 								session, rule, isRetrieve, packet);
269 						logger.debug("Runner before any action: {} {}", runner.shallIgnoreSave(), runner);
270 					} catch (WaarpDatabaseException e1) {
271 						session.setStatus(33);
272 						endInitRequestInError(channel, ErrorCode.QueryRemotelyUnknown,
273 								null, new OpenR66DatabaseGlobalException(e), packet);
274 						return;
275 					}
276 				} catch (WaarpDatabaseException e) {
277 					session.setStatus(34);
278 					endInitRequestInError(channel, ErrorCode.QueryRemotelyUnknown, null,
279 							new OpenR66DatabaseGlobalException(e), packet);
280 					return;
281 				}
282 				if (runner.isAllDone()) {
283 					// truly an error since done
284 					session.setStatus(31);
285 					endInitRequestInError(channel,
286 							ErrorCode.QueryAlreadyFinished, runner,
287 							new OpenR66ProtocolBusinessQueryAlreadyFinishedException(
288 									Messages.getString("LocalServerHandler.13") //$NON-NLS-1$
289 											+
290 											packet.getSpecialId()), packet);
291 					return;
292 				}
293 				LocalChannelReference lcr =
294 						Configuration.configuration.getLocalTransaction().
295 								getFromRequest(
296 										requested + " " + requester + " "
297 												+ packet.getSpecialId());
298 				if (lcr != null) {
299 					// truly an error since still running
300 					session.setStatus(32);
301 					endInitRequestInError(channel,
302 							ErrorCode.QueryStillRunning, runner,
303 							new OpenR66ProtocolBusinessQueryStillRunningException(
304 									Messages.getString("LocalServerHandler.14") //$NON-NLS-1$
305 											+
306 											packet.getSpecialId()), packet);
307 					return;
308 				}
309 				logger.debug("Runner before any action: {} {}", runner.shallIgnoreSave(), runner);
310 				// ok to restart
311 				try {
312 					if (runner.restart(false)) {
313 						runner.saveStatus();
314 					}
315 				} catch (OpenR66RunnerErrorException e) {
316 				}
317 				// Change the SpecialID! => could generate an error ?
318 				packet.setSpecialId(runner.getSpecialId());
319 			} else {
320 				// Id should be a reload
321 				try {
322 					runner = new DbTaskRunner(localChannelReference.getDbSession(),
323 							session, rule, packet.getSpecialId(),
324 							requester, requested);
325 				} catch (WaarpDatabaseException e) {
326 					if (localChannelReference.getDbSession() == null) {
327 						// Special case of no database client
328 						try {
329 							runner = new DbTaskRunner(localChannelReference.getDbSession(),
330 									session, rule, isRetrieve, packet);
331 							logger.debug("Runner before any action: {} {}", runner.shallIgnoreSave(), runner);
332 						} catch (WaarpDatabaseException e1) {
333 							session.setStatus(35);
334 							endInitRequestInError(channel, ErrorCode.QueryRemotelyUnknown, null,
335 									new OpenR66DatabaseGlobalException(e1), packet);
336 							return;
337 						}
338 					} else {
339 						endInitRequestInError(channel, ErrorCode.QueryRemotelyUnknown, null,
340 								new OpenR66DatabaseGlobalException(e), packet);
341 						session.setStatus(36);
342 						return;
343 					}
344 				}
345 				runner.setSender(isRetrieve);
346 				// FIX check for SelfRequest
347 				if (runner.isSelfRequest()) {
348 					runner.setFilename(runner.getOriginalFilename());
349 				}
350 				if (! runner.isSender()) {
351 					logger.debug("New filename ? :" +packet.getFilename());
352 					runner.setOriginalFilename(packet.getFilename());
353 					runner.setFilename(packet.getFilename());
354 				}
355 				logger.debug("Runner before any action: {} {}", runner.shallIgnoreSave(), runner);
356 				try {
357 					if (runner.restart(false)) {
358 						if (!runner.isSelfRequest()) {
359 							runner.saveStatus();
360 						}
361 					}
362 				} catch (OpenR66RunnerErrorException e) {
363 				}
364 			}
365 		} else {
366 			// Very new request
367 			// should not be the case (the requester should always set the id)
368 			logger.error("NO TransferID specified: SHOULD NOT BE THE CASE");
369 			try {
370 				runner = new DbTaskRunner(localChannelReference.getDbSession(),
371 						session, rule, isRetrieve, packet);
372 			} catch (WaarpDatabaseException e) {
373 				session.setStatus(37);
374 				endInitRequestInError(channel, ErrorCode.QueryRemotelyUnknown, null,
375 						new OpenR66DatabaseGlobalException(e), packet);
376 				return;
377 			}
378 			packet.setSpecialId(runner.getSpecialId());
379 		}
380 		logger.debug("Runner before any action: {} {}", runner.shallIgnoreSave(), runner);
381 		// Check now if request is a valid one
382 		if (packet.getCode() != ErrorCode.InitOk.code) {
383 			// not valid so create an error from there
384 			ErrorCode code = ErrorCode.getFromCode("" + packet.getCode());
385 			session.setBadRunner(runner, code);
386 			session.newState(ERROR);
387 			logger.error("Bad runner at startup {} {}", packet, session);
388 			ErrorPacket errorPacket = new ErrorPacket(code.mesg,
389 					code.getCode(), ErrorPacket.FORWARDCLOSECODE);
390 			errorMesg(channel, errorPacket);
391 			return;
392 		}
393 		// Receiver can specify a rank different from database
394 		if (runner.isSender()) {
395 			logger.debug("Rank was: " + runner.getRank() + " -> " + packet.getRank());
396 			runner.setRankAtStartup(packet.getRank());
397 		} else {
398 			if (runner.getRank() > packet.getRank()) {
399 				logger.debug("Recv Rank was: " + runner.getRank() + " -> " + packet.getRank());
400 				// if receiver, change only if current rank is upper proposed rank
401 				runner.setRankAtStartup(packet.getRank());
402 			}
403 			if (packet.getOriginalSize() > 0) {
404 				runner.setOriginalSize(packet.getOriginalSize());
405 			}
406 		}
407 		logger.debug("Filesize: "+packet.getOriginalSize()+":"+runner.isSender());
408 		boolean shouldInformBack = false;
409 		try {
410 			session.setRunner(runner);
411 			// Fix to ensure that recv request are not trying to access to not chroot files
412 			if (Configuration.configuration.chrootChecked && packet.isToValidate() && runner.isSender()) {
413 				session.startup(true);
414 			} else {
415 				session.startup(false);
416 			}
417 			if (runner.isSender() && ! runner.isSendThrough()) {
418 				if (packet.getOriginalSize() != runner.getOriginalSize()) {
419 					packet.setOriginalSize(runner.getOriginalSize());
420 					shouldInformBack = true;
421 					logger.debug("Filesize2: "+packet.getOriginalSize()+":"+runner.isSender());
422 				}
423 			}
424 		} catch (OpenR66RunnerErrorException e) {
425 			try {
426 				runner.saveStatus();
427 			} catch (OpenR66RunnerErrorException e1) {
428 				logger.error("Cannot save Status: " + runner, e1);
429 			}
430 			if (runner.getErrorInfo() == ErrorCode.InitOk ||
431 					runner.getErrorInfo() == ErrorCode.PreProcessingOk ||
432 					runner.getErrorInfo() == ErrorCode.TransferOk) {
433 				runner.setErrorExecutionStatus(ErrorCode.ExternalOp);
434 			}
435 			logger.error("PreTask in error {}", e.getMessage(), e);
436 			errorToSend("PreTask in error: " + e.getMessage(), runner.getErrorInfo(), channel, 38);
437 			return;
438 		}
439 		logger.debug("Filesize: "+packet.getOriginalSize()+":"+runner.isSender());
440 		if (runner.isFileMoved() && runner.isSender() && runner.isInTransfer()
441 				&& runner.getRank() == 0 && (!packet.isToValidate())) {
442 			// File was moved during PreTask and very beginning of the transfer
443 			// and the remote host has already received the request packet
444 			// => Informs the receiver of the new name
445 			sendFilenameFilesizeChanging(packet, runner, "Will send a modification of filename due to pretask: ",
446 					"Change Filename by Pre action on sender");
447 		} else if ((!packet.getFilename().equals(runner.getOriginalFilename())) 
448 				&& runner.isSender() && runner.isInTransfer()
449 				&& runner.getRank() == 0 && (!packet.isToValidate())) {
450 			// File was modify at the very beginning (using wildcards)
451 			// and the remote host has already received the request packet
452 			// => Informs the receiver of the new name
453 			sendFilenameFilesizeChanging(packet, runner, "Will send a modification of filename due to wildcard: ",
454 					"Change Filename by Wildcard on sender");
455 		} else if (runner.isSelfRequest() && runner.isSender() && runner.isInTransfer()
456 				&& runner.getRank() == 0 && (!packet.isToValidate())) {
457 			// FIX SelfRequest
458 			// File could be modified at the very beginning (using wildcards)
459 			// and the remote host has already received the request packet
460 			// => Informs the receiver of the new name
461 			sendFilenameFilesizeChanging(packet, runner, "Will send a modification of filename due to wildcard in SelfMode: ",
462 					"Change Filename by Wildcard on sender in SelfMode");
463 		} else if (shouldInformBack && (!packet.isToValidate())) {
464 			// Was only for (shouldInformBack)
465 			// File length is now known, so inform back
466 			sendFilenameFilesizeChanging(packet, runner, "Will send a modification of filesize: ",
467 					"Change Filesize on sender");
468 		}
469 		session.setReady(true);
470 		Configuration.configuration.getLocalTransaction().setFromId(runner, localChannelReference);
471 		// inform back
472 		if (packet.isToValidate()) {
473 			if (Configuration.configuration.monitoring != null) {
474 				Configuration.configuration.monitoring.lastInActiveTransfer =
475 						System.currentTimeMillis();
476 			}
477 			if (runner.isSender()) {
478 				// In case Wildcard was used
479 				logger.debug("New FILENAME: {}", runner.getOriginalFilename());
480 				packet.setFilename(runner.getOriginalFilename());
481 				logger.debug("Rank set: " + runner.getRank());
482 				packet.setRank(runner.getRank());
483 			} else {
484 				logger.debug("Rank set: " + runner.getRank());
485 				packet.setRank(runner.getRank());
486 			}
487 			packet.validate();
488 			session.newState(REQUESTD);
489 			ChannelUtils.writeAbstractLocalPacket(localChannelReference, packet, true);
490 		} else {
491 			session.newState(REQUESTD);
492 			// requester => might be a client
493 			// Save the runner into the session and validate the request so begin transfer
494 			session.getLocalChannelReference().getFutureRequest().runner = runner;
495 			localChannelReference.getFutureValidRequest().setSuccess();
496 			if (Configuration.configuration.monitoring != null) {
497 				Configuration.configuration.monitoring.lastOutActiveTransfer =
498 						System.currentTimeMillis();
499 			}
500 		}
501 		// if retrieve => START the retrieve operation except if in Send Through mode
502 		if (runner.isSender()) {
503 			if (runner.isSendThrough()) {
504 				// it is legal to send data from now
505 				logger.debug("Now ready to continue with send through");
506 				localChannelReference.validateEndTransfer(
507 						new R66Result(session, false, ErrorCode.PreProcessingOk, runner));
508 			} else {
509 				// Automatically send data now
510 				logger.debug("Now ready to continue with runRetrieve");
511 				NetworkTransaction.runRetrieve(session, channel);
512 			}
513 		}
514 		session.setStatus(39);
515 	}

517 	/**
518 	 * Send a Filename/Filesize change to the partner
519 	 * @param packet
520 	 * @param runner
521 	 * @throws OpenR66ProtocolPacketException

572 	/**
573 	 * Receive a data block
574 	 * 
575 	 * @param channel
576 	 * @param packet
577 	 * @throws OpenR66ProtocolNotAuthenticatedException
578 	 * @throws OpenR66ProtocolBusinessException
579 	 * @throws OpenR66ProtocolPacketException
580 	 */
581 	public void data(Channel channel, DataPacket packet)
582 			throws OpenR66ProtocolNotAuthenticatedException,
583 			OpenR66ProtocolBusinessException, OpenR66ProtocolPacketException {
584 		if (!session.isAuthenticated()) {
585 			logger.debug("Not authenticated while Data received");
586 			throw new OpenR66ProtocolNotAuthenticatedException(
587 					"Not authenticated while Data received");
588 		}
589 		if (!session.isReady()) {
590 			logger.debug("No request prepared");
591 			throw new OpenR66ProtocolBusinessException("No request prepared");
592 		}
593 		if (session.getRunner().isSender()) {
594 			logger.debug("Not in receive MODE but receive a packet");
595 			throw new OpenR66ProtocolBusinessException(
596 					"Not in receive MODE but receive a packet");
597 		}
598 		if (!session.getRunner().continueTransfer()) {
599 			logger.debug("EndTransfer failed ? "+localChannelReference.getFutureEndTransfer().isFailed());
600 			if (localChannelReference.getFutureEndTransfer().isFailed()) {
601 				// nothing to do since already done
602 				session.setStatus(94);
603 				return;
604 			}
605 			errorToSend("Transfer in error due previously aborted transmission", ErrorCode.TransferError, channel, 95);
606 			return;
607 		}
608 		if (packet.getPacketRank() != session.getRunner().getRank()) {
609 			logger.debug("Issue on rank: "+packet.getPacketRank() +":"+ session.getRunner().getRank());
610 			if (! session.addError()) {
611 				// cannot continue
612 				logger.error(Messages.getString("LocalServerHandler.15") + packet.getPacketRank() + " : " + //$NON-NLS-1$
613 						session.getRunner().getRank()+ " from {}", session.getRunner());
614 				errorToSend("Too much Bad Rank in transmission: " +
615 					packet.getPacketRank(), ErrorCode.TransferError, channel, 96);
616 				return;
617 			}
618 			// Fix the rank if possible
619 			if (packet.getPacketRank() < session.getRunner().getRank()) {
620 				logger.debug("Bad RANK: " + packet.getPacketRank() + " : " +
621 						session.getRunner().getRank());
622 				session.getRunner().setRankAtStartup(packet.getPacketRank());
623 				session.getRestart().restartMarker(
624 						session.getRunner().getBlocksize() *
625 								session.getRunner().getRank());
626 				try {
627 					session.getFile().restartMarker(session.getRestart());
628 				} catch (CommandAbstractException e) {
629 					logger.error("Bad RANK: " + packet.getPacketRank() + " : " +
630 							session.getRunner().getRank());
631 					errorToSend("Bad Rank in transmission even after retry: " +
632 							packet.getPacketRank(), ErrorCode.TransferError, channel, 96);
633 					return;
634 				}
635 			} else {
636 				// really bad
637 				logger.error("Bad RANK: " + packet.getPacketRank() + " : " +
638 						session.getRunner().getRank());
639 				errorToSend("Bad Rank in transmission: " +
640 						packet.getPacketRank()+ " > " +
641 								session.getRunner().getRank(), ErrorCode.TransferError, channel, 20);
642 				return;
643 			}
644 		}
645 		// Check global size
646 		long originalSize = session.getRunner().getOriginalSize();
647 		if (originalSize >= 0) {
648 			if (session.getRunner().getBlocksize() * (session.getRunner().getRank()-1) > originalSize) {
649 				// cannot continue
650 				logger.error(Messages.getString("LocalServerHandler.16") + packet.getPacketRank() + " : " + //$NON-NLS-1$
651 						(originalSize/session.getRunner().getBlocksize()+1)+" from {}", session.getRunner());
652 				errorToSend("Too much data transferred: " +
653 						packet.getPacketRank(), ErrorCode.TransferError, channel, 96);
654 				return;
655 			}
656 		}
657 		// if MD5 check MD5
658 		if (RequestPacket.isMD5Mode(session.getRunner().getMode())) {
659 			logger.debug("AlgoDigest: "+(localChannelReference.getPartner() != null ? localChannelReference.getPartner().getDigestAlgo() : "usual algo"));
660 			if (!packet.isKeyValid(localChannelReference.getPartner().getDigestAlgo())) {
661 				// Wrong packet
662 				logger.error(Messages.getString("LocalServerHandler.17"), packet, localChannelReference.getPartner().getDigestAlgo().name); //$NON-NLS-1$
663 				errorToSend("Transfer in error due to bad Hash on data packet ("+localChannelReference.getPartner().getDigestAlgo().name+")",
664 						ErrorCode.MD5Error, channel, 21);
665 				return;
666 			}
667 		}
668 		if (Configuration.configuration.globalDigest) {
669 			if (globalDigest == null) {
670 				try {
671 					// check if first block, since if not, digest will be only partial
672 					if (session.getRunner().getRank() > 0) {
673 						localChannelReference.setPartialHash();
674 					}
675 					if (localChannelReference.getPartner() != null) {
676 						if (localChannelReference.getPartner().useFinalHash()) {
677 							DigestAlgo algo = localChannelReference.getPartner().getDigestAlgo();
678 							if (algo != Configuration.configuration.digest) {
679 								globalDigest = new FilesystemBasedDigest(algo);
680 								localDigest = new FilesystemBasedDigest(Configuration.configuration.digest);
681 							}
682 						}
683 					}
684 					if (globalDigest == null) {
685 						globalDigest = new FilesystemBasedDigest(Configuration.configuration.digest);
686 						localDigest = null;
687 					}
688 				} catch (NoSuchAlgorithmException e) {
689 				}
690 				logger.debug("GlobalDigest: "+localChannelReference.getPartner().getDigestAlgo()+" different? "+(localDigest != null));
691 			}
692 			FileUtils.computeGlobalHash(globalDigest, packet.getData());
693 			if (localDigest != null) {
694 				FileUtils.computeGlobalHash(localDigest, packet.getData());
695 			}
696 		}
697 		DataBlock dataBlock = new DataBlock();
698 		if (session.getRunner().isRecvThrough() && localChannelReference.isRecvThroughMode()) {
699 			localChannelReference.getRecvThroughHandler().writeChannelBuffer(packet.getData());
700 			session.getRunner().incrementRank();
701 			if (packet.getPacketRank() % 100 == 1) {
702 				logger.debug("Good RANK: " + packet.getPacketRank() + " : " +
703 					session.getRunner().getRank());
704 			}
705 		} else {
706 			dataBlock.setBlock(packet.getData());
707 			try {
708 				session.getFile().writeDataBlock(dataBlock);
709 				session.getRunner().incrementRank();
710 				if (packet.getPacketRank() % 100 == 1) {
711 					logger.debug("Good RANK: " + packet.getPacketRank() + " : " +
712 							session.getRunner().getRank());
713 				}
714 			} catch (FileTransferException e) {
715 				errorToSend("Transfer in error",
716 						ErrorCode.TransferError, channel, 22);
717 				return;
718 			}
719 		}
720 	}

722 	/**
723 	 * Receive an End of Transfer
724 	 * 
725 	 * @param channel
726 	 * @param packet
727 	 * @throws OpenR66RunnerErrorException
728 	 * @throws OpenR66ProtocolSystemException
729 	 * @throws OpenR66ProtocolNotAuthenticatedException
730 	 */
731 	public void endTransfer(Channel channel, EndTransferPacket packet)
732 			throws OpenR66RunnerErrorException, OpenR66ProtocolSystemException,
733 			OpenR66ProtocolNotAuthenticatedException {
734 		if (!session.isAuthenticated()) {
735 			throw new OpenR66ProtocolNotAuthenticatedException(
736 					"Not authenticated while EndTransfer received");
737 		}
738 		// Check end of transfer
739 		long originalSize = session.getRunner().getOriginalSize();
740 		logger.debug("OSize: "+originalSize+" isSender: "+session.getRunner().isSender());
741 		if (packet.isToValidate()) {
742 			// check if possible originalSize
743 			if (originalSize > 0) {
744 				try {
745 					if (!session.getRunner().isRecvThrough() && session.getFile().length() != originalSize || 
746 							session.getFile().length() == 0) {
747 						R66Result result = new R66Result(new OpenR66RunnerErrorException(Messages.getString("LocalServerHandler.18")), //$NON-NLS-1$
748 								session, true, ErrorCode.TransferError, session.getRunner());
749 						try {
750 							session.setFinalizeTransfer(false, result);
751 						} catch (OpenR66RunnerErrorException e) {
752 						} catch (OpenR66ProtocolSystemException e) {
753 						}
754 						ErrorPacket error = new ErrorPacket(
755 								"Final size in error, transfer in error and rank should be reset to 0",
756 								ErrorCode.TransferError.getCode(), ErrorPacket.FORWARDCLOSECODE);
757 						try {
758 							ChannelUtils.writeAbstractLocalPacket(localChannelReference, error, true);
759 						} catch (OpenR66ProtocolPacketException e) {
760 						}
761 						session.setStatus(23);
762 						ChannelCloseTimer.closeFutureChannel(channel);
763 						return;
764 					}
765 				} catch (CommandAbstractException e) {
766 					// ignore
767 				}
768 			}
769 			// check if possible Global Digest
770 			String hash = packet.getOptional();
771 			logger.debug("GlobalDigest: "+localChannelReference.getPartner().getDigestAlgo()+" different? "+(localDigest != null)+" remoteHash? "+(hash != null));
772 			if (hash != null && globalDigest != null) {
773 				String localhash = FilesystemBasedDigest.getHex(globalDigest.Final());
774 				globalDigest = null;
775 				if (! localhash.equalsIgnoreCase(hash)) {
776 					// bad global Hash
777 					//session.getRunner().setRankAtStartup(0);
778 					R66Result result = new R66Result(new OpenR66RunnerErrorException(Messages.getString("LocalServerHandler.19")+ //$NON-NLS-1$
779 							localChannelReference.getPartner().getDigestAlgo().name+")"),
780 							session, true, ErrorCode.MD5Error, session.getRunner());
781 					try {
782 						session.setFinalizeTransfer(false, result);
783 					} catch (OpenR66RunnerErrorException e) {
784 					} catch (OpenR66ProtocolSystemException e) {
785 					}
786 					ErrorPacket error = new ErrorPacket(
787 							"Global Hash in error, transfer in error and rank should be reset to 0 (using "+
788 							localChannelReference.getPartner().getDigestAlgo().name+")",
789 							ErrorCode.MD5Error.getCode(), ErrorPacket.FORWARDCLOSECODE);
790 					try {
791 						ChannelUtils.writeAbstractLocalPacket(localChannelReference, error, true);
792 					} catch (OpenR66ProtocolPacketException e) {
793 					}
794 					session.setStatus(23);
795 					ChannelCloseTimer.closeFutureChannel(channel);
796 					return;
797 				} else {
798 					if (localDigest != null) {
799 						localhash = FilesystemBasedDigest.getHex(localDigest.Final());
800 					}
801 					localChannelReference.setHashComputeDuringTransfer(localhash);
802 					logger.debug("Global digest ok");
803 				}
804 			} else if (globalDigest != null) {
805 				String localhash = null;
806 				if (localDigest != null) {
807 					localhash = FilesystemBasedDigest.getHex(localDigest.Final());
808 				} else {
809 					localhash = FilesystemBasedDigest.getHex(globalDigest.Final());
810 				}
811 				globalDigest = null;
812 				localChannelReference.setHashComputeDuringTransfer(localhash);
813 			}
814 			localDigest = null;
815 			globalDigest = null;
816 			session.newState(ENDTRANSFERS);
817 			if (!localChannelReference.getFutureRequest().isDone()) {
818 				// Finish with post Operation
819 				R66Result result = new R66Result(session, false,
820 						ErrorCode.TransferOk, session.getRunner());
821 				session.newState(ENDTRANSFERR);
822 				try {
823 					session.setFinalizeTransfer(true, result);
824 				} catch (OpenR66RunnerErrorException e) {
825 					// TODO
826 					session.newState(ERROR);
827 					ErrorPacket error = null;
828 					if (localChannelReference.getFutureRequest().getResult() != null) {
829 						result = localChannelReference.getFutureRequest().getResult();
830 						error = new ErrorPacket(
831 							"Error while finalizing transfer: "+result.getMessage(),
832 							result.code.getCode(), ErrorPacket.FORWARDCLOSECODE);
833 					} else {
834 						error = new ErrorPacket(
835 								"Error while finalizing transfer",
836 								ErrorCode.FinalOp.getCode(), ErrorPacket.FORWARDCLOSECODE);
837 					}
838 					try {
839 						ChannelUtils.writeAbstractLocalPacket(localChannelReference, error, true);
840 					} catch (OpenR66ProtocolPacketException e1) {
841 					}
842 					session.setStatus(23);
843 					ChannelCloseTimer.closeFutureChannel(channel);
844 					return;
845 				} catch (OpenR66ProtocolSystemException e) {
846 					// TODO
847 					session.newState(ERROR);
848 					ErrorPacket error = null;
849 					if (localChannelReference.getFutureRequest().getResult() != null) {
850 						result = localChannelReference.getFutureRequest().getResult();
851 						error = new ErrorPacket(
852 							"Error while finalizing transfer: "+result.getMessage(),
853 							result.code.getCode(), ErrorPacket.FORWARDCLOSECODE);
854 					} else {
855 						error = new ErrorPacket(
856 								"Error while finalizing transfer",
857 								ErrorCode.FinalOp.getCode(), ErrorPacket.FORWARDCLOSECODE);
858 					}
859 					try {
860 						ChannelUtils.writeAbstractLocalPacket(localChannelReference, error, true);
861 					} catch (OpenR66ProtocolPacketException e1) {
862 					}
863 					session.setStatus(23);
864 					ChannelCloseTimer.closeFutureChannel(channel);
865 					return;
866 				}
867 				// Now can send validation
868 				packet.validate();
869 				try {
870 					ChannelUtils.writeAbstractLocalPacket(localChannelReference,
871 							packet, false);

873 					// ignore
874 				}
875 			} else {
876 				// in error due to a previous status (like bad MD5)
877 				logger
878 						.error(Messages.getString("LocalServerHandler.20")); //$NON-NLS-1$
879 				session.setStatus(23);
880 				Channels.close(channel);
881 				return;
882 			}
883 		} else {
884 			session.newState(ENDTRANSFERR);
885 			if (!localChannelReference.getFutureRequest().isDone()) {
886 				// Validation of end of transfer
887 				R66Result result = new R66Result(session, false,
888 						ErrorCode.TransferOk, session.getRunner());
889 				try {
890 					session.setFinalizeTransfer(true, result);

904 					}
905 					try {

907 					} catch (OpenR66ProtocolPacketException e1) {
908 					}

912 				} catch (OpenR66ProtocolSystemException e) {
913 					// TODO

938 	/**
939 	 * Receive an End of Request
940 	 * 
941 	 * @param channel
942 	 * @param packet
943 	 * @throws OpenR66RunnerErrorException
944 	 * @throws OpenR66ProtocolSystemException
945 	 * @throws OpenR66ProtocolNotAuthenticatedException
946 	 */
947 	public void endRequest(Channel channel, EndRequestPacket packet) {
948 		// Validate the last post action on a transfer from receiver remote host
949 		logger.info("Valid Request {} Packet {}",
950 				localChannelReference,
951 				packet);
952 		DbTaskRunner runner = session.getRunner();
953 		logger.debug("Runner endRequest: " + (session.getRunner() != null));
954 		if (runner != null) {
955 			runner.setAllDone();
956 			try {
957 				runner.saveStatus();
958 			} catch (OpenR66RunnerErrorException e) {
959 				// ignore
960 			}
961 		}
962 		String optional = null;
963 		if (session.getExtendedProtocol()) {
964 			optional = packet.getOptional();
965 		}
966 		if (!localChannelReference.getFutureRequest().isDone()) {
967 			// end of request
968 			R66Future transfer = localChannelReference.getFutureEndTransfer();
969 			try {
970 				transfer.await();
971 			} catch (InterruptedException e) {
972 			}
973 			if (transfer.isSuccess()) {
974 				if (session.getExtendedProtocol() && session.getBusinessObject() != null) {
975 					if (session.getBusinessObject().getInfo() == null) {
976 						session.getBusinessObject().setInfo(optional);
977 					} else {
978 						String temp = session.getBusinessObject().getInfo();
979 						session.getBusinessObject().setInfo(optional);
980 						optional = temp;
981 					}
982 				} else if (session.getExtendedProtocol() &&
983 						transfer.getResult().other == null && optional != null) {
984 					transfer.getResult().other = optional;
985 				}
986 				localChannelReference.validateRequest(transfer.getResult());
987 			}
988 		}
989 		session.setStatus(1);
990 		if (packet.isToValidate()) {
991 			session.newState(ENDREQUESTS);
992 			packet.validate();
993 			if (session.getExtendedProtocol()) {
994 				packet.setOptional(optional);
995 			}
996 			session.newState(ENDREQUESTR);
997 			try {
998 				ChannelUtils.writeAbstractLocalPacket(localChannelReference,
999 						packet, true);
1000 			} catch (OpenR66ProtocolPacketException e) {
1001 			}
1002 		} else {
1003 			session.newState(ENDREQUESTR);
1004 		}
1005 		if (runner != null && (runner.isSelfRequested() || runner.isSelfRequest())) {
1006 			ChannelCloseTimer.closeFutureChannel(channel);
1007 		}
1008 	}
