92      /**
93       * Finalize a request initialization in error
94       * 
95       * @param channel
96       * @param code
97       * @param runner
98       * @param e1
99       * @param packet
100      * @throws OpenR66ProtocolPacketException
101      */
102     private final void endInitRequestInError(Channel channel, ErrorCode code, DbTaskRunner runner,
103             OpenR66Exception e1, RequestPacket packet) throws OpenR66ProtocolPacketException {
104         logger.error("TaskRunner initialisation in error: " + code.mesg + " " + session
105                 + " {} runner {}",
106                 e1 != null ? e1.getMessage() : "no exception",
107                 (runner != null ? runner.toShortString() : "no runner"));
108         logger.debug("DEBUG Full stack", e1);
109         localChannelReference.invalidateRequest(new R66Result(
110                 e1, session, true, code, null));
111 
112         if (packet.isToValidate()) {
113             // / answer with a wrong request since runner is not set on remote host
114             if (runner != null) {
115                 if (runner.isSender()) {
116                     // In case Wildcard was used
117                     logger.debug("New FILENAME: {}", runner.getOriginalFilename());
118                     packet.setFilename(runner.getOriginalFilename());
119                     logger.debug("Rank set: " + runner.getRank());
120                     packet.setRank(runner.getRank());
121                 } else {
122                     logger.debug("Rank set: " + runner.getRank());
123                     packet.setRank(runner.getRank());
124                 }
125             }
126             packet.validate();
127             packet.setCode(code.code);
128             session.newState(ERROR);
129             ChannelUtils.writeAbstractLocalPacket(localChannelReference, packet, true);
130         } else {
131             session.newState(ERROR);
132             ErrorPacket error = new ErrorPacket(
133                     "TaskRunner initialisation in error: " + e1
134                             .getMessage() + " for " + packet.toString() + " since " + code.mesg,
135                     code.getCode(), ErrorPacket.FORWARDCLOSECODE);
136             ChannelUtils.writeAbstractLocalPacket(localChannelReference, error, true);
137         }
138         session.setStatus(47);
139         ChannelCloseTimer.closeFutureChannel(channel);
140     }

142     /**
143      * Receive a request of Transfer
144      * 
145      * @param channel
146      * @param packet
147      * @throws OpenR66ProtocolNoDataException
148      * @throws OpenR66ProtocolPacketException
149      * @throws OpenR66ProtocolBusinessException
150      * @throws OpenR66ProtocolSystemException
151      * @throws OpenR66RunnerErrorException
152      */
153     public void request(LocalChannel channel, RequestPacket packet)
154             throws OpenR66ProtocolNoDataException, OpenR66ProtocolPacketException,
155             OpenR66RunnerErrorException, OpenR66ProtocolSystemException,
156             OpenR66ProtocolBusinessException {
157         session.setStatus(99);
158         if (!session.isAuthenticated()) {
159             session.setStatus(48);
160             throw new OpenR66ProtocolNotAuthenticatedException(
161                     Messages.getString("LocalServerHandler.3")); //$NON-NLS-1$
162         }
163         if (packet.isToValidate()) {
164             session.newState(REQUESTR);
165         }
166         // XXX validLimit only on requested side
167         if (packet.isToValidate()) {
168             if (Configuration.configuration.isShutdown()) {
169                 logger.warn(Messages.getString("LocalServerHandler.7") //$NON-NLS-1$
170                         + packet.getRulename() + " from " + session.getAuth().toString());
171                 session.setStatus(100);
172                 endInitRequestInError(channel,
173                         ErrorCode.ServerOverloaded, null,
174                         new OpenR66ProtocolNotYetConnectionException(
175                                 "All new Request blocked"), packet);
176                 session.setStatus(100);
177                 return;
178             }
179             if (Configuration.configuration.getConstraintLimitHandler().checkConstraints()) {
180                 if (Configuration.configuration.getR66Mib() != null) {
181                     Configuration.configuration.getR66Mib().
182                             notifyOverloaded("Rule: " + packet.getRulename() + " from "
183                                     + session.getAuth().toString(),
184                                     Configuration.configuration.getConstraintLimitHandler().lastAlert);
185                 }
186                 logger.warn(Messages.getString("LocalServerHandler.8") //$NON-NLS-1$
187                         + packet.getRulename()
188                         + " while "
189                         + Configuration.configuration.getConstraintLimitHandler().lastAlert +
190                         " from " + session.getAuth().toString());
191                 session.setStatus(100);
192                 endInitRequestInError(channel,
193                         ErrorCode.ServerOverloaded, null,
194                         new OpenR66ProtocolNotYetConnectionException(
195                                 "Limit exceeded " + Configuration.configuration.getConstraintLimitHandler().lastAlert),
196                         packet);
197                 session.setStatus(100);
198                 return;
199             }
200         } else if (packet.getCode() == ErrorCode.ServerOverloaded.code) {
201             // XXX unvalid limit on requested host received
202             logger.info("TaskRunner initialisation in error: " + ErrorCode.ServerOverloaded.mesg);
203             localChannelReference.invalidateRequest(new R66Result(
204                     null, session, true, ErrorCode.ServerOverloaded, null));
205             session.setStatus(101);
206             ChannelCloseTimer.closeFutureChannel(channel);
207             return;
208         }
209         DbRule rule;
210         try {
211             rule = new DbRule(localChannelReference.getDbSession(), packet.getRulename());
212         } catch (WaarpDatabaseException e) {
213             logger.info("Rule is unknown: " + packet.getRulename() + " {}", e.getMessage());
214             session.setStatus(49);
215             endInitRequestInError(channel,
216                     ErrorCode.QueryRemotelyUnknown, null,
217                     new OpenR66ProtocolBusinessException(
218                             Messages.getString("LocalServerHandler.9") + //$NON-NLS-1$
219                                     packet.getRulename()), packet);
220             return;
221         }
222         int blocksize = packet.getBlocksize();
223         if (packet.isToValidate()) {
224             if (!rule.checkHostAllow(session.getAuth().getUser())) {
225                 session.setStatus(30);
226                 throw new OpenR66ProtocolNotAuthenticatedException(
227                         Messages.getString("LocalServerHandler.10")); //$NON-NLS-1$
228             }
229             // Check if the blocksize is greater than local value
230             if (Configuration.configuration.getBLOCKSIZE() < blocksize) {
231                 blocksize = Configuration.configuration.getBLOCKSIZE();
232                 String sep = localChannelReference.getPartner().getSeperator();
233                 packet = new RequestPacket(packet.getRulename(), packet.getMode(),
234                         packet.getFilename(), blocksize, packet.getRank(),
235                         packet.getSpecialId(), packet.getFileInformation(), packet.getOriginalSize(), sep);
236             }
237         }
238         if (!RequestPacket.isCompatibleMode(rule.getMode(), packet.getMode())) {
239             // not compatible Rule and mode in request
240             throw new OpenR66ProtocolNotAuthenticatedException(
241                     Messages.getString("LocalServerHandler.12") + rule.getMode() + " vs " //$NON-NLS-1$
242                             + packet.getMode());
243         }
244         session.setBlockSize(blocksize);
245         DbTaskRunner runner;
246         // requested
247         boolean isRetrieve = DbTaskRunner.getSenderByRequestPacket(packet);
248         if (packet.getSpecialId() != DbConstant.ILLEGALVALUE) {
249             // Reload or create
250             String requested = DbTaskRunner.getRequested(session, packet);
251             String requester = DbTaskRunner.getRequester(session, packet);
252             logger.debug("DEBUG: " + packet.getSpecialId() + ":" + isRetrieve);
253             if (packet.isToValidate()) {
254                 // Id could be a creation or a reload
255                 // Try reload
256                 try {
257                     runner = new DbTaskRunner(localChannelReference.getDbSession(),
258                             session, rule, packet.getSpecialId(),
259                             requester, requested);
260                     // Patch to prevent self request to be stored by sender
261                     boolean ignoreSave = runner.shallIgnoreSave();
262                     runner.setSender(isRetrieve);
263                     logger.debug("DEBUG: " + runner.getSpecialId() + ":" + ignoreSave + ":" + runner.shallIgnoreSave()
264                             + ":" + isRetrieve);
265                     if (ignoreSave && !runner.shallIgnoreSave() && !runner.checkFromDbForSubmit()) {
266                         // Since status changed, it means that object should be created and not reloaded
267                         // But in case of submit, item already exist so shall be loaded from database
268                         throw new WaarpDatabaseNoDataException("False load, must reopen and create DbTaskRunner");
269                     }
270                 } catch (WaarpDatabaseNoDataException e) {
271                     // Reception of request from requester host
272                     try {
273                         runner = new DbTaskRunner(localChannelReference.getDbSession(),
274                                 session, rule, isRetrieve, packet);
275                         logger.debug("Runner before any action: {} {}", runner.shallIgnoreSave(), runner);
276                     } catch (WaarpDatabaseException e1) {
277                         session.setStatus(33);
278                         endInitRequestInError(channel, ErrorCode.QueryRemotelyUnknown,
279                                 null, new OpenR66DatabaseGlobalException(e), packet);
280                         return;
281                     }
282                 } catch (WaarpDatabaseException e) {
283                     session.setStatus(34);
284                     endInitRequestInError(channel, ErrorCode.QueryRemotelyUnknown, null,
285                             new OpenR66DatabaseGlobalException(e), packet);
286                     return;
287                 }
288                 if (runner.isAllDone()) {
289                     // truly an error since done
290                     session.setStatus(31);
291                     endInitRequestInError(channel,
292                             ErrorCode.QueryAlreadyFinished, runner,
293                             new OpenR66ProtocolBusinessQueryAlreadyFinishedException(
294                                     Messages.getString("LocalServerHandler.13") //$NON-NLS-1$
295                                             +
296                                             packet.getSpecialId()), packet);
297                     return;
298                 }
299                 LocalChannelReference lcr =
300                         Configuration.configuration.getLocalTransaction().
301                                 getFromRequest(
302                                         requested + " " + requester + " "
303                                                 + packet.getSpecialId());
304                 if (lcr != null) {
305                     // truly an error since still running
306                     session.setStatus(32);
307                     endInitRequestInError(channel,
308                             ErrorCode.QueryStillRunning, runner,
309                             new OpenR66ProtocolBusinessQueryStillRunningException(
310                                     Messages.getString("LocalServerHandler.14") //$NON-NLS-1$
311                                             +
312                                             packet.getSpecialId()), packet);
313                     return;
314                 }
315                 logger.debug("Runner before any action: {} {}", runner.shallIgnoreSave(), runner);
316                 // ok to restart
317                 try {
318                     if (runner.restart(false)) {
319                         runner.saveStatus();
320                     }
321                 } catch (OpenR66RunnerErrorException e) {
322                 }
323                 // Change the SpecialID! => could generate an error ?
324                 packet.setSpecialId(runner.getSpecialId());
325             } else {
326                 // Id should be a reload
327                 try {
328                     runner = new DbTaskRunner(localChannelReference.getDbSession(),
329                             session, rule, packet.getSpecialId(),
330                             requester, requested);
331                 } catch (WaarpDatabaseException e) {
332                     if (localChannelReference.getDbSession() == null) {
333                         // Special case of no database client
334                         try {
335                             runner = new DbTaskRunner(localChannelReference.getDbSession(),
336                                     session, rule, isRetrieve, packet);
337                             logger.debug("Runner before any action: {} {}", runner.shallIgnoreSave(), runner);
338                         } catch (WaarpDatabaseException e1) {
339                             session.setStatus(35);
340                             endInitRequestInError(channel, ErrorCode.QueryRemotelyUnknown, null,
341                                     new OpenR66DatabaseGlobalException(e1), packet);
342                             return;
343                         }
344                     } else {
345                         endInitRequestInError(channel, ErrorCode.QueryRemotelyUnknown, null,
346                                 new OpenR66DatabaseGlobalException(e), packet);
347                         session.setStatus(36);
348                         return;
349                     }
350                 }
351                 runner.setSender(isRetrieve);
352                 // FIX check for SelfRequest
353                 if (runner.isSelfRequest()) {
354                     runner.setFilename(runner.getOriginalFilename());
355                 }
356                 if (!runner.isSender()) {
357                     logger.debug("New filename ? :" + packet.getFilename());
358                     runner.setOriginalFilename(packet.getFilename());
359                     if (runner.getRank() == 0) {
360                         runner.setFilename(packet.getFilename());
361                     }
362                 }
363                 logger.debug("Runner before any action: {} {}", runner.shallIgnoreSave(), runner);
364                 try {
365                     if (runner.restart(false)) {
366                         if (!runner.isSelfRequest()) {
367                             runner.saveStatus();
368                         }
369                     }
370                 } catch (OpenR66RunnerErrorException e) {
371                 }
372             }
373         } else {
374             // Very new request
375             // should not be the case (the requester should always set the id)
376             logger.error("NO TransferID specified: SHOULD NOT BE THE CASE");
377             try {
378                 runner = new DbTaskRunner(localChannelReference.getDbSession(),
379                         session, rule, isRetrieve, packet);
380             } catch (WaarpDatabaseException e) {
381                 session.setStatus(37);
382                 endInitRequestInError(channel, ErrorCode.QueryRemotelyUnknown, null,
383                         new OpenR66DatabaseGlobalException(e), packet);
384                 return;
385             }
386             packet.setSpecialId(runner.getSpecialId());
387         }
388         logger.debug("Runner before any action: {} {}", runner.shallIgnoreSave(), runner);
389         // Check now if request is a valid one
390         if (packet.getCode() != ErrorCode.InitOk.code) {
391             // not valid so create an error from there
392             ErrorCode code = ErrorCode.getFromCode("" + packet.getCode());
393             session.setBadRunner(runner, code);
394             if (!runner.shallIgnoreSave()) {
395                 runner.saveStatus();
396             }
397             session.newState(ERROR);
398             logger.error("Bad runner at startup {} {}", packet, session);
399             ErrorPacket errorPacket = new ErrorPacket(code.mesg,
400                     code.getCode(), ErrorPacket.FORWARDCLOSECODE);
401             errorMesg(channel, errorPacket);
402             return;
403         }
404         // Receiver can specify a rank different from database
405         if (runner.isSender()) {
406             logger.debug("Rank was: " + runner.getRank() + " -> " + packet.getRank());
407             runner.setRankAtStartup(packet.getRank());
408         } else {
409             if (runner.getRank() > packet.getRank()) {
410                 logger.debug("Recv Rank was: " + runner.getRank() + " -> " + packet.getRank());
411                 // if receiver, change only if current rank is upper proposed rank
412                 runner.setRankAtStartup(packet.getRank());
413             }
414             if (packet.getOriginalSize() > 0) {
415                 runner.setOriginalSize(packet.getOriginalSize());
416             }
417         }
418         logger.debug("Filesize: " + packet.getOriginalSize() + ":" + runner.isSender());
419         boolean shouldInformBack = false;
420         try {
421             session.setRunner(runner);
422             // Fix to ensure that recv request are not trying to access to not chroot files
423             if (Configuration.configuration.isChrootChecked() && packet.isToValidate() && runner.isSender()) {
424                 session.startup(true);
425             } else {
426                 session.startup(false);
427             }
428             if (runner.isSender() && !runner.isSendThrough()) {
429                 if (packet.getOriginalSize() != runner.getOriginalSize()) {
430                     packet.setOriginalSize(runner.getOriginalSize());
431                     shouldInformBack = true;
432                     logger.debug("Filesize2: " + packet.getOriginalSize() + ":" + runner.isSender());
433                 }
434             }
435         } catch (OpenR66RunnerErrorException e) {
436             try {
437                 runner.saveStatus();
438             } catch (OpenR66RunnerErrorException e1) {
439                 logger.error("Cannot save Status: " + runner, e1);
440             }
441             if (runner.getErrorInfo() == ErrorCode.InitOk ||
442                     runner.getErrorInfo() == ErrorCode.PreProcessingOk ||
443                     runner.getErrorInfo() == ErrorCode.TransferOk) {
444                 runner.setErrorExecutionStatus(ErrorCode.ExternalOp);
445             }
446             logger.error("PreTask in error {}", e.getMessage(), e);
447             errorToSend("PreTask in error: " + e.getMessage(), runner.getErrorInfo(), channel, 38);
448             return;
449         }
450         logger.debug("Filesize: " + packet.getOriginalSize() + ":" + runner.isSender());
451         if (! shouldInformBack) {
452             shouldInformBack = ! packet.getFileInformation().equals(runner.getFileInformation());
453         }
454         if (runner.isFileMoved() && runner.isSender() && runner.isInTransfer()
455                 && runner.getRank() == 0 && (!packet.isToValidate())) {
456             // File was moved during PreTask and very beginning of the transfer
457             // and the remote host has already received the request packet
458             // => Informs the receiver of the new name
459             sendFilenameFilesizeChanging(packet, runner, "Will send a modification of filename due to pretask: ",
460                     "Change Filename by Pre action on sender");
461         } else if ((!packet.getFilename().equals(runner.getOriginalFilename()))
462                 && runner.isSender() && runner.isInTransfer()
463                 && runner.getRank() == 0 && (!packet.isToValidate())) {
464             // File was modify at the very beginning (using wildcards)
465             // and the remote host has already received the request packet
466             // => Informs the receiver of the new name
467             sendFilenameFilesizeChanging(packet, runner, "Will send a modification of filename due to wildcard: ",
468                     "Change Filename by Wildcard on sender");
469         } else if (runner.isSelfRequest() && runner.isSender() && runner.isInTransfer()
470                 && runner.getRank() == 0 && (!packet.isToValidate())) {
471             // FIX SelfRequest
472             // File could be modified at the very beginning (using wildcards)
473             // and the remote host has already received the request packet
474             // => Informs the receiver of the new name
475             sendFilenameFilesizeChanging(packet, runner,
476                     "Will send a modification of filename due to wildcard in SelfMode: ",
477                     "Change Filename by Wildcard on sender in SelfMode");
478         } else if (shouldInformBack && (!packet.isToValidate())) {
479             // Was only for (shouldInformBack)
480             // File length is now known, so inform back
481             sendFilenameFilesizeChanging(packet, runner, "Will send a modification of filesize or fileInfo: ",
482                     "Change Filesize / FileInfo on sender");
483         }
484         session.setReady(true);
485         Configuration.configuration.getLocalTransaction().setFromId(runner, localChannelReference);
486         // inform back
487         if (packet.isToValidate()) {
488             if (Configuration.configuration.getMonitoring() != null) {
489                 Configuration.configuration.getMonitoring().lastInActiveTransfer =
490                         System.currentTimeMillis();
491             }
492             if (runner.isSender()) {
493                 // In case Wildcard was used
494                 logger.debug("New FILENAME: {}", runner.getOriginalFilename());
495                 packet.setFilename(runner.getOriginalFilename());
496                 logger.debug("Rank set: " + runner.getRank());
497                 packet.setRank(runner.getRank());
498             } else {
499                 logger.debug("Rank set: " + runner.getRank());
500                 packet.setRank(runner.getRank());
501             }
502             packet.validate();
503             session.newState(REQUESTD);
504             ChannelUtils.writeAbstractLocalPacket(localChannelReference, packet, true);
505         } else {
506             session.newState(REQUESTD);
507             // requester => might be a client
508             // Save the runner into the session and validate the request so begin transfer
509             session.getLocalChannelReference().getFutureRequest().setRunner(runner);
510             localChannelReference.getFutureValidRequest().setSuccess();
511             if (Configuration.configuration.getMonitoring() != null) {
512                 Configuration.configuration.getMonitoring().lastOutActiveTransfer =
513                         System.currentTimeMillis();
514             }
515         }
516         // if retrieve => START the retrieve operation except if in Send Through mode
517         if (runner.isSender()) {
518             if (runner.isSendThrough()) {
519                 // it is legal to send data from now
520                 logger.debug("Now ready to continue with send through");
521                 localChannelReference.validateEndTransfer(
522                         new R66Result(session, false, ErrorCode.PreProcessingOk, runner));
523             } else {
524                 // Automatically send data now
525                 logger.debug("Now ready to continue with runRetrieve");
526                 NetworkTransaction.runRetrieve(session, channel);
527             }
528         }
529         session.setStatus(39);
530     }

532     /**
533      * Send a Filename/Filesize change to the partner
534      * 
535      * @param packet
536      * @param runner
537      * @throws OpenR66ProtocolPacketException
538      */
539     private final void sendFilenameFilesizeChanging(RequestPacket packet, DbTaskRunner runner, String debug, String info)
540             throws OpenR66ProtocolPacketException {
541         logger.debug(debug + runner.getFilename());
542         session.newState(VALID);
543         if (localChannelReference.getPartner().useJson()) {
544             RequestJsonPacket request = new RequestJsonPacket();
545             request.setComment(info);
546             request.setFilename(runner.getFilename());
547             request.setFilesize(packet.getOriginalSize());
548             String infoTransfer = runner.getFileInformation();
549             if (infoTransfer != null && ! infoTransfer.equals(packet.getFileInformation())) {
550                 request.setFileInfo(runner.getFileInformation());
551             }
552             JsonCommandPacket validPacket = new JsonCommandPacket(request,
553                     LocalPacketFactory.REQUESTPACKET);
554             ChannelUtils.writeAbstractLocalPacket(localChannelReference,
555                     validPacket, true);
556         } else {
557             String infoTransfer = runner.getFileInformation();
558             ValidPacket validPacket;
559             if (infoTransfer != null && ! infoTransfer.equals(packet.getFileInformation())
560                     && localChannelReference.getPartner().changeFileInfoEnabled()) {
561                 validPacket = new ValidPacket(info,
562                     runner.getFilename() + PartnerConfiguration.BAR_SEPARATOR_FIELD + packet.getOriginalSize()
563                     + PartnerConfiguration.BAR_SEPARATOR_FIELD + packet.getFileInformation(),
564                     LocalPacketFactory.REQUESTPACKET);
565             } else {
566                 validPacket = new ValidPacket(info,
567                         runner.getFilename() + PartnerConfiguration.BAR_SEPARATOR_FIELD + packet.getOriginalSize(),
568                         LocalPacketFactory.REQUESTPACKET);
569             }
570             ChannelUtils.writeAbstractLocalPacket(localChannelReference,
571                     validPacket, true);
572         }
573     }

578      * @param message
579      * @param code
580      * @param channel
581      * @throws OpenR66ProtocolPacketException
582      */
583     private final void errorToSend(String message, ErrorCode code, Channel channel, int status)
584             throws OpenR66ProtocolPacketException {
585         session.newState(ERROR);
586         try {
587             session.setFinalizeTransfer(false, new R66Result(
588                     new OpenR66ProtocolPacketException(message), session, true,
589                     code, session.getRunner()));
590         } catch (OpenR66RunnerErrorException e1) {
591             localChannelReference.invalidateRequest(new R66Result(e1, session,
592                     true, code, session.getRunner()));
593         } catch (OpenR66ProtocolSystemException e1) {
594             localChannelReference.invalidateRequest(new R66Result(e1, session,
595                     true, code, session.getRunner()));
596         }
597         ErrorPacket error = new ErrorPacket(
598                 message,
599                 code.getCode(), ErrorPacket.FORWARDCLOSECODE);
600         ChannelUtils.writeAbstractLocalPacket(localChannelReference, error, true);
601         session.setStatus(status);
602         ChannelCloseTimer.closeFutureChannel(channel);
603     }

605     /**
606      * Receive a data block
607      * 
608      * @param channel
609      * @param packet
610      * @throws OpenR66ProtocolNotAuthenticatedException
611      * @throws OpenR66ProtocolBusinessException
612      * @throws OpenR66ProtocolPacketException
613      */
614     public void data(Channel channel, DataPacket packet)
615             throws OpenR66ProtocolNotAuthenticatedException,
616             OpenR66ProtocolBusinessException, OpenR66ProtocolPacketException {
617         if (!session.isAuthenticated()) {
618             logger.debug("Not authenticated while Data received");
619             packet.clear();
620             throw new OpenR66ProtocolNotAuthenticatedException(
621                     "Not authenticated while Data received");
622         }
623         if (!session.isReady()) {
624             logger.debug("No request prepared");
625             packet.clear();
626             throw new OpenR66ProtocolBusinessException("No request prepared");
627         }
628         if (session.getRunner().isSender()) {
629             logger.debug("Not in receive MODE but receive a packet");
630             packet.clear();
631             throw new OpenR66ProtocolBusinessException(
632                     "Not in receive MODE but receive a packet");
633         }
634         if (!session.getRunner().continueTransfer()) {
635             logger.debug("EndTransfer failed ? " + localChannelReference.getFutureEndTransfer().isFailed());
636             if (localChannelReference.getFutureEndTransfer().isFailed()) {
637                 // nothing to do since already done
638                 session.setStatus(94);
639                 packet.clear();
640                 return;
641             }
642             errorToSend("Transfer in error due previously aborted transmission", ErrorCode.TransferError, channel, 95);
643             packet.clear();
644             return;
645         }
646         if (packet.getPacketRank() != session.getRunner().getRank()) {
647             logger.debug("Issue on rank: " + packet.getPacketRank() + ":" + session.getRunner().getRank());
648             if (!session.addError()) {
649                 // cannot continue
650                 logger.error(Messages.getString("LocalServerHandler.15") + packet.getPacketRank() + " : " + //$NON-NLS-1$
651                         session.getRunner().getRank() + " from {}", session.getRunner());
652                 errorToSend("Too much Bad Rank in transmission: " +
653                         packet.getPacketRank(), ErrorCode.TransferError, channel, 96);
654                 packet.clear();
655                 return;
656             }
657             // Fix the rank if possible
658             if (packet.getPacketRank() < session.getRunner().getRank()) {
659                 logger.debug("Bad RANK: " + packet.getPacketRank() + " : " +
660                         session.getRunner().getRank());
661                 session.getRunner().setRankAtStartup(packet.getPacketRank());
662                 session.getRestart().restartMarker(
663                         session.getRunner().getBlocksize() *
664                                 session.getRunner().getRank());
665                 try {
666                     session.getFile().restartMarker(session.getRestart());
667                 } catch (CommandAbstractException e) {
668                     logger.error("Bad RANK: " + packet.getPacketRank() + " : " +
669                             session.getRunner().getRank());
670                     errorToSend("Bad Rank in transmission even after retry: " +
671                             packet.getPacketRank(), ErrorCode.TransferError, channel, 96);
672                     packet.clear();
673                     return;
674                 }
675             } else {
676                 // really bad
677                 logger.error("Bad RANK: " + packet.getPacketRank() + " : " +
678                         session.getRunner().getRank());
679                 errorToSend("Bad Rank in transmission: " +
680                         packet.getPacketRank() + " > " +
681                         session.getRunner().getRank(), ErrorCode.TransferError, channel, 20);
682                 packet.clear();
683                 return;
684             }
685         }
686         // Check global size
687         long originalSize = session.getRunner().getOriginalSize();
688         if (originalSize >= 0) {
689             if (session.getRunner().getBlocksize() * (session.getRunner().getRank() - 1) > originalSize) {
690                 // cannot continue
691                 logger.error(Messages.getString("LocalServerHandler.16") + packet.getPacketRank() + " : " + //$NON-NLS-1$
692                         (originalSize / session.getRunner().getBlocksize() + 1) + " from {}", session.getRunner());
693                 errorToSend("Too much data transferred: " +
694                         packet.getPacketRank(), ErrorCode.TransferError, channel, 96);
695                 packet.clear();
696                 return;
697             }
698         }
699         // if MD5 check MD5
700         if (RequestPacket.isMD5Mode(session.getRunner().getMode())) {
701             logger.debug("AlgoDigest: "
702                     + (localChannelReference.getPartner() != null ? localChannelReference.getPartner().getDigestAlgo()
703                             : "usual algo"));
704             if (!packet.isKeyValid(localChannelReference.getPartner().getDigestAlgo())) {
705                 // Wrong packet
706                 logger.error(
707                         Messages.getString("LocalServerHandler.17"), packet, localChannelReference.getPartner().getDigestAlgo().name); //$NON-NLS-1$
708                 errorToSend("Transfer in error due to bad Hash on data packet ("
709                         + localChannelReference.getPartner().getDigestAlgo().name + ")",
710                         ErrorCode.MD5Error, channel, 21);
711                 packet.clear();
712                 return;
713             }
714         }
715         if (Configuration.configuration.isGlobalDigest()) {
716             if (globalDigest == null) {
717                 try {
718                     // check if first block, since if not, digest will be only partial
719                     if (session.getRunner().getRank() > 0) {
720                         localChannelReference.setPartialHash();
721                     }
722                     if (localChannelReference.getPartner() != null) {
723                         if (localChannelReference.getPartner().useFinalHash()) {
724                             DigestAlgo algo = localChannelReference.getPartner().getDigestAlgo();
725                             if (algo != Configuration.configuration.getDigest()) {
726                                 globalDigest = new FilesystemBasedDigest(algo);
727                                 localDigest = new FilesystemBasedDigest(Configuration.configuration.getDigest());
728                             }
729                         }
730                     }
731                     if (globalDigest == null) {
732                         globalDigest = new FilesystemBasedDigest(Configuration.configuration.getDigest());
733                         localDigest = null;
734                     }
735                 } catch (NoSuchAlgorithmException e) {
736                 }
737                 logger.debug("GlobalDigest: " + localChannelReference.getPartner().getDigestAlgo() + " different? "
738                         + (localDigest != null));
739             }
740             FileUtils.computeGlobalHash(globalDigest, packet.getData());
741             if (localDigest != null) {
742                 FileUtils.computeGlobalHash(localDigest, packet.getData());
743             }
744         }
745         DataBlock dataBlock = new DataBlock();
746         if (session.getRunner().isRecvThrough() && localChannelReference.isRecvThroughMode()) {
747             try {
748                 localChannelReference.getRecvThroughHandler().writeByteBuf(packet.getData());
749                 session.getRunner().incrementRank();
750                 if (packet.getPacketRank() % 100 == 1) {
751                     logger.debug("Good RANK: " + packet.getPacketRank() + " : " +
752                             session.getRunner().getRank());
753                 }
754             } finally {
755                 packet.clear();
756             }
757         } else {
758             dataBlock.setBlock(packet.getData());
759             try {
760                 session.getFile().writeDataBlock(dataBlock);
761                 session.getRunner().incrementRank();
762                 if (packet.getPacketRank() % 100 == 1) {
763                     logger.debug("Good RANK: " + packet.getPacketRank() + " : " +
764                             session.getRunner().getRank());
765                 }
766             } catch (FileTransferException e) {
767                 errorToSend("Transfer in error",
768                         ErrorCode.TransferError, channel, 22);
769                 return;
770             } finally {
771                 dataBlock.clear();
772                 packet.clear();
773             }
774         }
775     }

779      *

788             OpenR66ProtocolNotAuthenticatedException {
789         if (!session.isAuthenticated()) {
790             throw new OpenR66ProtocolNotAuthenticatedException(
791                     "Not authenticated while EndTransfer received");
792         }
793         // Check end of transfer
794         long originalSize = session.getRunner().getOriginalSize();
795         logger.debug("OSize: " + originalSize + " isSender: " + session.getRunner().isSender());
796         if (packet.isToValidate()) {
797             // check if possible originalSize
798             if (originalSize > 0) {
799                 try {
800                     if (!session.getRunner().isRecvThrough() && session.getFile().length() != originalSize ||
801                             session.getFile().length() == 0) {
802                         R66Result result = new R66Result(new OpenR66RunnerErrorException(
803                                 Messages.getString("LocalServerHandler.18")), //$NON-NLS-1$
804                                 session, true, ErrorCode.TransferError, session.getRunner());
805                         try {
806                             session.setFinalizeTransfer(false, result);
807                         } catch (OpenR66RunnerErrorException e) {
808                         } catch (OpenR66ProtocolSystemException e) {
809                         }
810                         ErrorPacket error = new ErrorPacket(
811                                 "Final size in error, transfer in error and rank should be reset to 0",
812                                 ErrorCode.TransferError.getCode(), ErrorPacket.FORWARDCLOSECODE);
813                         try {
814                             ChannelUtils.writeAbstractLocalPacket(localChannelReference, error, true);
815                         } catch (OpenR66ProtocolPacketException e) {
816                         }
817                         session.setStatus(23);
818                         ChannelCloseTimer.closeFutureChannel(channel);
819                         return;

821                 } catch (CommandAbstractException e) {
822                     // ignore
823                 }
824             }
825             // check if possible Global Digest
826             String hash = packet.getOptional();
827             logger.debug("GlobalDigest: " + localChannelReference.getPartner().getDigestAlgo() + " different? "
828                     + (localDigest != null) + " remoteHash? " + (hash != null));
829             if (hash != null && globalDigest != null) {
830                 String localhash = FilesystemBasedDigest.getHex(globalDigest.Final());
831                 globalDigest = null;
832                 if (!localhash.equalsIgnoreCase(hash)) {
833                     // bad global Hash
834                     //session.getRunner().setRankAtStartup(0);
835                     R66Result result = new R66Result(new OpenR66RunnerErrorException(
836                             Messages.getString("LocalServerHandler.19") + //$NON-NLS-1$
837                                     localChannelReference.getPartner().getDigestAlgo().name + ")"),
838                             session, true, ErrorCode.MD5Error, session.getRunner());
839                     try {
840                         session.setFinalizeTransfer(false, result);
841                     } catch (OpenR66RunnerErrorException e) {
842                     } catch (OpenR66ProtocolSystemException e) {
843                     }
844                     ErrorPacket error = new ErrorPacket(
845                             "Global Hash in error, transfer in error and rank should be reset to 0 (using " +
846                                     localChannelReference.getPartner().getDigestAlgo().name + ")",
847                             ErrorCode.MD5Error.getCode(), ErrorPacket.FORWARDCLOSECODE);
848                     try {
849                         ChannelUtils.writeAbstractLocalPacket(localChannelReference, error, true);
850                     } catch (OpenR66ProtocolPacketException e) {
851                     }
852                     session.setStatus(23);
853                     ChannelCloseTimer.closeFutureChannel(channel);
854                     return;
855                 } else {
856                     if (localDigest != null) {
857                         localhash = FilesystemBasedDigest.getHex(localDigest.Final());
858                     }
859                     localChannelReference.setHashComputeDuringTransfer(localhash);
860                     logger.debug("Global digest ok");
861                 }
862             } else if (globalDigest != null) {
863                 String localhash = null;
864                 if (localDigest != null) {
865                     localhash = FilesystemBasedDigest.getHex(localDigest.Final());
866                 } else {
867                     localhash = FilesystemBasedDigest.getHex(globalDigest.Final());
868                 }
869                 globalDigest = null;
870                 localChannelReference.setHashComputeDuringTransfer(localhash);
871             }
872             localDigest = null;
873             globalDigest = null;
874             session.newState(ENDTRANSFERS);
875             if (!localChannelReference.getFutureRequest().isDone()) {
876                 // Finish with post Operation
877                 R66Result result = new R66Result(session, false,
878                         ErrorCode.TransferOk, session.getRunner());
879                 session.newState(ENDTRANSFERR);
880                 try {
881                     session.setFinalizeTransfer(true, result);
882                 } catch (OpenR66RunnerErrorException e) {
883                     // TODO
884                     session.newState(ERROR);
885                     ErrorPacket error = null;
886                     if (localChannelReference.getFutureRequest().getResult() != null) {
887                         result = localChannelReference.getFutureRequest().getResult();
888                         error = new ErrorPacket(
889                                 "Error while finalizing transfer: " + result.getMessage(),
890                                 result.getCode().getCode(), ErrorPacket.FORWARDCLOSECODE);
891                     } else {
892                         error = new ErrorPacket(
893                                 "Error while finalizing transfer",
894                                 ErrorCode.FinalOp.getCode(), ErrorPacket.FORWARDCLOSECODE);
895                     }
896                     try {
897                         ChannelUtils.writeAbstractLocalPacket(localChannelReference, error, true);
898                     } catch (OpenR66ProtocolPacketException e1) {
899                     }
900                     session.setStatus(23);
901                     ChannelCloseTimer.closeFutureChannel(channel);
902                     return;
903                 } catch (OpenR66ProtocolSystemException e) {
904                     // TODO
905                     session.newState(ERROR);
906                     ErrorPacket error = null;

908                         result = localChannelReference.getFutureRequest().getResult();

929                             packet, false);
930                 } catch (OpenR66ProtocolPacketException e) {
931                     // ignore
932                 }
933             } else {

939                 return;

954                         result = localChannelReference.getFutureRequest().getResult();
955                         error = new ErrorPacket(
956                                 "Error while finalizing transfer: " + result.getMessage(),
957                                 result.getCode().getCode(), ErrorPacket.FORWARDCLOSECODE);
958                     } else {
959                         error = new ErrorPacket(
960                                 "Error while finalizing transfer",
961                                 ErrorCode.FinalOp.getCode(), ErrorPacket.FORWARDCLOSECODE);
962                     }
963                     try {
964                         ChannelUtils.writeAbstractLocalPacket(localChannelReference, error, true);
965                     } catch (OpenR66ProtocolPacketException e1) {
966                     }
967                     session.setStatus(23);

996     /**
997      * Receive an End of Request
998      * 
999      * @param channel
1000      * @param packet
1001      * @throws OpenR66RunnerErrorException
1002      * @throws OpenR66ProtocolSystemException
1003      * @throws OpenR66ProtocolNotAuthenticatedException
1004      */
1005     public void endRequest(Channel channel, EndRequestPacket packet) {
1006         // Validate the last post action on a transfer from receiver remote host
1007         logger.info("Valid Request {} Packet {}",
1008                 localChannelReference,
1009                 packet);
1010         DbTaskRunner runner = session.getRunner();
1011         logger.debug("Runner endRequest: " + (session.getRunner() != null));
1012         if (runner != null) {
1013             runner.setAllDone();
1014             try {
1015                 runner.saveStatus();
1016             } catch (OpenR66RunnerErrorException e) {
1017                 // ignore
1018             }
1019             runner.clean();
1020         }
1021         String optional = null;
1022         if (session.getExtendedProtocol()) {
1023             optional = packet.getOptional();
1024         }
1025         if (!localChannelReference.getFutureRequest().isDone()) {
1026             // end of request
1027             R66Future transfer = localChannelReference.getFutureEndTransfer();
1028             try {
1029                 transfer.await();
1030             } catch (InterruptedException e) {
1031             }
1032             if (transfer.isSuccess()) {
1033                 if (session.getExtendedProtocol() && session.getBusinessObject() != null) {
1034                     if (session.getBusinessObject().getInfo(session) == null) {
1035                         session.getBusinessObject().setInfo(session, optional);
1036                     } else {
1037                         String temp = session.getBusinessObject().getInfo(session);
1038                         session.getBusinessObject().setInfo(session, optional);
1039                         optional = temp;
1040                     }
1041                 } else if (session.getExtendedProtocol() &&
1042                         transfer.getResult().getOther() == null && optional != null) {
1043                     transfer.getResult().setOther(optional);
1044                 }
1045                 localChannelReference.validateRequest(transfer.getResult());
1046             }
1047         }
1048         session.setStatus(1);
1049         if (packet.isToValidate()) {
1050             session.newState(ENDREQUESTS);
1051             packet.validate();
1052             if (session.getExtendedProtocol()) {
1053                 packet.setOptional(optional);
1054             }
1055             session.newState(ENDREQUESTR);
1056             try {
1057                 ChannelUtils.writeAbstractLocalPacket(localChannelReference,
1058                         packet, true);
1059             } catch (OpenR66ProtocolPacketException e) {
1060             }
1061         } else {
1062             session.newState(ENDREQUESTR);
1063         }
1064         if (runner != null && (runner.isSelfRequested() || runner.isSelfRequest())) {
1065             ChannelCloseTimer.closeFutureChannel(channel);
1066         }
1067     }
