186 	 * (non-Javadoc)
187 	 * @see org.jboss.netty.channel.SimpleChannelHandler#channelClosed(org.jboss.
188 	 * netty.channel.ChannelHandlerContext, org.jboss.netty.channel.ChannelStateEvent)
189 	 */
190 	@Override
191 	public void channelClosed(ChannelHandlerContext ctx, ChannelStateEvent e) {
192 		logger.debug("Local Server Channel Closed: {} {}",
193 				(localChannelReference != null ? localChannelReference
194 						: "no LocalChannelReference"), (session.getRunner() != null ?
195 						session.getRunner().toShortString() : "no runner"));
196 		// clean session objects like files
197 		DbTaskRunner runner = session.getRunner();
198 		boolean mustFinalize = true;
199 		if (localChannelReference != null &&
200 				localChannelReference.getFutureRequest().isDone()) {
201 			// already done
202 		} else {
203 			if (localChannelReference != null) {
204 				R66Future fvr = localChannelReference.getFutureValidRequest();
205 				try {
206 					fvr.await(Configuration.configuration.TIMEOUTCON*2, TimeUnit.MILLISECONDS);
207 				} catch (InterruptedException e1) {
208 				}
209 				if (fvr.isDone()) {
210 					if (!fvr.isSuccess()) {
211 						// test if remote server was Overloaded
212 						if (fvr.getResult().code == ErrorCode.ServerOverloaded) {
213 							// ignore
214 							mustFinalize = false;
215 						}
216 					}
217 				}
218 				logger.debug("Must Finalize: " + mustFinalize);
219 				if (mustFinalize) {
220 					session.newState(ERROR);
221 					R66Result finalValue = new R66Result(
222 							new OpenR66ProtocolSystemException(
223 									Messages.getString("LocalServerHandler.4")), //$NON-NLS-1$
224 							session, true, ErrorCode.FinalOp, runner); // True since closed
225 					try {
226 						tryFinalizeRequest(finalValue);
227 					} catch (OpenR66Exception e2) {
228 					}
229 				}
230 			}
231 		}
232 		if (mustFinalize && runner != null) {
233 			if (runner.isSelfRequested() && localChannelReference != null) {
234 				R66Future transfer = localChannelReference.getFutureRequest();
235 				// Since requested : log
236 				R66Result result = transfer.getResult();
237 				if (transfer.isDone() && transfer.isSuccess()) {
238 					logger.info("TRANSFER REQUESTED RESULT:     SUCCESS     " +
239 							(result != null ? result.toString() : "no result"));
240 				} else {
241 					logger.error("TRANSFER REQUESTED RESULT:     FAILURE     " +
242 							(result != null ? result.toString() : "no result"));
243 				}
244 			}
245 		}
246 		session.setStatus(50);
247 		session.newState(CLOSEDCHANNEL);
248 		session.clear();
249 		session.setStatus(51);
250 		if (localChannelReference != null) {
251 			if (localChannelReference.getDbSession() != null) {
252 				localChannelReference.getDbSession().endUseConnection();
253 				logger.debug("End Use Connection");
254 			}
255 			String requester =
256 					(runner != null && runner.isSelfRequested() &&
257 					localChannelReference.getNetworkChannelObject() != null) ?
258 							runner.getRequester() : null;
259 			NetworkTransaction.removeNetworkChannel(localChannelReference
260 					.getNetworkChannel(), e.getChannel(), requester);
261 			/*
262 			 * // Only requested can has a remote client if (runner != null &&
263 			 * runner.isSelfRequested() && localChannelReference.getNetworkChannelObject() != null
264 			 * && localChannelReference.getNetworkChannelObject().count <= 0) {
265 			 * NetworkTransaction.removeClient(runner.getRequester(),
266 			 * localChannelReference.getNetworkChannelObject()); }
267 			 */
268 			session.setStatus(52);
269 			Configuration.configuration.getLocalTransaction().remove(e.getChannel());
270 		} else {
271 			logger
272 					.error("Local Server Channel Closed but no LocalChannelReference: " +
273 							e.getChannel().getId());
274 		}
275 		// Now if runner is not yet finished, finish it by force
276 		if (mustFinalize && localChannelReference != null
277 				&& (!localChannelReference.getFutureRequest().isDone())) {
278 			R66Result finalValue = new R66Result(
279 					new OpenR66ProtocolSystemException(
280 							Messages.getString("LocalServerHandler.11")), //$NON-NLS-1$
281 					session, true, ErrorCode.FinalOp, runner);
282 			localChannelReference.invalidateRequest(finalValue);
283 			// In case stop the attached thread if any
284 			ClientRunner clientRunner = localChannelReference.getClientRunner();
285 			if (clientRunner != null) {
286 				try {
287 					Thread.sleep(Configuration.WAITFORNETOP);
288 				} catch (InterruptedException e1) {
289 				}
290 				clientRunner.interrupt();
291 			}
292 		}
293 		if (localChannelReference != null) {
294 			localChannelReference.close();
295 		}
296 	}

310 	 * (non-Javadoc)
311 	 * @see org.jboss.netty.channel.SimpleChannelHandler#messageReceived(org.jboss
312 	 * .netty.channel.ChannelHandlerContext, org.jboss.netty.channel.MessageEvent)
313 	 */
314 	@Override
315 	public void messageReceived(ChannelHandlerContext ctx, MessageEvent e)
316 			throws OpenR66Exception {
317 		// action as requested and answer if necessary
318 		final AbstractLocalPacket packet = (AbstractLocalPacket) e.getMessage();
319 		if (packet.getType() == LocalPacketFactory.STARTUPPACKET) {
320 			startup(e.getChannel(), (StartupPacket) packet);
321 		} else {
322 			if (localChannelReference == null) {
323 				logger.error("No LocalChannelReference at " +
324 						packet.getClass().getName());
325 				session.newState(ERROR);
326 				final ErrorPacket errorPacket = new ErrorPacket(
327 						"No LocalChannelReference at " +
328 								packet.getClass().getName(),
329 						ErrorCode.ConnectionImpossible.getCode(),
330 						ErrorPacket.FORWARDCLOSECODE);
331 				try {
332 					Channels.write(e.getChannel(), errorPacket).await();
333 				} catch (InterruptedException e1) {
334 				}
335 				ChannelUtils.close(e.getChannel());
336 				if (Configuration.configuration.r66Mib != null) {
337 					Configuration.configuration.r66Mib.notifyWarning(
338 							"No LocalChannelReference", packet.getClass().getSimpleName());
339 				}
340 				return;
341 			}
342 			switch (packet.getType()) {
343 				case LocalPacketFactory.AUTHENTPACKET: {
344 					authent(e.getChannel(), (AuthentPacket) packet);
345 					break;
346 				}
347 				// Already done case LocalPacketFactory.STARTUPPACKET:
348 				case LocalPacketFactory.DATAPACKET: {
349 					session.newState(DATAR);
350 					logger.debug("DATA RANK: " + ((DataPacket) packet).getPacketRank() + " : " +
351 							session.getRunner().getRank());
352 					data(e.getChannel(), (DataPacket) packet);
353 					break;
354 				}
355 				case LocalPacketFactory.VALIDPACKET: {
356 					valid(e.getChannel(), (ValidPacket) packet);
357 					break;
358 				}
359 				case LocalPacketFactory.ERRORPACKET: {
360 					session.newState(ERROR);
361 					errorMesg(e.getChannel(), (ErrorPacket) packet);
362 					break;
363 				}
364 				case LocalPacketFactory.CONNECTERRORPACKET: {
365 					connectionError(e.getChannel(),
366 							(ConnectionErrorPacket) packet);
367 					break;
368 				}
369 				case LocalPacketFactory.REQUESTPACKET: {
370 					request(e.getChannel(), (RequestPacket) packet);
371 					break;
372 				}
373 				case LocalPacketFactory.SHUTDOWNPACKET: {
374 					session.newState(SHUTDOWN);
375 					shutdown(e.getChannel(), (ShutdownPacket) packet);
376 					break;
377 				}
378 				case LocalPacketFactory.STOPPACKET:
379 				case LocalPacketFactory.CANCELPACKET:
380 				case LocalPacketFactory.CONFIMPORTPACKET:
381 				case LocalPacketFactory.CONFEXPORTPACKET:
382 				case LocalPacketFactory.BANDWIDTHPACKET: {
383 					logger.error("Unimplemented Mesg: " +
384 							packet.getClass().getName());
385 					session.newState(ERROR);
386 					localChannelReference.invalidateRequest(new R66Result(
387 							new OpenR66ProtocolSystemException(
388 									"Not implemented"), session, true,
389 							ErrorCode.Unimplemented, null));
390 					final ErrorPacket errorPacket = new ErrorPacket(
391 							"Unimplemented Mesg: " +
392 									packet.getClass().getName(),
393 							ErrorCode.Unimplemented.getCode(),
394 							ErrorPacket.FORWARDCLOSECODE);
395 					ChannelUtils.writeAbstractLocalPacket(localChannelReference, errorPacket, true);
396 					ChannelUtils.close(e.getChannel());
397 					break;
398 				}
399 				case LocalPacketFactory.TESTPACKET: {
400 					session.newState(TEST);
401 					test(e.getChannel(), (TestPacket) packet);
402 					break;
403 				}
404 				case LocalPacketFactory.ENDTRANSFERPACKET: {
405 					endTransfer(e.getChannel(), (EndTransferPacket) packet);
406 					break;
407 				}
408 				case LocalPacketFactory.INFORMATIONPACKET: {
409 					session.newState(INFORMATION);
410 					information(e.getChannel(), (InformationPacket) packet);
411 					break;
412 				}
413 				case LocalPacketFactory.ENDREQUESTPACKET: {
414 					endRequest(e.getChannel(), (EndRequestPacket) packet);
415 					break;
416 				}
417 				case LocalPacketFactory.BUSINESSREQUESTPACKET: {
418 					businessRequest(e.getChannel(), (BusinessRequestPacket) packet);
419 					break;
420 				}
421 				case LocalPacketFactory.BLOCKREQUESTPACKET: {
422 					blockRequest(e.getChannel(), (BlockRequestPacket) packet);
423 					break;
424 				}
425 				case LocalPacketFactory.JSONREQUESTPACKET: {
426 					jsonCommand(e.getChannel(), (JsonCommandPacket) packet);
427 					break;
428 				}
429 				default: {
430 					logger
431 							.error("Unknown Mesg: " +
432 									packet.getClass().getName());
433 					session.newState(ERROR);
434 					localChannelReference.invalidateRequest(new R66Result(
435 							new OpenR66ProtocolSystemException(
436 									"Unknown Message"), session, true,
437 							ErrorCode.Unimplemented, null));
438 					final ErrorPacket errorPacket = new ErrorPacket(
439 							"Unkown Mesg: " + packet.getClass().getName(),
440 							ErrorCode.Unimplemented.getCode(), ErrorPacket.FORWARDCLOSECODE);
441 					ChannelUtils.writeAbstractLocalPacket(localChannelReference, errorPacket, true);
442 					ChannelUtils.close(e.getChannel());
443 				}
444 			}
445 		}

448 	/*
449 	 * (non-Javadoc)
450 	 * @see org.jboss.netty.channel.SimpleChannelHandler#exceptionCaught(org.jboss
451 	 * .netty.channel.ChannelHandlerContext, org.jboss.netty.channel.ExceptionEvent)
452 	 */
453 	@Override
454 	public void exceptionCaught(ChannelHandlerContext ctx, ExceptionEvent e) {
455 		// inform clients
456 		logger.debug("Exception and isFinished: "+
457 				(localChannelReference != null && localChannelReference.getFutureRequest().isDone()), e.getCause());
458 		if (localChannelReference != null && localChannelReference.getFutureRequest().isDone()) {
459 			Channels.close(e.getChannel());
460 			return;
461 		}
462 		OpenR66Exception exception = OpenR66ExceptionTrappedFactory
463 				.getExceptionFromTrappedException(e.getChannel(), e);
464 		ErrorCode code = null;
465 		if (exception != null) {
466 			session.newState(ERROR);
467 			boolean isAnswered = false;
468 			if (exception instanceof OpenR66ProtocolShutdownException) {
469 				logger.warn(Messages.getString("LocalServerHandler.0") + //$NON-NLS-1$
470 						session.getAuth().getUser());
471 				if (localChannelReference != null) {
472 					R66Result finalValue = new R66Result(exception, session, true,
473 							ErrorCode.Shutdown, session.getRunner());
474 					try {
475 						tryFinalizeRequest(finalValue);
476 					} catch (OpenR66RunnerErrorException e2) {
477 					} catch (OpenR66ProtocolSystemException e2) {
478 					}
479 					if (!localChannelReference.getFutureRequest().isDone()) {
480 						try {
481 							session.setFinalizeTransfer(false, finalValue);
482 						} catch (OpenR66RunnerErrorException e1) {
483 							localChannelReference.invalidateRequest(finalValue);
484 						} catch (OpenR66ProtocolSystemException e1) {
485 							localChannelReference.invalidateRequest(finalValue);
486 						}
487 					}
488 				}
489 				// dont'close, thread will do
490 				ChannelUtils.startShutdown();
491 				// set global shutdown info and before close, send a valid
492 				// shutdown to all
493 				session.setStatus(54);
494 				return;
495 			} else {
496 				if (localChannelReference != null
497 						&& localChannelReference.getFutureRequest() != null) {
498 					if (localChannelReference.getFutureRequest().isDone()) {
499 						R66Result result = localChannelReference.getFutureRequest()
500 								.getResult();
501 						if (result != null) {
502 							isAnswered = result.isAnswered;
503 						}
504 					}
505 				}
506 				if (exception instanceof OpenR66ProtocolNoConnectionException) {
507 					code = ErrorCode.ConnectionImpossible;
508 					DbTaskRunner runner = session.getRunner();
509 					if (runner != null) {
510 						runner.stopOrCancelRunner(code);
511 					}
512 				} else if (exception instanceof OpenR66ProtocolBusinessCancelException) {
513 					code = ErrorCode.CanceledTransfer;
514 					DbTaskRunner runner = session.getRunner();
515 					if (runner != null) {
516 						runner.stopOrCancelRunner(code);
517 					}
518 				} else if (exception instanceof OpenR66ProtocolBusinessStopException) {
519 					code = ErrorCode.StoppedTransfer;
520 					DbTaskRunner runner = session.getRunner();
521 					if (runner != null) {
522 						runner.stopOrCancelRunner(code);
523 					}
524 				} else if (exception instanceof OpenR66ProtocolBusinessQueryAlreadyFinishedException) {
525 					code = ErrorCode.QueryAlreadyFinished;
526 					try {
527 						tryFinalizeRequest(new R66Result(session, true, code, session.getRunner()));
528 						return;
529 					} catch (OpenR66RunnerErrorException e1) {
530 					} catch (OpenR66ProtocolSystemException e1) {
531 					}
532 				} else if (exception instanceof OpenR66ProtocolBusinessQueryStillRunningException) {
533 					code = ErrorCode.QueryStillRunning;
534 					// nothing is to be done
535 					logger.error("Will close channel since ", exception);
536 					Channels.close(e.getChannel());
537 					session.setStatus(56);
538 					return;
539 				} else if (exception instanceof OpenR66ProtocolBusinessRemoteFileNotFoundException) {
540 					code = ErrorCode.FileNotFound;
541 				} else if (exception instanceof OpenR66RunnerException) {
542 					code = ErrorCode.ExternalOp;
543 				} else if (exception instanceof OpenR66RunnerErrorException) {
544 					code = ErrorCode.ExternalOp;
545 				} else if (exception instanceof OpenR66ProtocolNotAuthenticatedException) {
546 					code = ErrorCode.BadAuthent;
547 				} else if (exception instanceof OpenR66ProtocolNetworkException) {
548 					code = ErrorCode.Disconnection;
549 					DbTaskRunner runner = session.getRunner();
550 					if (runner != null) {
551 						R66Result finalValue = new R66Result(
552 								new OpenR66ProtocolSystemException(
553 										Messages.getString("LocalServerHandler.2")), //$NON-NLS-1$
554 								session, true, code, session.getRunner());
555 						try {
556 							tryFinalizeRequest(finalValue);
557 						} catch (OpenR66Exception e2) {
558 						}
559 					}
560 				} else if (exception instanceof OpenR66ProtocolRemoteShutdownException) {
561 					code = ErrorCode.RemoteShutdown;
562 					DbTaskRunner runner = session.getRunner();
563 					if (runner != null) {
564 						runner.stopOrCancelRunner(code);
565 					}
566 				} else {
567 					DbTaskRunner runner = session.getRunner();
568 					if (runner != null) {
569 						switch (runner.getErrorInfo()) {
570 							case InitOk:
571 							case PostProcessingOk:
572 							case PreProcessingOk:
573 							case Running:
574 							case TransferOk:
575 								code = ErrorCode.Internal;
576 								break;
577 							default:
578 								code = runner.getErrorInfo();
579 						}
580 					} else {
581 						code = ErrorCode.Internal;
582 					}
583 				}
584 				if ((!isAnswered) &&
585 						(!(exception instanceof OpenR66ProtocolBusinessNoWriteBackException)) &&
586 						(!(exception instanceof OpenR66ProtocolNoConnectionException))) {
587 					if (code == null || code == ErrorCode.Internal) {
588 						code = ErrorCode.RemoteError;
589 					}
590 					final ErrorPacket errorPacket = new ErrorPacket(exception
591 							.getMessage(),
592 							code.getCode(), ErrorPacket.FORWARDCLOSECODE);
593 					try {
594 						if (localChannelReference != null) {
595 							ChannelUtils.writeAbstractLocalPacket(localChannelReference,
596 								errorPacket, true);
597 						}
598 					} catch (OpenR66ProtocolPacketException e1) {
599 						// should not be
600 					}
601 				}
602 				R66Result finalValue =
603 						new R66Result(
604 								exception, session, true, code, session.getRunner());
605 				try {
606 					session.setFinalizeTransfer(false, finalValue);
607 					if (localChannelReference != null) {
608 						localChannelReference.invalidateRequest(finalValue);
609 					}
610 				} catch (OpenR66RunnerErrorException e1) {
611 					if (localChannelReference != null)
612 						localChannelReference.invalidateRequest(finalValue);
613 				} catch (OpenR66ProtocolSystemException e1) {
614 					if (localChannelReference != null)
615 						localChannelReference.invalidateRequest(finalValue);
616 				}
617 			}
618 			if (exception instanceof OpenR66ProtocolBusinessNoWriteBackException) {
619 				logger.error("Will close channel {}", exception.getMessage());
620 				Channels.close(e.getChannel());
621 				session.setStatus(56);
622 				return;
623 			} else if (exception instanceof OpenR66ProtocolNoConnectionException) {
624 				logger.error("Will close channel {}", exception.getMessage());
625 				Channels.close(e.getChannel());
626 				session.setStatus(57);
627 				return;
628 			}
629 			session.setStatus(58);
630 			ChannelCloseTimer.closeFutureChannel(e.getChannel());
631 		} else {
632 			// Nothing to do
633 			session.setStatus(59);
634 			return;
635 		}

713 	 * Authentication
714 	 * 
715 	 * @param channel
716 	 * @param packet
717 	 * @throws OpenR66ProtocolPacketException
718 	 */
719 	private void authent(Channel channel, AuthentPacket packet)
720 			throws OpenR66ProtocolPacketException {
721 		if (packet.isToValidate()) {
722 			session.newState(AUTHENTR);
723 		}
724 
725 		if (localChannelReference.getDbSession() != null) {
726 			localChannelReference.getDbSession().useConnection();
727 		}
728 		localChannelReference.getNetworkChannelObject().hostId = packet.getHostId();
729 		try {
730 			session.getAuth().connection(localChannelReference.getDbSession(),
731 					packet.getHostId(), packet.getKey());
732 		} catch (Reply530Exception e1) {
733 			refusedConnection(channel, packet, e1);
734 			session.setStatus(42);
735 			return;
736 		} catch (Reply421Exception e1) {
737 			session.newState(ERROR);
738 			logger.error("Service unavailable: " + packet.getHostId(), e1);
739 			R66Result result = new R66Result(
740 					new OpenR66ProtocolSystemException("Service unavailable",
741 							e1), session, true,
742 					ErrorCode.ConnectionImpossible, null);
743 			localChannelReference.invalidateRequest(result);
744 			ErrorPacket error = new ErrorPacket("Service unavailable",
745 					ErrorCode.ConnectionImpossible.getCode(),
746 					ErrorPacket.FORWARDCLOSECODE);
747 			ChannelUtils.writeAbstractLocalPacket(localChannelReference, error, true);
748 			localChannelReference.validateConnection(false, result);
749 			ChannelCloseTimer.closeFutureChannel(channel);
750 			session.setStatus(43);
751 			return;
752 		}
753 		localChannelReference.setPartner(packet.getHostId());
754 		// Now if configuration say to do so: check remote ip address
755 		if (Configuration.configuration.checkRemoteAddress && ! localChannelReference.getPartner().isProxified()) {
756 			DbHostAuth host = R66Auth.getServerAuth(localChannelReference.getDbSession(),
757 					packet.getHostId());
758 			boolean toTest = false;
759 			if (host.isClient()) {
760 				if (Configuration.configuration.checkClientAddress) {
761 					if (host.isNoAddress()) {
762 						// 0.0.0.0 so nothing
763 						toTest = false;
764 					} else {
765 						toTest = true;
766 					}
767 				}
768 			} else {
769 				toTest = true;
770 			}
771 			if (toTest) {
772 				// Real address so compare
773 				String address = host.getAddress();
774 				InetAddress[] inetAddress = null;
775 				try {
776 					inetAddress = InetAddress.getAllByName(address);
777 				} catch (UnknownHostException e) {
778 					inetAddress = null;
779 				}
780 				if (inetAddress != null) {
781 					InetSocketAddress socketAddress = (InetSocketAddress) session
782 							.getRemoteAddress();
783 					boolean found = false;
784 					for (int i = 0; i < inetAddress.length; i++) {
785 						if (socketAddress.getAddress().equals(inetAddress[i])) {
786 							found = true;
787 							break;
788 						}
789 					}
790 					if (!found) {
791 						// error
792 						refusedConnection(channel, packet,
793 								new OpenR66ProtocolNotAuthenticatedException(
794 										"Server IP not authenticated: " +
795 												inetAddress[0].toString() + " compare to "
796 												+ socketAddress.getAddress().toString()));
797 						session.setStatus(104);
798 						return;
799 					}
800 				}
801 			}
802 		}
803 		R66Result result = new R66Result(session, true, ErrorCode.InitOk, null);
804 		session.newState(AUTHENTD);
805 		localChannelReference.validateConnection(true, result);
806 		logger.debug("Local Server Channel Validated: {} ",
807 				(localChannelReference != null ? localChannelReference
808 						: "no LocalChannelReference"));
809 		session.setStatus(44);
810 		if (packet.isToValidate()) {
811 			// only requested
812 			NetworkTransaction.addClient(localChannelReference.getNetworkChannel(),
813 					packet.getHostId());
814 			packet.validate(session.getAuth().isSsl());
815 			ChannelUtils.writeAbstractLocalPacket(localChannelReference, packet, false);
816 			session.setStatus(98);
817 		}
818 		logger.debug("Partner: {} from {}", localChannelReference.getPartner(), Configuration.configuration.versions);
819 	}

875 	 */
876 	private void errorMesg(Channel channel, ErrorPacket packet)
877 			throws OpenR66RunnerErrorException, OpenR66ProtocolSystemException,
878 			OpenR66ProtocolBusinessException {
879 		// do something according to the error
880 		if (session.getLocalChannelReference().getFutureRequest().isDone()) {
881 			// already canceled or successful
882 			return;
883 		}
884 		logger.error(channel.getId() + ": " + packet.toString());
885 		session.setStatus(46);
886 		ErrorCode code = ErrorCode.getFromCode(packet.getSmiddle());
887 		session.getLocalChannelReference().setErrorMessage(packet.getSheader(), code);
888 		OpenR66ProtocolBusinessException exception;
889 		if (code.code == ErrorCode.CanceledTransfer.code) {
890 			exception =
891 					new OpenR66ProtocolBusinessCancelException(packet.getSheader());
892 			int rank = 0;
893 			DbTaskRunner runner = this.session.getRunner();
894 			if (runner != null) {
895 				runner.setRankAtStartup(rank);
896 				runner.stopOrCancelRunner(code);
897 			}
898 			R66Result result = new R66Result(exception, session,
899 					true, code, runner);
900 			// now try to inform other
901 			try {
902 				ChannelUtils.writeAbstractLocalPacket(localChannelReference, packet, false).
903 						addListener(new RunnerChannelFutureListener(localChannelReference, result));
904 			} catch (OpenR66ProtocolPacketException e) {
905 			}
906 			session.setFinalizeTransfer(false, result);
907 			return;
908 		} else if (code.code == ErrorCode.StoppedTransfer.code) {
909 			exception =
910 					new OpenR66ProtocolBusinessStopException(packet.getSheader());
911 			String[] vars = packet.getSheader().split(" ");
912 			String var = vars[vars.length - 1];
913 			int rank = Integer.parseInt(var);
914 			DbTaskRunner runner = this.session.getRunner();
915 			if (runner != null) {
916 				if (rank < runner.getRank()) {
917 					runner.setRankAtStartup(rank);
918 				}
919 				runner.stopOrCancelRunner(code);
920 			}
921 			R66Result result = new R66Result(exception, session,
922 					true, code, runner);
923 			// now try to inform other
924 			try {
925 				ChannelUtils.writeAbstractLocalPacket(localChannelReference, packet, false).
926 						addListener(new RunnerChannelFutureListener(localChannelReference, result));
927 			} catch (OpenR66ProtocolPacketException e) {
928 			}
929 			session.setFinalizeTransfer(false, result);
930 			return;
931 		} else if (code.code == ErrorCode.QueryAlreadyFinished.code) {
932 			DbTaskRunner runner = session.getRunner();
933 			if (runner == null) {
934 				exception =
935 						new OpenR66ProtocolBusinessCancelException(packet.toString());
936 			} else {
937 				if (runner.isSender()) {
938 					exception =
939 							new OpenR66ProtocolBusinessQueryAlreadyFinishedException(
940 									packet.getSheader());
941 					runner.finishTransferTask(code);
942 					tryFinalizeRequest(new R66Result(exception, session, true, code, runner));
943 				} else {
944 					exception =
945 							new OpenR66ProtocolBusinessCancelException(packet.toString());
946 				}
947 			}
948 			throw exception;
949 		} else if (code.code == ErrorCode.QueryStillRunning.code) {
950 			exception =
951 					new OpenR66ProtocolBusinessQueryStillRunningException(packet.getSheader());
952 			throw exception;
953 		} else if (code.code == ErrorCode.BadAuthent.code) {
954 			exception =
955 					new OpenR66ProtocolNotAuthenticatedException(packet.toString());
956 		} else if (code.code == ErrorCode.QueryRemotelyUnknown.code) {
957 			exception =
958 					new OpenR66ProtocolBusinessCancelException(packet.toString());
959 		} else if (code.code == ErrorCode.FileNotFound.code) {
960 			exception =
961 					new OpenR66ProtocolBusinessRemoteFileNotFoundException(packet.toString());
962 		} else {
963 			exception =
964 					new OpenR66ProtocolBusinessNoWriteBackException(packet.toString());
965 		}
966 		session.setFinalizeTransfer(false, new R66Result(exception, session,
967 				true, code, session.getRunner()));
968 		throw exception;
969 	}

1514 			throw new OpenR66ProtocolBusinessException("No request prepared");

1517 			logger.debug("Not in receive MODE but receive a packet");
1518 			throw new OpenR66ProtocolBusinessException(
1519 					"Not in receive MODE but receive a packet");

1524 				// nothing to do since already done

1526 				return;
1527 			}
1528 			errorToSend("Transfer in error due previously aborted transmission", ErrorCode.TransferError, channel, 95);

1531 		if (packet.getPacketRank() != session.getRunner().getRank()) {

1541 			// Fix the rank if possible
1542 			if (packet.getPacketRank() < session.getRunner().getRank()) {

1544 						session.getRunner().getRank());

1546 				session.getRestart().restartMarker(
1547 						session.getRunner().getBlocksize() *
1548 								session.getRunner().getRank());

1563 						packet.getPacketRank()+ " > " +
1564 								session.getRunner().getRank(), ErrorCode.TransferError, channel, 20);
1565 				return;
1566 			}
1567 		}
1568 		// Check global size
1569 		long originalSize = session.getRunner().getOriginalSize();
1570 		if (originalSize > 0) {
1571 			if (session.getRunner().getBlocksize() * (session.getRunner().getRank()-1) > originalSize) {
1572 				// cannot continue
1573 				logger.error(Messages.getString("LocalServerHandler.16") + packet.getPacketRank() + " : " + //$NON-NLS-1$

1578 			}

1580 		// if MD5 check MD5
