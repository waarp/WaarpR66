26  import com.fasterxml.jackson.annotation.JsonTypeInfo;

32  /**
33   * Json Object Command Message class for JsonCommandPacket
34   * 
35   * 1 string = comment
36   * 
37   * @author frederic bregier
38   */
39  @JsonTypeInfo(use=JsonTypeInfo.Id.CLASS, include=JsonTypeInfo.As.PROPERTY, property="@class")
40  public class JsonPacket {
41  	public static final AdaptativeJsonHandler handler = new AdaptativeJsonHandler(JsonCodec.JSON);
42  	/**
43  	 * @return the ObjectNode corresponding to this object
44  	 * @throws OpenR66ProtocolPacketException
45  	 */
46  	public ObjectNode createObjectNode() throws OpenR66ProtocolPacketException {
47  		try {
48  			String value = handler.mapper.writeValueAsString(this);
49  			return (ObjectNode) handler.mapper.readTree(value);
50  		} catch (JsonProcessingException e) {
51  			throw new OpenR66ProtocolPacketException("Json exception", e);
52  		} catch (IOException e) {
53  			throw new OpenR66ProtocolPacketException("Json exception", e);
54  		}
55  	}
56  	
57  	private String comment;
58  	private byte requestUserPacket;
59  
60  	/**
61  	 * @return the requestUserPacket
62  	 */
63  	public byte getRequestUserPacket() {
64  		return requestUserPacket;
65  	}
66  
67  
68  	/**
69  	 * @param requestUserPacket the requestUserPacket to set
70  	 */
71  	public void setRequestUserPacket(byte requestUserPacket) {
72  		this.requestUserPacket = requestUserPacket;
73  	}
74  
75  	/**
76  	 * @return the comment
77  	 */
78  	public String getComment() {
79  		return comment;
80  	}
81  
82  
83  	/**
84  	 * @param comment the comment to set
85  	 */
86  	public void setComment(String comment) {
87  		this.comment = comment;
88  	}
89  
90  	@Override
91  	public String toString() {
92  		return handler.writeAsString(this);
93  	}
94  
95  	/**
96  	 * @param value
97  	 * @return the new JsonPacket from buffer
98  	 * @throws IOException 
99  	 * @throws JsonMappingException 
100 	 * @throws JsonParseException 
101 	 */
102 	public static JsonPacket createFromBuffer(String value) throws JsonParseException, JsonMappingException, IOException {
103 		if (value != null && ! value.isEmpty()) {
104 			return handler.mapper.readValue(value, JsonPacket.class);
105 		}
106 		return null;
107 	}
108 
109 	/**
110 	 * Set from other Json
111 	 * @param json
112 	 */
113 	public void fromJson(JsonPacket json) {
114 		this.comment = json.comment;
115 		this.requestUserPacket = json.requestUserPacket;
116 	}
