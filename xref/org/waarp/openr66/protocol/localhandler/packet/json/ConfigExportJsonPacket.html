21  package org.waarp.openr66.protocol.localhandler.packet.json;
22  
23  import org.waarp.openr66.protocol.localhandler.packet.LocalPacketFactory;
24  
25  /**
26   * Exporting Configuration JSON packet
27   * 
28   * @author "Frederic Bregier"
29   *
30   */
31  public class ConfigExportJsonPacket extends JsonPacket {
32  
33      protected boolean host, rule, business, alias, roles;
34  
35      /**
36       * @return the host
37       */
38      public boolean isHost() {
39          return host;
40      }
41  
42      /**
43       * @param host
44       *            the host to set
45       */
46      public void setHost(boolean host) {
47          this.host = host;
48      }
49  
50      /**
51       * @return the rule
52       */
53      public boolean isRule() {
54          return rule;
55      }
56  
57      /**
58       * @param rule
59       *            the rule to set
60       */
61      public void setRule(boolean rule) {
62          this.rule = rule;
63      }
64  
65      /**
66       * @return the business
67       */
68      public boolean isBusiness() {
69          return business;
70      }
71  
72      /**
73       * @param business
74       *            the business to set
75       */
76      public void setBusiness(boolean business) {
77          this.business = business;
78      }
79  
80      /**
81       * @return the alias
82       */
83      public boolean isAlias() {
84          return alias;
85      }
86  
87      /**
88       * @param alias
89       *            the alias to set
90       */
91      public void setAlias(boolean alias) {
92          this.alias = alias;
93      }
94  
95      /**
96       * @return the roles
97       */
98      public boolean isRoles() {
99          return roles;
100     }
101 
102     /**
103      * @param roles
104      *            the roles to set
105      */
106     public void setRoles(boolean roles) {
107         this.roles = roles;
108     }
109 
110     @Override
111     public void fromJson(JsonPacket other) {
112         super.fromJson(other);
113         if (other instanceof ConfigExportJsonPacket) {
114             ConfigExportJsonPacket other2 = (ConfigExportJsonPacket) other;
115             this.host = other2.host;
116             this.rule = other2.rule;
117             this.business = other2.business;
118             this.alias = other2.alias;
119             this.roles = other2.roles;
120         }
121     }
122 
123     public void setRequestUserPacket() {
124         super.setRequestUserPacket(LocalPacketFactory.CONFEXPORTPACKET);
125     }
126 }
