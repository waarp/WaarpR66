18  package org.waarp.openr66.protocol.localhandler.packet;
19  
20  import java.util.List;
21  
22  import io.netty.buffer.ByteBuf;
23  import io.netty.channel.ChannelHandlerContext;
24  import io.netty.handler.codec.ByteToMessageCodec;
25  
26  import org.waarp.openr66.protocol.exception.OpenR66ProtocolPacketException;
27  
28  /**
29   * Local Packet Decoder
30   * 
31   * @author Frederic Bregier
32   */
33  public class LocalPacketCodec extends ByteToMessageCodec<AbstractLocalPacket> {
34  
35      @Override
36      protected void decode(ChannelHandlerContext ctx, ByteBuf buf, List<Object> out) throws Exception {
37          // Make sure if the length field was received.
38          if (buf.readableBytes() < 4) {
39              // The length field was not received yet - return null.
40              // This method will be invoked again when more packets are
41              // received and appended to the buffer.
42              return;
43          }
44          AbstractLocalPacket newbuf = decodeNetworkPacket(buf);
45          if (newbuf != null) {
46              out.add(newbuf);
47          }
48      }
49  
50      public static AbstractLocalPacket decodeNetworkPacket(ByteBuf buf)
51              throws OpenR66ProtocolPacketException {
52          // Mark the current buffer position
53          buf.markReaderIndex();
54          // Read the length field
55          final int length = buf.readInt();
56          if (buf.readableBytes() < length) {
57              buf.resetReaderIndex();
58              return null;
59          }
60          // Now we can read the header
61          // Header: Header length field (4 bytes) = Middle length field (4
62          // bytes), End length field (4 bytes), type field (1 byte), ...
63          final int middleLength = buf.readInt();
64          final int endLength = buf.readInt();
65          // check if the packet is complete
66          if (middleLength + endLength + length - 8 > buf.readableBytes()) {
67              buf.resetReaderIndex();
68              return null;
69          }
70          // createPacketFromByteBuf read the buffer
71          return LocalPacketFactory.createPacketFromByteBuf(length - 8,
72                  middleLength, endLength, buf);
73      }
74  
75      @Override
76      protected void encode(ChannelHandlerContext ctx, AbstractLocalPacket msg, ByteBuf out) throws Exception {
77          /*if (msg instanceof ByteBuf) {
78              out.writeBytes((ByteBuf) msg);
79              return;
80          }*/
81          final ByteBuf buf = msg.getLocalPacket(null);
82          out.writeBytes(buf);
83          buf.release();
84      }
85  
86  }
