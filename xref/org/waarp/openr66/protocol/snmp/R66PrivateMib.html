70      @Override
71      protected void agentRegisterWaarpMib()
72              throws DuplicateRegistrationException {
73          logger.debug("registerGGMib");
74          // register Static info
75          rowInfo = new WaarpMORow(this, rootOIDWaarpInfo, WaarpDefinition,
76                  MibLevel.staticInfo.ordinal());
77          rowInfo.setValue(WaarpDefinitionIndex.applName.ordinal(),
78                  "Waarp OpenR66");
79          rowInfo.setValue(WaarpDefinitionIndex.applServerName.ordinal(),
80                  Configuration.configuration.getHOST_ID());
81          rowInfo.setValue(WaarpDefinitionIndex.applVersion.ordinal(),
82                  Version.ID);
83          rowInfo.setValue(WaarpDefinitionIndex.applDescription.ordinal(),
84                  "Waarp OpenR66: File Transfer Monitor");
85          rowInfo.setValue(WaarpDefinitionIndex.applURL.ordinal(),
86                  "http://waarp.github.com/Waarp");
87          rowInfo.setValue(WaarpDefinitionIndex.applApplicationProtocol.ordinal(),
88                  applicationProtocol);
89  
90          rowInfo.registerMOs(agent.getServer(), null);
91          // register General info
92          rowGlobal = new WaarpMORow(this, rootOIDWaarpGlobal, WaarpGlobalValues,
93                  MibLevel.globalInfo.ordinal());
94          WaarpMOScalar memoryScalar = rowGlobal.getRow()[WaarpGlobalValuesIndex.memoryTotal.ordinal()];
95          memoryScalar.setValue(new MemoryGauge32(MemoryType.TotalMemory));
96          memoryScalar = rowGlobal.getRow()[WaarpGlobalValuesIndex.memoryFree.ordinal()];
97          memoryScalar.setValue(new MemoryGauge32(MemoryType.FreeMemory));
98          memoryScalar = rowGlobal.getRow()[WaarpGlobalValuesIndex.memoryUsed.ordinal()];
99          memoryScalar.setValue(new MemoryGauge32(MemoryType.UsedMemory));
100         rowGlobal.registerMOs(agent.getServer(), null);
101         // setup UpTime to SysUpTime and change status
102         scalarUptime = rowGlobal.getRow()[WaarpGlobalValuesIndex.applUptime.ordinal()];
103         scalarUptime.setValue(new WaarpUptime(upTime));
104         changeStatus(OperStatus.restarting);
105         changeStatus(OperStatus.up);
106         // register Detailed info
107         rowDetailed = new WaarpMORow(this, rootOIDWaarpDetailed, WaarpDetailedValues,
108                 MibLevel.detailedInfo.ordinal());
109         rowDetailed.registerMOs(agent.getServer(), null);
110         // register Error info
111         rowError = new WaarpMORow(this, rootOIDWaarpError, WaarpErrorValues,
112                 MibLevel.errorInfo.ordinal());
113         rowError.registerMOs(agent.getServer(), null);
114     }

164     /**
165      * Send a notification (trap or inform)
166      * 
167      * @param message
168      * @param runner
169      */
170     public void notifyInternalTask(String message, DbTaskRunner runner) {
171         try {
172             long delay = (runner.getStart().getTime() -
173                     agent.getUptimeSystemTime()) / 10;
174             if (delay < 0)
175                 delay = 0;
176             agent.getNotificationOriginator()
177                     .notify(new OctetString("public"),
178                             NotificationElements.InfoTask
179                                     .getOID(rootOIDWaarpNotif),
180                             new VariableBinding[] {
181                                     new VariableBinding(
182                                             NotificationElements.InfoTask.getOID(
183                                                     rootOIDWaarpNotif, 1),
184                                             new OctetString(
185                                                     NotificationElements.InfoTask
186                                                             .name())),
187                                     new VariableBinding(
188                                             NotificationElements.InfoTask.getOID(
189                                                     rootOIDWaarpNotif, 1),
190                                             new OctetString(message)),
191                                     // Start of Task
192                                     new VariableBinding(
193                                             NotificationElements.InfoTask
194                                                     .getOID(rootOIDWaarpNotif,
195                                                             NotificationTasks.globalStepInfo
196                                                                     .getOID()),
197                                             new Gauge32(runner.getGloballaststep())),
198                                     new VariableBinding(
199                                             NotificationElements.InfoTask
200                                                     .getOID(rootOIDWaarpNotif,
201                                                             NotificationTasks.stepInfo
202                                                                     .getOID()),
203                                             new Gauge32(runner.getStep() + 1)),
204                                     new VariableBinding(
205                                             NotificationElements.InfoTask
206                                                     .getOID(rootOIDWaarpNotif,
207                                                             NotificationTasks.rankFileInfo
208                                                                     .getOID()),
209                                             new Gauge32(runner.getRank())),
210                                     new VariableBinding(
211                                             NotificationElements.InfoTask
212                                                     .getOID(rootOIDWaarpNotif,
213                                                             NotificationTasks.stepStatusInfo
214                                                                     .getOID()),
215                                             new OctetString(runner.getStatus().getMesg())),
216                                     new VariableBinding(
217                                             NotificationElements.InfoTask
218                                                     .getOID(rootOIDWaarpNotif,
219                                                             NotificationTasks.filenameInfo
220                                                                     .getOID()),
221                                             new OctetString(runner.getFilename())),
222                                     new VariableBinding(
223                                             NotificationElements.InfoTask
224                                                     .getOID(rootOIDWaarpNotif,
225                                                             NotificationTasks.originalNameInfo
226                                                                     .getOID()),
227                                             new OctetString(runner.getOriginalFilename())),
228                                     new VariableBinding(
229                                             NotificationElements.InfoTask
230                                                     .getOID(rootOIDWaarpNotif,
231                                                             NotificationTasks.idRuleInfo
232                                                                     .getOID()),
233                                             new OctetString(runner.getRuleId())),
234                                     new VariableBinding(
235                                             NotificationElements.InfoTask
236                                                     .getOID(rootOIDWaarpNotif,
237                                                             NotificationTasks.modeTransInfo
238                                                                     .getOID()),
239                                             new OctetString(TRANSFERMODE.values()[runner.getMode()]
240                                                     .name())),
241                                     new VariableBinding(
242                                             NotificationElements.InfoTask
243                                                     .getOID(rootOIDWaarpNotif,
244                                                             NotificationTasks.retrieveModeInfo
245                                                                     .getOID()),
246                                             new OctetString(runner.isSender() ? "Sender" : "Receiver")),
247                                     new VariableBinding(
248                                             NotificationElements.InfoTask
249                                                     .getOID(rootOIDWaarpNotif,
250                                                             NotificationTasks.startTransInfo
251                                                                     .getOID()),
252                                             new TimeTicks(delay)),
253                                     new VariableBinding(
254                                             NotificationElements.InfoTask
255                                                     .getOID(rootOIDWaarpNotif,
256                                                             NotificationTasks.infoStatusInfo
257                                                                     .getOID()),
258                                             new OctetString(runner.getErrorInfo().getMesg())),
259                                     new VariableBinding(
260                                             NotificationElements.InfoTask
261                                                     .getOID(rootOIDWaarpNotif,
262                                                             NotificationTasks.requesterInfo
263                                                                     .getOID()),
264                                             new OctetString(runner.getRequester())),
265                                     new VariableBinding(
266                                             NotificationElements.InfoTask
267                                                     .getOID(rootOIDWaarpNotif,
268                                                             NotificationTasks.requestedInfo
269                                                                     .getOID()),
270                                             new OctetString(runner.getRequested())),
271                                     new VariableBinding(
272                                             NotificationElements.InfoTask
273                                                     .getOID(rootOIDWaarpNotif,
274                                                             NotificationTasks.specialIdInfo
275                                                                     .getOID()),
276                                             new OctetString("" + runner.getSpecialId())),
277                                     // End of Task
278                                     new VariableBinding(SnmpConstants.sysDescr,
279                                             snmpv2.getDescr()),
280                                     new VariableBinding(SnmpConstants.sysObjectID,
281                                             snmpv2.getObjectID()),
282                                     new VariableBinding(SnmpConstants.sysContact,
283                                             snmpv2.getContact()),
284                                     new VariableBinding(SnmpConstants.sysName,
285                                             snmpv2.getName()),
286                                     new VariableBinding(SnmpConstants.sysLocation,
287                                             snmpv2.getLocation()) });
288         } catch (NullPointerException e) {
289         }
290     }

292     /**
293      * Send a notification (trap or inform) for Warning/Error event on a single Transfer Task
294      * 
295      * @param message
296      * @param runner
297      */
298     public void notifyInfoTask(String message, DbTaskRunner runner) {
299         if (!TrapLevel.All.isLevelValid(agent.getTrapLevel()))
300             return;
301         if (logger.isDebugEnabled())
302             logger.debug("Notify: " + NotificationElements.InfoTask + ":" + message +
303                     ":" + runner.toShortString());
304         notifyInternalTask(message, runner);
305     }

317             logger.debug("Notify: " + NotificationElements.InfoTask + ":" + message +
318                     ":" + runner.toShortString());

320     }

322     /**
323      * Trap/Notification
324      * 
325      * @param element
326      * @param message
327      * @param message2
328      */
329     private void notify(NotificationElements element, String message, String message2) {
330         try {
331             if (logger.isDebugEnabled())
332                 logger.debug("Notify: " + element + ":" + message + ":" + message2);
333             agent.getNotificationOriginator().notify(
334                     new OctetString("public"),
335                     element.getOID(rootOIDWaarpNotif),
336                     new VariableBinding[] {
337                             new VariableBinding(
338                                     element.getOID(rootOIDWaarpNotif, 1),
339                                     new OctetString(element.name())),
340                             new VariableBinding(
341                                     element.getOID(rootOIDWaarpNotif, 1),
342                                     new OctetString(message)),
343                             new VariableBinding(
344                                     element.getOID(rootOIDWaarpNotif, 1),
345                                     new OctetString(message2)),
346                             new VariableBinding(SnmpConstants.sysDescr, snmpv2.getDescr()),
347                             new VariableBinding(SnmpConstants.sysObjectID, snmpv2.getObjectID()),
348                             new VariableBinding(SnmpConstants.sysContact, snmpv2.getContact()),
349                             new VariableBinding(SnmpConstants.sysName, snmpv2.getName()),
350                             new VariableBinding(SnmpConstants.sysLocation, snmpv2.getLocation())
351                     });
352         } catch (NullPointerException e) {
353         }
354     }
