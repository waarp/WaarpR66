55  	 * Copy one file to another one
56  	 * 
57  	 * @param from
58  	 * @param to
59  	 * @param move
60  	 *            True if the copy is in fact a move operation
61  	 * @param append
62  	 *            True if the copy is in append
63  	 * @throws OpenR66ProtocolSystemException
64  	 */
65  	public static void copy(File from, File to, boolean move, boolean append)
66  			throws OpenR66ProtocolSystemException {
67  		if (from == null || to == null) {
68  			throw new OpenR66ProtocolSystemException(
69  					"Source or Destination is null");
70  		}
71  		File directoryTo = to.getParentFile();
72  		if (createDir(directoryTo)) {
73  			if (move && from.renameTo(to)) {
74  				return;
75  			}
76  			FileChannel fileChannelIn = getFileChannel(from, false, false);
77  			if (fileChannelIn == null) {
78  				throw new OpenR66ProtocolSystemException(
79  						"Cannot read source file");
80  				// return false;
81  			}
82  			FileChannel fileChannelOut = getFileChannel(to, true, append);
83  			if (fileChannelOut == null) {
84  				try {
85  					fileChannelIn.close();
86  				} catch (IOException e) {
87  				}
88  				throw new OpenR66ProtocolSystemException(
89  						"Cannot write destination file");
90  			}
91  			if (write(fileChannelIn, fileChannelOut) > 0) {
92  				if (move) {
93  					// do not test the delete
94  					from.delete();
95  				}
96  				return;
97  			}
98  			throw new OpenR66ProtocolSystemException("Cannot copy");
99  		}
100 		throw new OpenR66ProtocolSystemException(
101 				"Cannot access to parent dir of destination");
102 	}

105 	 * Copy a group of files to a directory
106 	 * 
107 	 * @param from
108 	 * @param directoryTo
109 	 * @param move
110 	 *            True if the copy is in fact a move operation
111 	 * @return the group of copy files or null (partially or totally) if an error occurs
112 	 * @throws OpenR66ProtocolSystemException
113 	 */
114 	public static File[] copy(File[] from, File directoryTo, boolean move)
115 			throws OpenR66ProtocolSystemException {
116 		if (from == null || directoryTo == null) {
117 			return null;
118 		}
119 		File[] to = null;
120 		if (createDir(directoryTo)) {
121 			to = new File[from.length];
122 			for (int i = 0; i < from.length; i++) {
123 				try {
124 					to[i] = copyToDir(from[i], directoryTo, move);
125 				} catch (OpenR66ProtocolSystemException e) {
126 					throw e;
127 				}
128 			}
129 		}
130 		return to;
131 	}

249 		if (directory == null) {
250 			return true;
251 		}
252 		boolean retour = true;
253 		if (!directory.exists()) {
254 			return true;
255 		}
256 		if (!directory.isDirectory()) {
257 			return false;
258 		}
259 		File[] list = directory.listFiles();
260 		if (list == null || list.length == 0) {
261 			list = null;
262 			retour = directory.delete();
263 			return retour;
264 		}
265 		int len = list.length;
266 		for (int i = 0; i < len; i++) {
267 			if (list[i].isDirectory()) {
268 				if (!deleteRecursiveFileDir(list[i])) {
269 					retour = false;
270 				}
271 			} else {
272 				retour = false;
273 			}
274 		}
275 		list = null;
276 		if (retour) {
277 			retour = directory.delete();
278 		}
279 		return retour;
280 	}

321 	 * Change or create the R66File associated with the context
322 	 * @param logger
323 	 * @param session
324 	 * @param filenameSrc new filename
325 	 * @param isPreStart
326 	 * @param isSender
327 	 * @param isThrough
328 	 * @param file old R66File if any (might be null)
329 	 * @return the R66File
330 	 * @throws OpenR66RunnerErrorException
331 	 */
332 	public final static R66File getFile(WaarpInternalLogger logger, R66Session session, String filenameSrc, 
333 			boolean isPreStart, boolean isSender, boolean isThrough, R66File file) throws OpenR66RunnerErrorException {
334 		String filename;
335 		logger.debug("PreStart: "+isPreStart);
336 		logger.debug("Dir is: "+session.getDir().getFullPath());
337 		logger.debug("File is: "+filenameSrc);
338 		if (isPreStart) {
339 			filename = R66Dir.normalizePath(filenameSrc);
340 			if (filename.startsWith("file:/")) {
341 				if (DetectionUtils.isWindows()) {
342 					filename = filename.substring("file:/".length());
343 				} else {
344 					filename = filename.substring("file:".length());
345 				}
346 				try {
347 					filename = URLDecoder.decode(filename, WaarpStringUtils.UTF8.name());
348 				} catch (UnsupportedEncodingException e) {
349 					throw new OpenR66RunnerErrorException("File cannot be read: " +
350 							filename);
351 				}
352 			}
353 			logger.debug("File becomes: "+filename);
354 		} else {
355 			filename = filenameSrc;
356 		}
357 		if (isSender) {
358 			try {
359 				if (file == null) {
360 					try {
361 						file = (R66File) session.getDir().setFile(filename, false);
362 					} catch (CommandAbstractException e) {
363 						logger.warn("File not placed in normal directory", e);
364 						// file is not under normal base directory, so is external
365 						// File should already exist but can be using special code ('*?')
366 						file = session.getDir().setFileNoCheck(filename);
367 					}
368 				}
369 				if (isThrough) {
370 					// no test on file since it does not really exist
371 					logger.debug("File is in through mode: {}", file);
372 				} else if (!file.canRead()) {
373 					// file is not under normal base directory, so is external
374 					// File should already exist but cannot use special code ('*?')
375 					file = new R66File(session, session.getDir(), filename);
376 					if (!file.canRead()) {
377 						throw new OpenR66RunnerErrorException("File cannot be read: " +
378 								file.getTrueFile().getAbsolutePath());
379 					}
380 				}
381 			} catch (CommandAbstractException e) {
382 				throw new OpenR66RunnerErrorException(e);
383 			}
384 		} else {
385 			// not sender so file is just registered as is but no test of existence
386 			file = new R66File(session, session.getDir(), filename);
387 		}
388 		return file;
389 	}

407 	 * Returns the FileChannel in Out MODE (if isOut is True) or in In MODE (if isOut is False)
408 	 * associated with the file. In out MODE, it can be in append MODE.
409 	 * 
410 	 * @param isOut
411 	 * @param append
412 	 * @return the FileChannel (OUT or IN)
413 	 * @throws OpenR66ProtocolSystemException
414 	 */
415 	private static FileChannel getFileChannel(File file, boolean isOut,
416 			boolean append) throws OpenR66ProtocolSystemException {
417 		FileChannel fileChannel = null;
418 		try {
419 			if (isOut) {
420 				@SuppressWarnings("resource")
421 				FileOutputStream fileOutputStream = new FileOutputStream(file
422 						.getPath(), append);
423 				fileChannel = fileOutputStream.getChannel();
424 				if (append) {
425 					// Bug in JVM since it does not respect the API (position
426 					// should be set as length)
427 					try {
428 						fileChannel.position(file.length());
429 					} catch (IOException e) {
430 					}
431 				}
432 			} else {
433 				if (!file.exists()) {
434 					throw new OpenR66ProtocolSystemException(
435 							"File does not exist");
436 				}
437 				@SuppressWarnings("resource")
438 				FileInputStream fileInputStream = new FileInputStream(file
439 						.getPath());
440 				fileChannel = fileInputStream.getChannel();
441 			}
442 		} catch (FileNotFoundException e) {
443 			throw new OpenR66ProtocolSystemException("File not found", e);
444 		}
445 		return fileChannel;
446 	}

507 	/**
508 	 * Compute global hash (if possible)
509 	 * @param digest
510 	 * @param buffer
511 	 */
512 	public static void computeGlobalHash(FilesystemBasedDigest digest, ChannelBuffer buffer) {
513 		if (digest == null) {
514 			return;
515 		}
516 		byte[] bytes = null;
517 		int start = 0;
518 		int length = buffer.readableBytes();
519 		if (buffer.hasArray()) {
520 			start = buffer.arrayOffset();
521 			bytes = buffer.array();
522 			if (bytes.length > start + length) {
523 				byte[] temp = new byte[length];
524 				System.arraycopy(bytes, start, temp, 0, length);
525 				start = 0;
526 				bytes = temp;
527 			}
528 		} else {
529 			bytes = new byte[length];
530 			buffer.getBytes(buffer.readerIndex(), bytes);
531 		}
532 		digest.Update(bytes, start, length);
533 	}

535 	/**
536 	 * Compute global hash (if possible) from a file but up to length
537 	 * @param digest
538 	 * @param file
539 	 * @param length
540 	 */
541 	public static void computeGlobalHash(FilesystemBasedDigest digest, File file, int length) {
542 		if (digest == null) {
543 			return;
544 		}
545 		byte[] bytes = new byte[65536];
546 		int still = length;
547 		int len = still > 65536 ? 65536 : still;
548 		FileInputStream inputStream = null;
549 		try {
550 			inputStream = new FileInputStream(file);
551 			while (inputStream.read(bytes, 0, len) > 0) {
552 				digest.Update(bytes, 0, len);
553 				still -= length;
554 				if (still <= 0) {
555 					break;
556 				}
557 				len = still > 65536 ? 65536 : still;
558 			}
559 		} catch (FileNotFoundException e) {
560 			// error
561 		} catch (IOException e) {
562 			// error
563 		} finally {
564 			if (inputStream != null) {
565 				try {
566 					inputStream.close();
567 				} catch (IOException e) {
568 				}
569 			}
570 		}
571 	}

573 	/**
574 	 * Write one fileChannel to another one. Close the fileChannels
575 	 * 
576 	 * @param fileChannelIn
577 	 *            source of file
578 	 * @param fileChannelOut
579 	 *            destination of file
580 	 * @return The size of copy if is OK
581 	 * @throws AtlasIoException
582 	 */
583 	private static long write(FileChannel fileChannelIn,
584 			FileChannel fileChannelOut) throws OpenR66ProtocolSystemException {
585 		if (fileChannelIn == null) {
586 			if (fileChannelOut != null) {
587 				try {
588 					fileChannelOut.close();
589 				} catch (IOException e) {
590 				}
591 			}
592 			throw new OpenR66ProtocolSystemException("FileChannelIn is null");
593 		}
594 		if (fileChannelOut == null) {
595 			try {
596 				fileChannelIn.close();
597 			} catch (IOException e) {
598 			}
599 			throw new OpenR66ProtocolSystemException("FileChannelOut is null");
600 		}
601 		long size = 0;
602 		long transfert = 0;
603 		try {
604 			transfert = fileChannelOut.position();
605 			size = fileChannelIn.size();
606 			int chunkSize = 8192;
607 			while (transfert < size) {
608 				if (chunkSize < size - transfert) {
609 					chunkSize = (int) (size - transfert);
610 				}
611 				transfert += fileChannelOut.transferFrom(fileChannelIn, transfert, chunkSize);
612 			}
613 		} catch (IOException e) {
614 			try {
615 				fileChannelOut.close();
616 				fileChannelIn.close();
617 			} catch (IOException e1) {
618 			}
619 			throw new OpenR66ProtocolSystemException(
620 					"An error during copy occurs", e);
621 		}
622 		try {
623 			fileChannelOut.close();
624 			fileChannelIn.close();
625 		} catch (IOException e) {// Close error can be ignored
626 		}
627 		boolean retour = size == transfert;
628 		if (!retour) {
629 			throw new OpenR66ProtocolSystemException("Copy is not complete: " +
630 					transfert + " bytes instead of " + size + " original bytes");
631 		}
632 		return size;
633 	}
