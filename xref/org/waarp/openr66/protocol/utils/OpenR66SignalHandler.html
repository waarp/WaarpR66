95  	/**
96  	 * Finalize resources attached to handlers
97  	 * 
98  	 * @author Frederic Bregier
99  	 */
100 	public static class R66TimerTask extends TimerTask {
101 		/**
102 		 * Internal Logger
103 		 */
104 		private static final WaarpInternalLogger logger = WaarpInternalLoggerFactory
105 				.getLogger(R66TimerTask.class);
106 
107 		/**
108 		 * EXIT type (System.exit(1))
109 		 */
110 		public static final int TIMER_EXIT = 1;
111 
112 		/**
113 		 * Type of execution in run() method
114 		 */
115 		private final int type;
116 
117 		/**
118 		 * Constructor from type
119 		 * 
120 		 * @param type
121 		 */
122 		public R66TimerTask(int type) {
123 			this.type = type;
124 		}
125 
126 		@Override
127 		public void run() {
128 			switch (type) {
129 				case TIMER_EXIT:
130 					logger.error("System will force EXIT");
131 					if (logger.isDebugEnabled()) {
132 						Map<Thread, StackTraceElement[]> map = Thread
133 								.getAllStackTraces();
134 						for (Thread thread : map.keySet()) {
135 							printStackTrace(thread, map.get(thread));
136 						}
137 					}
138 					System.exit(0);
139 					break;
140 				default:
141 					logger.error("Type unknown in TimerTask");
142 			}
143 		}
144 	}
145 
146 	/**
147 	 * Function to terminate IoSession and Connection.
148 	 */
149 	private static void terminate() {
150 		shutdown = true;
151 		if (immediate) {
152 			ChannelUtils.exit();
153 			// Force exit!
154 			try {
155 				Thread.sleep(Configuration.configuration.TIMEOUTCON);
156 			} catch (InterruptedException e) {
157 			}
158 			if (R66TimerTask.logger.isDebugEnabled()) {
159 				Map<Thread, StackTraceElement[]> map = Thread
160 						.getAllStackTraces();
161 				for (Thread thread : map.keySet()) {
162 					printStackTrace(thread, map.get(thread));
163 				}
164 			}
165 			try {
166 				Thread.sleep(1000);
167 			} catch (InterruptedException e) {
168 			}
169 			System.err.println("Halt System");
170 			Runtime.getRuntime().halt(0);
171 		} else {
172 			launchFinalExit();
173 			immediate = true;
174 			ChannelUtils.exit();
175 			System.err.println("Exit System");
176 			// System.exit(0);
177 		}
178 	}

180 	public static void launchFinalExit() {
181 		Timer timer = new Timer("R66FinalExit", true);
182 		R66TimerTask timerTask = new R66TimerTask(R66TimerTask.TIMER_EXIT);
183 		timer.schedule(timerTask, Configuration.configuration.TIMEOUTCON * 4);
184 	}

186 	/**
187 	 * Function to initialized the SignalHandler
188 	 */
189 	public static void initSignalHandler() {
190 		if (initialized) {
191 			return;
192 		}
193 		Signal diagSignal = new Signal("TERM");
194 		OpenR66SignalHandler diagHandler = new OpenR66SignalHandler();
195 		diagHandler.oldHandler = Signal.handle(diagSignal, diagHandler);
196 		// Not on WINDOWS ?
197 		Configuration.ISUNIX = (!System.getProperty("os.name")
198 				.toLowerCase().startsWith("win"));
199 		System.out.println("ISUNIX: " + Configuration.ISUNIX);
200 		if (Configuration.ISUNIX) {
201 			String vendor = SystemPropertyUtil.get("java.vm.vendor");
202 			vendor = vendor.toLowerCase();
203 			if (vendor.indexOf("ibm") >= 0) {
204 				diagSignal = new Signal("USR1");
205 				diagHandler = new OpenR66SignalHandler();
206 				diagHandler.oldHandler = Signal.handle(diagSignal, diagHandler);
207 			}
208 		}
209 		initialized = true;
210 	}

212 	/**
213 	 * Handle signal
214 	 * 
215 	 * @param signal
216 	 */
217 	public void handle(Signal signal) {
218 		try {
219 			terminate();
220 			// Chain back to previous handler, if one exists
221 			if (oldHandler != SIG_DFL && oldHandler != SIG_IGN) {
222 				oldHandler.handle(signal);
223 			}
224 		} catch (final Exception e) {
225 		}
226 		// ChannelUtils.stopLogger();
227 		System.err.println("Signal: " + signal.getNumber());
228 		System.exit(0);
229 	}
