118     /**
119      * Finalize resources attached to handlers
120      * 
121      * @author Frederic Bregier
122      */
123     private static class R66ChannelGroupFutureListener implements
124             ChannelGroupFutureListener {
125         String name;
126         EventLoopGroup group;
127 
128         public R66ChannelGroupFutureListener(String name, EventLoopGroup group) {
129             this.name = name;
130             this.group = group;
131         }
132 
133         public void operationComplete(ChannelGroupFuture future)
134                 throws Exception {
135             logger.info("Start with shutdown external resources for " + name);
136             if (group != null) {
137                 group.shutdownGracefully();
138             }
139             logger.info("Done with shutdown " + name);
140         }
141     }

143     /**
144      * Terminate all registered channels
145      * 
146      * @return the number of previously registered network channels
147      */
148     private static int terminateCommandChannels() {
149         if (Configuration.configuration.getServerChannelGroup() == null) {
150             return 0;
151         }
152         final int result = Configuration.configuration.getServerChannelGroup().size();
153         logger.info("ServerChannelGroup: " + result);
154         Configuration.configuration.getServerChannelGroup().close()
155                 .addListener(
156                         new R66ChannelGroupFutureListener(
157                                 "ServerChannelGroup",
158                                 Configuration.configuration.getHandlerGroup()));
159         return result;
160     }

200     /**
201      * 
202      * @param localChannelReference
203      * @param block
204      * @return the ChannelFuture of this write operation
205      * @throws OpenR66ProtocolPacketException
206      */
207     public static ChannelFuture writeBackDataBlock(
208             LocalChannelReference localChannelReference, DataBlock block)
209             throws OpenR66ProtocolPacketException {
210         ByteBuf md5 = Unpooled.EMPTY_BUFFER;
211         DbTaskRunner runner = localChannelReference.getSession().getRunner();
212         if (RequestPacket.isMD5Mode(runner.getMode())) {
213             md5 = FileUtils.getHash(block.getBlock(), Configuration.configuration.getDigest());
214         }
215         if (runner.getRank() % 100 == 1 || localChannelReference.getSessionState() != R66FiniteDualStates.DATAS) {
216             localChannelReference.sessionNewState(R66FiniteDualStates.DATAS);
217         }
218         logger.trace("sending data block {}", runner.getRank());
219         DataPacket data = new DataPacket(runner.getRank(), block.getBlock(), md5);// was block.getBlock().copy()
220         ChannelFuture future = writeAbstractLocalPacket(localChannelReference, data, false);
221         runner.incrementRank();
222         return future;
223     }

255     /**
256      * Write an AbstractLocalPacket to the network Channel
257      * 
258      * @param localChannelReference
259      * @param packet
260      * @param wait
261      * @return the ChannelFuture on write operation
262      * @throws OpenR66ProtocolPacketException
263      */
264     public static ChannelFuture writeAbstractLocalPacket(
265             LocalChannelReference localChannelReference, AbstractLocalPacket packet,
266             boolean wait)
267             throws OpenR66ProtocolPacketException {
268         final NetworkPacket networkPacket;
269         try {
270             networkPacket = new NetworkPacket(localChannelReference
271                     .getLocalId(), localChannelReference.getRemoteId(), packet, localChannelReference);
272         } catch (OpenR66ProtocolPacketException e) {
273             logger.error(Messages.getString("ChannelUtils.6") + packet.toString(), //$NON-NLS-1$
274                     e);
275             throw e;
276         }
277         if (wait) {
278             ChannelFuture future = localChannelReference.getNetworkChannel().writeAndFlush(networkPacket);
279             localChannelReference.getNetworkChannelObject().use();
280             try {
281                 future.await(Configuration.configuration.getTIMEOUTCON());
282                 return future;
283             } catch (InterruptedException e) {
284                 return future;
285             }
286         } else {
287             return localChannelReference.getNetworkChannel().writeAndFlush(networkPacket);
288         }
289     }

318     /**
319      * Compute Wait for Traffic in Write (ugly turn around)
320      * 
321      * @param cts
322      * @param size
323      * @return the wait in ms
324      */
325     public static final long willBeWaitingWriting(ChannelTrafficShapingHandler cts, int size) {
326         long currentTime = System.currentTimeMillis();
327         if (cts != null && Configuration.configuration.getServerChannelWriteLimit() > 0) {
328             TrafficCounter tc = cts.trafficCounter();
329             if (tc != null) {
330                 long wait = waitTraffic(Configuration.configuration.getServerChannelWriteLimit(),
331                         tc.currentWrittenBytes() + size,
332                         tc.lastTime(), currentTime);
333                 if (wait > 0) {
334                     return wait;
335                 }
336             }
337         }
338         if (Configuration.configuration.getServerGlobalWriteLimit() > 0) {
339             TrafficCounter tc = Configuration.configuration
340                 .getGlobalTrafficShapingHandler().trafficCounter();
341             if (tc != null) {
342                 long wait = waitTraffic(Configuration.configuration.getServerGlobalWriteLimit(),
343                         tc.currentWrittenBytes() + size,
344                         tc.lastTime(), currentTime);
345                 if (wait > 0) {
346                     return wait;
347                 }
348             }
349         }
350         return 0;
351     }
352 
353     private static final long waitTraffic(long limit, long bytes, long lastTime,
354             long curtime) {
355         long interval = curtime - lastTime;
356         if (interval == 0) {
357             // Time is too short, so just lets continue
358             return 0;
359         }
360         return ((bytes * 1000 / limit - interval) / 10) * 10;
361     }

363     /**
364      * Exit global ChannelFactory
365      */
366     public static void exit() {
367         logger.info("Current launched threads before exit: " + ManagementFactory.getThreadMXBean().getThreadCount());
368         if (Configuration.configuration.getConstraintLimitHandler() != null) {
369             Configuration.configuration.getConstraintLimitHandler().release();
370         }
371         // First try to StopAll
372         TransferUtils.stopSelectedTransfers(DbConstant.admin.getSession(), 0,
373                 null, null, null, null, null, null, null, null, null, true, true, true);
374         Configuration.configuration.setShutdown(true);
375         Configuration.configuration.prepareServerStop();
376         final long delay = Configuration.configuration.getTIMEOUTCON();
377         // Inform others that shutdown
378         if (Configuration.configuration.getLocalTransaction() != null) {
379             Configuration.configuration.getLocalTransaction().shutdownLocalChannels();
380         }
381         logger.info("Unbind server network services");
382         Configuration.configuration.unbindServer();
383         logger.warn(Messages.getString("ChannelUtils.7") + delay + " ms"); //$NON-NLS-1$
384         try {
385             Thread.sleep(delay);
386         } catch (final InterruptedException e) {
387         }
388         NetworkTransaction.closeRetrieveExecutors();
389         if (Configuration.configuration.getLocalTransaction() != null) {
390             Configuration.configuration.getLocalTransaction().debugPrintActiveLocalChannels();
391         }
392         if (Configuration.configuration.getGlobalTrafficShapingHandler() != null) {
393             Configuration.configuration.getGlobalTrafficShapingHandler().release();
394         }
395         logger.info("Exit Shutdown Http");
396         terminateHttpChannels();
397         logger.info("Exit Shutdown Local");
398         if (Configuration.configuration.getLocalTransaction() != null) {
399             Configuration.configuration.getLocalTransaction().closeAll();
400         }
401         logger.info("Exit Shutdown LocalExec");
402         if (Configuration.configuration.isUseLocalExec()) {
403             LocalExecClient.releaseResources();
404         }
405         logger.info("Exit Shutdown Command");
406         terminateCommandChannels();
407         logger.info("Exit Shutdown Db Connection");
408         DbAdmin.closeAllConnection();
409         logger.info("Exit Shutdown ServerStop");
410         Configuration.configuration.serverStop();
411         logger.warn(Messages.getString("ChannelUtils.15")); //$NON-NLS-1$
412         System.err.println(Messages.getString("ChannelUtils.15")); //$NON-NLS-1$
413         stopLogger();
414         //Thread.currentThread().interrupt();
415     }

418         if (WaarpLoggerFactory.getDefaultFactory() instanceof WaarpSlf4JLoggerFactory) {
419             LoggerContext lc = (LoggerContext) LoggerFactory.getILoggerFactory();
420             lc.stop();
421         }

438         if (R66ShutdownHook.isInShutdown()) {
439             return;
440         }
441         Thread thread = new Thread(new ChannelUtils(), "R66 Shutdown Thread");
442         thread.setDaemon(false);
