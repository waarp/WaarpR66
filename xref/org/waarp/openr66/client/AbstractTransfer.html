56  public abstract class AbstractTransfer implements Runnable {

58       * Internal Logger
59       */
60      static protected volatile WaarpLogger logger;

62      protected static String _INFO_ARGS =
63              Messages.getString("AbstractTransfer.0") + Messages.getString("Message.OutputFormat"); //$NON-NLS-1$
64  
65      protected static final String NO_INFO_ARGS = "noinfo";
66  
67      protected final R66Future future;
68  
69      protected final String filename;
70  
71      protected final String rulename;
72  
73      protected final String fileinfo;
74  
75      protected final boolean isMD5;
76  
77      protected final String remoteHost;
78  
79      protected final int blocksize;
80  
81      protected final long id;
82  
83      protected final Timestamp startTime;
84  
85      protected boolean normalInfoAsWarn = true;

87      /**
88       * @param clasz
89       *            Class of Client Transfer
90       * @param future
91       * @param filename
92       * @param rulename
93       * @param fileinfo
94       * @param isMD5
95       * @param remoteHost
96       * @param blocksize
97       * @param id
98       */
99      public AbstractTransfer(Class<?> clasz, R66Future future, String filename,
100             String rulename, String fileinfo,
101             boolean isMD5, String remoteHost, int blocksize, long id, Timestamp timestart) {
102         if (logger == null) {
103             logger = WaarpLoggerFactory.getLogger(clasz);
104         }
105         this.future = future;
106         this.filename = filename;
107         this.rulename = rulename;
108         this.fileinfo = fileinfo;
109         this.isMD5 = isMD5;
110         if (Configuration.configuration.getAliases().containsKey(remoteHost)) {
111             this.remoteHost = Configuration.configuration.getAliases().get(remoteHost);
112         } else {
113             this.remoteHost = remoteHost;
114         }
115         this.blocksize = blocksize;
116         this.id = id;
117         this.startTime = timestart;
118     }

120     /**
121      * Initiate the Request and return a potential DbTaskRunner
122      * 
123      * @return null if an error occurs or a DbTaskRunner
124      */
125     protected DbTaskRunner initRequest() {
126         DbRule rule;
127         try {
128             rule = new DbRule(rulename);
129         } catch (WaarpDatabaseException e) {
130             logger.error("Cannot get Rule: " + rulename, e);
131             future.setResult(new R66Result(new OpenR66DatabaseGlobalException(e), null, true,
132                     ErrorCode.Internal, null));
133             future.setFailure(e);
134             return null;
135         }
136         int mode = rule.getMode();
137         if (isMD5) {
138             mode = RequestPacket.getModeMD5(mode);
139         }
140         DbTaskRunner taskRunner = null;
141         if (id != DbConstant.ILLEGALVALUE) {
142             try {
143                 taskRunner = new DbTaskRunner(id, remoteHost);
144             } catch (WaarpDatabaseException e) {
145                 logger.error("Cannot get task", e);
146                 future.setResult(new R66Result(new OpenR66DatabaseGlobalException(e), null, true,
147                         ErrorCode.QueryRemotelyUnknown, null));
148                 future.setFailure(e);
149                 return null;
150             }
151             // requested
152             taskRunner.setSenderByRequestToValidate(true);
153             if (fileinfo != null && !fileinfo.equals(NO_INFO_ARGS)) {
154                 taskRunner.setFileInformation(fileinfo);
155             }
156             if (startTime != null) {
157                 taskRunner.setStart(startTime);
158             }
159         } else {
160             long originalSize = -1;
161             if (RequestPacket.isSendMode(mode) && !RequestPacket.isThroughMode(mode)) {
162                 File file = new File(filename);
163                 // Change dir
164                 try {
165                     R66Session session = new R66Session();
166                     session.getAuth().specialNoSessionAuth(false, Configuration.configuration.getHOST_ID());
167                     session.getDir().changeDirectory(rule.getSendPath());
168                     R66File filer66 = FileUtils.getFile(logger, session, filename, true, true, false, null);
169                     file = filer66.getTrueFile();
170                 } catch (CommandAbstractException e) {
171                 } catch (OpenR66RunnerErrorException e) {
172                 }
173                 if (file.canRead()) {
174                     originalSize = file.length();
175                     if (originalSize == 0) {
176                         originalSize = -1;
177                     }
178                 }
179             }
180             logger.debug("Filesize: " + originalSize);
181             String sep = PartnerConfiguration.getSeparator(remoteHost);
182             RequestPacket request = new RequestPacket(rulename,
183                     mode, filename, blocksize, 0,
184                     id, fileinfo, originalSize, sep);
185             // Not isRecv since it is the requester, so send => isRetrieve is true
186             boolean isRetrieve = !RequestPacket.isRecvMode(request.getMode());
187             try {
188                 taskRunner =
189                         new DbTaskRunner(rule, isRetrieve, request, remoteHost,
190                                 startTime);
191             } catch (WaarpDatabaseException e) {
192                 logger.error("Cannot get task", e);
193                 future.setResult(new R66Result(new OpenR66DatabaseGlobalException(e), null, true,
194                         ErrorCode.Internal, null));
195                 future.setFailure(e);
196                 return null;
197             }
198         }
199         return taskRunner;
200     }

205     static protected String fileInfo = null;
206     static protected boolean ismd5 = false;
207     static protected int block = 0x10000; // 64K
208                                           // as

211     static protected long idt = DbConstant.ILLEGALVALUE;

215     /**
216      * Parse the parameter and set current values
217      * 
218      * @param args
219      * @param submitOnly
220      *            True if the client is only a submitter (through database)
221      * @return True if all parameters were found and correct
222      */
223     protected static boolean getParams(String[] args, boolean submitOnly) {
224         _INFO_ARGS = Messages.getString("AbstractTransfer.0") + Messages.getString("Message.OutputFormat"); //$NON-NLS-1$
225         if (args.length < 2) {
226             logger.error(_INFO_ARGS);
227             return false;
228         }
229         if (submitOnly) {
230             if (!FileBasedConfiguration
231                     .setSubmitClientConfigurationFromXml(Configuration.configuration, args[0])) {
232                 logger.error(Messages.getString("Configuration.NeedCorrectConfig")); //$NON-NLS-1$
233                 return false;
234             }
235         } else if (!FileBasedConfiguration
236                 .setClientConfigurationFromXml(Configuration.configuration, args[0])) {
237             logger.error(Messages.getString("Configuration.NeedCorrectConfig")); //$NON-NLS-1$
238             return false;
239         }
240         // Now set default values from configuration
241         block = Configuration.configuration.getBLOCKSIZE();
242         int i = 1;
243         try {
244             for (i = 1; i < args.length; i++) {
245                 if (args[i].equalsIgnoreCase("-to")) {
246                     i++;
247                     rhost = args[i];
248                     if (Configuration.configuration.getAliases().containsKey(rhost)) {
249                         rhost = Configuration.configuration.getAliases().get(rhost);
250                     }
251                 } else if (args[i].equalsIgnoreCase("-file")) {
252                     i++;
253                     localFilename = args[i];
254                     localFilename = localFilename.replace('§', '*');
255                 } else if (args[i].equalsIgnoreCase("-rule")) {
256                     i++;
257                     rule = args[i];
258                 } else if (args[i].equalsIgnoreCase("-info")) {
259                     i++;
260                     fileInfo = args[i];
261                 } else if (args[i].equalsIgnoreCase("-md5")) {
262                     ismd5 = true;
263                 } else if (args[i].equalsIgnoreCase("-logWarn")) {
264                     snormalInfoAsWarn = true;
265                 } else if (args[i].equalsIgnoreCase("-notlogWarn")) {
266                     snormalInfoAsWarn = false;
267                 } else if (args[i].equalsIgnoreCase("-block")) {
268                     i++;
269                     block = Integer.parseInt(args[i]);
270                     if (block < 100) {
271                         logger.error(Messages.getString("AbstractTransfer.1") + block); //$NON-NLS-1$
272                         return false;
273                     }
274                 } else if (args[i].equalsIgnoreCase("-nolog")) {
275                     nolog = true;
276                 } else if (args[i].equalsIgnoreCase("-id")) {
277                     i++;
278                     idt = Long.parseLong(args[i]);
279                 } else if (args[i].equalsIgnoreCase("-start")) {
280                     i++;
281                     Date date;
282                     SimpleDateFormat dateFormat = new SimpleDateFormat("yyyyMMddHHmmss");
283                     try {
284                         date = dateFormat.parse(args[i]);
285                         ttimestart = new Timestamp(date.getTime());
286                     } catch (ParseException e) {
287                     }
288                 } else if (args[i].equalsIgnoreCase("-delay")) {
289                     i++;
290                     if (args[i].charAt(0) == '+') {
291                         ttimestart = new Timestamp(System.currentTimeMillis() +
292                                 Long.parseLong(args[i].substring(1)));
293                     } else {
294                         ttimestart = new Timestamp(Long.parseLong(args[i]));
295                     }
296                 }
297             }
298             OutputFormat.getParams(args);
299         } catch (NumberFormatException e) {
300             logger.error(Messages.getString("AbstractTransfer.20") + i); //$NON-NLS-1$
301             return false;
302         }
303         if (fileInfo == null) {
304             fileInfo = NO_INFO_ARGS;
305         }
306         if (rhost != null && rule != null && localFilename != null) {
307             return true;
308         } else if (idt != DbConstant.ILLEGALVALUE && rhost != null) {
309             try {
310                 DbTaskRunner runner = new DbTaskRunner(idt, rhost);
311                 rule = runner.getRuleId();
312                 localFilename = runner.getOriginalFilename();
313                 return true;
314             } catch (WaarpDatabaseException e) {
315                 logger.error(
316                         Messages.getString("AbstractBusinessRequest.NeedMoreArgs", "(-to -rule -file | -to -id)") //$NON-NLS-1$
317                         , e);
318                 return false;
319             }
320 
321         }
322         logger.error(Messages.getString("AbstractBusinessRequest.NeedMoreArgs", "(-to -rule -file | -to -id)") + //$NON-NLS-1$
323                 _INFO_ARGS);
324         return false;
325     }
326     /**
327      * Shared code for finalize one Transfer request in error
328      * @param runner
329      * @param taskRunner
330      */
331     protected void finalizeInErrorTransferRequest(ClientRunner runner, DbTaskRunner taskRunner, ErrorCode code) {
332         if (runner.getLocalChannelReference() != null) {
333             runner.getLocalChannelReference().setErrorMessage(code.getMesg(), code);
334         }
335         taskRunner.setErrorTask(runner.getLocalChannelReference());
336         try {
337             taskRunner.forceSaveStatus();
338             taskRunner.run();
339         } catch (OpenR66RunnerErrorException e1) {
340             runner.changeUpdatedInfo(UpdatedInfo.INERROR, code, true);
341         }
342     }
