48  /**
49   * Abstract class for Transfer operation
50   * 
51   * @author Frederic Bregier
52   * 
53   */
54  public abstract class AbstractTransfer implements Runnable {
55      /**
56       * Internal Logger
57       */
58      static protected volatile WaarpLogger logger;
59  
60      protected static String _INFO_ARGS =
61              Messages.getString("AbstractTransfer.0") + Messages.getString("Message.OutputFormat"); //$NON-NLS-1$
62  
63      protected static final String NO_INFO_ARGS = "noinfo";
64  
65      protected final R66Future future;
66  
67      protected final String filename;
68  
69      protected final String rulename;
70  
71      protected final String fileinfo;
72  
73      protected final boolean isMD5;
74  
75      protected final String remoteHost;
76  
77      protected final int blocksize;
78  
79      protected final long id;
80  
81      protected final Timestamp startTime;
82  
83      protected boolean normalInfoAsWarn = true;
84  
85      /**
86       * @param clasz
87       *            Class of Client Transfer
88       * @param future
89       * @param filename
90       * @param rulename
91       * @param fileinfo
92       * @param isMD5
93       * @param remoteHost
94       * @param blocksize
95       * @param id
96       */
97      public AbstractTransfer(Class<?> clasz, R66Future future, String filename,
98              String rulename, String fileinfo,
99              boolean isMD5, String remoteHost, int blocksize, long id, Timestamp timestart) {
100         if (logger == null) {
101             logger = WaarpLoggerFactory.getLogger(clasz);
102         }
103         this.future = future;
104         this.filename = filename;
105         this.rulename = rulename;
106         this.fileinfo = fileinfo;
107         this.isMD5 = isMD5;
108         if (Configuration.configuration.aliases.containsKey(remoteHost)) {
109             this.remoteHost = Configuration.configuration.aliases.get(remoteHost);
110         } else {
111             this.remoteHost = remoteHost;
112         }
113         this.blocksize = blocksize;
114         this.id = id;
115         this.startTime = timestart;
116     }
117 
118     /**
119      * Initiate the Request and return a potential DbTaskRunner
120      * 
121      * @return null if an error occurs or a DbTaskRunner
122      */
123     protected DbTaskRunner initRequest() {
124         DbRule rule;
125         try {
126             rule = new DbRule(DbConstant.admin.session, rulename);
127         } catch (WaarpDatabaseException e) {
128             logger.error("Cannot get Rule: " + rulename, e);
129             future.setResult(new R66Result(new OpenR66DatabaseGlobalException(e), null, true,
130                     ErrorCode.Internal, null));
131             future.setFailure(e);
132             return null;
133         }
134         int mode = rule.mode;
135         if (isMD5) {
136             mode = RequestPacket.getModeMD5(mode);
137         }
138         DbTaskRunner taskRunner = null;
139         if (id != DbConstant.ILLEGALVALUE) {
140             try {
141                 taskRunner = new DbTaskRunner(DbConstant.admin.session, id,
142                         remoteHost);
143             } catch (WaarpDatabaseException e) {
144                 logger.error("Cannot get task", e);
145                 future.setResult(new R66Result(new OpenR66DatabaseGlobalException(e), null, true,
146                         ErrorCode.QueryRemotelyUnknown, null));
147                 future.setFailure(e);
148                 return null;
149             }
150             // requested
151             taskRunner.setSenderByRequestToValidate(true);
152             if (fileinfo != null && !fileinfo.equals(NO_INFO_ARGS)) {
153                 taskRunner.setFileInformation(fileinfo);
154             }
155             if (startTime != null) {
156                 taskRunner.setStart(startTime);
157             }
158         } else {
159             long originalSize = -1;
160             if (RequestPacket.isSendMode(mode) && !RequestPacket.isThroughMode(mode)) {
161                 File file = new File(filename);
162                 // Change dir
163                 try {
164                     R66Session session = new R66Session();
165                     session.getAuth().specialNoSessionAuth(false, Configuration.configuration.HOST_ID);
166                     session.getDir().changeDirectory(rule.getSendPath());
167                     R66File filer66 = FileUtils.getFile(logger, session, filename, true, true, false, null);
168                     file = filer66.getTrueFile();
169                 } catch (CommandAbstractException e) {
170                 } catch (OpenR66RunnerErrorException e) {
171                 }
172                 if (file.canRead()) {
173                     originalSize = file.length();
174                     if (originalSize == 0) {
175                         originalSize = -1;
176                     }
177                 }
178             }
179             logger.debug("Filesize: " + originalSize);
180             String sep = PartnerConfiguration.getSeparator(remoteHost);
181             RequestPacket request = new RequestPacket(rulename,
182                     mode, filename, blocksize, 0,
183                     id, fileinfo, originalSize, sep);
184             // Not isRecv since it is the requester, so send => isRetrieve is true
185             boolean isRetrieve = !RequestPacket.isRecvMode(request.getMode());
186             try {
187                 taskRunner =
188                         new DbTaskRunner(DbConstant.admin.session, rule, isRetrieve, request,
189                                 remoteHost, startTime);
190             } catch (WaarpDatabaseException e) {
191                 logger.error("Cannot get task", e);
192                 future.setResult(new R66Result(new OpenR66DatabaseGlobalException(e), null, true,
193                         ErrorCode.Internal, null));
194                 future.setFailure(e);
195                 return null;
196             }
197         }
198         return taskRunner;
199     }

201     static protected String rhost = null;
202     static protected String localFilename = null;
203     static protected String rule = null;
204     static protected String fileInfo = null;
205     static protected boolean ismd5 = false;
206     static protected int block = 0x10000; // 64K
207                                           // as
208                                           // default
209     static protected boolean nolog = false;
210     static protected long idt = DbConstant.ILLEGALVALUE;
211     static protected Timestamp ttimestart = null;
212     static protected boolean snormalInfoAsWarn = true;

214     /**
215      * Parse the parameter and set current values
216      * 
217      * @param args
218      * @param submitOnly
219      *            True if the client is only a submitter (through database)
220      * @return True if all parameters were found and correct
221      */
222     protected static boolean getParams(String[] args, boolean submitOnly) {
223         _INFO_ARGS = Messages.getString("AbstractTransfer.0") + Messages.getString("Message.OutputFormat"); //$NON-NLS-1$
224         if (args.length < 2) {
225             logger.error(_INFO_ARGS);
226             return false;
227         }
228         if (submitOnly) {
229             if (!FileBasedConfiguration
230                     .setSubmitClientConfigurationFromXml(Configuration.configuration, args[0])) {
231                 logger.error(Messages.getString("Configuration.NeedCorrectConfig")); //$NON-NLS-1$
232                 return false;
233             }
234         } else if (!FileBasedConfiguration
235                 .setClientConfigurationFromXml(Configuration.configuration, args[0])) {
236             logger.error(Messages.getString("Configuration.NeedCorrectConfig")); //$NON-NLS-1$
237             return false;
238         }
239         // Now set default values from configuration
240         block = Configuration.configuration.BLOCKSIZE;
241         int i = 1;
242         try {
243             for (i = 1; i < args.length; i++) {
244                 if (args[i].equalsIgnoreCase("-to")) {
245                     i++;
246                     rhost = args[i];
247                 } else if (args[i].equalsIgnoreCase("-file")) {
248                     i++;
249                     localFilename = args[i];
250                     localFilename = localFilename.replace('ยง', '*');
251                 } else if (args[i].equalsIgnoreCase("-rule")) {
252                     i++;
253                     rule = args[i];
254                 } else if (args[i].equalsIgnoreCase("-info")) {
255                     i++;
256                     fileInfo = args[i];
257                 } else if (args[i].equalsIgnoreCase("-md5")) {
258                     ismd5 = true;
259                 } else if (args[i].equalsIgnoreCase("-logWarn")) {
260                     snormalInfoAsWarn = true;
261                 } else if (args[i].equalsIgnoreCase("-notlogWarn")) {
262                     snormalInfoAsWarn = false;
263                 } else if (args[i].equalsIgnoreCase("-block")) {
264                     i++;
265                     block = Integer.parseInt(args[i]);
266                     if (block < 100) {
267                         logger.error(Messages.getString("AbstractTransfer.1") + block); //$NON-NLS-1$
268                         return false;
269                     }
270                 } else if (args[i].equalsIgnoreCase("-nolog")) {
271                     nolog = true;
272                 } else if (args[i].equalsIgnoreCase("-id")) {
273                     i++;
274                     idt = Long.parseLong(args[i]);
275                 } else if (args[i].equalsIgnoreCase("-start")) {
276                     i++;
277                     Date date;
278                     SimpleDateFormat dateFormat = new SimpleDateFormat("yyyyMMddHHmmss");
279                     try {
280                         date = dateFormat.parse(args[i]);
281                         ttimestart = new Timestamp(date.getTime());
282                     } catch (ParseException e) {
283                     }
284                 } else if (args[i].equalsIgnoreCase("-delay")) {
285                     i++;
286                     if (args[i].charAt(0) == '+') {
287                         ttimestart = new Timestamp(System.currentTimeMillis() +
288                                 Long.parseLong(args[i].substring(1)));
289                     } else {
290                         ttimestart = new Timestamp(Long.parseLong(args[i]));
291                     }
292                 }
293             }
294             OutputFormat.getParams(args);
295         } catch (NumberFormatException e) {
296             logger.error(Messages.getString("AbstractTransfer.20") + i); //$NON-NLS-1$
297             return false;
298         }
299         if (fileInfo == null) {
300             fileInfo = NO_INFO_ARGS;
301         }
302         if (rhost != null && rule != null && localFilename != null) {
303             return true;
304         } else if (idt != DbConstant.ILLEGALVALUE && rhost != null) {
305             try {
306                 DbTaskRunner runner = new DbTaskRunner(DbConstant.admin.session, idt,
307                         rhost);
308                 rule = runner.getRuleId();
309                 localFilename = runner.getOriginalFilename();
310                 return true;
311             } catch (WaarpDatabaseException e) {
312                 logger.error(
313                         Messages.getString("AbstractBusinessRequest.NeedMoreArgs", "(-to -rule -file | -to -id)") //$NON-NLS-1$
314                         , e);
315                 return false;
316             }
317 
318         }
319         logger.error(Messages.getString("AbstractBusinessRequest.NeedMoreArgs", "(-to -rule -file | -to -id)") + //$NON-NLS-1$
320                 _INFO_ARGS);
321         return false;
322     }
