44  /**
45   * Direct Transfer from a client with or without database connection to transfer for multiple files to multiple hosts at once.<br>
46   * Files will have to be separated by ','.<br>
47   * Hosts will have to be separated by ','.<br>
48   * <br>
49   * For instance: -to host1,host2,host3 -file file1,file2 <br>
50   * Will generate: <br>
51   * -to host1 -file file1<br>
52   * -to host1 -file file2<br>
53   * -to host2 -file file1<br>
54   * -to host2 -file file2<br>
55   * -to host3 -file file1<br>
56   * -to host3 -file file2<br>
57   * 
58   * @author Frederic Bregier
59   * 
60   */
61  public class MultipleDirectTransfer extends DirectTransfer {
62      private int errorMultiple = 0;
63      private int doneMultiple = 0;
64      private List<OutputFormat> results = new ArrayList<OutputFormat>();
65  
66      public MultipleDirectTransfer(R66Future future, String remoteHost,
67              String filename, String rulename, String fileinfo, boolean isMD5, int blocksize,
68              long id,
69              NetworkTransaction networkTransaction) {
70          // no starttime since it is direct (blocking request, no delay)
71          super(future, remoteHost, filename, rulename, fileinfo, isMD5, blocksize, id, networkTransaction);
72      }

74      public static List<String> getRemoteFiles(DbRule dbrule, String[] localfilenames, String requested,
75              NetworkTransaction networkTransaction) {
76          List<String> files = new ArrayList<String>();
77          for (String filename : localfilenames) {
78              if (!(filename.contains("*") || filename.contains("?") || filename.contains("~"))) {
79                  files.add(filename);
80              } else {
81                  // remote query
82                  R66Future futureInfo = new R66Future(true);
83                  logger.info(Messages.getString("Transfer.3") + filename + " to " + requested); //$NON-NLS-1$
84                  RequestInformation info = new RequestInformation(futureInfo, requested, rule, filename,
85                          (byte) InformationPacket.ASKENUM.ASKLIST.ordinal(), -1, false, networkTransaction);
86                  info.run();
87                  futureInfo.awaitUninterruptibly();
88                  if (futureInfo.isSuccess()) {
89                      ValidPacket valid = (ValidPacket) futureInfo.getResult().getOther();
90                      if (valid != null) {
91                          String line = valid.getSheader();
92                          String[] lines = line.split("\n");
93                          for (String string : lines) {
94                              File tmpFile = new File(string);
95                              files.add(tmpFile.getPath());
96                          }
97                      }
98                  } else {
99                      logger.error(Messages.getString("Transfer.6") + filename + " to " + requested + ": " +
100                             (futureInfo.getCause() == null ? "" : futureInfo.getCause().getMessage())); //$NON-NLS-1$
101                 }
102             }
103         }
104         return files;
105     }

107     public static List<String> getLocalFiles(DbRule dbrule, String[] localfilenames) {
108         List<String> files = new ArrayList<String>();
109         R66Session session = new R66Session();
110         session.getAuth().specialNoSessionAuth(false, Configuration.configuration.getHOST_ID());
111         R66Dir dir = new R66Dir(session);
112         try {
113             dir.changeDirectory(dbrule.getSendPath());
114         } catch (CommandAbstractException e) {
115         }
116         if (localfilenames != null) {
117             for (String filename : localfilenames) {
118                 if (!(filename.contains("*") || filename.contains("?") || filename.contains("~"))) {
119                     logger.info("Direct add: " + filename);
120                     files.add(filename);
121                 } else {
122                     // local: must check
123                     logger.info("Local Ask for " + filename + " from " + dir.getFullPath());
124                     List<String> list;
125                     try {
126                         list = dir.list(filename);
127                         if (list != null) {
128                             files.addAll(list);
129                         }
130                     } catch (CommandAbstractException e) {
131                         logger.warn(Messages.getString("Transfer.14") + filename + " : " + e.getMessage()); //$NON-NLS-1$
132                     }
133                 }
134             }
135         }
136         return files;
137     }

139     @Override
140     public void run() {
141         String[] localfilenames = filename.split(",");
142         String[] rhosts = remoteHost.split(",");
143         boolean inError = false;
144         R66Result resultError = null;
145         // first check if filenames contains wildcards
146         DbRule dbrule = null;
147         try {
148             dbrule = new DbRule(DbConstant.admin.getSession(), rulename);
149         } catch (WaarpDatabaseException e1) {
150             logger.error(Messages.getString("Transfer.18"), e1); //$NON-NLS-1$
151             this.future.setFailure(e1);
152             return;
153         }
154         List<String> files = null;
155         if (dbrule.isSendMode()) {
156             files = getLocalFiles(dbrule, localfilenames);
157         }
158         for (String host : rhosts) {
159             host = host.trim();
160             if (host != null && !host.isEmpty()) {
161                 if (dbrule.isRecvMode()) {
162                     files = getRemoteFiles(dbrule, localfilenames, host, networkTransaction);
163                 }
164                 for (String filename : files) {
165                     filename = filename.trim();
166                     if (filename != null && !filename.isEmpty()) {
167                         logger.info("Launch transfer to " + host + " with file " + filename);
168                         long time1 = System.currentTimeMillis();
169                         R66Future future = new R66Future(true);
170                         DirectTransfer transaction = new DirectTransfer(future,
171                                 host, filename, rule, fileInfo, ismd5, block, idt,
172                                 networkTransaction);
173                         transaction.normalInfoAsWarn = normalInfoAsWarn;
174                         logger.debug("rhost: " + host + ":" + transaction.remoteHost);
175                         transaction.run();
176                         future.awaitUninterruptibly();
177                         long time2 = System.currentTimeMillis();
178                         logger.debug("finish transfer: " + future.isSuccess());
179                         long delay = time2 - time1;
180                         R66Result result = future.getResult();
181                         OutputFormat outputFormat = new OutputFormat("Unique "
182                                 + MultipleDirectTransfer.class.getSimpleName(), null);
183                         if (future.isSuccess()) {
184                             if (result.getRunner().getErrorInfo() == ErrorCode.Warning) {
185                                 outputFormat.setValue(FIELDS.status.name(), 1);
186                                 outputFormat
187                                         .setValue(
188                                                 FIELDS.statusTxt.name(),
189                                                 Messages.getString("Transfer.Status") + Messages.getString("RequestInformation.Warned")); //$NON-NLS-1$
190                             } else {
191                                 outputFormat.setValue(FIELDS.status.name(), 0);
192                                 outputFormat
193                                         .setValue(
194                                                 FIELDS.statusTxt.name(),
195                                                 Messages.getString("Transfer.Status") + Messages.getString("RequestInformation.Success")); //$NON-NLS-1$
196                             }
197                             outputFormat.setValue(FIELDS.remote.name(), host);
198                             outputFormat.setValueString(result.getRunner().getJson());
199                             outputFormat.setValue("filefinal", (result.getFile() != null ? result.getFile().toString()
200                                     : "no file"));
201                             outputFormat.setValue("delay", delay);
202                             getResults().add(outputFormat);
203                             setDoneMultiple(getDoneMultiple() + 1);
204                             if (transaction.normalInfoAsWarn) {
205                                 logger.warn(outputFormat.loggerOut());
206                             } else {
207                                 logger.info(outputFormat.loggerOut());
208                             }
209                             if (nolog || result.getRunner().shallIgnoreSave()) {
210                                 // In case of success, delete the runner
211                                 try {
212                                     result.getRunner().delete();
213                                 } catch (WaarpDatabaseException e) {
214                                     logger.warn("Cannot apply nolog to     " + result.getRunner().toShortString(),
215                                             e);
216                                 }
217                             }
218                         } else {
219                             if (result == null || result.getRunner() == null) {
220                                 outputFormat.setValue(FIELDS.status.name(), 2);
221                                 outputFormat.setValue(FIELDS.statusTxt.name(),
222                                         Messages.getString("Transfer.FailedNoId")); //$NON-NLS-1$
223                                 outputFormat.setValue(FIELDS.remote.name(), host);
224                                 logger.error(outputFormat.loggerOut(), future.getCause());
225                                 outputFormat.setValue(FIELDS.error.name(), future.getCause().getMessage());
226                                 outputFormat.sysout();
227                                 networkTransaction.closeAll();
228                                 System.exit(ErrorCode.Unknown.ordinal());
229                             }
230                             if (result.getRunner().getErrorInfo() == ErrorCode.Warning) {
231                                 outputFormat.setValue(FIELDS.status.name(), 1);
232                                 outputFormat
233                                         .setValue(
234                                                 FIELDS.statusTxt.name(),
235                                                 Messages.getString("Transfer.Status") + Messages.getString("RequestInformation.Warned")); //$NON-NLS-1$
236                             } else {
237                                 outputFormat.setValue(FIELDS.status.name(), 2);
238                                 outputFormat
239                                         .setValue(
240                                                 FIELDS.statusTxt.name(),
241                                                 Messages.getString("Transfer.Status") + Messages.getString("RequestInformation.Failure")); //$NON-NLS-1$
242                             }
243                             outputFormat.setValue(FIELDS.remote.name(), host);
244                             outputFormat.setValueString(result.getRunner().getJson());
245                             if (result.getRunner().getErrorInfo() == ErrorCode.Warning) {
246                                 logger.warn(outputFormat.loggerOut(), future.getCause());
247                             } else {
248                                 logger.error(outputFormat.loggerOut(), future.getCause());
249                             }
250                             outputFormat.setValue(FIELDS.error.name(), future.getCause().getMessage());
251                             getResults().add(outputFormat);
252                             setErrorMultiple(getErrorMultiple() + 1);
253                             inError = true;
254                             if (result != null) {
255                                 inError = true;
256                                 resultError = result;
257                             }
258                         }
259                     }
260                 }
261             }
262         }
263         if (inError) {
264             if (resultError != null) {
265                 this.future.setResult(resultError);
266             }
267             this.future.cancel();
268         } else {
269             this.future.setSuccess();
270         }
271     }

273     public static void main(String[] args) {
274         WaarpLoggerFactory.setDefaultFactory(new WaarpSlf4JLoggerFactory(null));
275         if (logger == null) {
276             logger = WaarpLoggerFactory.getLogger(MultipleDirectTransfer.class);
277         }
278         if (!getParams(args, false)) {
279             logger.error(Messages.getString("Configuration.WrongInit")); //$NON-NLS-1$
280             if (!OutputFormat.isQuiet()) {
281                 System.out.println(Messages.getString("Configuration.WrongInit")); //$NON-NLS-1$
282             }
283             if (DbConstant.admin != null && DbConstant.admin.isActive()) {
284                 DbConstant.admin.close();
285             }
286             ChannelUtils.stopLogger();
287             System.exit(2);
288         }
289 
290         Configuration.configuration.pipelineInit();
291         NetworkTransaction networkTransaction = new NetworkTransaction();
292         try {
293             R66Future future = new R66Future(true);
294             long time1 = System.currentTimeMillis();
295             MultipleDirectTransfer multipleDirectTransfer =
296                     new MultipleDirectTransfer(future, rhost, localFilename,
297                             rule, fileInfo, ismd5, block, idt,
298                             networkTransaction);
299             multipleDirectTransfer.normalInfoAsWarn = snormalInfoAsWarn;
300             multipleDirectTransfer.run();
301             future.awaitUninterruptibly();
302             long time2 = System.currentTimeMillis();
303             logger.debug("finish all transfers: " + future.isSuccess());
304             long delay = time2 - time1;
305             OutputFormat outputFormat = new OutputFormat(MultipleDirectTransfer.class.getSimpleName(), args);
306             if (future.isSuccess()) {
307                 outputFormat.setValue(FIELDS.status.name(), 0);
308                 outputFormat
309                         .setValue(
310                                 FIELDS.statusTxt.name(),
311                                 "Multiple " + Messages.getString("Transfer.Status") + Messages.getString("RequestInformation.Success")); //$NON-NLS-1$
312                 outputFormat.setValue(FIELDS.remote.name(), rhost);
313                 outputFormat.setValue("ok", multipleDirectTransfer.getDoneMultiple());
314                 outputFormat.setValue("delay", delay);
315                 if (multipleDirectTransfer.normalInfoAsWarn) {
316                     logger.warn(outputFormat.loggerOut());
317                 } else {
318                     logger.info(outputFormat.loggerOut());
319                 }
320                 if (!OutputFormat.isQuiet()) {
321                     outputFormat.sysout();
322                     for (OutputFormat result : multipleDirectTransfer.getResults()) {
323                         System.out.println();
324                         result.sysout();
325                     }
326                 }
327             } else {
328                 outputFormat.setValue(FIELDS.status.name(), 2);
329                 outputFormat
330                         .setValue(
331                                 FIELDS.statusTxt.name(),
332                                 "Multiple " + Messages.getString("Transfer.Status") + Messages.getString("RequestInformation.Failure")); //$NON-NLS-1$
333                 outputFormat.setValue(FIELDS.remote.name(), rhost);
334                 outputFormat.setValue("ok", multipleDirectTransfer.getDoneMultiple());
335                 outputFormat.setValue("ko", multipleDirectTransfer.getErrorMultiple());
336                 outputFormat.setValue("delay", delay);
337                 logger.error(outputFormat.loggerOut());
338                 if (!OutputFormat.isQuiet()) {
339                     outputFormat.sysout();
340                     for (OutputFormat result : multipleDirectTransfer.getResults()) {
341                         System.out.println();
342                         result.sysout();
343                     }
344                 }
345                 networkTransaction.closeAll();
346                 System.exit(multipleDirectTransfer.getErrorMultiple());
347             }
348         } catch (Throwable e) {
349             logger.error("Exception", e);
350         } finally {
351             networkTransaction.closeAll();
352             System.exit(0);
353         }
354     }
