35  /**
36   * Client to submit a transfer
37   * 
38   * @author Frederic Bregier
39   * 
40   */
41  public class SubmitTransfer extends AbstractTransfer {
42  
43  	public SubmitTransfer(R66Future future, String remoteHost,
44  			String filename, String rulename, String fileinfo, boolean isMD5, int blocksize,
45  			long id,
46  			Timestamp starttime) {
47  		super(SubmitTransfer.class,
48  				future, filename, rulename, fileinfo, isMD5, remoteHost, blocksize, id, starttime);
49  	}
50  
51  	public void run() {
52  		if (logger == null) {
53  			logger = WaarpInternalLoggerFactory.getLogger(SubmitTransfer.class);
54  		}
55  		DbTaskRunner taskRunner = this.initRequest();
56  		taskRunner.changeUpdatedInfo(AbstractDbData.UpdatedInfo.TOSUBMIT);
57  		try {
58  			taskRunner.update();
59  		} catch (WaarpDatabaseException e) {
60  			logger.debug("Cannot prepare task", e);
61  			R66Result result = new R66Result(new OpenR66DatabaseGlobalException(e), null, true,
62  					ErrorCode.Internal, taskRunner);
63  			future.setResult(result);
64  			future.setFailure(e);
65  			return;
66  		}
67  		R66Result result = new R66Result(null, false, ErrorCode.InitOk, taskRunner);
68  		future.setResult(result);
69  		future.setSuccess();
70  	}
71  
72  	/**
73  	 * 
74  	 * @param args
75  	 *            configuration file, the remoteHost Id, the file to transfer, the rule, file
76  	 *            transfer information as arguments and optionally isMD5=1 for true or 0 for
77  	 *            false(default) and the blocksize if different than default
78  	 */
79  	public static void main(String[] args) {
80  		InternalLoggerFactory.setDefaultFactory(new WaarpSlf4JLoggerFactory(null));
81  		if (logger == null) {
82  			logger = WaarpInternalLoggerFactory.getLogger(SubmitTransfer.class);
83  		}
84  		if (!getParams(args, true)) {
85  			logger.error("Wrong initialization");
86  			if (DbConstant.admin != null && DbConstant.admin.isConnected) {
87  				DbConstant.admin.close();
88  			}
89  			ChannelUtils.stopLogger();
90  			System.exit(1);
91  		}
92  		R66Future future = new R66Future(true);
93  		SubmitTransfer transaction = new SubmitTransfer(future,
94  				rhost, localFilename, rule, fileInfo, ismd5, block, DbConstant.ILLEGALVALUE,
95  				ttimestart);
96  		transaction.run();
97  		future.awaitUninterruptibly();
98  		DbTaskRunner runner = future.getResult().runner;
99  		if (future.isSuccess()) {
100 			logger.warn("Prepare transfer in\n    SUCCESS\n    " + runner.toShortString() +
101 					"<REMOTE>" + rhost + "</REMOTE>");
102 		} else {
103 			if (runner != null) {
104 				logger.error("Prepare transfer in\n    FAILURE\n     " + runner.toShortString() +
105 						"<REMOTE>" + rhost + "</REMOTE>", future.getCause());
106 			} else {
107 				logger.error("Prepare transfer in\n    FAILURE\n     ", future.getCause());
108 			}
109 			DbConstant.admin.close();
110 			ChannelUtils.stopLogger();
111 			System.exit(future.getResult().code.ordinal());
112 		}
113 		DbConstant.admin.close();
114 	}
