85  	/**
86  	 * Parse the parameter and set current values
87  	 * 
88  	 * @param args
89  	 * @return True if all parameters were found and correct
90  	 */
91  	protected static boolean getParams(String[] args) {
92  		if (args.length < 5) {
93  			logger
94  					.error("Needs at least 5 arguments:\n" +
95  							"  the XML client configuration file,\n" +
96  							"  '-id' the transfer Id,\n" +
97  							"  '-to' the requested host Id or '-from' the requester host Id " +
98  							"(localhost will be the opposite),\n" +
99  							"Other options (only one):\n" +
100 							"  '-cancel' to cancel completely the transfer,\n" +
101 							"  '-stop' to stop the transfer (maybe restarted),\n" +
102 							"  '-restart' to restart if possible a transfer and optionnally the following arguments may be specified for a restart:\n"+
103 							"      '-start' \"time start\" as yyyyMMddHHmmss (override previous -delay options)\n"
104 							+
105 							"      '-delay' \"+delay in ms\" as delay in ms from current time(override previous -start options)\n"
106 							+
107 							"      '-delay' \"delay in ms\" as time in ms (override previous -start options)");
108 			return false;
109 		}
110 		if (!FileBasedConfiguration
111 				.setClientConfigurationFromXml(Configuration.configuration, args[0])) {
112 			logger
113 					.error("Needs a correct configuration file as first argument");
114 			return false;
115 		}
116 		for (int i = 1; i < args.length; i++) {
117 			if (args[i].equalsIgnoreCase("-id")) {
118 				i++;
119 				try {
120 					sspecialId = Long.parseLong(args[i]);
121 				} catch (NumberFormatException e) {
122 					logger.error("Cannot value Id: "+args[i],e);
123 					return false;
124 				}
125 			} else if (args[i].equalsIgnoreCase("-to")) {
126 				i++;
127 				srequested = args[i];
128 				try {
129 					srequester = Configuration.configuration.getHostId(DbConstant.admin.session,
130 							srequested);
131 				} catch (WaarpDatabaseException e) {
132 					logger.error("Cannot get Host Id: " + srequester, e);
133 					return false;
134 				}
135 			} else if (args[i].equalsIgnoreCase("-from")) {
136 				i++;
137 				srequester = args[i];
138 				try {
139 					srequested = Configuration.configuration.getHostId(DbConstant.admin.session,
140 							srequester);
141 				} catch (WaarpDatabaseException e) {
142 					logger.error("Cannot get Host Id: " + srequested, e);
143 					return false;
144 				}
145 			} else if (args[i].equalsIgnoreCase("-cancel")) {
146 				scancel = true;
147 			} else if (args[i].equalsIgnoreCase("-stop")) {
148 				sstop = true;
149 			} else if (args[i].equalsIgnoreCase("-restart")) {
150 				srestart = true;
151 			} else if (args[i].equalsIgnoreCase("-start")) {
152 				i++;
153 				srestarttime = args[i];
154 			} else if (args[i].equalsIgnoreCase("-delay")) {
155 				i++;
156 				if (args[i].charAt(0) == '+') {
157 					Date date = new Date(System.currentTimeMillis() +
158 							Long.parseLong(args[i].substring(1)));
159 					SimpleDateFormat dateFormat = new SimpleDateFormat("yyyyMMddHHmmss");
160 					srestarttime = dateFormat.format(date);
161 				} else {
162 					Date date = new Date(Long.parseLong(args[i]));
163 					SimpleDateFormat dateFormat = new SimpleDateFormat("yyyyMMddHHmmss");
164 					srestarttime = dateFormat.format(date);
165 				}
166 			}
167 		}
168 		if ((scancel && srestart) || (scancel && sstop) || (srestart && sstop)) {
169 			logger.error("Cannot cancel or restart or stop at the same time");
170 			return false;
171 		}
172 		if (sspecialId == DbConstant.ILLEGALVALUE || srequested == null) {
173 			logger.error("TransferId and Requested/Requester HostId must be set");
174 			return false;
175 		}
176 
177 		return true;
178 	}

207 	public RequestTransfer(R66Future future, long specialId, String requested, String requester,
208 			boolean cancel, boolean stop, boolean restart, String restarttime,
209 			NetworkTransaction networkTransaction) {
210 		this.future = future;
211 		this.specialId = specialId;
212 		this.requested = requested;
213 		this.requester = requester;
214 		this.cancel = cancel;
215 		this.stop = stop;
216 		this.restart = restart;
217 		this.restarttime = restarttime;
218 		this.networkTransaction = networkTransaction;

221 	public void run() {
222 		if (logger == null) {
223 			logger = WaarpInternalLoggerFactory.getLogger(RequestTransfer.class);
224 		}
225 		DbTaskRunner runner = null;
226 		try {
227 			runner = new DbTaskRunner(DbConstant.admin.session, null, null,
228 					specialId, requester, requested);
229 			logger.info("Found previous Runner: "+runner.toString());
230 		} catch (WaarpDatabaseException e) {
231 			// Maybe we can ask to the remote
232 			R66Future futureInfo = new R66Future(true);
233 			RequestInformation requestInformation = new RequestInformation(futureInfo, srequested, null, null, (byte) -1, specialId, true, networkTransaction);
234 			requestInformation.run();
235 			futureInfo.awaitUninterruptibly();
236 			if (futureInfo.isSuccess()) {
237 				R66Result r66result = futureInfo.getResult();
238 				ValidPacket info = (ValidPacket) r66result.other;
239 				String xml = info.getSheader();
240 				Document document;
241 				try {
242 					document = DocumentHelper.parseText(xml);
243 				} catch (DocumentException e1) {
244 					logger.error("Cannot find the transfer");
245 					future.setResult(new R66Result(new OpenR66DatabaseGlobalException(e1), null, true,
246 							ErrorCode.Internal, null));
247 					future.setFailure(e);
248 					return;
249 				}
250 				try {
251 					runner = DbTaskRunner.fromXml(DbConstant.admin.session, document.getRootElement(), true);
252 					logger.info("Get Runner from remote: "+runner.toString());
253 					if (runner.getSpecialId() == DbConstant.ILLEGALVALUE || ! runner.isSender()) {
254 						logger.error("Cannot find the transfer");
255 						future.setResult(new R66Result(new OpenR66DatabaseGlobalException(e), null, true,
256 								ErrorCode.Internal, null));
257 						future.setFailure(e);
258 						return;
259 					}
260 					if (runner.isAllDone()) {
261 						logger.error("Transfer already finished");
262 						future.setResult(new R66Result(new OpenR66DatabaseGlobalException(e), null, true,
263 								ErrorCode.Internal, null));
264 						future.setFailure(e);
265 						return;
266 					}
267 				} catch (OpenR66ProtocolBusinessException e1) {
268 					logger.error("Cannot find the transfer");
269 					future.setResult(new R66Result(new OpenR66DatabaseGlobalException(e1), null, true,
270 							ErrorCode.Internal, null));
271 					future.setFailure(e);
272 					return;
273 				}
274 			} else {
275 				logger.error("Cannot find the transfer");
276 				future.setResult(new R66Result(new OpenR66DatabaseGlobalException(e), null, true,
277 						ErrorCode.Internal, null));
278 				future.setFailure(e);
279 				return;
280 			}
281 		}
282 		if (cancel || stop || restart) {
283 			if (cancel) {
284 				// Cancel the task and delete any file if in retrieve
285 				if (runner.isAllDone()) {
286 					// nothing to do since already finished
287 					setDone(runner);
288 					logger.info("Transfer already finished: " + runner.toString());
289 					future.setResult(new R66Result(null, true, ErrorCode.TransferOk, runner));
290 					future.getResult().runner = runner;
291 					future.setSuccess();
292 					return;
293 				} else {
294 					// Send a request of cancel
295 					ErrorCode code = sendValid(runner, LocalPacketFactory.CANCELPACKET);
296 					switch (code) {
297 						case CompleteOk:
298 							logger.info("Transfer cancel requested and done: {}",
299 									runner);
300 							break;
301 						case TransferOk:
302 							logger.info("Transfer cancel requested but already finished: {}",
303 									runner);
304 							break;
305 						default:
306 							logger.info("Transfer cancel requested but internal error: {}",
307 									runner);
308 							break;
309 					}
310 				}
311 			} else if (stop) {
312 				// Just stop the task
313 				// Send a request
314 				ErrorCode code = sendValid(runner, LocalPacketFactory.STOPPACKET);
315 				switch (code) {
316 					case CompleteOk:
317 						logger.info("Transfer stop requested and done: {}", runner);
318 						break;
319 					case TransferOk:
320 						logger.info("Transfer stop requested but already finished: {}",
321 								runner);
322 						break;
323 					default:
324 						logger.info("Transfer stop requested but internal error: {}",
325 								runner);
326 						break;
327 				}
328 			} else if (restart) {
329 				// Restart if already stopped and not finished
330 				ErrorCode code = sendValid(runner, LocalPacketFactory.VALIDPACKET);
331 				switch (code) {
332 					case QueryStillRunning:
333 						logger.info(
334 								"Transfer restart requested but already active and running: {}",
335 								runner);
336 						break;
337 					case Running:
338 						logger.info("Transfer restart requested but already running: {}",
339 								runner);
340 						break;
341 					case PreProcessingOk:
342 						logger.info("Transfer restart requested and restarted: {}",
343 								runner);
344 						break;
345 					case CompleteOk:
346 						logger.info("Transfer restart requested but already finished: {}",
347 								runner);
348 						break;
349 					case RemoteError:
350 						logger.info("Transfer restart requested but remote error: {}",
351 								runner);
352 						break;
353 					case PassThroughMode:
354 						logger.info("Transfer not restarted since it is in PassThrough mode: {}",
355 								runner);
356 						break;
357 					default:
358 						logger.info("Transfer restart requested but internal error: {}",
359 								runner);
360 						break;
361 				}
362 			}
363 		} else {
364 			// Only request
365 			logger.info("Transfer information:\n    " + runner.toShortString());
366 			future.setResult(new R66Result(null, true, runner.getErrorInfo(), runner));
367 			future.setSuccess();
368 		}
369 	}

371 	/**
372 	 * Set the runner to DONE
373 	 * 
374 	 * @param runner
375 	 */
376 	private void setDone(DbTaskRunner runner) {
377 		if (runner.getUpdatedInfo() != UpdatedInfo.DONE) {
378 			runner.changeUpdatedInfo(UpdatedInfo.DONE);
379 			runner.forceSaveStatus();
380 		}
381 	}

383 	private ErrorCode sendValid(DbTaskRunner runner, byte code) {
384 		DbHostAuth host;
385 		host = R66Auth.getServerAuth(DbConstant.admin.session,
386 				this.requester);
387 		if (host == null) {
388 			logger.error("Requester host cannot be found: " + this.requester);
389 			OpenR66Exception e =
390 					new OpenR66RunnerErrorException("Requester host cannot be found");
391 			future.setResult(new R66Result(
392 					e,
393 					null, true,
394 					ErrorCode.TransferError, null));
395 			future.setFailure(e);
396 			return ErrorCode.Internal;
397 		}
398 		// check if requester is "client" so no connect from him but direct action
399 		logger.debug("Requester Host isClient: "+host.isClient());
400 		if (host.isClient()) {
401 			if (code == LocalPacketFactory.VALIDPACKET) {
402 				logger.warn("RequestTransfer from Client as requester, so use DirectTransfer instead: "+
403 						runner.toShortString());
404 				R66Future transfer = new R66Future(true);
405 				DirectTransfer transaction = new DirectTransfer(transfer,
406 						runner.getRequested(), runner.getOriginalFilename(), 
407 						runner.getRuleId(), runner.getFileInformation(), false, 
408 						runner.getBlocksize(), runner.getSpecialId(), networkTransaction);
409 				transaction.run();
410 				transfer.awaitUninterruptibly();
411 				logger.info("Request done with " + (transfer.isSuccess() ? "success" : "error"));
412 				if (transfer.isSuccess()) {
413 					future.setResult(new R66Result(null, true, ErrorCode.PreProcessingOk, runner));
414 					future.getResult().runner = runner;
415 					future.setSuccess();
416 					return ErrorCode.PreProcessingOk;
417 				} else {
418 					R66Result result = transfer.getResult();
419 					ErrorCode error = ErrorCode.Internal;
420 					if (result != null) {
421 						error = result.code;
422 					}
423 					OpenR66Exception e =
424 							new OpenR66RunnerErrorException("Transfer in direct mode failed: "+error.mesg);
425 					future.setFailure(e);
426 					return error;
427 				}
428 			} else {
429 				// get remote host instead
430 				host = R66Auth.getServerAuth(DbConstant.admin.session,
431 						this.requested);
432 				if (host == null) {
433 					logger.error("Requested host cannot be found: " + this.requested);
434 					OpenR66Exception e =
435 							new OpenR66RunnerErrorException("Requested host cannot be found");
436 					future.setResult(new R66Result(
437 							e,
438 							null, true,
439 							ErrorCode.TransferError, null));
440 					future.setFailure(e);
441 					return ErrorCode.Internal;
442 				}
443 			}
444 		}
445 
446 		logger.info("Try RequestTransfer to "+host.toString());
447 		SocketAddress socketAddress = host.getSocketAddress();
448 		boolean isSSL = host.isSsl();
449 
450 		LocalChannelReference localChannelReference = networkTransaction
451 				.createConnectionWithRetry(socketAddress, isSSL, future);
452 		socketAddress = null;
453 		if (localChannelReference == null) {
454 			logger.debug("Cannot connect to " + host.toString());
455 			host = null;
456 			future.setResult(new R66Result(null, true,
457 					ErrorCode.ConnectionImpossible, null));
458 			future.cancel();
459 			return ErrorCode.Internal;
460 		}
461 		ValidPacket packet = null;
462 		if (restarttime != null && code == LocalPacketFactory.VALIDPACKET) {
463 			// restart time set
464 			logger.debug("Restart with time: "+restarttime);
465 			packet = new ValidPacket("Request on Transfer",
466 					this.requested + " " + this.requester + " " + this.specialId+" "+restarttime,
467 					code);
468 		} else {
469 			packet = new ValidPacket("Request on Transfer",
470 				this.requested + " " + this.requester + " " + this.specialId,
471 				code);
472 		}
473 		localChannelReference.sessionNewState(R66FiniteDualStates.VALIDOTHER);
474 		try {
475 			ChannelUtils.writeAbstractLocalPacket(localChannelReference, packet, false);
476 		} catch (OpenR66ProtocolPacketException e) {
477 			logger.error("Cannot transfer request to " + host.toString());
478 			Channels.close(localChannelReference.getLocalChannel());
479 			localChannelReference = null;
480 			host = null;
481 			packet = null;
482 			logger.debug("Bad Protocol", e);
483 			future.setResult(new R66Result(e, null, true,
484 					ErrorCode.TransferError, null));
485 			future.setFailure(e);
486 			return ErrorCode.Internal;
487 		}
488 		packet = null;
489 		host = null;
490 		future.awaitUninterruptibly();
491 
492 		Channels.close(localChannelReference.getLocalChannel());
493 		localChannelReference = null;
494 
495 		logger.info("Request done with " + (future.isSuccess() ? "success" : "error"));
496 		R66Result result = future.getResult();
497 		if (result != null) {
498 			return result.code;
499 		}
500 		return ErrorCode.Internal;
501 	}

503 	/**
504 	 * @param args
505 	 */
506 	public static void main(String[] args) {
507 		InternalLoggerFactory.setDefaultFactory(new WaarpSlf4JLoggerFactory(null));
508 		if (logger == null) {
509 			logger = WaarpInternalLoggerFactory.getLogger(RequestTransfer.class);
510 		}
511 		if (!getParams(args)) {
512 			logger.error("Wrong initialization");
513 			if (DbConstant.admin != null && DbConstant.admin.isConnected) {
514 				DbConstant.admin.close();
515 			}
516 			ChannelUtils.stopLogger();
517 			System.exit(1);
518 		}
519 		int value = 99;
520 		try {
521 			Configuration.configuration.pipelineInit();
522 			NetworkTransaction networkTransaction = new NetworkTransaction();
523 			R66Future result = new R66Future(true);
524 			RequestTransfer requestTransfer =
525 					new RequestTransfer(result, sspecialId, srequested, srequester,
526 							scancel, sstop, srestart, srestarttime,
527 							networkTransaction);
528 			requestTransfer.run();
529 			result.awaitUninterruptibly();
530 			R66Result finalValue = result.getResult();
531 			if (scancel || sstop || srestart) {
532 				if (scancel) {
533 					if (result.isSuccess()) {
534 						value = 0;
535 						logger.warn("Transfer already finished:\n    " +
536 								finalValue.runner.toShortString());
537 					} else {
538 						switch (finalValue.code) {
539 							case CompleteOk:
540 								value = 0;
541 								logger.warn("Transfer cancel requested and done:\n    " +
542 										finalValue.runner.toShortString());
543 								break;
544 							case TransferOk:
545 								value = 3;
546 								logger.warn("Transfer cancel requested but already finished:\n    "
547 										+
548 										finalValue.runner.toShortString());
549 								break;
550 							default:
551 								value = 4;
552 								logger.error("Transfer cancel requested but internal error:\n    " +
553 										finalValue.runner.toShortString());
554 								break;
555 						}
556 					}
557 				} else if (sstop) {
558 					switch (finalValue.code) {
559 						case CompleteOk:
560 							value = 0;
561 							logger.warn("Transfer stop requested and done:\n    " +
562 									finalValue.runner.toShortString());
563 							break;
564 						case TransferOk:
565 							value = 0;
566 							logger.warn("Transfer stop requested but already finished:\n    " +
567 									finalValue.runner.toShortString());
568 							break;
569 						default:
570 							value = 3;
571 							logger.error("Transfer stop requested but internal error:\n    " +
572 									finalValue.runner.toShortString());
573 							break;
574 					}
575 				} else if (srestart) {
576 					switch (finalValue.code) {
577 						case QueryStillRunning:
578 							value = 0;
579 							logger.warn("Transfer restart requested but already active and running:\n    "
580 									+
581 									finalValue.runner.toShortString());
582 							break;
583 						case Running:
584 							value = 0;
585 							logger.warn("Transfer restart requested but already running:\n    " +
586 									finalValue.runner.toShortString());
587 							break;
588 						case PreProcessingOk:
589 							value = 0;
590 							logger.warn("Transfer restart requested and restarted:\n    " +
591 									finalValue.runner.toShortString());
592 							break;
593 						case CompleteOk:
594 							value = 4;
595 							logger.warn("Transfer restart requested but already finished:\n    " +
596 									finalValue.runner.toShortString());
597 							break;
598 						case RemoteError:
599 							value = 5;
600 							logger.error("Transfer restart requested but remote error:\n    " +
