1   package org.waarp.openr66.dao.database.oracle;
2   
3   import java.sql.Connection;
4   import java.sql.PreparedStatement;
5   import java.sql.ResultSet;
6   import java.sql.SQLException;
7   
8   import org.waarp.openr66.dao.database.DBTransferDAO;
9   import org.waarp.openr66.dao.exception.DAOException;
10  
11  public class OracleTransferDAO extends DBTransferDAO {
12  
13      protected static String SQL_GET_ID = "SELECT runseq.nextval FROM DUAL";
14  
15      public OracleTransferDAO(Connection con) throws DAOException {
16          super(con);
17      }
18  
19      @Override
20      protected long getNextId() throws DAOException {
21          PreparedStatement ps = null;
22          try {
23              ps = connection.prepareStatement(SQL_GET_ID);
24              ResultSet rs = ps.executeQuery();
25              if (rs.next()) {
26                  return rs.getLong(1);
27              } else {
28                  throw new DAOException(
29                          "Error no id available, you should purge the database.");
30              }
31          } catch (SQLException e) {
32              throw new DAOException(e);
33          } finally {
34              closeStatement(ps);
35          }
36      }
37  }
