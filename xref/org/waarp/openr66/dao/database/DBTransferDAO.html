214     private String prepareFindQuery(List<Filter> filters, Object[] params) {
215         StringBuilder query = new StringBuilder(getGetAllRequest());
216         Iterator<Filter> it = filters.listIterator();
217         if (it.hasNext()) {
218             query.append(" WHERE ");
219         }
220         String prefix = "";
221         int i = 0;
222         while (it.hasNext()) {
223             query.append(prefix);
224             Filter filter = it.next();
225             query.append(filter.key + " " + filter.operand + " ?");
226             params[i] = filter.value;
227             i++;
228             prefix = " AND ";
229         }
230         return query.toString();
231     }

576     private Transfer getFromResultSet(ResultSet set) throws SQLException {
577         return new Transfer(
578                 set.getLong(ID_FIELD),
579                 set.getString(ID_RULE_FIELD),
580                 set.getInt(TRANSFER_MODE_FIELD),
581                 set.getString(FILENAME_FIELD),
582                 set.getString(ORIGINAL_NAME_FIELD),
583                 set.getString(FILE_INFO_FIELD),
584                 set.getBoolean(IS_MOVED_FIELD),
585                 set.getInt(BLOCK_SIZE_FIELD),
586                 set.getBoolean(RETRIEVE_MODE_FIELD),
587                 set.getString(OWNER_REQUEST_FIELD),
588                 set.getString(REQUESTER_FIELD),
589                 set.getString(REQUESTED_FIELD),
590                 set.getString(TRANSFER_INFO_FIELD),
591                 Transfer.TASKSTEP.valueOf(set.getInt(GLOBAL_STEP_FIELD)),
592                 Transfer.TASKSTEP.valueOf(set.getInt(GLOBAL_LAST_STEP_FIELD)),
593                 set.getInt(STEP_FIELD),
594                 ErrorCode.getFromCode(set.getString(STEP_STATUS_FIELD)),
595                 ErrorCode.getFromCode(set.getString(INFO_STATUS_FIELD)),
596                 set.getInt(RANK_FIELD),
597                 set.getTimestamp(TRANSFER_START_FIELD),
598                 set.getTimestamp(TRANSFER_STOP_FIELD),
599                 UpdatedInfo.valueOf(set.getInt(UPDATED_INFO_FIELD)));
