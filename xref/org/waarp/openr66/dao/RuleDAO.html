8   /**
9    * Interface to interact with Rule objects in the persistance layer
10   */
11  public interface RuleDAO {
12  
13      /**
14       * Retrieve all Rule objects in a List from the persistance layer
15       *
16       * @throws DAOException If data access error occurs
17       */
18      List<Rule> getAll() throws DAOException;
19  
20      /**
21       * Retrieve all Rule objects corresponding to the given filters
22       * a List from the persistance layer
23       *
24       * @param filters List of filter
25       * @throws DAOException If data access error occurs
26       */
27      List<Rule> find(List<Filter> filters) throws DAOException;
28  
29      /**
30       * Retrieve the Rule object with the specified Rulename from the
31       * persistance layer
32       *
33       * @param rulename rulename of the Rule object requested
34       * @throws DAOException If a data access error occurs
35       */
36      Rule select(String rulename) throws DAOException;
37  
38      /**
39       * Verify if a Rule object with the specified Rulename exists in
40       * the persistance layer
41       *
42       * @param rulename rulename of the Rule object verified
43       * @return true if a Rule object with the specified Rulename exist; false
44       * if no Rule object correspond to the specified Rulename.
45       * @throws DAOException If a data access error occurs
46       */
47      boolean exist(String rulename) throws DAOException;
48  
49      /**
50       * Insert the specified Rule object in the persistance layer
51       *
52       * @param rule Rule object to insert
53       * @throws DAOException If a data access error occurs
54       */
55      void insert(Rule rule) throws DAOException;
56  
57      /**
58       * Update the specified Rule object in the persistance layer
59       *
60       * @param rule Rule object to update
61       * @throws DAOException If a data access error occurs
62       */
63      void update(Rule rule) throws DAOException;
64  
65      /**
66       * Remove the specified Rule object from the persistance layer
67       *
68       * @param rule Rule object to insert
69       * @throws DAOException If a data access error occurs
70       */
71      void delete(Rule rule) throws DAOException;
72  
73      /**
74       * Remove all Rule objects from the persistance layer
75       *
76       * @throws DAOException If a data access error occurs
77       */
78      void deleteAll() throws DAOException;
79  
80      void close();
81  }
