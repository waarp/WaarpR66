64      private File[] getRuleFiles() {
65          File ruleDir = new File(Configuration.configuration.getBaseDirectory() +
66                  Configuration.configuration.getConfigPath());
67          List<File> res = new ArrayList<File>();
68          if(ruleDir.isDirectory()) {
69              res.addAll(Arrays.asList(
70                      ruleDir.listFiles(new ExtensionFilter(EXT_RULE))));
71              res.addAll(Arrays.asList(
72                      ruleDir.listFiles(new ExtensionFilter(EXT_RULES))));
73          }
74          return res.toArray(new File[0]);
75      }

85      public List<Rule> getAll() throws DAOException {
86          List<Rule> res = new ArrayList<Rule>();
87  
88          File[] files = getRuleFiles();
89          for (File ruleFile : files) {
90              try {
91                  DocumentBuilderFactory dbf = DocumentBuilderFactory.newInstance();
92                  Document document = dbf.newDocumentBuilder().parse(ruleFile);
93                  // Setup XPath query
94                  XPath xPath = XPathFactory.newInstance().newXPath();
95                  XPathExpression xpe = xPath.compile(XML_GET_ALL);
96                  NodeList listNode = (NodeList) xpe.evaluate(document,
97                          XPathConstants.NODESET);
98                  // Iterate through all found nodes
99  
100                 for (int i = 0; i < listNode.getLength(); i++) {
101                     Node node = listNode.item(i);
102                     res.add(getFromNode(node));
103                 }
104             } catch (SAXException e) {
105                 throw new DAOException(e);
106             } catch (XPathExpressionException e) {
107                 throw new DAOException(e);
108             } catch (ParserConfigurationException e) {
109                 throw new DAOException(e);
110             } catch (IOException e) {
111                 throw new DAOException(e);
112             }
113         }
114         return res;
115     }

117     public boolean exist(String rulename) throws DAOException {
118         File[] files = getRuleFiles();
119         for (File ruleFile : files) {
120             try {
121                 DocumentBuilderFactory dbf = DocumentBuilderFactory.newInstance();
122                 Document document = dbf.newDocumentBuilder().parse(ruleFile);
123                 // Setup XPath variable
124                 SimpleVariableResolver resolver = new SimpleVariableResolver();
125                 resolver.addVariable(new QName(null, "idrule"), rulename);
126                 // Setup XPath query
127                 XPath xPath = XPathFactory.newInstance().newXPath();
128                 xPath.setXPathVariableResolver(resolver);
129                 XPathExpression xpe = xPath.compile(XML_SELECT);
130                 // Query will return "" if nothing is found
131                 if (!"".equals(xpe.evaluate(document))) {
132                     return true;
133                 }
134             } catch (SAXException e) {
135                 throw new DAOException(e);
136             } catch (XPathExpressionException e) {
137                 throw new DAOException(e);
138             } catch (ParserConfigurationException e) {
139                 throw new DAOException(e);
140             } catch (IOException e) {
141                 throw new DAOException(e);
142             }
143         }
144         return false;
145     }

155     public Rule select(String rulename) throws DAOException {
156         File[] files = getRuleFiles();
157         for (File ruleFile : files) {
158             try {
159                 DocumentBuilderFactory dbf = DocumentBuilderFactory.newInstance();
160                 Document document = dbf.newDocumentBuilder().parse(ruleFile);
161                 // Setup XPath variable
162                 SimpleVariableResolver resolver = new SimpleVariableResolver();
163                 resolver.addVariable(new QName(null, "idrule"), rulename);
164                 // Setup XPath query
165                 XPath xPath = XPathFactory.newInstance().newXPath();
166                 xPath.setXPathVariableResolver(resolver);
167                 XPathExpression xpe = xPath.compile(XML_SELECT);
168                 // Retrieve node and instantiate object
169                 Node node = (Node) xpe.evaluate(document, XPathConstants.NODE);
170                 if (node != null) {
171                     return getFromNode(node);
172                 }
173                 return null;
174             } catch (SAXException e) {
175                 throw new DAOException(e);
176             } catch (XPathExpressionException e) {
177                 throw new DAOException(e);
178             } catch (ParserConfigurationException e) {
179                 throw new DAOException(e);
180             } catch (IOException e) {
181                 throw new DAOException(e);
182             }
183         }
184         return null;
185     }

191     private Rule getFromNode(Node parent) throws DAOException {
192         Rule res = new Rule();
193 
194         NodeList children = parent.getChildNodes();
195         for (int j = 0; j < children.getLength(); j++) {
196             Node node = children.item(j);
197             if (node.getNodeName().equals(RULENAME_FIELD)) {
198                 res.setName(node.getTextContent());
199             } else if (node.getNodeName().equals(HOSTIDS_LIST)) {
200                 res.setHostids(retrieveHostids(node.getTextContent()));
201             } else if (node.getNodeName().equals(MODE_FIELD)) {
202                 res.setMode(Integer.parseInt(node.getTextContent()));
203             } else if (node.getNodeName().equals(SEND_PATH_FIELD)) {
204                 res.setSendPath(node.getTextContent());
205             } else if (node.getNodeName().equals(RECV_PATH_FIELD)) {
206                 res.setRecvPath(node.getTextContent());
207             } else if (node.getNodeName().equals(ARCH_PATH_FIELD)) {
208                 res.setArchivePath(node.getTextContent());
209             } else if (node.getNodeName().equals(WORK_PATH_FIELD)) {
210                 res.setWorkPath(node.getTextContent());
211             } else if (node.getNodeName().equals(RPRE_TASKS_FIELD)) {
212                 res.setRPreTasks(retrieveTasks(node));
213             } else if (node.getNodeName().equals(RPOST_TASKS_FIELD)) {
214                 res.setRPostTasks(retrieveTasks(node));
215             } else if (node.getNodeName().equals(RERROR_TASKS_FIELD)) {
216                 res.setRErrorTasks(retrieveTasks(node));
217             } else if (node.getNodeName().equals(SPRE_TASKS_FIELD)) {
218                 res.setSPreTasks(retrieveTasks(node));
219             } else if (node.getNodeName().equals(SPOST_TASKS_FIELD)) {
220                 res.setSPostTasks(retrieveTasks(node));
221             } else if (node.getNodeName().equals(SERROR_TASKS_FIELD)) {
222                 res.setSErrorTasks(retrieveTasks(node));
223             }
224         }
225         return res;
226     }

262     private List<String> retrieveHostids(String xml) throws DAOException {
263         ArrayList<String> res = new ArrayList<String>();
264         if ((xml == null) || xml.equals("")) {
265             return res;
266         }
267         Document document = null;
268         InputStream stream = null;
269         try {
270             stream = new ByteArrayInputStream(xml.getBytes("UTF-8"));
271             document = DocumentBuilderFactory.newInstance().
272                     newDocumentBuilder().parse(stream);
273         } catch (IOException e) {
274             throw new DAOException(e);
275         } catch (ParserConfigurationException e) {
276             throw new DAOException(e);
277         } catch (SAXException e) {
278             throw new DAOException(e);
279         } finally {
280             if (stream != null) {
281                 try {
282                     stream.close();
283                 } catch (IOException e) {
284                     logger.warn("Cannot properly close input stream", e);
285                 }
286             }
287         }
288         document.getDocumentElement().normalize();
289 
290         NodeList hostsList = document.getElementsByTagName(HOSTID_FIELD);
291         for (int i = 0; i < hostsList.getLength(); i++) {
292             res.add(hostsList.item(i).getTextContent());
293         }
294         return res;
295     }

297     public static final String TASK_NODE = "task";
298     public static final String TYPE_FIELD= "type";
299     public static final String PATH_FIELD = "path";
300     public static final String DELAY_FIELD = "delay";
301 
302     private List<RuleTask> retrieveTasks(Node src) throws DAOException {
303         List<RuleTask> res = new ArrayList<RuleTask>();
304         NodeList feed = src.getChildNodes();
305         for (int i = 0; i < feed.getLength(); i++) {
306             Node mainnode = feed.item(i);
307             if(mainnode.getNodeType() == Node.ELEMENT_NODE) {
308                 Element e = (Element) mainnode;
309                 NodeList tasksList = e.getElementsByTagName(TASK_NODE);
310                 for (int j = 0; j < tasksList.getLength(); j++) {
311                     Node taskNode = tasksList.item(j);
312                     if (taskNode.getNodeType() == Node.ELEMENT_NODE) {
313                         Element task = (Element) taskNode;
314                         String type = task.getElementsByTagName(TYPE_FIELD)
315                                 .item(0).getTextContent();
316                         String path = task.getElementsByTagName(PATH_FIELD)
317                                 .item(0).getTextContent();
318                         int delay = Integer.parseInt(
319                                 task.getElementsByTagName(DELAY_FIELD)
320                                 .item(0).getTextContent());
321                         res.add(new RuleTask(type, path, delay));
322                     }
323                 }
324             }
325         }
326         return res;
327     }
