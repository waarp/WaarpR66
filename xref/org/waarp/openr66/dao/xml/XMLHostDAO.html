59      public List<Host> getAll() throws DAOException {
60          if (!file.exists()) {
61              throw new DAOException("File doesn't exist");
62          }
63          try {
64              DocumentBuilderFactory dbf = DocumentBuilderFactory.newInstance();
65              Document document = dbf.newDocumentBuilder().parse(file);
66              // Setup XPath query
67              XPath xPath = XPathFactory.newInstance().newXPath();
68              XPathExpression xpe = xPath.compile(XML_GET_ALL);
69              NodeList listNode = (NodeList) xpe.evaluate(document,
70                      XPathConstants.NODESET);
71              // Iterate through all found nodes
72              List<Host> res = new ArrayList<Host>(listNode.getLength());
73              for (int i = 0; i < listNode.getLength(); i++) {
74                  Node node = listNode.item(i);
75                  res.add(getFromNode(node));
76              }
77              return res;
78          } catch (SAXException e) {
79              throw new DAOException(e);
80          } catch (XPathExpressionException e) {
81              throw new DAOException(e);
82          } catch (ParserConfigurationException e) {
83              throw new DAOException(e);
84          } catch (IOException e) {
85              throw new DAOException(e);
86          }
87      }

89      public boolean exist(String hostid) throws DAOException {
90          if (!file.exists()) {
91              throw new DAOException("File doesn't exist");
92          }
93          try {
94              DocumentBuilderFactory dbf = DocumentBuilderFactory.newInstance();
95              Document document = dbf.newDocumentBuilder().parse(file);
96              // Setup XPath variable
97              SimpleVariableResolver resolver = new SimpleVariableResolver();
98              resolver.addVariable(new QName(null, "hostid"), hostid);
99              // Setup XPath query
100             XPath xPath = XPathFactory.newInstance().newXPath();
101             xPath.setXPathVariableResolver(resolver);
102             XPathExpression xpe = xPath.compile(XML_SELECT);
103             // Query will return "" if nothing is found
104             return(!"".equals(xpe.evaluate(document)));
105         } catch (SAXException e) {
106             throw new DAOException(e);
107         } catch (XPathExpressionException e) {
108             throw new DAOException(e);
109         } catch (ParserConfigurationException e) {
110             throw new DAOException(e);
111         } catch (IOException e) {
112             throw new DAOException(e);
113         }
114     }

124     public Host select(String hostid) throws DAOException {
125         if (!file.exists()) {
126             throw new DAOException("File " + file.getPath() + " doesn't exist");
127         }
128         try {
129             DocumentBuilderFactory dbf = DocumentBuilderFactory.newInstance();
130             Document document = dbf.newDocumentBuilder().parse(file);
131             // Setup XPath variable
132             SimpleVariableResolver resolver = new SimpleVariableResolver();
133             resolver.addVariable(new QName(null, "hostid"), hostid);
134             // Setup XPath query
135             XPath xPath = XPathFactory.newInstance().newXPath();
136             xPath.setXPathVariableResolver(resolver);
137             XPathExpression xpe = xPath.compile(XML_SELECT);
138             // Retrieve node and instantiate object
139             Node node = (Node) xpe.evaluate(document, XPathConstants.NODE);
140             if (node != null) {
141                  return getFromNode(node);
142             }
143             return null;
144         } catch (SAXException e) {
145             throw new DAOException(e);
146         } catch (XPathExpressionException e) {
147             throw new DAOException(e);
148         } catch (ParserConfigurationException e) {
149             throw new DAOException(e);
150         } catch (IOException e) {
151             throw new DAOException(e);
152         }
153     }

159     private Host getFromNode(Node parent) {
160         Host res = new Host();
161 
162         NodeList children = parent.getChildNodes();
163         for (int j = 0; j < children.getLength(); j++) {
164             Node node = children.item(j);
165             if (node.getNodeName().equals(HOSTID_FIELD)) {
166                 res.setHostid(node.getTextContent());
167             } else if (node.getNodeName().equals(ADDRESS_FIELD)) {
168                 res.setAddress(node.getTextContent());
169             } else if (node.getNodeName().equals(PORT_FIELD)) {
170                 res.setPort(Integer.parseInt(node.getTextContent()));
171             } else if (node.getNodeName().equals(IS_SSL_FIELD)) {
172                 res.setSSL(Boolean.parseBoolean(node.getTextContent()));
173             } else if (node.getNodeName().equals(IS_CLIENT_FIELD)) {
174                 res.setClient(Boolean.parseBoolean(node.getTextContent()));
175             } else if (node.getNodeName().equals(IS_PROXIFIED_FIELD)) {
176                 res.setProxified(Boolean.parseBoolean(node.getTextContent()));
177             } else if (node.getNodeName().equals(ADMINROLE_FIELD)) {
178                 res.setAdmin(Boolean.parseBoolean(node.getTextContent()));
179             } else if (node.getNodeName().equals(HOSTKEY_FIELD)) {
180                 res.setHostkey(node.getTextContent().getBytes());
181             }
182         }
183         return res;
184     }

186     private Node getNode(Document doc, Host host) {
187         Node res = doc.createElement("entry");
188         res.appendChild(XMLUtils.createNode(doc, HOSTID_FIELD,
189                 host.getHostid()));
190         res.appendChild(XMLUtils.createNode(doc, ADDRESS_FIELD,
191                 host.getAddress()));
192         res.appendChild(XMLUtils.createNode(doc, PORT_FIELD,
193                 Integer.toString(host.getPort())));
194         res.appendChild(XMLUtils.createNode(doc, IS_SSL_FIELD,
195                 Boolean.toString(host.isSSL())));
196         res.appendChild(XMLUtils.createNode(doc, IS_CLIENT_FIELD,
197                 Boolean.toString(host.isClient())));
198         res.appendChild(XMLUtils.createNode(doc, IS_PROXIFIED_FIELD,
199                 Boolean.toString(host.isProxified())));
200         res.appendChild(XMLUtils.createNode(doc, ADMINROLE_FIELD,
201                 Boolean.toString(host.isAdmin())));
202         try {
203             res.appendChild(XMLUtils.createNode(doc, HOSTKEY_FIELD,
204                     new String(host.getHostkey(), "UTF-8")));
205         } catch (UnsupportedEncodingException e) {
206             logger.error("Unsupported charset ! Should not happened");
207         }
208         return res;
209     }
