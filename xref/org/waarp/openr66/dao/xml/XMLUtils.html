11  import javax.xml.transform.dom.DOMSource;
12  import javax.xml.transform.stream.StreamResult;
13  import java.io.File;
14  
15  public class XMLUtils {
16  
17      private static final WaarpLogger logger = WaarpLoggerFactory
18              .getLogger(XMLUtils.class);
19  
20      public static Node createNode(Document document, String tag, String text) {
21          Node res = document.createElement(tag);
22          res.setTextContent(text);
23          return res;
24      }
25  
26      public static void writeToFile(File file, Document document) {
27          TransformerFactory transformerFactory = TransformerFactory.newInstance();
28          try {
29              Transformer transformer = transformerFactory.newTransformer();
30              DOMSource domSource = new DOMSource(document);
31              StreamResult streamResult = new StreamResult(file);
32              transformer.transform(domSource, streamResult);
33          } catch (TransformerException e) {
34              logger.error("Error while writing document to file", e);
35          }
36      }
37  }
