37  /**
38   * Utility class to initiate the database for a server
39   * 
40   * @author Frederic Bregier
41   * 
42   */
43  public class ServerInitDatabase {
44  	/**
45  	 * Internal Logger
46  	 */
47  	static volatile WaarpInternalLogger logger;
48  
49  	static String sxml = null;
50  	static boolean database = false;
51  	static String sdirconfig = null;
52  	static String shostauth = null;
53  	static String slimitconfig = null;
54  
55  	protected static boolean getParams(String[] args) {
56  		if (args.length < 1) {
57  			logger.error("Need at least the configuration file as first argument then optionally\n"
58  					+
59  					"    -initdb\n" +
60  					"    -dir directory for rules configuration\n" +
61  					"    -limit xmlfile containing limit of bandwidth\n" +
62  					"    -auth xml file containing the authentication of hosts");
63  			return false;
64  		}
65  		sxml = args[0];
66  		for (int i = 1; i < args.length; i++) {
67  			if (args[i].equalsIgnoreCase("-initdb")) {
68  				database = true;
69  			} else if (args[i].equalsIgnoreCase("-dir")) {
70  				i++;
71  				sdirconfig = args[i];
72  			} else if (args[i].equalsIgnoreCase("-limit")) {
73  				i++;
74  				slimitconfig = args[i];
75  			} else if (args[i].equalsIgnoreCase("-auth")) {
76  				i++;
77  				shostauth = args[i];
78  			}
79  		}
80  		return true;
81  	}
82  
83  	/**
84  	 * @param args
85  	 *            as config_database file [rules_directory host_authent limit_configuration]
86  	 */
87  	public static void main(String[] args) {
88  		InternalLoggerFactory.setDefaultFactory(new WaarpSlf4JLoggerFactory(null));
89  		if (logger == null) {
90  			logger = WaarpInternalLoggerFactory.getLogger(ServerInitDatabase.class);
91  		}
92  		if (!getParams(args)) {
93  			logger.error("Need at least the configuration file as first argument then optionally\n"
94  					+
95  					"    -initdb\n" +
96  					"    -dir directory for rules configuration\n" +
97  					"    -limit xmlfile containing limit of bandwidth\n" +
98  					"    -auth xml file containing the authentication of hosts");
99  			if (DbConstant.admin != null && DbConstant.admin.isConnected) {
100 				DbConstant.admin.close();
101 			}
102 			ChannelUtils.stopLogger();
103 			System.exit(1);
104 		}
105 
106 		try {
107 			if (!FileBasedConfiguration
108 					.setConfigurationInitDatabase(Configuration.configuration, args[0])) {
109 				logger
110 						.error("Needs a correct configuration file as first argument");
111 				if (DbConstant.admin != null) {
112 					DbConstant.admin.close();
113 				}
114 				ChannelUtils.stopLogger();
115 				System.exit(1);
116 				return;
117 			}
118 			if (database) {
119 				// Init database
120 				try {
121 					initdb();
122 				} catch (WaarpDatabaseNoConnectionException e) {
123 					logger.error("Cannot connect to database");
124 					return;
125 				}
126 				System.out.println("End creation");
127 			}
128 			if (sdirconfig != null) {
129 				// load Rules
130 				File dirConfig = new File(sdirconfig);
131 				if (dirConfig.isDirectory()) {
132 					loadRules(dirConfig);
133 				} else {
134 					System.err.println("Dir is not a directory: " + sdirconfig);
135 				}
136 			}
137 			if (shostauth != null) {
138 				// Load Host Authentications
139 				if (args.length > 2) {
140 					loadHostAuth(shostauth);
141 				}
142 			}
143 			if (slimitconfig != null) {
144 				// Load configuration
145 				if (args.length > 3) {
146 					FileBasedConfiguration.setConfigurationLoadLimitFromXml(
147 							Configuration.configuration,
148 							slimitconfig);
149 				}
150 			}
151 			System.out.println("Load done");
152 		} finally {
153 			if (DbConstant.admin != null) {
154 				DbConstant.admin.close();
155 			}
156 		}
157 	}
158 
159 	public static void initdb() throws WaarpDatabaseNoConnectionException {
160 		// Create tables: configuration, hosts, rules, runner, cptrunner
161 		DbModelFactory.dbModel.createTables(DbConstant.admin.session);
162 	}
163 
164 	public static void loadRules(File dirConfig) {
165 		try {
166 			RuleFileBasedConfiguration.importRules(dirConfig);
167 		} catch (OpenR66ProtocolSystemException e3) {
168 			e3.printStackTrace();
169 		} catch (WaarpDatabaseException e) {
170 			e.printStackTrace();
171 		}
172 	}
173 
174 	public static void loadHostAuth(String filename) {
175 		AuthenticationFileBasedConfiguration.loadAuthentication(Configuration.configuration,
176 				filename);
177 	}
