21  package org.waarp.openr66.service;
22  
23  import org.waarp.common.future.WaarpFuture;
24  import org.waarp.common.logging.WaarpLogger;
25  import org.waarp.common.logging.WaarpLoggerFactory;
26  import org.waarp.common.service.EngineAbstract;
27  import org.waarp.common.utility.SystemPropertyUtil;
28  import org.waarp.openr66.protocol.configuration.Configuration;
29  import org.waarp.openr66.protocol.configuration.R66SystemProperties;
30  import org.waarp.openr66.protocol.utils.ChannelUtils;
31  import org.waarp.openr66.server.R66Server;
32  
33  /**
34   * Engine used to start and stop the real R66 service
35   * 
36   * @author Frederic Bregier
37   *
38   */
39  public class R66Engine extends EngineAbstract {
40      /**
41       * Internal Logger
42       */
43      private static final WaarpLogger logger = WaarpLoggerFactory
44              .getLogger(R66Engine.class);
45  
46      static WaarpFuture closeFuture = new WaarpFuture(true);
47  
48      @Override
49      public void run() {
50          String config = SystemPropertyUtil.get(R66SystemProperties.OPENR66_CONFIGFILE);
51          if (config == null) {
52              logger.error("Cannot find " + R66SystemProperties.OPENR66_CONFIGFILE + " parameter");
53              closeFuture.cancel();
54              shutdown();
55              return;
56          }
57          Configuration.configuration.getShutdownConfiguration().serviceFuture = closeFuture;
58          try {
59              if (!R66Server.initialize(config)) {
60                  throw new Exception("Initialization in error");
61              }
62          } catch (Throwable e) {
63              logger.error("Cannot start R66", e);
64              closeFuture.cancel();
65              shutdown();
66              return;
67          }
68          logger.warn("Service started with " + config);
69      }
70  
71      @Override
72      public void shutdown() {
73          logger.warn("Shutdown");
74          ChannelUtils.startShutdown();
75          closeFuture.setSuccess();
76          logger.info("Service stopped");
77      }
78  
79      @Override
80      public boolean isShutdown() {
81          return closeFuture.isDone();
82      }
83  
84      @Override
85      public boolean waitShutdown() throws InterruptedException {
86          closeFuture.await();
87          logger.info("Shutdown on going: " + closeFuture.isSuccess());
88          return closeFuture.isSuccess();
89      }
90  }
