18  package org.waarp.openr66.context.filesystem;
19  
20  import org.waarp.common.command.exception.CommandAbstractException;
21  import org.waarp.common.command.exception.Reply502Exception;
22  import org.waarp.common.file.filesystembased.FilesystemBasedRestartImpl;
23  import org.waarp.openr66.context.R66Session;
24  
25  /**
26   * Restart implementation
27   * 
28   * @author frederic bregier
29   * 
30   */
31  public class R66Restart extends FilesystemBasedRestartImpl {
32  
33      /**
34       * @param session
35       */
36      public R66Restart(R66Session session) {
37          super(session);
38      }
39  
40      /*
41       * (non-Javadoc)
42       * @see org.waarp.common.file.Restart#restartMarker(java.lang.String)
43       */
44      @Override
45      public boolean restartMarker(String marker) throws CommandAbstractException {
46          long newposition = 0;
47          try {
48              newposition = Long.parseLong(marker);
49          } catch (NumberFormatException e) {
50              throw new Reply502Exception(
51                      "Marker must be length in byte as a position");
52          }
53          return restartMarker(newposition);
54      }
55  
56      /**
57       * Same as restartMarker with String
58       * 
59       * @param newposition
60       * @return True if OK
61       */
62      public boolean restartMarker(long newposition) {
63          position = newposition;
64          setSet(true);
65          return true;
66      }
67  }
