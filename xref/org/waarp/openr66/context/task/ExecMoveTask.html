40  /**
41   * Execute an external command and Rename the file (using the new name from the result).<br>
42   * 
43   * The move of the file (if any) should be done by the external command itself.<br>
44   * <br>
45   * 
46   * waitForValidation (#NOWAIT#) must not be set since it will prevent to have the MOVE TASK to occur
47   * normally. So even if set, the #NOWAIT# will be ignored.
48   * 
49   * @author Frederic Bregier
50   * 
51   */
52  public class ExecMoveTask extends AbstractTask {
53      /**
54       * Internal Logger
55       */
56      private static final WaarpLogger logger = WaarpLoggerFactory
57              .getLogger(ExecMoveTask.class);
58  
59      /**
60       * @param argRule
61       * @param delay
62       * @param argTransfer
63       * @param session
64       */
65      public ExecMoveTask(String argRule, int delay, String argTransfer,
66              R66Session session) {
67          super(TaskType.EXECMOVE, delay, argRule, argTransfer, session);
68      }
69  
70      @Override
71      public void run() {
72          /*
73           * First apply all replacements and format to argRule from context and argTransfer. Will
74           * call exec (from first element of resulting string) with arguments as the following value
75           * from the replacements. Return 0 if OK, else 1 for a warning else as an error. The last
76           * line of stdout will be the new name given to the R66File in case of status 0. The
77           * previous file should be deleted by the script or will be deleted in case of status 0. If
78           * the status is 1, no change is made to the file.
79           */
80          logger.info("ExecMove with " + argRule + ":" + argTransfer + " and {}",
81                  session);
82          String finalname = argRule;
83          finalname = getReplacedValue(finalname, argTransfer.split(" "));
84          // Force the WaitForValidation
85          waitForValidation = true;
86          if (Configuration.configuration.isUseLocalExec() && useLocalExec) {
87              LocalExecClient localExecClient = new LocalExecClient();
88              if (localExecClient.connect()) {
89                  localExecClient
90                          .runOneCommand(finalname, delay, waitForValidation, futureCompletion);
91                  LocalExecResult result = localExecClient.getLocalExecResult();
92                  move(result.getStatus(), result.getResult(), finalname);
93                  localExecClient.disconnect();
94                  return;
95              } // else continue
96          }
97          String[] args = finalname.split(" ");
98          File exec = new File(args[0]);
99          if (exec.isAbsolute()) {
100             if (!exec.canExecute()) {
101                 logger.error("Exec command is not executable: " + finalname);
102                 R66Result result = new R66Result(session, false,
103                         ErrorCode.CommandNotFound, session.getRunner());
104                 futureCompletion.setResult(result);
105                 futureCompletion.cancel();
106                 return;
107             }
108         }
109         CommandLine commandLine = new CommandLine(args[0]);
110         for (int i = 1; i < args.length; i++) {
111             commandLine.addArgument(args[i]);
112         }
113         DefaultExecutor defaultExecutor = new DefaultExecutor();
114         PipedInputStream inputStream = new PipedInputStream();
115         PipedOutputStream outputStream = null;
116         try {
117             outputStream = new PipedOutputStream(inputStream);
118         } catch (IOException e1) {
119             try {
120                 inputStream.close();
121             } catch (IOException e) {
122             }
123             logger.error("Exception: " + e1.getMessage() +
124                     " Exec in error with " + commandLine.toString(), e1);
125             futureCompletion.setFailure(e1);
126             return;
127         }
128         PumpStreamHandler pumpStreamHandler = new PumpStreamHandler(
129                 outputStream, null);
130         defaultExecutor.setStreamHandler(pumpStreamHandler);
131         int[] correctValues = {
132                 0, 1 };
133         defaultExecutor.setExitValues(correctValues);
134         ExecuteWatchdog watchdog = null;
135 
136         if (delay > 0) {
137             watchdog = new ExecuteWatchdog(delay);
138             defaultExecutor.setWatchdog(watchdog);
139         }
140         LastLineReader lastLineReader = new LastLineReader(inputStream);
141         Thread thread = new Thread(lastLineReader, "ExecRename"
142                 + session.getRunner().getSpecialId());
143         thread.setDaemon(true);
144         Configuration.configuration.getExecutorService().execute(thread);
145         int status = -1;
146         try {
147             status = defaultExecutor.execute(commandLine);
148         } catch (ExecuteException e) {
149             if (e.getExitValue() == -559038737) {
150                 // Cannot run immediately so retry once
151                 try {
152                     Thread.sleep(Configuration.RETRYINMS);
153                 } catch (InterruptedException e1) {
154                 }
155                 try {
156                     status = defaultExecutor.execute(commandLine);
157                 } catch (ExecuteException e1) {
158                     try {
159                         outputStream.close();
160                     } catch (IOException e2) {
161                     }
162                     thread.interrupt();
163                     try {
164                         inputStream.close();
165                     } catch (IOException e2) {
166                     }
167                     try {
168                         pumpStreamHandler.stop();
169                     } catch (IOException e2) {
170                     }
171                     logger.error("ExecuteException: " + e.getMessage() +
172                             " . Exec in error with " + commandLine.toString());
173                     futureCompletion.setFailure(e);
174                     return;
175                 } catch (IOException e1) {
176                     try {
177                         outputStream.close();
178                     } catch (IOException e2) {
179                     }
180                     thread.interrupt();
181                     try {
182                         inputStream.close();
183                     } catch (IOException e2) {
184                     }
185                     try {
186                         pumpStreamHandler.stop();
187                     } catch (IOException e2) {
188                     }
189                     logger.error("IOException: " + e.getMessage() +
190                             " . Exec in error with " + commandLine.toString());
191                     futureCompletion.setFailure(e);
192                     return;
193                 }
194             } else {
195                 try {
196                     outputStream.close();
197                 } catch (IOException e1) {
198                 }
199                 thread.interrupt();
200                 try {
201                     inputStream.close();
202                 } catch (IOException e1) {
203                 }
204                 try {
205                     pumpStreamHandler.stop();
206                 } catch (IOException e2) {
207                 }
208                 logger.error("ExecuteException: " + e.getMessage() +
209                         " . Exec in error with " + commandLine.toString());
210                 futureCompletion.setFailure(e);
211                 return;
212             }
213         } catch (IOException e) {
214             try {
215                 outputStream.close();
216             } catch (IOException e1) {
217             }
218             thread.interrupt();
219             try {
220                 inputStream.close();
221             } catch (IOException e1) {
222             }
223             try {
224                 pumpStreamHandler.stop();
225             } catch (IOException e2) {
226             }
227             logger.error("IOException: " + e.getMessage() +
228                     " . Exec in error with " + commandLine.toString());
229             futureCompletion.setFailure(e);
230             return;
231         }
232         try {
233             outputStream.flush();
234         } catch (IOException e) {
235         }
236         try {
237             outputStream.close();
238         } catch (IOException e) {
239         }
240         try {
241             pumpStreamHandler.stop();
242         } catch (IOException e2) {
243         }
244         try {
245             if (delay > 0) {
246                 thread.join(delay);
247             } else {
248                 thread.join();
249             }
250         } catch (InterruptedException e) {
251             Thread.currentThread().interrupt();
252         }
253         try {
254             inputStream.close();
255         } catch (IOException e1) {
256         }
257         String newname = null;
258         if (defaultExecutor.isFailure(status) && watchdog != null &&
259                 watchdog.killedProcess()) {
260             // kill by the watchdoc (time out)
261             status = -1;
262             newname = "TimeOut";
263         } else {
264             newname = lastLineReader.getLastLine();
265             if (status == 0 && (newname == null || newname.isEmpty())) {
266                 status = 1;
267             }
268         }
269         move(status, newname, commandLine.toString());
270     }
271 
272     private void move(int status, String newName, String commandLine) {
273         String newname = newName.replace('\\', '/');
274         if (status == 0) {
275             if (newname.indexOf(' ') > 0) {
276                 logger.warn("Exec returns a multiple string in final line: " +
277                         newname);
278                 // XXX FIXME: should not split String[] args = newname.split(" ");
279                 //newname = args[args.length - 1];
280             }
281             // now test if the previous file was deleted (should be)
282             File file = new File(newname);
283             if (!file.exists()) {
284                 logger.warn("New file does not exist at the end of the exec: " + newname);
285             }
286             // now replace the file with the new one
287             try {
288                 session.getFile().replaceFilename(newname, true);
289             } catch (CommandAbstractException e) {
290                 logger
291                         .warn("Exec in warning with " + commandLine,
292                                 e);
293             }
294             session.getRunner().setFileMoved(newname, true);
295             R66Result result = new R66Result(session, true, ErrorCode.CompleteOk, this.session.getRunner());
296             result.setOther(newname);
297             futureCompletion.setResult(result);
298             futureCompletion.setSuccess();
299             logger.info("Exec OK with {} returns {}", commandLine,
300                     newname);
301         } else if (status == 1) {
302             logger.warn("Exec in warning with " + commandLine +
303                     " returns " + newname);
304             session.getRunner().setErrorExecutionStatus(ErrorCode.Warning);
305             R66Result result = new R66Result(session, true, ErrorCode.Warning, this.session.getRunner());
306             result.setOther(newname);
307             futureCompletion.setResult(result);
308             futureCompletion.setSuccess();
309         } else {
310             logger.error("Status: " + status + " Exec in error with " +
311                     commandLine + " returns " + newname);
312             futureCompletion.cancel();
313         }
314     }
