33  /**
34   * Dummy Runnable Task that only logs
35   * 
36   * @author Frederic Bregier
37   * 
38   */
39  public abstract class AbstractExecJavaTask implements R66Runnable {
40  	/**
41  	 * Internal Logger
42  	 */
43  	private static final WaarpInternalLogger logger = WaarpInternalLoggerFactory
44  			.getLogger(AbstractExecJavaTask.class);
45  
46  	protected int delay;
47  	protected int status = -1;
48  	protected R66Session session;
49  	protected boolean waitForValidation;
50  	protected boolean useLocalExec;
51  
52  	protected String classname;
53  	protected String fullarg;
54  	protected boolean isToValidate;
55  	protected boolean callFromBusiness;
56  
57  	protected String finalInformation;
58  	
59  	/**
60  	 * Server side method to validate the request
61  	 * 
62  	 * @param packet
63  	 */
64  	public void validate(BusinessRequestPacket packet) {
65  		this.status = 0;
66  		packet.validate();
67  		if (callFromBusiness) {
68  			LocalChannelReference localChannelReference = session.getLocalChannelReference();
69  			if (localChannelReference != null) {
70  				R66Result result = new R66Result(session, true,
71  						ErrorCode.CompleteOk, null);
72  				localChannelReference.validateRequest(result);
73  				try {
74  					ChannelUtils.writeAbstractLocalPacket(localChannelReference,
75  							packet, true);
76  				} catch (OpenR66ProtocolPacketException e) {
77  				}
78  			} else {
79  				finalInformation = packet.toString();
80  			}
81  		}
82  	}
83  
84  	/**
85  	 * To be called by the requester when finished
86  	 * 
87  	 * @param object
88  	 *            special object to get back
89  	 */
90  	public void finalValidate(Object object) {
91  		this.status = 0;
92  		if (callFromBusiness) {
93  			LocalChannelReference localChannelReference = session.getLocalChannelReference();
94  			if (localChannelReference != null) {
95  				R66Result result = new R66Result(session, true,
96  						ErrorCode.CompleteOk, null);
97  				result.other = object;
98  				localChannelReference.validateRequest(result);
99  				ChannelUtils.close(localChannelReference.getLocalChannel());
100 			} else {
101 				finalInformation = JsonHandler.writeAsString(object);
102 			}
103 		}
104 	}
105 
106 	/**
107 	 * To be used if abnormal usage is made of one Java Method
108 	 */
109 	public void invalid() {
110 		this.status = 2;
111 		if (!callFromBusiness) {
112 			return;
113 		}
114 		R66Result result = new R66Result(null, session, true,
115 				ErrorCode.Unimplemented, session.getRunner());
116 		LocalChannelReference localChannelReference = session.getLocalChannelReference();
117 		if (localChannelReference != null) {
118 			localChannelReference.sessionNewState(R66FiniteDualStates.ERROR);
119 			ErrorPacket error = new ErrorPacket("Command Incompatible",
120 					ErrorCode.ExternalOp.getCode(), ErrorPacket.FORWARDCLOSECODE);
121 			try {
122 				ChannelUtils.writeAbstractLocalPacket(localChannelReference, error, true);
123 			} catch (OpenR66ProtocolPacketException e1) {
124 			}
125 			localChannelReference.invalidateRequest(result);
126 			ChannelUtils.close(localChannelReference.getLocalChannel());
127 		}
128 	}
129 
130 	public void run() {
131 		if (callFromBusiness) {
132 			// Business Request to validate?
133 			if (isToValidate) {
134 				BusinessRequestPacket packet =
135 						new BusinessRequestPacket(this.classname + " " + this.fullarg, 0);
136 				validate(packet);
137 			}
138 		}
139 		StringBuilder builder = new StringBuilder(this.getClass().getSimpleName() + ":");
140 		builder.append("args(");
141 		builder.append(fullarg);
142 		builder.append(")");
143 		logger.warn(builder.toString());
144 		this.status = 0;
145 	}
146 
147 	public void setArgs(R66Session session, boolean waitForValidation,
148 			boolean useLocalExec, int delay, String classname, String arg, boolean callFromBusiness, boolean isToValidate) {
149 		this.session = session;
150 		this.waitForValidation = waitForValidation;
151 		this.useLocalExec = useLocalExec;
152 		this.delay = delay;
153 		this.classname = classname;
154 		this.callFromBusiness = callFromBusiness;
155 		this.fullarg = arg;
156 		this.isToValidate = isToValidate;
157 	}
158 
159 	public int getFinalStatus() {
160 		return status;
161 	}
162 
163 	@Override
164 	public String toString() {
165 		if (status == -1 || finalInformation == null) {
166 			StringBuilder builder = new StringBuilder(this.getClass().getSimpleName() + ": [");
167 			builder.append(fullarg);
168 			builder.append(']');
169 			return builder.toString();
170 		} else {
171 			return finalInformation;
172 		}
173 	}
