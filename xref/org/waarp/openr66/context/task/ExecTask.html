18  package org.waarp.openr66.context.task;
19  
20  import java.io.File;
21  import java.io.IOException;
22  
23  import org.apache.commons.exec.CommandLine;
24  import org.apache.commons.exec.DefaultExecutor;
25  import org.apache.commons.exec.ExecuteException;
26  import org.apache.commons.exec.ExecuteWatchdog;
27  import org.apache.commons.exec.PumpStreamHandler;
28  import org.waarp.common.logging.WaarpLogger;
29  import org.waarp.common.logging.WaarpLoggerFactory;
30  import org.waarp.openr66.context.ErrorCode;
31  import org.waarp.openr66.context.R66Result;
32  import org.waarp.openr66.context.R66Session;
33  import org.waarp.openr66.context.task.localexec.LocalExecClient;
34  import org.waarp.openr66.protocol.configuration.Configuration;
35  
36  /**
37   * Execute an external command
38   * 
39   * 
40   * @author Frederic Bregier
41   * 
42   */
43  public class ExecTask extends AbstractTask {
44  
45      /**
46       * Internal Logger
47       */
48      private static final WaarpLogger logger = WaarpLoggerFactory
49              .getLogger(ExecTask.class);
50  
51      /**
52       * @param argRule
53       * @param delay
54       * @param argTransfer
55       * @param session
56       */
57      public ExecTask(String argRule, int delay, String argTransfer,
58              R66Session session) {
59          super(TaskType.EXEC, delay, argRule, argTransfer, session);
60      }
61  
62      @Override
63      public void run() {
64          /*
65           * First apply all replacements and format to argRule from context and argTransfer. Will
66           * call exec (from first element of resulting string) with arguments as the following value
67           * from the replacements. Return 0 if OK, else 1 for a warning else as an error. No change
68           * should be done in the FILENAME
69           */
70          logger.debug("Exec with " + argRule + ":" + argTransfer + " and {}",
71                  session);
72          String finalname = argRule;
73          finalname = getReplacedValue(finalname, argTransfer.split(" "));
74          // Check if the execution will be done through LocalExec daemon
75          if (Configuration.configuration.isUseLocalExec() && useLocalExec) {
76              LocalExecClient localExecClient = new LocalExecClient();
77              if (localExecClient.connect()) {
78                  localExecClient
79                          .runOneCommand(finalname, delay, waitForValidation, futureCompletion);
80                  localExecClient.disconnect();
81                  return;
82              }// else continue
83          }
84          // Execution is done internally
85          String[] args = finalname.split(" ");
86          File exec = new File(args[0]);
87          if (exec.isAbsolute()) {
88              if (!exec.canExecute()) {
89                  logger.error("Exec command is not executable: " + finalname);
90                  R66Result result = new R66Result(session, false,
91                          ErrorCode.CommandNotFound, session.getRunner());
92                  futureCompletion.setResult(result);
93                  futureCompletion.cancel();
94                  return;
95              }
96          }
97          CommandLine commandLine = new CommandLine(args[0]);
98          for (int i = 1; i < args.length; i++) {
99              commandLine.addArgument(args[i]);
100         }
101         DefaultExecutor defaultExecutor = new DefaultExecutor();
102         PumpStreamHandler pumpStreamHandler = new PumpStreamHandler(null, null);
103         defaultExecutor.setStreamHandler(pumpStreamHandler);
104         int[] correctValues = {
105                 0, 1 };
106         defaultExecutor.setExitValues(correctValues);
107         ExecuteWatchdog watchdog = null;
108         if (delay > 0 && waitForValidation) {
109             watchdog = new ExecuteWatchdog(delay);
110             defaultExecutor.setWatchdog(watchdog);
111         }
112         if (!waitForValidation) {
113             // Do not wait for validation
114             futureCompletion.setSuccess();
115             logger.info("Exec will start but no WAIT with {}", commandLine);
116         }
117         int status = -1;
118         try {
119             status = defaultExecutor.execute(commandLine);
120         } catch (ExecuteException e) {
121             if (e.getExitValue() == -559038737) {
122                 // Cannot run immediately so retry once
123                 try {
124                     Thread.sleep(Configuration.RETRYINMS);
125                 } catch (InterruptedException e1) {
126                 }
127                 try {
128                     status = defaultExecutor.execute(commandLine);
129                 } catch (ExecuteException e1) {
130                     try {
131                         pumpStreamHandler.stop();
132                     } catch (IOException e2) {
133                     }
134                     logger.error("Exception: " + e.getMessage() +
135                             " Exec in error with " + commandLine.toString());
136                     if (waitForValidation) {
137                         futureCompletion.setFailure(e);
138                     }
139                     return;
140                 } catch (IOException e1) {
141                     try {
142                         pumpStreamHandler.stop();
143                     } catch (IOException e2) {
144                     }
145                     logger.error("Exception: " + e.getMessage() +
146                             " Exec in error with " + commandLine.toString());
147                     if (waitForValidation) {
148                         futureCompletion.setFailure(e);
149                     }
150                     return;
151                 }
152             } else {
153                 try {
154                     pumpStreamHandler.stop();
155                 } catch (IOException e2) {
156                 }
157                 logger.error("Exception: " + e.getMessage() +
158                         " Exec in error with " + commandLine.toString());
159                 if (waitForValidation) {
160                     futureCompletion.setFailure(e);
161                 }
162                 return;
163             }
164         } catch (IOException e) {
165             try {
166                 pumpStreamHandler.stop();
167             } catch (IOException e2) {
168             }
169             logger.error("Exception: " + e.getMessage() +
170                     " Exec in error with " + commandLine.toString());
171             if (waitForValidation) {
172                 futureCompletion.setFailure(e);
173             }
174             return;
175         }
176         try {
177             pumpStreamHandler.stop();
178         } catch (IOException e2) {
179         }
180         if (defaultExecutor.isFailure(status) && watchdog != null &&
181                 watchdog.killedProcess()) {
182             // kill by the watchdoc (time out)
183             logger.error("Exec is in Time Out");
184             status = -1;
185         }
186         if (status == 0) {
187             if (waitForValidation) {
188                 futureCompletion.setSuccess();
189             }
190             logger.info("Exec OK with {}", commandLine);
191         } else if (status == 1) {
192             logger.warn("Exec in warning with " + commandLine.toString());
193             if (waitForValidation) {
194                 futureCompletion.setSuccess();
195             }
196         } else {
197             logger.error("Status: " + status + " Exec in error with " +
198                     commandLine.toString());
199             if (waitForValidation) {
200                 futureCompletion.cancel();
201             }
202         }
203     }
