41  /**
42   * Execute an external command and Use the output if an error occurs.<br>
43   * 
44   * The output is ignored if the command has a correct status.<br>
45   * If the output finishes with <tt>NEWFINALNAME:xxx</tt> then this part is removed from the output
46   * and the xxx is used as the last valid name for the file (meaning the file was moved or renamed)<br>
47   * <br>
48   * 
49   * waitForValidation (#NOWAIT#) must not be set since it will prevent to have the feedback in case
50   * of error. So it is ignored.
51   * 
52   * @author Frederic Bregier
53   * 
54   */
55  public class ExecOutputTask extends AbstractTask {
56  	/**
57  	 * Internal Logger
58  	 */
59  	private static final WaarpInternalLogger logger = WaarpInternalLoggerFactory
60  			.getLogger(ExecOutputTask.class);

62  	public static final String DELIMITER = "NEWFINALNAME:";

66  	 * @param delay
67  	 * @param argTransfer
68  	 * @param session
69  	 */
70  	public ExecOutputTask(String argRule, int delay, String argTransfer,

75  	/*
76  	 * (non-Javadoc)
77  	 * @see org.waarp.openr66.context.task.AbstractTask#run()
78  	 */
79  	@Override
80  	public void run() {
81  		/*
82  		 * First apply all replacements and format to argRule from context and argTransfer. Will
83  		 * call exec (from first element of resulting string) with arguments as the following value
84  		 * from the replacements. Return 0 if OK, else 1 for a warning else as an error. In case of
85  		 * an error (> 0), all the line from output will be send back to the partner with the Error
86  		 * code. No change is made to the file.
87  		 */
88  		logger.info("ExecOutput with " + argRule + ":" + argTransfer + " and {}",
89  				session);
90  		String finalname = argRule;
91  		finalname = getReplacedValue(finalname, argTransfer.split(" "));
92  		// Force the WaitForValidation
93  		waitForValidation = true;
94  		if (Configuration.configuration.useLocalExec && useLocalExec) {
95  			LocalExecClient localExecClient = new LocalExecClient();
96  			if (localExecClient.connect()) {
97  				localExecClient
98  						.runOneCommand(finalname, delay, waitForValidation, futureCompletion);
99  				LocalExecResult result = localExecClient.getLocalExecResult();
100 				finalize(result.status, result.result, finalname);
101 				localExecClient.disconnect();
102 				return;
103 			} // else continue
104 		}
105 		String[] args = finalname.split(" ");
106 		File exec = new File(args[0]);
107 		if (exec.isAbsolute()) {
108 			if (!exec.canExecute()) {
109 				logger.error("Exec command is not executable: " + finalname);
110 				R66Result result = new R66Result(session, false,
111 						ErrorCode.CommandNotFound, session.getRunner());
112 				futureCompletion.setResult(result);
113 				futureCompletion.cancel();
114 				return;
115 			}
116 		}
117 		CommandLine commandLine = new CommandLine(args[0]);
118 		for (int i = 1; i < args.length; i++) {
119 			commandLine.addArgument(args[i]);
120 		}
121 		DefaultExecutor defaultExecutor = new DefaultExecutor();
122 		PipedInputStream inputStream = new PipedInputStream();
123 		PipedOutputStream outputStream = null;
124 		try {
125 			outputStream = new PipedOutputStream(inputStream);
126 		} catch (IOException e1) {
127 			try {
128 				inputStream.close();
129 			} catch (IOException e) {
130 			}
131 			logger.error("Exception: " + e1.getMessage() +
132 					" Exec in error with " + commandLine.toString(), e1);
133 			futureCompletion.setFailure(e1);
134 			return;
135 		}
136 		PumpStreamHandler pumpStreamHandler = new PumpStreamHandler(
137 				outputStream, null);
138 		defaultExecutor.setStreamHandler(pumpStreamHandler);
139 		int[] correctValues = {
140 				0, 1 };
141 		defaultExecutor.setExitValues(correctValues);
142 		ExecuteWatchdog watchdog = null;
143 		if (delay > 0) {
144 			watchdog = new ExecuteWatchdog(delay);
145 			defaultExecutor.setWatchdog(watchdog);
146 		}
147 		AllLineReader allLineReader = new AllLineReader(inputStream);
148 		Thread thread = new Thread(allLineReader, "ExecRename" + session.getRunner().getSpecialId());
149 		thread.setDaemon(true);
150 		Configuration.configuration.getExecutorService().execute(thread);
151 		int status = -1;
152 		try {
153 			status = defaultExecutor.execute(commandLine);
154 		} catch (ExecuteException e) {
155 			if (e.getExitValue() == -559038737) {
156 				// Cannot run immediately so retry once
157 				try {
158 					Thread.sleep(Configuration.RETRYINMS);
159 				} catch (InterruptedException e1) {
160 				}
161 				try {
162 					status = defaultExecutor.execute(commandLine);
163 				} catch (ExecuteException e1) {
164 					finalizeFromError(outputStream,
165 							pumpStreamHandler,
166 							inputStream,
167 							allLineReader,
168 							thread,
169 							status,
170 							commandLine);
171 					return;
172 				} catch (IOException e1) {
173 					try {
174 						outputStream.flush();
175 					} catch (IOException e2) {
176 					}
177 					try {
178 						outputStream.close();
179 					} catch (IOException e2) {
180 					}
181 					thread.interrupt();
182 					try {
183 						inputStream.close();
184 					} catch (IOException e2) {
185 					}
186 					pumpStreamHandler.stop();
187 					logger.error("IOException: " + e.getMessage() +
188 							" . Exec in error with " + commandLine.toString());
189 					futureCompletion.setFailure(e);
190 					return;
191 				}
192 			} else {
193 				finalizeFromError(outputStream,
194 						pumpStreamHandler,
195 						inputStream,
196 						allLineReader,
197 						thread,
198 						status,
199 						commandLine);
200 				return;
201 			}
202 		} catch (IOException e) {
203 			try {
204 				outputStream.close();
205 			} catch (IOException e1) {
206 			}
207 			thread.interrupt();
208 			try {
209 				inputStream.close();
210 			} catch (IOException e1) {
211 			}
212 			pumpStreamHandler.stop();
213 			logger.error("IOException: " + e.getMessage() +
214 					" . Exec in error with " + commandLine.toString());
215 			futureCompletion.setFailure(e);
216 			return;
217 		}
218 		try {
219 			outputStream.flush();
220 		} catch (IOException e) {
221 		}
222 		try {
223 			outputStream.close();
224 		} catch (IOException e) {
225 		}
226 		pumpStreamHandler.stop();
227 		try {
228 			if (delay > 0) {
229 				thread.join(delay);
230 			} else {
231 				thread.join();
232 			}
233 		} catch (InterruptedException e) {
234 			Thread.currentThread().interrupt();
235 		}
236 		try {
237 			inputStream.close();
238 		} catch (IOException e1) {
239 		}
240 		String newname = null;
241 		if (defaultExecutor.isFailure(status) && watchdog != null &&
242 				watchdog.killedProcess()) {
243 			// kill by the watchdoc (time out)
244 			status = -1;
245 			newname = "TimeOut";
246 		} else {
247 			newname = allLineReader.lastLine.toString();
248 		}
249 		finalize(status, newname, commandLine.toString());
250 	}

252 	private void finalize(int status, String newName, String commandLine) {
253 		String newname = newName;
254 		if (status == 0) {
255 			futureCompletion.setSuccess();
256 			logger.info("Exec OK with {} returns {}", commandLine,
257 					newname);
258 		} else if (status == 1) {
259 			logger.warn("Exec in warning with " + commandLine +
260 					" returns " + newname);
261 			session.getRunner().setErrorExecutionStatus(ErrorCode.Warning);
262 			futureCompletion.setSuccess();
263 		} else {
264 			int pos = newname.lastIndexOf(DELIMITER);
265 			if (pos >= 0) {
266 				String newfilename = newname.substring(pos + DELIMITER.length());
267 				newname = newname.substring(0, pos);
268 				if (newfilename.indexOf(' ') > 0) {
269 					logger.warn("Exec returns a multiple string in final line: " +
270 							newfilename);
271 					String[] args = newfilename.split(" ");
272 					newfilename = args[args.length - 1];
273 				}
274 				// now test if the previous file was deleted (should be)
275 				File file = new File(newfilename);
276 				if (!file.exists()) {
277 					logger.warn("New file does not exist at the end of the exec: " + newfilename);
278 				}
279 				// now replace the file with the new one
280 				try {
281 					session.getFile().replaceFilename(newfilename, true);
282 				} catch (CommandAbstractException e) {
283 					logger
284 							.warn("Exec in warning with " + commandLine,
285 									e);
286 				}
287 				session.getRunner().setFileMoved(newfilename, true);
288 			}
289 			logger.error("Status: " + status + " Exec in error with " +
290 					commandLine + " returns " + newname);
291 			OpenR66RunnerErrorException exc =
292 					new OpenR66RunnerErrorException("<STATUS>" + status + "</STATUS><ERROR>"
293 							+ newname + "</ERROR>");
294 			futureCompletion.setFailure(exc);
295 		}
296 	}

298 	private void finalizeFromError(PipedOutputStream outputStream,
299 			PumpStreamHandler pumpStreamHandler,
300 			PipedInputStream inputStream, AllLineReader allLineReader, Thread thread,
301 			int status, CommandLine commandLine) {
302 		try {
303 			Thread.sleep(Configuration.RETRYINMS);
304 		} catch (InterruptedException e) {
305 		}
306 		try {
307 			outputStream.flush();
308 		} catch (IOException e2) {
309 		}
310 		try {
311 			Thread.sleep(Configuration.RETRYINMS);
312 		} catch (InterruptedException e) {
313 		}
314 		try {
315 			outputStream.close();
316 		} catch (IOException e1) {
317 		}
318 		thread.interrupt();
319 		try {
320 			inputStream.close();
321 		} catch (IOException e1) {
322 		}
323 		try {
324 			Thread.sleep(Configuration.RETRYINMS);
325 		} catch (InterruptedException e) {
326 		}
327 		pumpStreamHandler.stop();
328 		try {
329 			Thread.sleep(Configuration.RETRYINMS);
330 		} catch (InterruptedException e) {
331 		}
332 		String result = allLineReader.lastLine.toString();
333 		logger.error("Status: " + status + " Exec in error with " +
334 				commandLine + " returns\n" + result);
335 		OpenR66RunnerErrorException exc =
336 				new OpenR66RunnerErrorException("<STATUS>" + status + "</STATUS><ERROR>" + result
337 						+ "</ERROR>");
338 		futureCompletion.setFailure(exc);
339 	}
