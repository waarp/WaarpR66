50      /**
51       * Internal Logger
52       */
53      private static final WaarpLogger logger = WaarpLoggerFactory
54              .getLogger(SpooledInformTask.class);
55  
56      public static final TreeMap<String, SpooledInformation> spooledInformationMap = new TreeMap<String, SpooledInformTask.SpooledInformation>();
57  
58      public static class SpooledInformation {
59          public String host;
60          public FileMonitorInformation fileMonitorInformation;
61          public Date lastUpdate = new Date();
62  
63          /**
64           * @param host
65           * @param fileItemHashMap
66           */
67          private SpooledInformation(String host, FileMonitorInformation fileMonitorInformation) {
68              this.host = host;
69              this.fileMonitorInformation = fileMonitorInformation;
70          }
71      }
72  
73      @Override
74      public void run() {
75          if (callFromBusiness) {
76              // Business Request to validate?
77              String validated = SpooledDirectoryTransfer.PARTIALOK;
78              if (isToValidate) {
79                  try {
80                      FileMonitorInformation fileMonitorInformation =
81                              JsonHandler.mapper.readValue(fullarg, FileMonitorInformation.class);
82                      logger.info("Receive SpooledInform of size: "
83                              + fullarg.length()
84                              + " ("
85                              + fileMonitorInformation.fileItems.size()
86                              + ", "
87                              +
88                              (fileMonitorInformation.removedFileItems != null ? fileMonitorInformation.removedFileItems
89                                      .size() : -1) + ")");
90                      String host = this.session.getAuth().getUser();
91                      synchronized (spooledInformationMap) {
92                          if (fileMonitorInformation.removedFileItems == null
93                                  || fileMonitorInformation.removedFileItems.isEmpty()) {
94                              SpooledInformation old = spooledInformationMap.put(fileMonitorInformation.name,
95                                      new SpooledInformation(host, fileMonitorInformation));
96                              if (old != null && old.fileMonitorInformation != null) {
97                                  if (old.fileMonitorInformation.directories != null) {
98                                      old.fileMonitorInformation.directories.clear();
99                                  }
100                                 if (old.fileMonitorInformation.fileItems != null) {
101                                     old.fileMonitorInformation.fileItems.clear();
102                                 }
103                                 old.fileMonitorInformation = null;
104                             }
105                             old = null;
106                         } else {
107                             // partial update
108                             SpooledInformation update = spooledInformationMap.get(fileMonitorInformation.name);
109                             if (update == null) {
110                                 // Issue since update is not existing so full update is needed next time
111                                 spooledInformationMap.put(fileMonitorInformation.name, new SpooledInformation(host,
112                                         fileMonitorInformation));
113                                 validated = SpooledDirectoryTransfer.NEEDFULL;
114                             } else {
115                                 for (String item : fileMonitorInformation.removedFileItems) {
116                                     update.fileMonitorInformation.fileItems.remove(item);
117                                 }
118                                 update.fileMonitorInformation.fileItems.putAll(fileMonitorInformation.fileItems);
119                                 update.lastUpdate = new Date();
120                             }
121                         }
122                     }
123                 } catch (JsonParseException e1) {
124                     logger.warn("Cannot parse SpooledInformation: " + fullarg + " " + e1.getMessage());
125                 } catch (JsonMappingException e1) {
126                     logger.warn("Cannot parse SpooledInformation: " + fullarg + " " + e1.getMessage());
127                 } catch (IOException e1) {
128                     logger.warn("Cannot parse SpooledInformation: " + e1.getMessage());
129                 }
130                 BusinessRequestPacket packet =
131                         new BusinessRequestPacket(this.getClass().getName() + " informed", 0);
132                 validate(packet);
133                 try {
134                     ChannelUtils.writeAbstractLocalPacket(session.getLocalChannelReference(),
135                             packet, true);
136                 } catch (OpenR66ProtocolPacketException e) {
137                 }
138                 this.status = 0;
139             }
140             finalValidate(validated);
141         } else {
142             // unallowed
143             logger.warn("SpooledInformTask not allowed as Java Task: " + fullarg);
144             invalid();
145         }
146     }

148     /**
149      * @param detailed
150      * @param status
151      *            1 for ok, -1 for ko, 0 for all
152      * @param uri
153      * @return the StringBuilder containing the HTML format as a Table of the current Spooled information
154      */
155     public static StringBuilder buildSpooledTable(boolean detailed, int status, String uri) {
156         StringBuilder builder = beginSpooledTable(detailed, uri);
157         // get current information
158         synchronized (spooledInformationMap) {
159             Set<String> names = spooledInformationMap.keySet();
160             for (String name : names) {
161                 // per Name
162                 buildSpooledTableElement(detailed, status, builder, name);
163             }
164         }
165         endSpooledTable(builder);
166         return builder;
167     }
168 
169     /**
170      * @param name
171      * @param uri
172      * @return the StringBuilder containing the HTML format as a Table of the current Spooled information
173      */
174     public static StringBuilder buildSpooledUniqueTable(String uri, String name) {
175         StringBuilder builder = beginSpooledTable(false, uri);
176         // get current information
177         synchronized (spooledInformationMap) {
178             // per Name
179             SpooledInformation inform = buildSpooledTableElement(false, 0, builder, name);
180             endSpooledTable(builder);
181             builder.append("<BR>");
182             if (inform != null) {
183                 buildSpooledTableFiles(builder, inform);
184             }
185         }
186         return builder;
187     }

189     /**
190      * @param builder
191      */
192     private static void endSpooledTable(StringBuilder builder) {
193         builder.append("</TABLE>");
194     }

225     /**
226      * @param detailed
227      * @param status
228      * @param builder
229      * @param name
230      */
231     private static SpooledInformation buildSpooledTableElement(boolean detailed, int status, StringBuilder builder,
232             String name) {
233         SpooledInformation inform = spooledInformationMap.get(name);
234         if (inform == null) {
235             return null;
236         }
237         long time = inform.lastUpdate.getTime() + Configuration.configuration.TIMEOUTCON;
238         long curtime = System.currentTimeMillis();
239         if (time + Configuration.configuration.TIMEOUTCON < curtime) {
240             if (status > 0) {
241                 return inform;
242             }
243         } else {
244             if (status < 0) {
245                 return inform;
246             }
247         }
248         builder.append("<TR><TH>").append(name.replace(',', ' ')).append("</TH><TD>").append(inform.host)
249                 .append("</TD>");
250         if (time + Configuration.configuration.TIMEOUTCON < curtime) {
251             builder.append("<TD bgcolor=Red>");
252         } else if (time < curtime) {
253             builder.append("<TD bgcolor=Orange>");
254         } else {
255             builder.append("<TD bgcolor=LightGreen>");
256         }
257         builder.append(inform.lastUpdate).append("</TD>");
258         if (inform.fileMonitorInformation != null) {
259             builder.append(Messages.getString("SpooledInformTask.AllOk")) //$NON-NLS-1$
260                     .append(inform.fileMonitorInformation.globalok)
261                     .append(Messages.getString("SpooledInformTask.AllError")) //$NON-NLS-1$
262                     .append(inform.fileMonitorInformation.globalerror)
263                     .append(Messages.getString("SpooledInformTask.TodayOk")) //$NON-NLS-1$
264                     .append(inform.fileMonitorInformation.todayok)
265                     .append(Messages.getString("SpooledInformTask.TodayError")) //$NON-NLS-1$
266                     .append(inform.fileMonitorInformation.todayerror)
267                     .append("</TD><TD>")
268                     .append(inform.fileMonitorInformation.elapseTime)
269                     .append("</TD><TD>")
270                     .append(inform.fileMonitorInformation.stopFile)
271                     .append("</TD><TD>")
272                     .append(inform.fileMonitorInformation.statusFile)
273                     .append("</TD><TD>")
274                     .append(inform.fileMonitorInformation.scanSubDir)
275                     .append("</TD>");
276             String dirs = "";
277             for (File dir : inform.fileMonitorInformation.directories) {
278                 dirs += dir + "<br>";
279             }
280             builder.append("<TD>").append(dirs).append("</TD><TD>");
281             if (detailed && inform.fileMonitorInformation.fileItems != null) {
282                 buildSpooledTableFiles(builder, inform);
283             } else {
284                 // simply print number of files
285                 if (inform.fileMonitorInformation.fileItems != null) {
286                     builder.append(inform.fileMonitorInformation.fileItems.size());
287                 } else {
288                     builder.append(0);
289                 }
290                 // Form GET to ensure encoding
291                 builder.append(
292                         "<FORM name='DETAIL' method='GET' action='/SpooledDetailed.html'><input type=hidden name='name' value='")
293                         .append(name).append("'/><INPUT type='submit' value='DETAIL'/></FORM>");
294             }
295         }
296         builder.append("</TD></TR>");
297         return inform;
298     }

300     /**
301      * @param builder
302      * @param inform
303      */
304     private static void buildSpooledTableFiles(StringBuilder builder, SpooledInformation inform) {
305         builder.append("<TABLE BORDER=1><TR><TH>").append(Messages.getString("SpooledInformTask.10")) //$NON-NLS-1$
306                 .append("</TH><TH>").append(Messages.getString("SpooledInformTask.11")) //$NON-NLS-1$
307                 .append("</TH><TH>").append(Messages.getString("SpooledInformTask.12")) //$NON-NLS-1$
308                 .append("</TH><TH>").append(Messages.getString("SpooledInformTask.13")) //$NON-NLS-1$
309                 .append("</TH><TH>").append(Messages.getString("SpooledInformTask.14")) //$NON-NLS-1$
310                 .append("</TH><TH>").append(Messages.getString("SpooledInformTask.15")) //$NON-NLS-1$
311                 .append("</TH></TR>");
312         for (FileItem fileItem : inform.fileMonitorInformation.fileItems.values()) {
313             builder.append("<TR><TD>").append(fileItem.file).append("</TD><TD>");
314             if (fileItem.hash != null) {
315                 builder.append(FilesystemBasedDigest.getHex(fileItem.hash));
316             }
317             builder.append("</TD><TD>");
318             if (fileItem.lastTime > 0) {
319                 builder.append(new Date(fileItem.lastTime));
320             }
321             builder.append("</TD><TD>");
322             if (fileItem.timeUsed > 0) {
323                 builder.append(new Date(fileItem.timeUsed));
324             }
325             builder.append("</TD><TD>").append(fileItem.used).append("</TD><TD>")
326                     .append(fileItem.specialId).append("</TD></TR>");
327         }
328         builder.append("</TABLE>");
329     }
