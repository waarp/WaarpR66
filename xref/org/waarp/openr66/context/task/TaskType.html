18  package org.waarp.openr66.context.task;
19  
20  import org.waarp.common.logging.WaarpLogger;
21  import org.waarp.common.logging.WaarpLoggerFactory;
22  import org.waarp.openr66.context.R66Session;
23  import org.waarp.openr66.context.task.exception.OpenR66RunnerErrorException;

25  /**
26   * This enum class references all available tasks.
27   * 
28   * If a new task type is to be created, this is the place where it should be referenced.
29   * 
30   * @author Frederic Bregier
31   * 
32   */
33  public enum TaskType {
34      LOG, MOVE, MOVERENAME, COPY, COPYRENAME, EXEC, EXECMOVE, LINKRENAME, TRANSFER,
35      VALIDFILEPATH, DELETE, TAR, ZIP, EXECOUTPUT, RESCHEDULE, EXECJAVA, TRANSCODE, SNMP, FTP,
36      RENAME, RESTART, UNZEROED, CHMOD, CHKFILE;
37  
38      int type;
39  
40      String name;
41      /**
42       * Internal Logger
43       */
44      private static final WaarpLogger logger = WaarpLoggerFactory
45              .getLogger(TaskType.class);
46  
47      private TaskType() {
48          type = ordinal();
49          name = name();
50      }
51  
52      /**
53       * 
54       * @param type
55       * @param argRule
56       * @param delay
57       * @param session
58       * @return the corresponding AbstractTask
59       * @throws OpenR66RunnerErrorException
60       */
61      public static AbstractTask getTaskFromId(TaskType type, String argRule,
62              int delay, R66Session session)
63              throws OpenR66RunnerErrorException {
64          switch (type) {
65              case LOG:
66                  return new LogTask(argRule, delay, session.getRunner()
67                          .getFileInformation(), session);
68              case MOVE:
69                  return new MoveTask(argRule, delay, session.getRunner()
70                          .getFileInformation(), session);
71              case MOVERENAME:
72                  return new MoveRenameTask(argRule, delay, session.getRunner()
73                          .getFileInformation(), session);
74              case COPY:
75                  return new CopyTask(argRule, delay, session.getRunner()
76                          .getFileInformation(), session);
77              case COPYRENAME:
78                  return new CopyRenameTask(argRule, delay, session.getRunner()
79                          .getFileInformation(), session);
80              case EXEC:
81                  return new ExecTask(argRule, delay, session.getRunner()
82                          .getFileInformation(), session);
83              case EXECMOVE:
84                  return new ExecMoveTask(argRule, delay, session.getRunner()
85                          .getFileInformation(), session);
86              case LINKRENAME:
87                  return new LinkRenameTask(argRule, delay, session.getRunner()
88                          .getFileInformation(), session);
89              case TRANSFER:
90                  return new TransferTask(argRule, delay, session.getRunner()
91                          .getFileInformation(), session);
92              case VALIDFILEPATH:
93                  return new ValidFilePathTask(argRule, delay, session.getRunner()
94                          .getFileInformation(), session);
95              case DELETE:
96                  return new DeleteTask(argRule, delay, session.getRunner()
97                          .getFileInformation(), session);
98              case TAR:
99                  return new TarTask(argRule, delay, session.getRunner()
100                         .getFileInformation(), session);
101             case ZIP:
102                 return new ZipTask(argRule, delay, session.getRunner()
103                         .getFileInformation(), session);
104             case EXECOUTPUT:
105                 return new ExecOutputTask(argRule, delay, session.getRunner().
106                         getFileInformation(), session);
107             case RESCHEDULE:
108                 return new RescheduleTransferTask(argRule, delay, session.getRunner().
109                         getFileInformation(), session);
110             case EXECJAVA:
111                 return new ExecJavaTask(argRule, delay, session.getRunner().
112                         getFileInformation(), session);
113             case TRANSCODE:
114                 return new TranscodeTask(argRule, delay, session.getRunner().
115                         getFileInformation(), session);
116             case SNMP:
117                 return new SnmpTask(argRule, delay, session.getRunner().
118                         getFileInformation(), session);
119             case FTP:
120                 return new FtpTransferTask(argRule, delay, session.getRunner().
121                         getFileInformation(), session);
122             case RENAME:
123                 return new RenameTask(argRule, delay, session.getRunner().
124                         getFileInformation(), session);
125             case RESTART:
126                 return new RestartServerTask(argRule, delay, session.getRunner().
127                         getFileInformation(), session);
128             case UNZEROED:
129                 return new UnzeroedFileTask(argRule, delay, session.getRunner().
130                         getFileInformation(), session);
131             case CHMOD:
132                 return new ChModTask(argRule, delay, session.getRunner().
133                         getFileInformation(), session);
134             case CHKFILE:
135                 return new FileCheckTask(argRule, delay, session.getRunner().
136                         getFileInformation(), session);
137             default:
138                 logger.error("name unknown: " + type.name);
139                 throw new OpenR66RunnerErrorException("Unvalid Task: " +
140                         type.name);
141         }
142     }

144     /**
145      * 
146      * @param name
147      * @param argRule
148      * @param delay
149      * @param session
150      * @return the corresponding AbstractTask
151      * @throws OpenR66RunnerErrorException
152      */
153     public static AbstractTask getTaskFromId(String name, String argRule,
154             int delay, R66Session session) throws OpenR66RunnerErrorException {
155         TaskType type;
156         try {
157             type = valueOf(name);
158         } catch (NullPointerException e) {
159             logger.error("name empty " + name);
160             throw new OpenR66RunnerErrorException("Unvalid Task: " + name);
161         } catch (IllegalArgumentException e) {
162             logger.error("name unknown: " + name);
163             throw new OpenR66RunnerErrorException("Unvalid Task: " + name);
164         }
165         return getTaskFromId(type, argRule, delay, session);
166     }

168     /**
169      * For usage in ExecBusinessTask
170      * 
171      * @param name
172      * @param argRule
173      * @param delay
174      * @param session
175      * @return the corresponding AbstractTask
176      * @throws OpenR66RunnerErrorException
177      */
178     public static AbstractTask getTaskFromIdForBusiness(String name, String argRule,
179             int delay, R66Session session)
180             throws OpenR66RunnerErrorException {
181         TaskType type;
182         try {
183             type = valueOf(name);
184         } catch (NullPointerException e) {
185             logger.error("name empty " + name);
186             throw new OpenR66RunnerErrorException("Unvalid Task: " + name);
187         } catch (IllegalArgumentException e) {
188             logger.error("name unknown: " + name);
189             throw new OpenR66RunnerErrorException("Unvalid Task: " + name);
190         }
191         switch (type) {
192             case LOG:
193                 int newdelay = delay;
194                 if (newdelay == 0) {
195                     newdelay = 1;
196                 }
197                 return new LogTask(argRule, newdelay, "", session);
198             case EXEC:
199                 return new ExecTask(argRule, delay, "", session);
200             case TRANSFER:
201                 return new TransferTask(argRule, delay, "", session);
202             case TAR:
203                 return new TarTask(argRule, delay, "", session);
204             case ZIP:
205                 return new ZipTask(argRule, delay, "", session);
206             case EXECOUTPUT:
207                 return new ExecOutputTask(argRule, delay, "", session);
208             case EXECJAVA:
209                 return new ExecJavaTask(argRule, delay, "", session);
210             case SNMP:
211                 return new SnmpTask(argRule, delay, "", session);
212             case FTP:
213                 return new FtpTransferTask(argRule, delay, "", session);
214             case RESTART:
215                 return new RestartServerTask(argRule, delay, "", session);
216             case MOVE:
217             case MOVERENAME:
218             case COPY:
219             case COPYRENAME:
220             case EXECMOVE:
221             case LINKRENAME:
222             case VALIDFILEPATH:
223             case DELETE:
224             case RESCHEDULE:
225             case TRANSCODE:
226             case RENAME:
227             case UNZEROED:
228             case CHMOD:
229             case CHKFILE:
230                 throw new OpenR66RunnerErrorException("Unvalid Task: " +
231                         type.name);
232             default:
233                 logger.error("name unknown: " + type.name);
234                 throw new OpenR66RunnerErrorException("Unvalid Task: " +
235                         type.name);
236         }
237     }
