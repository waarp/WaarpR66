31  /**
32   * Command to Restart the R66 server (for instance after upgrade of jar sent by administrative operations, unzipped in the library directory)
33   * 
34   * 
35   * @author Frederic Bregier
36   * 
37   */
38  public class RestartServerTask extends AbstractTask {
39  
40  	/**
41  	 * Internal Logger
42  	 */
43  	private static final WaarpInternalLogger logger = WaarpInternalLoggerFactory
44  			.getLogger(RestartServerTask.class);
45  
46  	/**
47  	 * @param argRule
48  	 * @param delay
49  	 * @param argTransfer
50  	 * @param session
51  	 */
52  	public RestartServerTask(String argRule, int delay, String argTransfer,
53  			R66Session session) {
54  		super(TaskType.RESTART, delay, argRule, argTransfer, session);
55  	}
56  
57  	@Override
58  	public void run() {
59  		// check if allowed to do restart
60  		// SYSTEM authorization
61  		boolean isAdmin = session.getAuth().isValidRole(ROLE.SYSTEM);
62  		if (! isAdmin) {
63  			// not allowed
64  			logger.error("Shutdown order received but unallowed: " +
65  					session.getAuth().getUser());
66  			futureCompletion.setFailure(new OpenR66ProtocolSystemException("Shutdown order received but unallowed: " +
67  					session.getAuth().getUser()));
68  			return;
69  		}
70  		// now start the process
71  		logger.warn("Shutdown order received and going from: " +
72  				session.getAuth().getUser());
73  		R66ShutdownHook.setRestart(true);
74  		futureCompletion.setSuccess();
75  		Thread thread = new Thread(new ChannelUtils(), "R66 Shutdown Thread");
76  		thread.setDaemon(true);
77  		// give time for the task to finish correctly
78  		Configuration.configuration.launchInFixedDelay(thread, Configuration.WAITFORNETOP, TimeUnit.MILLISECONDS);
79  	}
80  
81  }
