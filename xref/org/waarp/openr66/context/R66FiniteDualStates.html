32  public enum R66FiniteDualStates {

39  	ENDTRANSFERS, ENDREQUESTS,
40  	TEST, INFORMATION, VALIDOTHER,

46  	private enum R66Transition {
47  		tOPENEDCHANNEL(OPENEDCHANNEL, EnumSet.of(STARTUP, CLOSEDCHANNEL, ERROR)),
48  		tSTARTUP(STARTUP, EnumSet.of(AUTHENTR, AUTHENTD, CLOSEDCHANNEL, ERROR)),
49  		tAUTHENTR(AUTHENTR, EnumSet.of(AUTHENTD, CLOSEDCHANNEL, ERROR)),
50  		tAUTHENTD(AUTHENTD, EnumSet.of(REQUESTR, VALIDOTHER, INFORMATION, SHUTDOWN, TEST,
51  				BUSINESSR, BUSINESSD, ENDREQUESTS, CLOSEDCHANNEL, ERROR)),

53  		tREQUESTD(REQUESTD, EnumSet.of(DATAS, DATAR, CLOSEDCHANNEL, ERROR)),

55  		tDATAS(DATAS, EnumSet.of(DATAS, ENDTRANSFERS, CLOSEDCHANNEL, ERROR)),
56  		tDATAR(DATAR, EnumSet.of(DATAR, ENDTRANSFERS, CLOSEDCHANNEL, ERROR)),
57  		tENDTRANSFERS(ENDTRANSFERS, EnumSet.of(ENDTRANSFERR, CLOSEDCHANNEL, ERROR)),
58  		tENDTRANSFERR(ENDTRANSFERR, EnumSet.of(ENDREQUESTS, CLOSEDCHANNEL, ERROR)),
59  		tENDREQUESTS(ENDREQUESTS, EnumSet.of(ENDREQUESTR, CLOSEDCHANNEL, ERROR)),
60  		tENDREQUESTR(ENDREQUESTR, EnumSet.of(CLOSEDCHANNEL, ERROR)),
61  		tINFORMATION(INFORMATION, EnumSet.of(VALIDOTHER, CLOSEDCHANNEL, ERROR)),
62  		tTEST(TEST, EnumSet.of(TEST, VALIDOTHER)),
63  		tVALIDOTHER(VALIDOTHER, EnumSet.of(VALIDOTHER, CLOSEDCHANNEL, ERROR)),
64  		tSHUTDOWN(SHUTDOWN, EnumSet.of(CLOSEDCHANNEL, SHUTDOWN, ERROR)),
65  		tERROR(ERROR, EnumSet.of(ERROR, CLOSEDCHANNEL)),
66  		tCLOSEDCHANNEL(CLOSEDCHANNEL, EnumSet.noneOf(R66FiniteDualStates.class)),
67  		tBUSINESSR(BUSINESSR, EnumSet.of(ERROR, BUSINESSD, CLOSEDCHANNEL, VALIDOTHER)),
68  		tBUSINESSD(BUSINESSD, EnumSet.of(ERROR, BUSINESSD, BUSINESSR, CLOSEDCHANNEL, VALIDOTHER));

70  		public Transition<R66FiniteDualStates> elt;
71  
72  		private R66Transition(R66FiniteDualStates state, EnumSet<R66FiniteDualStates> set) {
73  			this.elt = new Transition<R66FiniteDualStates>(state, set);
74  		}
75  	}
76  
77  	private static ConcurrentHashMap<R66FiniteDualStates, EnumSet<?>> stateMap =
78  			new ConcurrentHashMap<R66FiniteDualStates, EnumSet<?>>();
79  
80  	/**
81  	 * This method should be called once at startup to initialize the Finite States association.
82  	 */
83  	public static void initR66FiniteStates() {
84  		for (R66Transition trans : R66Transition.values()) {
85  			stateMap.put(trans.elt.state, trans.elt.set);
86  		}
87  	}

93  	public static MachineState<R66FiniteDualStates> newSessionMachineState() {
94  		MachineState<R66FiniteDualStates> machine =
95  				new MachineState<R66FiniteDualStates>(OPENEDCHANNEL, stateMap);
96  		return machine;
97  	}
