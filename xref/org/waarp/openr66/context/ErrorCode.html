22  /**
23   * This enum class keeps all code that will be returned into the result and store (char
24   * representation) into the runner.
25   *

27   * 
28   */
29  public enum ErrorCode {
30      /**
31       * Code stands for initialization ok (internal connection, authentication)
32       */
33      InitOk(Messages.getString("ErrorCode.0"), 'i'), //$NON-NLS-1$
34      /**
35       * Code stands for pre processing ok
36       */
37      PreProcessingOk(
38              Messages.getString("ErrorCode.1"), //$NON-NLS-1$
39              'B'),
40      /**
41       * Code stands for transfer OK
42       */
43      TransferOk(Messages.getString("ErrorCode.2"), 'X'), //$NON-NLS-1$
44      /**
45       * Code stands for post processing ok
46       */
47      PostProcessingOk(
48              Messages.getString("ErrorCode.3"), //$NON-NLS-1$
49              'P'),
50      /**
51       * Code stands for All action are completed ok
52       */
53      CompleteOk(Messages.getString("ErrorCode.4"), 'O'), //$NON-NLS-1$
54      /**
55       * Code stands for connection is impossible (remote or local reason)
56       */
57      ConnectionImpossible(
58              Messages.getString("ErrorCode.5"), //$NON-NLS-1$
59              'C'),
60      /**
61       * Code stands for connection is impossible now due to limits(remote or local reason)
62       */
63      ServerOverloaded(
64              Messages.getString("ErrorCode.6"), //$NON-NLS-1$
65              'l'),
66      /**
67       * Code stands for bad authentication (remote or local)

69      BadAuthent(Messages.getString("ErrorCode.7"), 'A'), //$NON-NLS-1$
70      /**
71       * Code stands for External operation in error (pre, post or error processing)
72       */
73      ExternalOp(

75              'E'),
76      /**
77       * Code stands for Transfer is in error
78       */
79      TransferError(Messages.getString("ErrorCode.9"), 'T'), //$NON-NLS-1$
80      /**
81       * Code stands for Transfer in error due to MD5
82       */
83      MD5Error(
84              Messages.getString("ErrorCode.10"), //$NON-NLS-1$
85              'M'),
86      /**
87       * Code stands for Network disconnection
88       */
89      Disconnection(Messages.getString("ErrorCode.11"), 'D'), //$NON-NLS-1$
90      /**
91       * Code stands for Remote Shutdown

93      RemoteShutdown(Messages.getString("ErrorCode.12"), 'r'), //$NON-NLS-1$

114     /**
115      * Code stands for an internal error
116      */
117     Internal(
118             Messages.getString("ErrorCode.17"), //$NON-NLS-1$
119             'I'),
120     /**
121      * Code stands for a request of stopping transfer
122      */
123     StoppedTransfer(Messages.getString("ErrorCode.18"), 'H'), //$NON-NLS-1$
124     /**
125      * Code stands for a request of canceling transfer
126      */
127     CanceledTransfer(Messages.getString("ErrorCode.19"), 'K'), //$NON-NLS-1$
128     /**
129      * Warning in execution
130      */
131     Warning(Messages.getString("ErrorCode.20"), 'W'), //$NON-NLS-1$
132     /**
133      * Code stands for unknown type of error
134      */
135     Unknown(Messages.getString("ErrorCode.21"), '-'), //$NON-NLS-1$
136     /**
137      * Code stands for a request that is already remotely finished
138      */
139     QueryAlreadyFinished(Messages.getString("ErrorCode.22"), 'Q'), //$NON-NLS-1$
140     /**
141      * Code stands for request that is still running

143     QueryStillRunning(Messages.getString("ErrorCode.23"), 's'), //$NON-NLS-1$
144     /**
145      * Code stands for not known host
146      */
147     NotKnownHost(Messages.getString("ErrorCode.24"), 'N'), //$NON-NLS-1$
148     /**
149      * Code stands for self requested host starting request is invalid
150      */
151     LoopSelfRequestedHost(Messages.getString("ErrorCode.25"), 'L'), //$NON-NLS-1$
152     /**
153      * Code stands for request should exist but is not found on remote host

155     QueryRemotelyUnknown(Messages.getString("ErrorCode.26"), 'u'), //$NON-NLS-1$
156     /**
157      * Code stands for File not found error
158      */
159     FileNotFound(Messages.getString("ErrorCode.27"), 'f'), //$NON-NLS-1$
160     /**
161      * Code stands for Command not found error

169     /**
170      * Code stands for running step
171      */

173     /**
174      * Code stands for Incorrect command
175      */
176     IncorrectCommand(Messages.getString("ErrorCode.31"), 'n'), //$NON-NLS-1$

178      * Code stands for File not allowed
179      */
180     FileNotAllowed(Messages.getString("ErrorCode.32"), 'a'), //$NON-NLS-1$

183      */
184     SizeNotAllowed(Messages.getString("ErrorCode.33"), 'd'); //$NON-NLS-1$

186     /**

196         this.mesg = mesg;

244     /**
245      * Code is either the 1 char code or the exact name in Enum
246      * 
247      * @param code
248      * @return the ErrorCode according to the code
249      */
250     public static ErrorCode getFromCode(String code) {
251         if (code.isEmpty()) {
252             return Unknown;
253         }
254         switch (code.charAt(0)) {
255             case 'i':
256                 return InitOk;
257             case 'B':
258                 return PreProcessingOk;
259             case 'P':
260                 return PostProcessingOk;
261             case 'X':
262                 return TransferOk;
263             case 'O':
264                 return CompleteOk;
265             case 'C':
266                 return ConnectionImpossible;
267             case 'A':
268                 return BadAuthent;
269             case 'E':
270                 return ExternalOp;
271             case 'T':
272                 return TransferError;
273             case 'M':
274                 return MD5Error;
275             case 'D':
276                 return Disconnection;
277             case 'r':
278                 return RemoteShutdown;
279             case 'F':
280                 return FinalOp;
281             case 'U':
282                 return Unimplemented;
283             case 'S':
284                 return Shutdown;
285             case 'R':
286                 return RemoteError;
287             case 'I':
288                 return Internal;
289             case 'H':
290                 return StoppedTransfer;
291             case 'K':
292                 return CanceledTransfer;
293             case 'W':
294                 return Warning;
295             case '-':
296                 return Unknown;
297             case 'Q':
298                 return QueryAlreadyFinished;
299             case 's':
300                 return QueryStillRunning;
301             case 'N':
302                 return NotKnownHost;
303             case 'L':
304                 return LoopSelfRequestedHost;
305             case 'u':
306                 return QueryRemotelyUnknown;
307             case 'f':
308                 return FileNotFound;
309             case 'z':
310                 return Running;
311             case 'c':
312                 return CommandNotFound;
313             case 'p':
314                 return PassThroughMode;
315             case 'l':
316                 return ServerOverloaded;
317             case 'n':
318                 return IncorrectCommand;
319             case 'a':
320                 return FileNotAllowed;
321             case 'd':
322                 return SizeNotAllowed;
323             default:
324                 ErrorCode ecode = Unknown;
325                 try {
326                     ecode = ErrorCode.valueOf(code.trim());
327                 } catch (IllegalArgumentException e) {
328                     return Unknown;
329                 }
330                 return ecode;
331         }
332     }

334     public static boolean isErrorCode(ErrorCode code) {
335         switch (code) {
336             case BadAuthent:
337             case CanceledTransfer:
338             case CommandNotFound:
339             case ConnectionImpossible:
340             case Disconnection:
341             case ExternalOp:
342             case FileNotFound:
343             case FinalOp:
344             case Internal:
345             case LoopSelfRequestedHost:
346             case MD5Error:
347             case NotKnownHost:
348             case PassThroughMode:
349             case QueryAlreadyFinished:
350             case QueryRemotelyUnknown:
351             case QueryStillRunning:
352             case RemoteError:
353             case RemoteShutdown:
354             case ServerOverloaded:
355             case Shutdown:
356             case StoppedTransfer:
357             case TransferError:
358             case Unimplemented:
359             case IncorrectCommand:
360             case FileNotAllowed:
361             case SizeNotAllowed:
362                 return true;
363             case CompleteOk:
364             case InitOk:
365             case PostProcessingOk:
366             case PreProcessingOk:
367             case Running:
368             case TransferOk:
369             case Unknown:
370             case Warning:
371                 return false;
372             default:
373                 break;
374         }
375         return true;
376     }
