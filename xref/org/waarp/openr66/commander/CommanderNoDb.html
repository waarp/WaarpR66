37  /**
38   * Commander is responsible to read list of updated data from time to time in order to achieve new
39   * runner or new configuration updates.
40   * 
41   * Based on no Database support
42   * 
43   * @author Frederic Bregier
44   * 
45   */
46  public class CommanderNoDb implements CommanderInterface {
47  	/**
48  	 * Internal Logger
49  	 */
50  	private static final WaarpInternalLogger logger = WaarpInternalLoggerFactory
51  			.getLogger(CommanderNoDb.class);
52  
53  	private InternalRunner internalRunner = null;
54  	public static final ConcurrentLinkedQueue<AbstractDbData> todoList = new ConcurrentLinkedQueue<AbstractDbData>();
55  
56  	/**
57  	 * Prepare requests that will be executed from time to time
58  	 * 
59  	 * @param runner
60  	 */
61  	public CommanderNoDb(InternalRunner runner) {
62  		this.internalConstructor(runner);
63  	}
64  
65  	/**
66  	 * Prepare requests that will be executed from time to time
67  	 * 
68  	 * @param runner
69  	 * @param fromStartup
70  	 *            True if call from startup of the server
71  	 */
72  	public CommanderNoDb(InternalRunner runner, boolean fromStartup) {
73  		this.internalConstructor(runner);
74  		if (fromStartup) {
75  			ClientRunner.activeRunners = new ConcurrentLinkedQueue<ClientRunner>();
76  			// Change RUNNING or INTERRUPTED to TOSUBMIT since they should be ready
77  			/*
78  			 * Configuration.configuration.baseDirectory+
79  			 * Configuration.configuration.archivePath+R66Dir.SEPARATOR+
80  			 * this.requesterHostId+"_"+this.requestedHostId+"_"+this.ruleId+"_"+this.specialId
81  			 * +XMLEXTENSION;
82  			 */
83  			File directory = new File(Configuration.configuration.baseDirectory +
84  					Configuration.configuration.archivePath);
85  			File[] files = FileUtils.getFiles(directory,
86  					new ExtensionFilter(DbTaskRunner.XMLEXTENSION));
87  			for (File file : files) {
88  				String shortname = file.getName();
89  				String[] info = shortname.split("_");
90  				if (info.length < 5) {
91  					continue;
92  				}
93  				DbRule rule;
94  				try {
95  					rule = new DbRule(null, info[2]);
96  				} catch (WaarpDatabaseException e) {
97  					logger.warn("Cannot find the rule named: " + info[2]);
98  					continue;
99  				}
100 				long id = Long.parseLong(info[3]);
101 				try {
102 					DbTaskRunner task = new DbTaskRunner(null, null, rule, id, info[0], info[1]);
103 					UpdatedInfo status = task.getUpdatedInfo();
104 					if (status == UpdatedInfo.RUNNING || status == UpdatedInfo.INTERRUPTED) {
105 						task.changeUpdatedInfo(UpdatedInfo.TOSUBMIT);
106 						task.update();
107 					}
108 				} catch (WaarpDatabaseException e) {
109 					logger.warn("Cannot reload the task named: " + shortname);
110 					continue;
111 				}
112 			}
113 		}
114 	}

116 	private void internalConstructor(InternalRunner runner) {
117 		internalRunner = runner;
118 	}

121 	 * Finalize internal data
122 	 */
123 	public void finalize() {
124 		// no since it will be reloaded
125 		// todoList.clear();
126 	}

128 	/*
129 	 * (non-Javadoc)
130 	 * @see java.lang.Runnable#run()
131 	 */
132 	@Override
133 	public void run() {
134 		Thread.currentThread().setName("OpenR66Commander");
135 		while (!todoList.isEmpty()) {
136 			try {
137 				AbstractDbData data = todoList.poll();
138 				// First check Configuration
139 				if (data instanceof DbConfiguration) {
140 					// should be only one...
141 					DbConfiguration configuration = (DbConfiguration) data;
142 					if (configuration.isOwnConfiguration()) {
143 						configuration.updateConfiguration();
144 					}
145 					configuration.changeUpdatedInfo(AbstractDbData.UpdatedInfo.NOTUPDATED);
146 					configuration.update();
147 				}
148 				// Check HostAuthent
149 				else if (data instanceof DbHostAuth) {
150 					DbHostAuth hostAuth = (DbHostAuth) data;
151 					// Nothing to do except validate
152 					hostAuth.changeUpdatedInfo(AbstractDbData.UpdatedInfo.NOTUPDATED);
153 					hostAuth.update();
154 				}
155 				// Check Rules
156 				else if (data instanceof DbRule) {
157 					// Nothing to do except validate
158 					DbRule rule = (DbRule) data;
159 					rule.changeUpdatedInfo(AbstractDbData.UpdatedInfo.NOTUPDATED);
160 					rule.update();
161 				}
162 				// Check TaskRunner
163 				else if (data instanceof DbTaskRunner) {
164 					DbTaskRunner taskRunner = (DbTaskRunner) data;
165 					logger.debug("get a task: {}", taskRunner);
166 					// Launch if possible this task
167 					String key = taskRunner.getRequested() + " " + taskRunner.getRequester() +
168 							" " + taskRunner.getSpecialId();
169 					if (Configuration.configuration.getLocalTransaction().
170 							getFromRequest(key) != null) {
171 						// already running
172 						continue;
173 					}
174 					if (taskRunner.isSelfRequested()) {
175 						// cannot schedule a request where the host is the requested host
176 						taskRunner.changeUpdatedInfo(UpdatedInfo.INTERRUPTED);
177 						taskRunner.update();
178 						continue;
179 					}
180 					taskRunner.changeUpdatedInfo(UpdatedInfo.RUNNING);
181 					taskRunner.update();
182 					internalRunner.submitTaskRunner(taskRunner);
183 					try {
184 						Thread.sleep(Configuration.RETRYINMS);
185 					} catch (InterruptedException e) {
186 					}
187 					taskRunner = null;
188 				}
189 				if (R66ShutdownHook.isInShutdown()) {
190 					// no more task to submit
191 					return;
192 				}
193 			} catch (WaarpDatabaseException e) {
194 				// TODO Auto-generated catch block
195 				e.printStackTrace();
196 			}
197 		}
198 	}
