109 	 * (non-Javadoc)
110 	 * @see java.lang.Runnable#run()
111 	 */
112 	@Override
113 	public void run() {
114 		if (Configuration.configuration.isShutdown) {
115 			taskRunner.changeUpdatedInfo(UpdatedInfo.TOSUBMIT);
116 			try {
117 				taskRunner.update();
118 			} catch (WaarpDatabaseException e) {
119 			}
120 			return;
121 		}
122 		try {
123 			if (activeRunners != null) {
124 				activeRunners.add(this);
125 			}
126 			R66Future transfer;
127 			try {
128 				transfer = this.runTransfer();
129 			} catch (OpenR66RunnerErrorException e) {
130 				logger.error("Runner Error: {} {}", e.getMessage(),
131 						taskRunner.toShortString());
132 				return;
133 			} catch (OpenR66ProtocolNoConnectionException e) {
134 				logger.error("No connection Error {}", e.getMessage());
135 				if (localChannelReference != null) {
136 					localChannelReference.setErrorMessage(
137 							ErrorCode.ConnectionImpossible.mesg,
138 							ErrorCode.ConnectionImpossible);
139 				}
140 				taskRunner.setErrorTask(localChannelReference);
141 				try {
142 					taskRunner.saveStatus();
143 					taskRunner.run();
144 				} catch (OpenR66RunnerErrorException e1) {
145 					this.changeUpdatedInfo(UpdatedInfo.INERROR,
146 							ErrorCode.ConnectionImpossible);
147 				}
148 				return;
149 			} catch (OpenR66ProtocolPacketException e) {
150 				logger.error("Protocol Error", e);
151 				return;
152 			} catch (OpenR66ProtocolNotYetConnectionException e) {
153 				logger.warn("No connection warning {}", e.getMessage());
154 				return;
155 			}
156 			R66Result result = transfer.getResult();
157 			if (result != null) {
158 				if (result.code == ErrorCode.QueryAlreadyFinished) {
159 					logger.warn("TRANSFER RESULT:\n    " +
160 							(transfer.isSuccess() ? "SUCCESS" : "FAILURE") +
161 							"\n    " + ErrorCode.QueryAlreadyFinished.mesg +
162 							":" +
163 							(result != null ? result.toString() : "no result"));
164 				} else {
165 					if (transfer.isSuccess()) {
166 						logger.info("TRANSFER RESULT:\n    SUCCESS\n    " +
167 								(result != null ? result.toString()
168 										: "no result"));
169 					} else {
170 						logger.error("TRANSFER RESULT:\n    FAILURE\n    " +
171 								(result != null ? result.toString()
172 										: "no result"));
173 					}
174 				}
175 			} else {
176 				if (transfer.isSuccess()) {
177 					logger.warn("TRANSFER REQUESTED RESULT:\n    SUCCESS\n    no result");
178 				} else {
179 					logger.error("TRANSFER REQUESTED RESULT:\n    FAILURE\n    no result");
180 				}
181 			}
182 			transfer = null;
183 			Thread.currentThread().setName(
184 					"Finished_" + Thread.currentThread().getName());
185 		} finally {
186 			if (activeRunners != null) {
187 				activeRunners.remove(this);
188 			}
189 		}
190 	}

192 	/**
193 	 * 
194 	 * @param runner
195 	 * @param limit
196 	 * @return True if the task was run less than limit, else False
197 	 */
198 	public boolean incrementTaskRunerTry(DbTaskRunner runner, int limit) {
199 		String key = runner.getKey();
200 		Integer tries = taskRunnerRetryHashMap.get(key);
201 		logger.debug("try to find integer: " + tries);
202 		if (tries == null) {
203 			tries = new Integer(1);
204 		} else {
205 			tries = tries + 1;
206 		}
207 		if (limit <= tries) {
208 			taskRunnerRetryHashMap.remove(key);
209 			return false;
210 		} else {
211 			taskRunnerRetryHashMap.put(key, tries);
212 			return true;
213 		}
214 	}

216 	/**
217 	 * True transfer run (can be called directly to enable exception outside any executors)
218 	 * 
219 	 * @return The R66Future of the transfer operation
220 	 * @throws OpenR66RunnerErrorException
221 	 * @throws OpenR66ProtocolNoConnectionException
222 	 * @throws OpenR66ProtocolPacketException
223 	 * @throws OpenR66ProtocolNotYetConnectionException
224 	 */
225 	public R66Future runTransfer() throws OpenR66RunnerErrorException,
226 			OpenR66ProtocolNoConnectionException,
227 			OpenR66ProtocolPacketException,
228 			OpenR66ProtocolNotYetConnectionException {
229 		logger.debug("Start attempt Transfer");
230 		localChannelReference = initRequest();
231 		try {
232 			localChannelReference.getFutureValidRequest().await();
233 		} catch (InterruptedException e) {
234 		}
235 		if (localChannelReference.getFutureValidRequest().isSuccess()) {
236 			return finishTransfer(true, localChannelReference);
237 		} else if (localChannelReference.getFutureValidRequest().getResult().code == ErrorCode.ServerOverloaded) {
238 			return tryAgainTransferOnOverloaded(true, localChannelReference);
239 		} else
240 			return finishTransfer(true, localChannelReference);
241 	}

243 	/**
244 	 * In case an overloaded signal is returned by the requested
245 	 * 
246 	 * @param retry
247 	 *            if True, it will retry in case of overloaded remote server, else it just stops
248 	 * @param localChannelReference
249 	 * @return The R66Future of the transfer operation
250 	 * @throws OpenR66RunnerErrorException
251 	 * @throws OpenR66ProtocolNoConnectionException
252 	 * @throws OpenR66ProtocolPacketException
253 	 * @throws OpenR66ProtocolNotYetConnectionException
254 	 */
255 	public R66Future tryAgainTransferOnOverloaded(boolean retry,
256 			LocalChannelReference localChannelReference)
257 			throws OpenR66RunnerErrorException,
258 			OpenR66ProtocolNoConnectionException,
259 			OpenR66ProtocolPacketException,
260 			OpenR66ProtocolNotYetConnectionException {
261 		if (this.localChannelReference == null) {
262 			this.localChannelReference = localChannelReference;
263 		}
264 		boolean incRetry = incrementTaskRunerTry(taskRunner,
265 				Configuration.RETRYNB);
266 		logger.debug("tryAgainTransferOnOverloaded: " + retry + ":" + incRetry);
267 		switch (taskRunner.getUpdatedInfo()) {
268 			case DONE:
269 			case INERROR:
270 			case INTERRUPTED:
271 				break;
272 			default:
273 				this.changeUpdatedInfo(UpdatedInfo.INERROR,
274 						ErrorCode.ServerOverloaded);
275 		}
276 		// redo if possible
277 		if (retry && incRetry) {
278 			try {
279 				Thread.sleep(Configuration.configuration.constraintLimitHandler
280 						.getSleepTime());
281 			} catch (InterruptedException e) {
282 			}
283 			return runTransfer();
284 		} else {
285 			if (localChannelReference == null) {
286 				taskRunner
287 						.setLocalChannelReference(new LocalChannelReference());
288 			}
289 			taskRunner.getLocalChannelReference().setErrorMessage(
290 					ErrorCode.ConnectionImpossible.mesg,
291 					ErrorCode.ConnectionImpossible);
292 			this.taskRunner.setErrorTask(localChannelReference);
293 			this.taskRunner.run();
294 			throw new OpenR66ProtocolNoConnectionException(
295 					"End of retry on ServerOverloaded");
296 		}
297 	}

299 	/**
300 	 * Finish the transfer (called at the end of runTransfer)
301 	 * 
302 	 * @param retry
303 	 *            if True, it will retry in case of overloaded remote server, else it just stops
304 	 * @param localChannelReference
305 	 * @return The R66Future of the transfer operation
306 	 * @throws OpenR66ProtocolNotYetConnectionException
307 	 * @throws OpenR66ProtocolPacketException
308 	 * @throws OpenR66ProtocolNoConnectionException
309 	 * @throws OpenR66RunnerErrorException
310 	 */
311 	public R66Future finishTransfer(boolean retry,
312 			LocalChannelReference localChannelReference)
313 			throws OpenR66RunnerErrorException {
314 		if (this.localChannelReference == null) {
315 			this.localChannelReference = localChannelReference;
316 		}
317 		R66Future transfer = localChannelReference.getFutureRequest();
318 		try {
319 			transfer.await();
320 		} catch (InterruptedException e1) {
321 		}
322 		taskRunnerRetryHashMap.remove(taskRunner.getKey());
323 		logger.info("Request done with {}", (transfer.isSuccess() ? "success"
324 				: "error"));
325 		Channels.close(localChannelReference.getLocalChannel());
326 		// now reload TaskRunner if it still exists (light client can forget it)
327 		if (transfer.isSuccess()) {
328 			try {
329 				taskRunner.select();
330 			} catch (WaarpDatabaseException e) {
331 				logger.debug("Not a problem but cannot find at the end the task");
332 				taskRunner.setFrom(transfer.runner);
333 			}
334 			this.changeUpdatedInfo(UpdatedInfo.DONE, ErrorCode.CompleteOk);
335 		} else {
336 			try {
337 				taskRunner.select();
338 			} catch (WaarpDatabaseException e) {
339 				logger.debug("Not a problem but cannot find at the end the task");
340 				taskRunner.setFrom(transfer.runner);
341 			}
342 			// Case when we were interrupted
343 			if (transfer.getResult() == null) {
344 				switch (taskRunner.getUpdatedInfo()) {
345 					case DONE:
346 						R66Result ok = new R66Result(null, true,
347 								ErrorCode.CompleteOk, taskRunner);
348 						transfer.setResult(ok);
349 						transfer.setSuccess();
350 						this.changeUpdatedInfo(UpdatedInfo.DONE,
351 								ErrorCode.CompleteOk);
352 						break;
353 					case INERROR:
354 					case INTERRUPTED:
355 					default:
356 						R66Result error = new R66Result(null, true,
357 								ErrorCode.Internal, taskRunner);
358 						transfer.setResult(error);
359 						transfer.cancel();
360 						this.changeUpdatedInfo(UpdatedInfo.INERROR,
361 								ErrorCode.Internal);
362 				}
363 				return transfer;
364 			}
365 			if (transfer.getResult().code == ErrorCode.QueryAlreadyFinished) {
366 				// check if post task to execute
367 				logger.warn("WARN QueryAlreadyFinished:\n    " +
368 						transfer.toString() + "\n    " +
369 						taskRunner.toShortString());
370 				try {
371 					TransferUtils.finalizeTaskWithNoSession(taskRunner,
372 							localChannelReference);
373 				} catch (OpenR66RunnerErrorException e) {
374 					this.taskRunner.changeUpdatedInfo(UpdatedInfo.INERROR);
375 					try {
376 						this.taskRunner.update();
377 					} catch (WaarpDatabaseException e1) {
378 					}
379 				}
380 			} else {
381 				switch (taskRunner.getUpdatedInfo()) {
382 					case DONE:
383 					case INERROR:
384 					case INTERRUPTED:
385 						break;
386 					default:
387 						this.changeUpdatedInfo(UpdatedInfo.INERROR,
388 								transfer.getResult().code);
389 				}
390 			}
391 		}
392 		return transfer;
393 	}

395 	/**
396 	 * Initialize the request
397 	 * 
398 	 * @return the localChannelReference holding the transfer request
399 	 * @throws OpenR66ProtocolNoConnectionException
400 	 * @throws OpenR66RunnerErrorException
401 	 * @throws OpenR66ProtocolPacketException
402 	 * @throws OpenR66ProtocolNotYetConnectionException
403 	 */
404 	public LocalChannelReference initRequest()
405 			throws OpenR66ProtocolNoConnectionException,
406 			OpenR66RunnerErrorException, OpenR66ProtocolPacketException,
407 			OpenR66ProtocolNotYetConnectionException {
408 		this.changeUpdatedInfo(UpdatedInfo.RUNNING, ErrorCode.Running);
409 		long id = taskRunner.getSpecialId();
410 		String tid;
411 		if (id == DbConstant.ILLEGALVALUE) {
412 			tid = taskRunner.getRuleId() + "_" + taskRunner.getMode() +
413 					"_NEWTRANSFER";
414 		} else {
415 			tid = taskRunner.getRuleId() + "_" + taskRunner.getMode() + "_" +
416 					id;
417 		}
418 		Thread.currentThread().setName(tid);
419 		logger.debug("Will run {}", this.taskRunner);
420 		boolean restartPost = false;
421 		if (taskRunner.getGloballaststep() == TASKSTEP.POSTTASK.ordinal()) {
422 			// Send a validation to requested
423 			if (!taskRunner.isSelfRequested()) {
424 				// restart
425 				restartPost = true;
426 			}
427 		}
428 		if (taskRunner.isSelfRequested()) {
429 			// Don't have to restart a task for itself (or should use requester)
430 			logger.warn("Requested host cannot initiate itself the request");
431 			this.changeUpdatedInfo(UpdatedInfo.INERROR,
432 					ErrorCode.LoopSelfRequestedHost);
433 			throw new OpenR66ProtocolNoConnectionException(
434 					"Requested host cannot initiate itself the request");
435 		}
436 		DbHostAuth host = R66Auth.getServerAuth(DbConstant.admin.session,
437 				taskRunner.getRequested());
438 		if (host == null) {
439 			logger.error("Requested host cannot be found: " +
440 					taskRunner.getRequested());
441 			this.changeUpdatedInfo(UpdatedInfo.INERROR, ErrorCode.NotKnownHost);
442 			throw new OpenR66ProtocolNoConnectionException(
443 					"Requested host cannot be found " +
444 							taskRunner.getRequested());
445 		}
446 		if (host.isClient()) {
447 			logger.debug("Cannot initiate a connection with a client: {}", host);
448 			this.changeUpdatedInfo(UpdatedInfo.INERROR,
449 					ErrorCode.ConnectionImpossible);
450 			throw new OpenR66ProtocolNoConnectionException(
451 					"Cannot connect to client " + host.toString());
452 		}
453 		SocketAddress socketAddress = host.getSocketAddress();
454 		boolean isSSL = host.isSsl();
455 
456 		LocalChannelReference localChannelReference = networkTransaction
457 				.createConnectionWithRetry(socketAddress, isSSL, futureRequest);
458 		taskRunner.setLocalChannelReference(localChannelReference);
459 		socketAddress = null;
460 		if (localChannelReference == null) {
461 			// propose to redo
462 			// See if reprogramming is ok (not too many tries)
463 			String retry;
464 			if (incrementTaskRunerTry(taskRunner, Configuration.RETRYNB)) {
465 				logger.debug("Will retry since Cannot connect to {}", host);
466 				retry = " but will retry";
467 				// now wait
468 				try {
469 					Thread.sleep(Configuration.configuration.delayRetry);
470 				} catch (InterruptedException e) {
471 				}
472 				this.changeUpdatedInfo(UpdatedInfo.TOSUBMIT,
473 						ErrorCode.ConnectionImpossible);
474 				throw new OpenR66ProtocolNotYetConnectionException(
475 						"Cannot connect to server " + host.toString() + retry);
476 			} else {
477 				logger.debug(
478 						"Will not retry since limit of connection attemtps is reached for {}",
479 						host);
480 				retry = " and retries limit is reached so stop here";
481 				this.changeUpdatedInfo(UpdatedInfo.INERROR,
482 						ErrorCode.ConnectionImpossible);
483 				taskRunner
484 						.setLocalChannelReference(new LocalChannelReference());
485 				throw new OpenR66ProtocolNoConnectionException(
486 						"Cannot connect to server " + host.toString() + retry);
487 			}
488 		}
489 		if (handler != null) {
490 			localChannelReference.setRecvThroughHandler(handler);
491 		}
492 		localChannelReference.setSendThroughMode(isSendThroughMode);
493 		if (restartPost) {
494 			RequestPacket request = taskRunner.getRequest();
495 			logger.debug("Will send request {} ", request);
496 			localChannelReference.setClientRunner(this);
497 			localChannelReference.sessionNewState(R66FiniteDualStates.REQUESTR);
498 			try {
499 				ChannelUtils.writeAbstractLocalPacket(localChannelReference,
500 						request, false);
501 			} catch (OpenR66ProtocolPacketException e) {
502 				// propose to redo
503 				logger.warn("Cannot transfer request to " + host.toString());
504 				this.changeUpdatedInfo(UpdatedInfo.INTERRUPTED,
505 						ErrorCode.Internal);
506 				Channels.close(localChannelReference.getLocalChannel());
507 				localChannelReference = null;
508 				host = null;
509 				request = null;
510 				throw e;
511 			}
512 			logger.debug("Wait for request to {}", host);
513 			request = null;
514 			host = null;
515 			return localChannelReference;
516 		}
517 		// If Requester is NOT Sender, and if TransferTask then decrease now if
518 		// possible the rank
519 		if (!taskRunner.isSender() &&
520 				(taskRunner.getGloballaststep() == TASKSTEP.TRANSFERTASK
521 						.ordinal())) {
522 			logger.debug(
523 					"Requester is not Sender so decrease if possible the rank {}",
524 					taskRunner);
525 			taskRunner.restartRank();
526 			taskRunner.saveStatus();
527 			logger.debug(
528 					"Requester is not Sender so new rank is " +
529 							taskRunner.getRank() + " {}", taskRunner);
530 		}
531 		RequestPacket request = taskRunner.getRequest();
532 		logger.debug("Will send request {} {}", request, localChannelReference);
533 		localChannelReference.setClientRunner(this);
534 		localChannelReference.sessionNewState(R66FiniteDualStates.REQUESTR);
535 		try {
536 			ChannelUtils.writeAbstractLocalPacket(localChannelReference,
537 					request, false);
538 		} catch (OpenR66ProtocolPacketException e) {
539 			// propose to redo
540 			logger.warn("Cannot transfer request to " + host.toString());
541 			this.changeUpdatedInfo(UpdatedInfo.INTERRUPTED, ErrorCode.Internal);
542 			Channels.close(localChannelReference.getLocalChannel());
543 			localChannelReference = null;
544 			host = null;
545 			request = null;
546 			throw e;
547 		}
548 		logger.debug("Wait for request to {}", host);
549 		request = null;
550 		host = null;
551 		return localChannelReference;
552 	}
