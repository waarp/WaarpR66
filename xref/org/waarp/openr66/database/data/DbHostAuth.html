286 	 * @param adminrole
287 	 * @param isClient

289 	public DbHostAuth(DbSession dbSession, String hostid, String address, int port,
290 			boolean isSSL, byte[] hostkey, boolean adminrole, boolean isClient) {
291 		super(dbSession);
292 		this.hostid = hostid;
293 		this.address = address;
294 		this.port = port;
295 		this.isSsl = isSSL;
296 		if (hostkey == null) {
297 			this.hostkey = null;
298 		} else {
299 			try {
300 				// Save as crypted with the local Key and HEX
301 				this.hostkey = Configuration.configuration.cryptoKey.cryptToHex(hostkey).getBytes(WaarpStringUtils.UTF8);
302 			} catch (Exception e) {
303 				this.hostkey = new byte[0];
304 			}
305 		}
306 		this.adminrole = adminrole;
307 		this.isClient = isClient;
308 		if (this.port < 0) {
309 			this.isClient = true;
310 			this.address = DEFAULT_CLIENT_ADDRESS;
311 		}
312 		setToArray();
313 		isSaved = false;
314 	}

488 				dbSession);

492 					getWherePrimaryKey());

494 			setValues(preparedStatement, primaryKey);
495 			preparedStatement.executeQuery();
496 			if (preparedStatement.getNext()) {

548 	 * 
549 	 * @param dbSession
550 	 *            may be null
551 	 * @return the array of DbHostAuth
552 	 * @throws WaarpDatabaseNoConnectionException
553 	 * @throws WaarpDatabaseSqlException
554 	 */
555 	public static DbHostAuth[] getAllHosts(DbSession dbSession)
556 			throws WaarpDatabaseNoConnectionException, WaarpDatabaseSqlException {
557 		if (dbSession == null) {
558 			DbHostAuth[] result = new DbHostAuth[0];
559 			return dbR66HostAuthHashMap.values().toArray(result);
560 		}
561 		String request = "SELECT " + selectAllFields;
562 		request += " FROM " + table;
563 		DbPreparedStatement preparedStatement = new DbPreparedStatement(dbSession, request);
564 		ArrayList<DbHostAuth> dbArrayList = new ArrayList<DbHostAuth>();
565 		preparedStatement.executeQuery();
566 		while (preparedStatement.getNext()) {
567 			DbHostAuth hostAuth = getFromStatement(preparedStatement);
568 			dbArrayList.add(hostAuth);
569 		}
570 		preparedStatement.realClose();
571 		DbHostAuth[] result = new DbHostAuth[0];
572 		return dbArrayList.toArray(result);
573 	}

608 	/**
609 	 * 
610 	 * @param session
611 	 * @param host
612 	 * @param addr
613 	 * @param ssl
614 	 * @return the DbPreparedStatement according to the filter
615 	 * @throws WaarpDatabaseNoConnectionException
616 	 * @throws WaarpDatabaseSqlException
617 	 */
618 	public static DbPreparedStatement getFilterPrepareStament(DbSession session,
619 			String host, String addr, boolean ssl, boolean active)
620 			throws WaarpDatabaseNoConnectionException, WaarpDatabaseSqlException {
621 		DbPreparedStatement preparedStatement = new DbPreparedStatement(session);
622 		String request = "SELECT " + selectAllFields + " FROM " + table + " WHERE ";
623 		String condition = null;
624 		if (host != null) {
625 			condition = Columns.HOSTID.name() + " LIKE '%" + host + "%' ";
626 		}
627 		if (addr != null) {
628 			if (condition != null) {
629 				condition += " AND ";
630 			} else {
631 				condition = "";
632 			}
633 			condition += Columns.ADDRESS.name() + " LIKE '%" + addr + "%' ";
634 		}
635 		if (condition != null) {
636 			condition += " AND ";
637 		} else {
638 			condition = "";
639 		}
640 		condition += Columns.ISSSL.name() + " = ? AND ";
641 		condition += Columns.ISACTIVE.name() + " = ? ";
642 		preparedStatement.createPrepareStatement(request + condition +
643 				" ORDER BY " + Columns.HOSTID.name());
644 		try {
645 			preparedStatement.getPreparedStatement().setBoolean(1, ssl);
646 			preparedStatement.getPreparedStatement().setBoolean(2, active);
647 		} catch (SQLException e) {
648 			preparedStatement.realClose();
649 			throw new WaarpDatabaseSqlException(e);
650 		}
651 		return preparedStatement;
652 	}

809 		String remoteHost = host;
810 		String alias = "";
811 		if (Configuration.configuration.aliases.containsKey(remoteHost)) {
812 			remoteHost = Configuration.configuration.aliases.get(remoteHost);
813 			alias += "(Alias: "+remoteHost+") ";
814 		}
815 		if (Configuration.configuration.reverseAliases.containsKey(remoteHost)) {
816 			String alias2 = "(Also alias: ";
817 			String[] list = Configuration.configuration.reverseAliases.get(remoteHost);
818 			boolean found = false;
819 			for (String string : list) {
820 				if (string.equals(host)) {
821 					continue;
822 				}
823 				found = true;
824 				alias2 += string+" ";
825 			}
826 			if (found) {
827 				alias += alias2+") ";
828 			}
829 		}
830 		if (Configuration.configuration.businessWhiteSet.contains(remoteHost)) {
831 			alias += "(Business Allowed) ";
832 		}
833 		if (Configuration.configuration.roles.containsKey(remoteHost)) {
834 			RoleDefault item = Configuration.configuration.roles.get(remoteHost);
835 			alias += "(Role: "+item.toString()+") ";
836 		}
837 		return alias+(Configuration.configuration.versions.containsKey(remoteHost) ? 
838 				Configuration.configuration.versions.get(remoteHost).toString() :
839 					"Version Unknown");
840 	}
