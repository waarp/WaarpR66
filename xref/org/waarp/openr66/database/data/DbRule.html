284     protected final void checkPath() {
285         if (getRecvPath() != null && !getRecvPath().isEmpty()
286                 && getRecvPath().charAt(0) != DirInterface.SEPARATORCHAR) {
287             rule.setRecvPath(DirInterface.SEPARATOR + getRecvPath());
288         }
289         if (getSendPath() != null && !getSendPath().isEmpty()
290                 && getSendPath().charAt(0) != DirInterface.SEPARATORCHAR) {
291             rule.setSendPath(DirInterface.SEPARATOR + getSendPath());
292         }
293         if (getArchivePath() != null && !getArchivePath().isEmpty()
294                 && getArchivePath().charAt(0) != DirInterface.SEPARATORCHAR) {
295             rule.setArchivePath(DirInterface.SEPARATOR + getArchivePath());
296         }
297         if (getWorkPath() != null && !getWorkPath().isEmpty()
298                 && getWorkPath().charAt(0) != DirInterface.SEPARATORCHAR) {
299             rule.setWorkPath(DirInterface.SEPARATOR + getWorkPath());
300         }
301     }

503     public static DbRule[] deleteAll() throws WaarpDatabaseException {
504         RuleDAO ruleAccess = null;
505         List<Rule> rules = null;
506         try {
507             ruleAccess = DAOFactory.getInstance().getRuleDAO();
508             rules = ruleAccess.getAll();
509             ruleAccess.deleteAll();
510         } catch (DAOException e) {
511             throw new WaarpDatabaseException(e);
512         } finally {
513             if (ruleAccess != null) {
514                 ruleAccess.close();
515             }
516         }
517         DbRule[] res = new DbRule[rules.size()];
518         int i = 0;
519         for (Rule rule : rules) {
520             res[i] = new DbRule(rule);
521             i++;
522         }
523         return res;
524     }

642      * For instance from Commander when getting updated information
643      *
644      * @param preparedStatement
645      * @return the next updated DbRule
646      * @throws WaarpDatabaseNoConnectionException
647      * @throws WaarpDatabaseSqlException
648      */
649     public static DbRule getFromStatement(DbPreparedStatement preparedStatement)
650             throws WaarpDatabaseNoConnectionException, WaarpDatabaseSqlException {
651         DbRule dbRule = new DbRule();
652         dbRule.getValues(preparedStatement, dbRule.allFields);
653         dbRule.setFromArray();
654         dbRule.isSaved = true;
655         logger.debug("Get one Rule from Db: " + dbRule.getIdRule());
656         dbR66RuleHashMap.put(dbRule.getIdRule(), dbRule);
657         return dbRule;
658     }

698      * Get Ids from String. If it is not ok, then it sets the default values and return False, else
699      * returns True.
700      *
701      * @param idsref
702      * @return True if ok, else False (default values).
703      */
704     private String[] getIdsRule(String idsref) {
705         if (idsref == null) {
706             // No ids so setting to the default!
707             return new String[0];
708         }
709         StringReader reader = new StringReader(idsref);
710         Document document = null;
711         try {
712             document = new SAXReader().read(reader);
713             XmlValue[] values = XmlUtil.read(document,
714                     RuleFileBasedConfiguration.hostsDecls);
715             return RuleFileBasedConfiguration.getHostIds(values[0]);
716         } catch (DocumentException e) {
717             logger.warn("Unable to read the ids for Rule: " + idsref, e);
718             // No ids so setting to the default!
719             return new String[0];
720         } finally {
721             reader.close();
722         }
723     }

726      * Get Tasks from String. If it is not ok, then it sets the default values and return new array
727      * of Tasks or null if in error.
728      *
729      * @param tasks
730      * @return Array of tasks or empty array if in error.
731      */
732     private String[][] getTasksRule(String tasks) {
733         if (tasks == null) {
734             // No tasks so setting to the default!
735             return new String[0][0];
736         }
737         StringReader reader = new StringReader(tasks);
738         Document document = null;
739         try {
740             document = new SAXReader().read(reader);

974     public static DbPreparedStatement getFilterPrepareStament(DbSession session,
975             String rule, int mode)
976             throws WaarpDatabaseNoConnectionException, WaarpDatabaseSqlException {
977         DbPreparedStatement preparedStatement = new DbPreparedStatement(session);
978         String request = "SELECT " + selectAllFields + " FROM " + table;
979         String condition = null;
980         if (rule != null) {
981             condition = " WHERE " + Columns.IDRULE.name() + " LIKE '%" + rule + "%' ";
982         }
983         if (mode >= 0) {
984             if (condition != null) {
985                 condition += " AND ";
986             } else {
987                 condition = " WHERE ";
988             }
989             condition += Columns.MODETRANS.name() + " = ?";
990         } else {
991             condition = "";
992         }
993         preparedStatement.createPrepareStatement(request + condition +
994                 " ORDER BY " + Columns.IDRULE.name());
995         if (mode >= 0) {
996             try {
997                 preparedStatement.getPreparedStatement().setInt(1, mode);
998             } catch (SQLException e) {
999                 preparedStatement.realClose();
1000                 throw new WaarpDatabaseSqlException(e);
1001             }
1002         }
1003         return preparedStatement;
1004     }

1007      * Write selected DbRule to a Json String
1008      *
1009      * @param preparedStatement
1010      * @return the associated Json String
1011      * @throws WaarpDatabaseNoConnectionException
1012      * @throws WaarpDatabaseSqlException
1013      * @throws OpenR66ProtocolBusinessException
1014      */
1015     public static String getJson(DbPreparedStatement preparedStatement, int limit)
1016             throws WaarpDatabaseNoConnectionException, WaarpDatabaseSqlException,
1017             OpenR66ProtocolBusinessException {
1018         ArrayNode arrayNode = JsonHandler.createArrayNode();
1019         try {
1020             preparedStatement.executeQuery();
1021             int nb = 0;
1022             while (preparedStatement.getNext()) {
1023                 DbRule rule = DbRule.getFromStatement(preparedStatement);
1024                 ObjectNode node = rule.getInternalJson();
1025                 arrayNode.add(node);
1026                 nb++;
1027                 if (nb >= limit) {
1028                     break;
1029                 }
1030             }
1031         } finally {
1032             preparedStatement.realClose();
1033         }
1034         // \n is not correctly parsed within HTML so put double \\n in fine
1035         return WaarpStringUtils.cleanJsonForHtml(arrayNode.toString());
1036     }
