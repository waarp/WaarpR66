274 	@Override
275 	public void select() throws WaarpDatabaseException {
276 		if (dbSession == null) {
277 			DbMultipleMonitor conf = dbR66MMHashMap.get(this.hostid);
278 			if (conf == null) {
279 				throw new WaarpDatabaseNoDataException("No row found");
280 			} else {
281 				// copy info
282 				for (int i = 0; i < allFields.length; i++) {
283 					allFields[i].value = conf.allFields[i].value;
284 				}
285 				setFromArray();
286 				isSaved = true;
287 				return;
288 			}
289 		}
290 		super.select();
291 	}
