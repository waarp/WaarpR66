261     /**
262      * For instance from Commander when getting updated information
263      * 
264      * @param preparedStatement
265      * @return the next updated Configuration
266      * @throws WaarpDatabaseNoConnectionException
267      * @throws WaarpDatabaseSqlException
268      */
269     public static DbMultipleMonitor getFromStatement(DbPreparedStatement preparedStatement)
270             throws WaarpDatabaseNoConnectionException, WaarpDatabaseSqlException {
271         DbMultipleMonitor dbMm = new DbMultipleMonitor();
272         dbMm.getValues(preparedStatement, dbMm.allFields);
273         dbMm.setFromArray();
274         dbMm.isSaved = true;
275         return dbMm;
276     }

278     /**
279      * 
280      * @return the DbPreparedStatement for getting Updated Object in "FOR UPDATE" mode
281      * @throws WaarpDatabaseNoConnectionException
282      * @throws WaarpDatabaseSqlException
283      */
284     public static DbPreparedStatement getUpdatedPrepareStament(DbSession session)
285             throws WaarpDatabaseNoConnectionException, WaarpDatabaseSqlException {
286         DbMultipleMonitor multipleMonitor = new DbMultipleMonitor(
287                 Configuration.configuration.getHOST_ID(), 0, 0, 0);
288         try {
289             if (!multipleMonitor.exist()) {
290                 multipleMonitor.insert();
291                 session.commit();
292             }
293         } catch (WaarpDatabaseException e1) {
294         }
295         String request = "SELECT " + selectAllFields;
296         request += " FROM " + table + " WHERE " + Columns.HOSTID.name() + " = '"
297                 + Configuration.configuration.getHOST_ID() + "'" +
298                 " FOR UPDATE ";
299         DbPreparedStatement prep = new DbPreparedStatement(session, request);
300         return prep;
301     }

303     /**
304      * On Commander side
305      * 
306      * @return True if this is the last update
307      */
308     public boolean checkUpdateConfig() {
309         if (getCountConfig() <= 0) {
310             setCountConfig(Configuration.configuration.getMultipleMonitors());
311             setCountConfig(getCountConfig() - 1);
312             this.isSaved = false;
313         } else {
314             setCountConfig(getCountConfig() - 1);
315             this.isSaved = false;
316         }
317         return this.getCountConfig() <= 0;
318     }
