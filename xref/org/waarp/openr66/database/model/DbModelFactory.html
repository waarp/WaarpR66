18  package org.waarp.openr66.database.model;
19  
20  import org.waarp.common.database.DbAdmin;
21  import org.waarp.common.database.exception.WaarpDatabaseNoConnectionException;
22  import org.waarp.common.database.model.DbModel;
23  import org.waarp.common.database.model.DbType;
24  
25  /**
26   * Factory to store the Database Model object
27   * 
28   * @author Frederic Bregier
29   * 
30   */
31  public class DbModelFactory extends org.waarp.common.database.model.DbModelFactory {
32  
33      /**
34       * Initialize the Database Model according to arguments.
35       * 
36       * @param dbdriver
37       * @param dbserver
38       * @param dbuser
39       * @param dbpasswd
40       * @param write
41       * @throws WaarpDatabaseNoConnectionException
42       */
43      public static DbAdmin initialize(String dbdriver, String dbserver,
44              String dbuser, String dbpasswd, boolean write)
45              throws WaarpDatabaseNoConnectionException {
46          DbType type = DbType.getFromDriver(dbdriver);
47          DbModel dbModel;
48          switch (type) {
49              case H2:
50                  dbModel = new DbModelH2(dbserver, dbuser, dbpasswd);
51                  break;
52              case Oracle:
53                  dbModel = new DbModelOracle(dbserver, dbuser, dbpasswd);
54                  break;
55              case PostGreSQL:
56                  dbModel = new DbModelPostgresql();
57                  break;
58              case MySQL:
59                  dbModel = new DbModelMysql(dbserver, dbuser, dbpasswd);
60                  break;
61              case MariaDB:
62                  dbModel = new DbModelMariadb(dbserver, dbuser, dbpasswd);
63                  break;
64              default:
65                  throw new WaarpDatabaseNoConnectionException(
66                          "TypeDriver unknown: " + type);
67          }
68          dbModels.add(dbModel);
69          return new DbAdmin(dbModel, dbserver, dbuser, dbpasswd,
70                  write);
71      }
72  
73  }
