297     @Override
298     public synchronized long nextSequence(DbSession dbSession)
299         throws WaarpDatabaseNoConnectionException,
300             WaarpDatabaseSqlException, WaarpDatabaseNoDataException {
301         lock.lock();
302         try {
303             long result = DbConstant.ILLEGALVALUE;
304             String action = "SELECT seq FROM Sequences WHERE name = '" +
305                 DbTaskRunner.fieldseq + "' FOR UPDATE";
306             DbPreparedStatement preparedStatement = new DbPreparedStatement(
307                     dbSession);
308             try {
309                 dbSession.conn.setAutoCommit(false);
310             } catch (SQLException e1) {
311             }
312             try {
313                 preparedStatement.createPrepareStatement(action);
314                 // Limit the search
315                 preparedStatement.executeQuery();
316                 if (preparedStatement.getNext()) {
317                     try {
318                         result = preparedStatement.getResultSet().getLong(1);
319                     } catch (SQLException e) {
320                         throw new WaarpDatabaseSqlException(e);
321                     }
322                 } else {
323                     throw new WaarpDatabaseNoDataException(
324                             "No sequence found. Must be initialized first");
325                 }
326             } finally {
327                 preparedStatement.realClose();
328             }
329             action = "UPDATE Sequences SET seq = "+(result+1)+
330                 " WHERE name = '"+DbTaskRunner.fieldseq+"'";
331             try {
332                 preparedStatement.createPrepareStatement(action);
333                 // Limit the search
334                 preparedStatement.executeUpdate();
335             } finally {
336                 preparedStatement.realClose();
337             }
338             return result;
339         } finally {
340             try {
341                 dbSession.conn.setAutoCommit(true);
342             } catch (SQLException e1) {
343             }
344             lock.unlock();
345         }
346     }
