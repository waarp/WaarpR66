261     @Override
262     public long nextSequence(DbSession dbSession)
263         throws WaarpDatabaseNoConnectionException,
264             WaarpDatabaseSqlException, WaarpDatabaseNoDataException {
265         long result = DbConstant.ILLEGALVALUE;
266         String action = "SELECT NEXTVAL('" + DbTaskRunner.fieldseq + "')";
267         DbPreparedStatement preparedStatement = new DbPreparedStatement(
268                 dbSession);
269         try {
270             preparedStatement.createPrepareStatement(action);
271             // Limit the search
272             preparedStatement.executeQuery();
273             if (preparedStatement.getNext()) {
274                 try {
275                     result = preparedStatement.getResultSet().getLong(1);
276                 } catch (SQLException e) {
277                     throw new WaarpDatabaseSqlException(e);
278                 }
279                 return result;
280             } else {
281                 throw new WaarpDatabaseNoDataException(
282                         "No sequence found. Must be initialized first");
283             }
284         } finally {
285             preparedStatement.realClose();
286         }
