14  public class DBTransferMariaDBDAOIT extends DBTransferDAOIT {
15  
16      private String createScript = "mysql/create.sql";
17      private String populateScript = "mysql/populate.sql";
18      private String cleanScript = "mysql/clean.sql";
19  
20      @ClassRule
21      public static MariaDBContainer db = new MariaDBContainer();
22  
23      @Override
24      public DBTransferDAO getDAO(Connection con) throws DAOException {
25          return new MariaDBTransferDAO(con);
26      }
27  
28  
29      @Override
30      public Connection getConnection() throws SQLException {
31          return DriverManager.getConnection(
32                  db.getJdbcUrl(),
33                  db.getUsername(),
34                  db.getPassword());
35      }
36  
37      @Override
38      public void initDB() {
39          runScript(createScript);
40          runScript(populateScript);
41      }
42  
43      @Override
44      public void cleanDB() {
45          runScript(cleanScript);
46      }
47  }
