29      public void runScript(String script) {
30          try {
31              ScriptRunner runner = new ScriptRunner(con, false, true);
32              URL url = Thread.currentThread().getContextClassLoader().getResource(script);
33              runner.runScript(new BufferedReader(new FileReader(url.getPath())));
34          } catch (Exception e) {
35              fail(e.getMessage());
36          }
