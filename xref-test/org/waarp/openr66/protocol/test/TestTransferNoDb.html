21  import java.util.concurrent.Executors;

23  import org.waarp.common.command.exception.CommandAbstractException;
24  import org.waarp.common.logging.WaarpLoggerFactory;

28  import org.waarp.openr66.context.R66Result;
29  import org.waarp.openr66.database.DbConstant;
30  import org.waarp.openr66.protocol.configuration.Configuration;
31  import org.waarp.openr66.protocol.networkhandler.NetworkTransaction;
32  import org.waarp.openr66.protocol.utils.R66Future;
33  
34  /**
35   * Test class for multiple DirectTransfer
36   * 
37   * @author Frederic Bregier
38   * 
39   */
40  public class TestTransferNoDb extends DirectTransfer {
41      static int nb = 100;
42  
43      /**
44       * @param args
45       * @param rank
46       * @return True if OK
47       */
48      protected static boolean getSpecialParams(String[] args, int rank) {
49          for (int i = rank; i < args.length; i++) {
50              if (args[i].equalsIgnoreCase("-nb")) {
51                  i++;
52                  nb = Integer.parseInt(args[i]);
53              } else if (args[i].equalsIgnoreCase("-md5")) {
54              } else if (args[i].charAt(0) == '-') {
55                  i++;// jump one
56              }
57          }
58          return true;
59      }
60  
61      public TestTransferNoDb(R66Future future, String remoteHost,
62              String filename, String rulename, String fileinfo, boolean isMD5, int blocksize,
63              long id,
64              NetworkTransaction networkTransaction) {
65          super(future, remoteHost, filename, rulename, fileinfo, isMD5, blocksize, id,
66                  networkTransaction);
67      }
68  
69      public static void main(String[] args) {
70          WaarpLoggerFactory.setDefaultFactory(new WaarpSlf4JLoggerFactory(
71                  null));
72          if (logger == null) {
73              logger = WaarpLoggerFactory.getLogger(DirectTransfer.class);
74          }
75          if (!getParams(args, false)) {
76              logger.error("Wrong initialization");
77              if (DbConstant.admin != null && DbConstant.admin.isActive) {
78                  DbConstant.admin.close();
79              }
80              System.exit(1);
81          }
82          getSpecialParams(args, 1);
83          Configuration.configuration.CLIENT_THREAD = nb;
84          Configuration.configuration.pipelineInit();
85          NetworkTransaction networkTransaction = new NetworkTransaction();
86          try {
87              ExecutorService executorService = Executors.newCachedThreadPool();
88  
89              R66Future[] arrayFuture = new R66Future[nb];
90              logger.info("Start of Test Transfer");
91              long time1 = System.currentTimeMillis();
92              for (int i = 0; i < nb; i++) {
93                  arrayFuture[i] = new R66Future(true);
94                  TestTransferNoDb transaction = new TestTransferNoDb(arrayFuture[i],
95                          rhost, localFilename, rule, fileInfo, ismd5, block,
96                          DbConstant.ILLEGALVALUE,
97                          networkTransaction);
98                  transaction.normalInfoAsWarn = snormalInfoAsWarn;
99                  executorService.execute(transaction);
100                 try {
101                     Thread.sleep(10);
102                 } catch (InterruptedException e) {
103                 }
104             }
105             int success = 0;
106             int error = 0;
107             int warn = 0;
108             for (int i = 0; i < nb; i++) {
109                 arrayFuture[i].awaitUninterruptibly();
110                 R66Result result = arrayFuture[i].getResult();
111                 if (arrayFuture[i].isSuccess()) {
112                     if (result.runner.getErrorInfo() == ErrorCode.Warning) {
113                         warn++;
114                     } else {
115                         success++;
116                     }
117                 } else {
118                     if (result.runner != null &&
119                             result.runner.getErrorInfo() == ErrorCode.Warning) {
120                         warn++;
121                     } else {
122                         error++;
123                     }
124                 }
125             }
126             long time2 = System.currentTimeMillis();
127             long length = 0;
128             // Get the first result as testing only
129             R66Result result = arrayFuture[0].getResult();
130             logger.warn("Final file: " +
131                     (result.file != null ? result.file.toString() : "no file"));
132             try {
133                 length = result.file != null ? result.file.length() : 0L;
134             } catch (CommandAbstractException e) {
135             }
136             long delay = time2 - time1;
137             float nbs = success * 1000;
138             nbs /= delay;
139             float mbs = nbs * length / 1024;
140             logger.warn("Success: " + success + " Warning: " + warn + " Error: " +
141                     error + " delay: " + delay + " NB/s: " + nbs + " KB/s: " + mbs);
142             executorService.shutdown();
143         } finally {
144             networkTransaction.closeAll();
145         }
146     }
