430     public static void launchThreads() {
431         // init thread model
432         ExecutorService pool = Executors.newFixedThreadPool(NB);
433         HttpTestRestR66Client[] clients = new HttpTestRestR66Client[NB];
434         for (int i = 0; i < NB; i++) {
435             clients[i] = new HttpTestRestR66Client();
436         }
437         for (int i = 0; i < NB; i++) {
438             pool.execute(clients[i]);
439         }
440         pool.shutdown();
441         try {
442             while (!pool.awaitTermination(100000, TimeUnit.SECONDS))
443                 ;
444         } catch (InterruptedException e) {
445         }
446     }
